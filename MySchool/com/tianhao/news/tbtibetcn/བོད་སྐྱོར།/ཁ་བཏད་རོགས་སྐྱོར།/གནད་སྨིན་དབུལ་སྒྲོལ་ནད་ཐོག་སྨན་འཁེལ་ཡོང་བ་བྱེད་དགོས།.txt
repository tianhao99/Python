གནད་སྨིན་དབུལ་སྒྲོལ་ནད་ཐོག་སྨན་འཁེལ་ཡོང་བ་བྱེད་དགོས།
ཐོན་ལས་འཕེལ་རྒྱས་གཏོང་རྒྱུ་ནི་དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་གཞི་རྩའི་ཐབས་ཇུས་ཤིག་ཡིན་པ་དང་། ཐོན་ལས་སྐྱེད་སྲིང་བྱ་རྒྱུ་ནི་དབུལ་སྒྲོལ་འགག་སྒྲོལ་ལ་སྐུལ་འདེད་གཏོང་བའི་རྩ་བའི་ཐབས་ལམ་ཞིག་རེད། 
          འོན་ཀྱང་ཐོན་ལས་གང་འདྲ་ཞིག་འཕེལ་རྒྱས་གཏོང་དགོས་པ་དང་། ཇི་ལྟར་བྱས་ནས་ཐོན་ལས་འཕེལ་རྒྱས་གཏོང་དགོས་པ། ཇི་ལྟར་བྱས་ནས་ཐོན་ལས་ལ་བརྟེན་ནས་མང་ཚོགས་དབུལ་སྒྲོལ་ཕྱུག་འགྱུར་ཐུབ་རྒྱུར་རྒྱུན་མཐུད་སྣེ་ཁྲིད་བྱེད་པ་སོགས་ཀྱི་གནད་དོན་རབ་དང་རིམ་པ་འཕྲད་སྐབས། ཉམས་མྱོང་མེད་པ་དང་དོན་དངོས་ཀྱི་དཔྱད་གཞི་གང་ཡང་མེད་པའི་མཛོ་སྒང་རྫོང་གི་ངོས་ནས་བརྗོད་ན་བྱ་དཀའ་བྱེད་གཏད་དང་ཁ་ཚ་དགོས་གཏུགས་ཀྱི་གནད་དོན་ཞིག་ཏུ་གྱུར་ཡོད། དབུལ་སྒྲོལ་འགག་སྒྲོལ་བྱེད་པར་གནད་སྨིན་ངང་དབུལ་པོ་ཆགས་རྐྱེན་འཚོལ་ཐུབ་པ་བྱས་ཏེ་ཆེས་དོན་དངོས་ཀྱི་གནད་དོན་ཐག་གཅོད་བྱེད་དགོས། དཔེར་ན། ཁྲོམ་རའི་བྱེད་པོ་རྗེས་ལུས་དང་། མང་ཚོགས་ཀྱི་འདུ་ཤེས་རྙིང་ཞེན་བག་འཁུམ་དང་རྗེས་ལུས། ཁྲོམ་རའི་དཔལ་འབྱོར་གྱི་འདུ་ཤེས་ཟབ་མོ་མེད་པ། རང་ངོས་འཕེལ་རྒྱས་གཏོང་ནུས་ཞན་པ་བཅས་དང་། རང་བྱུང་ཆ་རྐྱེན་གྱི་ཚོད་འཛིན་ཕོག་པ་དང་། འཕེལ་རྒྱས་གཏོང་སྟངས་རགས་ལས། ཐོན་ལས་ཀྱི་གྲུབ་ཆ་རྐྱང་པ། ནང་སྐྱེས་སྒུལ་ཤུགས་མི་འདང་བ་སོགས་ཡིན། 
          “ས་ཁུལ་དབུལ་པོས་དོན་དངོས་ནས་འགོ་འཛུགས་པ་དང་། ཡུལ་བབ་དང་བསྟུན་པ། གང་ཞིག་འདེབས་འཛུགས་བྱེད་པ་དང་། གང་ཞིག་གསོ་ཚགས་བྱེད་པ། གང་ནས་འབབ་འཕར་ཡོང་བ་བྱེད་པ་བཅས་ཀྱི་ཐད་བསམ་བློ་ཡག་པོ་གཏོང་དགོས་པའི”སྐོར་གྱི་སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིན་ཕིང་གི་མཛུབ་སྟོན་གལ་ཆེན་ལས་མཛོ་སྒང་རྫོང་གི་ཐབས་ཇུས་ལག་བསྟར་མཁན་ཚོར་ལམ་སྟོན་གསལ་བཟོ་གནང་ཡོད། 
            མཛོ་སྒང་རྫོང་གིས་རྫོང་དེ་ཡོངས་ཀྱི་གྲོང་རྡལ་3དང་ཤང་7གྱི་གྲོང་ཚོ(གྲོང་ལྷན)128ཀྱི་རང་བྱུང་གྲོང་ཚོ་342ཀྱི་ཐོན་ལས་འཕེལ་རྒྱས་ཀྱི་ལེགས་ལྡན་ཐོན་ཁུངས་ལ་དབྱེ་ཞིབ་གཏིང་ཟབ་བྱས་པ་དང་། གནད་སྨིན་དབུལ་སྐྱོར་དང་གནད་སྨིན་དབུལ་སྒྲོལ་གྱི་ཀུན་ཕན་སྲིད་ཇུས་ལ་དམ་པོར་དམིགས་ནས། G214དང་། རྒྱལ་ལམ་318 ཞིང་ཆེན་ལམ་བུ་203བཅས་ས་ཁོངས་ཡོངས་སུ་སྦྲེལ་མཐུད་ཐུབ་པ་དང་། ཧྥུའུ་ཅན་ཞིང་ཆེན་གྱི་ཁ་གཏད་རོགས་སྐྱོར་སོགས་ཀྱི་ཕན་ཡོད་ཆ་རྐྱེན་ཁ་གསལ་དུ་བཏང་བ་དང་། ཐོན་ལས་ཀྱི་འཆར་འགོད་དང་། བཀོད་པ། འཛུགས་སྐྲུན། དོ་དམ་སོགས་ཀྱི་འགག་རྩའི་ལྷུ་ཚིགས་ལ་ཤུགས་སྣོན་ཏན་ཏིག་བརྒྱབ་སྟེ“ཀ་རྟེན་གྱི་ཐོན་ལས་ཆེན་པོ་ལྔ”སྐྱེད་སྲིང་བྱས་ནས་དབུལ་སྒྲོལ་ཡོང་བར་ཁྱབ་འགྱེད་དང་སྣེ་ཁྲིད་ཀྱི་ནུས་པ་ལྡན་པའི་གཞི་ཁྱོན་ཅན་དང་དེང་རབས་ཅན་གྱི་འདེབས་འཛུགས་གསོ་ཚགས་རྟེན་གཞི་ཁག་གཅིག་བཏོད་པ་མ་ཟད། ཐོན་ཁུངས་རྣམ་པའི་ལས་སྣོན་ཁེ་ལས་ཁག་གཅིག་སྐྱེད་སྲིང་དང་ཡུལ་སྐོར་ལ་བརྟེན་པའི་གྲོང་གསེབ་ཀྱི་ཐོན་ལས་ཁག་གཅིག་འཕེལ་རྒྱས་བཏང་ཡོད། 
            དེ་ཐོག་ནས་དབུལ་སྒྲོལ་འགག་སྒྲོལ་ནི་ཡ་ཡུག་གི་བྱ་བ་ཞིག་མིན་པར་དེ་ནི་ཡུལ་བབ་དང་བསྟུན་པ་དང་། ས་གནས་དེ་གར་འཚམ་པའི་བྱ་ཐབས་དང་བྱེད་སྟངས་གནད་ལ་འཁེལ་བ་ཞིག་བཙལ་འཚོལ་གྱིས་ས་གནས་དེ་གའི་འཕེལ་རྒྱས་ལ་སྐུལ་ཁྲིད་ནུས་ལྡན་བྱས་པས་དབུལ་པོའི་གནད་དོན་རྩ་བའི་ཆ་ནས་ཐག་གཅོད་ཐུབ་པ་མཚོན་ཡོད། ཤང་དང་གྲོང་ཚོ་སོ་སོར་རང་སོ་སོའི་དམིགས་བསལ་གྱི་ལེགས་ཆ་ཡོད་པ་དེ་དག་ནི་ཤང་གཅིག་དང་གྲོང་ཚོ་གཅིག་དབུལ་སྒྲོལ་ཕྱུག་འགྱུར་ཐུབ་པའི་མི་མངོན་པའི་སྟོབས་ཤུགས་གནས་ས་ཞིག་ཀྱང་རེད། 
           མཛོ་སྒང་རྫོང་གིས་ཕྱོགས་གཅིག་ནས་རྩེ་རིམ་ཇུས་འགོད་ཡག་པོ་བྱས་པ་དང་། ཕྱོགས་གཞན་ཞིག་ནས་མང་ཚོགས་དང་འབྲེལ་འདྲིས་དམ་ཟབ་ཀྱིས་དངོས་འཛིན་དངོས་སྒྲུབ་ཀྱི་བྱ་ཐབས་སྤྱད་པ་དེ་ཐོག་ནས་མང་ཚོགས་དབུལ་པོའི་ཧུར་སེམས་དང་། རང་འགུལ། ཞུགས་སེམས་སོགས་སྐུལ་སློང་གང་ལེགས་བྱས་པ་མ་ཟད། གཞི་རིམ་ལས་བྱེད་པས་ཀྱང་ལག་ལེན་ཁྲོད་མང་ཚོགས་ཀྱི་སྣེ་ཁྲིད་དེ་དབུལ་སྒྲོལ་ཕྱུག་འགྱུར་ཡོང་བའི་འཇོན་ཐང་ཆེ་རུ་དང་ཆུ་ཚད་མཐོ་རུ་བཏང་ནས་རང་ངོས་ཀྱི་མི་མངོན་པའི་སྟོབས་ཤུགས་སྔོག་འདོན་དང་། ཐོན་ཁུངས་ཤིག་གསར་སྤེལ་དང་། ཕྱོགས་གཅིག་གི་དཔལ་འབྱོར་ལ་སྣེ་ཁྲིད། ཕྱོགས་གཅིག་གི་མི་སེར་ཕྱུག་འགྱུར་བཅས་ཀྱི་དམིགས་ཚད་མངོན་འགྱུར་ཐུབ་ཡོད། 
རྩོམ་སྒྲིག་པ། བདེ་སྒྲོལ།
