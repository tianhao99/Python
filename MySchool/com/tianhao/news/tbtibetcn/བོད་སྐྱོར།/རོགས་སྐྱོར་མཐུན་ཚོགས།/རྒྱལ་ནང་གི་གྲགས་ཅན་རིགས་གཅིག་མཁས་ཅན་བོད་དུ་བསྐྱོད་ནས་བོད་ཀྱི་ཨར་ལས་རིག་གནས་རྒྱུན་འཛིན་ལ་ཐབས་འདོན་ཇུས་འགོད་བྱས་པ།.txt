རྒྱལ་ནང་གི་གྲགས་ཅན་རིགས་གཅིག་མཁས་ཅན་བོད་དུ་བསྐྱོད་ནས་བོད་ཀྱི་ཨར་ལས་རིག་གནས་རྒྱུན་འཛིན་ལ་ཐབས་འདོན་ཇུས་འགོད་བྱས་པ།
བོད་ལྗོངས་ཚོང་འཕྲིན་ཐོག་གནས་ཚུལ་སྤེལ་བ་གཞིར་བཟུང་། ཉེ་ཆར་སྡོད་ཁང་དང་གྲོང་ཁྱེར་དང་གྲོང་གསེབ་འཛུགས་སྐྲུན་པུས་ལྷ་སར་ཚད་གཞི་མཐོ་བའི་བོད་ཀྱི་བཟོ་སྐྲུན་རིག་གནས་བགྲོ་གླེང་ཚོགས་འདུ་དང་བོད་ཀྱི་སྤྱིའི་འཆར་འགོད་སྒྲིག་བཟོའི་ལས་དོན་བཞུགས་མོལ་ཚོགས་འདུ་ཚོགས་པ་རེད། སྡོད་ཁང་དང་གྲོང་ཁྱེར་དང་གྲོང་གསེབ་འཛུགས་སྐྲུན་པུའི་པུའུ་ཀྲང་གཞོན་པ་ཧོང་ཡན་དང་། ཀྲུང་གོ་འདུ་འགོད་ཁང་གི་ཡོན་ཧྲི་ཚུའེ་ཁའེ་དང་མོང་ཅན་མིན། ཀྲུང་གོ་གྲོང་ཁྱེར་འཆར་འགོད་ཇུས་འགོད་ཁང་གི་ཡོན་ཀྲང་དབྱང་པའོ་ཅུན་སོགས་རྒྱལ་ཡོངས་ཀྱི་བཟོ་སྐྲུན་ཁྱབ་ཁོངས་ཀྱི་སློབ་དཔོན་དང་། རིགས་གཅིག་མཁས་ཅན། མཁས་དབང་བཅས་10ལྷག་ཙམ་སླེབས་པ་དང་འབྲེལ། བོད་ཀྱི་ཨར་ལས་རིག་གནས་རྒྱུན་འཛིན་དང་བོད་ཀྱི་སྤྱིའི་འཆར་འགོད་སྒྲིག་བཟོའི་ཐད་ཐབས་འདོན་ཇུས་འགོད་བྱས་པ་རེད།
          བོད་ཀྱི་ཨར་ལས་རིག་གནས་ཀྱི་རྨང་གཞི་ཟབ་པ་དང་། ཨར་ལས་ཁྱད་ཆོས་ཐུན་མོང་མ་ཡིན་པ། ཨར་ལས་སྒྱུ་རྩལ་རིན་ཐང་ཤིན་ཏུ་མཐོ་བ་བཅས་ཡོད། བོད་ཀྱི་ཨར་ལས་རིག་གནས་བགྲོ་གླེང་ཚོགས་འདུའི་ཐོག་རང་སྐྱོང་ལྗོངས་སྡོད་ཁང་དང་གྲོང་ཁྱེར་དང་གྲོང་གསེབ་འཛུགས་སྐྲུན་ཐིང་གི་འབྲེལ་ཡོད་འགན་འཁུར་བས་བོད་ཀྱི་ལོ་རྒྱུས་རིག་གནས་ཤུལ་བཞག་སྲུང་སྐྱོབ་དང་ཨར་ལས་འཆར་འགོད་དོ་དམ་གྱི་གནས་ཚུལ་མདོར་བསྡུས་ངོ་སྤྲོད་བྱས་པ་དང་། ལྗོངས་ཨར་ལས་རྩད་ཞིབ་འཆར་འགོད་ཁང་གི་འབྲེལ་ཡོད་འགན་འཁུར་བས་བོད་ཀྱི་ཨར་ལས་ཁྱད་ཆོས་ངོ་སྤྲོད་བྱས་པ། ལྷ་ས་གྲོང་ཁྱེར་གྱི་ཚོགས་ཞུགས་མི་སྣས་གྲོང་ཁུལ་རྙིང་པའི་སྲུང་སྐྱོབ་ལས་དོན་ངོ་སྤྲོད་བྱས་པ། ཉིང་ཁྲི་གྲོང་ཁྱེར་གྱིས་སྨན་གླིང་རྫོང་གི་གྲོང་རྡལ་ཆུང་གྲས་སྒྱུར་བཀོད་བྱས་ཚུལ་ངོ་སྤྲོད་བཅས་བྱས་པ་རེད། ཀྲུང་གོ་འདུ་འགོད་ཁང་གི་ཡོན་ཧྲི་ཚུའེ་ཁའེ་དང་། མོང་ཅན་མིན། དེ་བཞིན་རྒྱལ་ནང་དང་ཡུལ་དེ་གའི་ཨར་ལས་ཁྱབ་ཁོངས་ཀྱི་རིགས་གཅིག་མཁས་ཅན་གྱིས་སྔ་རྗེས་སུ་རང་སོ་སོའི་ཆེད་ལས་ཀྱི་ཐད་ནས་བོད་ཀྱི་ཨར་ལས་རིག་གནས་རྒྱུན་འཛིན་དང་། གྲོང་ཁྱེར་གྱི་རྣམ་པ་གསར་པ་གཏོད་པ། ཨར་ལས་འཆར་འགོད་སོགས་ལ་དམིགས་ཏེ་དཔྱད་བསྡུར་ཞིབ་ཚགས་བྱས་པ་དང་འབྲེལ། བོད་ཀྱི་འཐབ་ཇུས་གནས་བབ་གལ་ཆེན་ཐོག་ནས་འགོ་ཚུགས་པ་དང་བོད་ཀྱི་དཔལ་འབྱོར་དང་སྤྱི་ཚོགས་འཕེལ་རྒྱས་ཀྱི་དོན་དངོས་དང་ཟུང་འབྲེལ་བྱས་ནས་བོད་ཀྱི་ཨར་ལས་རིག་གནས་རྒྱུན་འཛིན་དང་། གྲོང་ཁྱེར་གྱི་རྣམ་པ་གསར་གཏོད། བཟོ་སྐྲུན་འཆར་འགོད་བཅས་ལ་ཤུགས་སྣོན་རྒྱག་པའི་ཐད་གནད་ལ་འཕིགས་པའི་ལྟ་ཚུལ་གླེང་བ་དང་། རང་སྐྱོང་ལྗོངས་རིམ་པས་གྲོང་ཁྱེར་འཛུགས་སྐྲུན་ལ་ཤུགས་རྒྱག་པའི་རྩེ་རིམ་ཇུས་འགོད་དང་ལམ་ལུགས་ཚོད་འཛིན་བཏོན་ཏེ་ཨར་ལས་འཆར་འགོད་དང་གྲོང་ཁྱེར་འཕེལ་རྒྱས་ཁྲོད་གཅིག་མཚུངས་ཅན་དང་རང་སྣང་ཅན་མ་ཐོན་སྔོན་འགོག་བྱས་ཐོག བོད་མི་རིགས་ཀྱི་ཁྱད་ཆོས་ལྡན་པའི་དཔེ་སྟོན་ཨར་སྐྲུན་ཁག་གཅིག་ཡག་པོ་བཏོད་ནས་རིམ་བཞིན་སྐུལ་འདེད་དང་ཕྱོགས་ཡོངས་ནས་སྐུལ་འདེད་གཏོང་དགོས་ཞེས་བཤད།
          བོད་ཀྱི་སྤྱིའི་འཆར་འགོད་སྒྲིག་བཟོའི་ལས་དོན་བཞུགས་མོལ་ཚོགས་འདུའི་ཐོག་ཀྲུང་གོ་གྲོང་ཁྱེར་འཆར་འགོད་ཇུས་འགོད་ཞིབ་འཇུག་ཁང་གིས་བོད་ཀྱི་སྤྱིའི་འཆར་འགོད་སྒྲིག་བཟོའི་ལས་དོན་གྱི་གཞི་རྩའི་གནས་ཚུལ་སྙན་སེང་ཞུས་པ་དང་གོམ་སྟབས་རྗེས་མའི་བོད་ཀྱི་སྤྱིའི་འཆར་འགོད་སྒྲིག་བཟོའི་ལས་དོན་ཐད་བསམ་འཆར་དང་གྲོས་འགོ་བཏོན་པ་རེད།
          ཚོགས་འདུའི་ཐོག་འཆར་འགོད་སྒྲིག་བཟོ་ཚན་པས་བརྟག་ཞིབ་གཏིང་ཟབ་དང་ཕྱོགས་ཡོངས་ནས་བོད་ཀྱི་སྤྱིའི་བཀོད་པ་ཁོང་དུ་ཆུད་པ་བྱས་ཏེ་བོད་ཀྱི་ས་ཆ་ཁག་དང་འབྲེལ་མཐུད་ལ་ཤུགས་རྒྱག་པ་དང་སྤྱི་ཡོངས་ཀྱི་ཐོག་ནས་འགོ་བཙུགས་ཏེ་སྒྲིག་བཟོའི་ལས་དོན་ཡག་པོ་སྤེལ་དགོས་པ་དང་། འབྲེལ་མཐུད་དང་འབྲེལ་གཏུགས་ལ་ཤུགས་རྒྱག་པ་དང་བསམ་གཞིག་དང་ཞིབ་འཇུག་གཏིང་ཟབ་བྱས་ཏེ། སྣོད་བཅུད་སྲུང་སྐྱོང་དམིགས་ཚད་དུ་བྱས་ནས་འཆར་འགོད་སྒྲིག་བཟོའི་ཁྱབ་ཁོངས་སྔར་བས་གསལ་པོ་བཟོ་དགོས། ལྗོངས་ཡོངས་ཀྱི་སྡོད་ཁང་དང་གྲོང་ཁྱེར་དང་གྲོང་གསེབ་འཛུགས་སྐྲུན་སྡེ་ཚན་གྱིས་འཕེལ་སྒྱུར་དང་ནོར་སྲིད་སོགས་འབྲེལ་ཡོད་སྡེ་ཚན་དང་འབྲེལ་མཐུད་ལ་ཤུགས་བརྒྱབ་སྟེ་རང་འགུལ་ངང་འཆར་འགོད་སྒྲིག་བཟོའི་ཚན་པར་ཕྱོགས་ཡོངས་དང་དཔྱད་གཞིར་འཛིན་རུང་བའི་རྨང་གཞིའི་དཔྱད་གཞི་མཁོ་འདོན་བྱས་ནས་འཆར་འགོད་སྒྲིག་བཟོའི་ལས་དོན་གོ་རིམ་ལྡན་པའི་ངང་སྤེལ་ཐུབ་པའི་འགན་ལེན་བྱེད་དགོས་པ་བསྟན་པ་རེད།
