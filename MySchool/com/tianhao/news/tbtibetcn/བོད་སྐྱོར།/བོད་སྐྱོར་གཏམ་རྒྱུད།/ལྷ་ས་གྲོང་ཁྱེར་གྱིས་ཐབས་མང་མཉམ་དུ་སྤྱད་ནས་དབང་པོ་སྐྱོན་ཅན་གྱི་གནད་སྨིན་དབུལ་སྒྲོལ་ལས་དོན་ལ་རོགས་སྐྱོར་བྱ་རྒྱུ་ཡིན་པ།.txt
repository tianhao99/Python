ལྷ་ས་གྲོང་ཁྱེར་གྱིས་ཐབས་མང་མཉམ་དུ་སྤྱད་ནས་དབང་པོ་སྐྱོན་ཅན་གྱི་གནད་སྨིན་དབུལ་སྒྲོལ་ལས་དོན་ལ་རོགས་སྐྱོར་བྱ་རྒྱུ་ཡིན་པ།
ཉེ་ཆར་གསར་འགོད་པར་ལྷ་ས་གྲོང་ཁྱེར་དབང་པོ་སྐྱོན་ཅན་མཉམ་འབྲེལ་ལྷན་ཚོགས་ནས་ཤེས་རྟོགས་བྱུང་བར། དབང་པོ་སྐྱོན་ཅན་གྱི་གནད་སྨིན་དབུལ་སྐྱོར"དབུལ་སྒྲོལ་འགན་ལེན"ལས་དོན་ཡག་པོ་སྤེལ་ཆེད། ལྷ་ས་གྲོང་ཁྱེར་དབང་པོ་སྐྱོན་ཅན་མཉམ་འབྲེལ་ལྷན་ཚོགས་དབང་པོ་སྐྱོན་ཅན་གྱི་གནད་སྨིན་དབུལ་སྐྱོར་ལས་དོན་ཚོགས་ཆུང་གིས་ལྷ་ས་གྲོང་ཨུད་དང་གྲོང་ཁྱེར་སྲིད་གཞུང་གིས་བཏོན་པའི"ལོ་གཉིས་ལ་དབུལ་སྒྲོལ་དང་ལོ་གསུམ་ལ་སྲ་བརྟན་ཡོང་བ་བྱེད་དགོས་པའི"དམིགས་ཚད་གཞིར་བཟུང《ལྷ་ས་གྲོང་ཁྱེར་གྱི་དབང་པོ་སྐྱོན་ཅན་གྱི་གནད་སྨིན་དབུལ་སྐྱོར"དབུལ་སྒྲོལ་འགན་ལེན"གྱི་དབུལ་སྐྱོར་བྱ་ཡུལ་ཁེར་རྐྱང་ལ་རོགས་སྐྱོར་བྱ་རྒྱུའི་འཆར་འགོད་རེའུ་མིག》བཟོ་བ་དང་། ལོ་རེའི་ཡོང་འབབ་དབུལ་སྒྲོལ་ཚད་གཞི་ལས་དམའ་བའི་དབང་པོ་སྐྱོན་ཅན་1253གོམ་སྟབས་རྗེས་མའི་ལས་དོན་གྱི་གཙོ་གནད་ཅིག་ཏུ་བརྩི་བ། "ལུས་ལ་ཚད་བརྒྱབ་ནས་གོས་འཚེམ་པ"ཡི་དབུལ་སྐྱོར་བྱེད་ཐབས་ཞིབ་ཕྲ་བཟོ་བ་བཅས་བྱས་ནས་གནད་སྨིན་དབུལ་སྐྱོར་ཡག་པོ་བྱ་རྒྱུ་རེད།
          《ལྷ་ས་གྲོང་ཁྱེར་གྱི་དབང་པོ་སྐྱོན་ཅན་གྱི་གནད་སྨིན་དབུལ་སྐྱོར"དབུལ་སྒྲོལ་འགན་ལེན"གྱི་དབུལ་སྐྱོར་བྱ་ཡུལ་ཁེར་རྐྱང་ལ་རོགས་སྐྱོར་བྱ་རྒྱུའི་འཆར་འགོད་རེའུ་མིག》བཟོ་སྐབས་རང་རེའི་གྲོང་ཁྱེར་གྱི་ལོ་རེའི་ཡོང་འབབ་དབུལ་སྒྲོལ་ཚད་གཞི་ལས་དམའ་བའི་དབང་པོ་སྐྱོན་ཅན་1253ལ་གཙོ་བོར་དམིགས་ཏེ་དུད་ཚང་རེ་རེ་བཞིན་བརྟག་དཔྱད་བྱེད་པར་བསྐྱོད་དེ་དུད་ཚང་དང་མི་སྒེར་གྱི་གནས་ཚུལ་མི་འདྲ་བར་གཞིགས་ནས་རོགས་སྐྱོར་བྱ་རྒྱུ་རེད།
           གཞན་ཡང་། ལྷ་ས་གྲོང་ཁྱེར་དབང་པོ་སྐྱོན་ཅན་མཉམ་འབྲེལ་ལྷན་ཚོགས་དབང་པོ་སྐྱོན་ཅན་གྱི་གནད་སྨིན་དབུལ་སྐྱོར་ལས་དོན་ཚོགས་ཆུང་གིས་རྫོང(ཆུས)ཁག་གི་དུད་ཚང་དུ་སྐྱོད་ཚུལ་ལ་ལྟ་སྐུལ་མཛུབ་ཁྲིད་བྱ་རྒྱུ་དང་། རྫོང(ཆུས)ཁག་གི"ཁ་གསབ་རྣམ་པའི"གཟིགས་སྐྱོང་སྲིད་ཇུས་བཏོན་པ་དང་འབྲེལ་ལག་བསྟར་བྱས་ཚུལ་ཤེས་རྟོགས་བྱ་རྒྱུ། རྫོང(ཆུས)ཁག་གི་དབང་པོ་སྐྱོན་ཅན་མཉམ་འབྲེལ་ལྷན་ཚོགས་ལ་གཞོགས་འདེགས་ཀྱིས་རྫོང(ཆུས)ཏང་ཨུད་དང་སྲིད་གཞུང་དང་འབྲེལ་གཏུག་མཐུན་སྦྱོར་བྱ་རྒྱུ་མ་ཟད། ད་དུང་ལྷ་ས་གྲོང་ཨུད་དང་གྲོང་ཁྱེར་སྲིད་གཞུང་གི《དབང་པོ་སྐྱོན་ཅན་གྱི་བྱ་གཞག་འཕེལ་རྒྱས་ལ་སྐུལ་འདེད་གཏོང་རྒྱུའི་ལག་བསྟར་བྱ་ཐབས》ནན་ཏན་ངང་ལག་བསྟར་དོན་འཁྱོལ་བྱ་རྒྱུ་རེད།
