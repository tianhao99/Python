མཚོ་སྔོན་པོའི་ཉ་མོ་ཚེ་ཐར་དུས་ཆེན་གྱི་ཟུར་ཙམ།
ཁོར་ཡུག་རྒྱ་མཚོ་མེ་རེ་རེ།། ཉ་མོ་ཟས་འཚོལ་ཤ་ར་ར།། རྦ་རླབས་སྣ་ཚོགས་ཉི་ལི་ལི།། ཆུ་བྱ་སྐད་སྙན་ཀྱུ་རུ་རུ།། དེ་ལྟར་ངོ་མཚར་མཐའ་ཡས་པ།།ཞེས་ཞབས་དཀར་བ་ཚོགས་དྲུག་རང་གྲོལ་གྱིས་སྙན་དབང་གསོ་བའི་མགུར་གླུ་ཚན་པ་དེས་སྣེ་དྲངས་ནས་ང་ཚོའི་ཡིད་ཀྱི་ཆིབས་ཁ་ཡང་མཚོ་སྨན་ཁྲི་ཤོར་རྒྱལ་མོའི་བྱང་ངོགས་གྱི་གཡུ་གཞུང་རྐང་ཚར་བསྒྱུར་ནས། གནས་དེའི་དབྱར་ཟླ་དྲུག་པའི་མཚོ་སྔོན་པོའི་ཉ་མོ་ཚེ་ཐར་དུས་ཆེན་གྱི་ངོ་མཚར་ལྟད་མོར་རོལ་བར་ཕེབས།       གཡུ་ཡི་ཞུན་ཐིགས་ལྟར་འཁྱིལ་བའི་མཚོ་སྨན་ཁྲི་ཤོར་རྒྱལ་མོ་ནི་མཐོ་སྒང་གི་ཆེས་ཆེ་བའི་ཚྭ་མཚོ་ཡིན་ལ། གནས་འདིར་མཁའ་ལ་འདབ་ཆགས་རྩེན་ཞིང་ཆུ་ལ་ཉ་མོ་འཁྱུག་པའི་སྣོད་བཅུད་རྟེན་འབྲེལ་མ་འཆོལ་བའི་ངོ་མཚར་ལྟད་མོ་རང་བྱུང་སྒོས་ངོམས་ཡོད་ལ། སྒོས་སུ་ལོ་རེའི་མཚོ་སྔོན་པོའི་གསེར་ཉ་རྒྱུད་སྤེལ་ཕྱིར་རྒྱུའི་རྣམ་པ་ནི་མིའི་བློ་ཡིད་ལ་ཤོང་དཀའ་བའི་ངོ་མཚར་ཡུལ་ལྗོངས་ཤིག་ཏུ་རང་ཤུགས་སུ་གྲུབ་པས་ན། རྐང་ཚ་རྫོང་སྲིད་གཞུང་གིས་ཀྱང་ངོ་མཚར་གྱིས་ཕྱུག་པའི་རང་བྱུང་ལྟད་མོ་དེ་ཡུལ་སྐོར་མགྲོན་འགུག་གི་ལས་གཞི་གལ་ཆེན་ཞིག་དུ་བཀོད་ནས་གཞི་ཁྱོན་ཅན་གྱིས་དར་སྤེལ་བྱེད་བཞིན་ཡོད་ལ། “༢༠༡༡ལོའི་མཚོ་སྔོན་པོའི་ཙོང་ཉ་སྐྱོང་སྤེལ་ཆུར་གཏོང་དང་ཉ་གཟིགས་སྲོག་བླུའི་དུས་ཆེན”དེ་ནི་རྐང་ཚའི་མཚོ་མོའི་རིག་གནས་ཡུལ་སྐོར་གྱི་ནང་ཚན་གཙོ་བོར་བགྲང་ཡོད།  “༢༠༡༡ལོའི་མཚོ་སྔོན་པོའི་ཙོང་ཉ་སྐྱོང་སྤེལ་ཆུར་གཏོང་དང་ཉ་གཟིགས་སྲོག་བླུའི་དུས་ཆེན”གྱི་མཛད་སྒོ་༢༠༡༡ལོའི་ཟླ༧པའི་ཚེས༤ཉིན་གཡུ་གཞུང་རྐང་ཚའི་འོམ་ཆུ་འགྲམ་གྱི་གསེར་ཉའི་གཞིས་ཁྱིམ་དུ་འབོད་པའི་ཚོམས་རར་གཟབ་རྒྱས་ངང་བསྡུས། མཚོ་བྱང་བོད་རིགས་རང་སྐྱོང་ཁུལ་མི་དམངས་སྲིད་གཞུང་གི་ཁུལ་དཔོན་ཉི་མ་སྒྲོལ་མས་མཛད་སྒོར་འགོ་འཛིན་བྱས།མཚོ་སྔོན་པོའི་ཉ་མོ་ཚེ་ཐར་དུས་ཆེན་གྱི་ཟུར་ཙམ།
ཁོར་ཡུག་རྒྱ་མཚོ་མེ་རེ་རེ།། ཉ་མོ་ཟས་འཚོལ་ཤ་ར་ར།། རྦ་རླབས་སྣ་ཚོགས་ཉི་ལི་ལི།། ཆུ་བྱ་སྐད་སྙན་ཀྱུ་རུ་རུ།། དེ་ལྟར་ངོ་མཚར་མཐའ་ཡས་པ།།ཞེས་ཞབས་དཀར་བ་ཚོགས་དྲུག་རང་གྲོལ་གྱིས་སྙན་དབང་གསོ་བའི་མགུར་གླུ་ཚན་པ་དེས་སྣེ་དྲངས་ནས་ང་ཚོའི་ཡིད་ཀྱི་ཆིབས་ཁ་ཡང་མཚོ་སྨན་ཁྲི་ཤོར་རྒྱལ་མོའི་བྱང་ངོགས་གྱི་གཡུ་གཞུང་རྐང་ཚར་བསྒྱུར་ནས། གནས་དེའི་དབྱར་ཟླ་དྲུག་པའི་མཚོ་སྔོན་པོའི་ཉ་མོ་ཚེ་ཐར་དུས་ཆེན་གྱི་ངོ་མཚར་ལྟད་མོར་རོལ་བར་ཕེབས།       གཡུ་ཡི་ཞུན་ཐིགས་ལྟར་འཁྱིལ་བའི་མཚོ་སྨན་ཁྲི་ཤོར་རྒྱལ་མོ་ནི་མཐོ་སྒང་གི་ཆེས་ཆེ་བའི་ཚྭ་མཚོ་ཡིན་ལ། གནས་འདིར་མཁའ་ལ་འདབ་ཆགས་རྩེན་ཞིང་ཆུ་ལ་ཉ་མོ་འཁྱུག་པའི་སྣོད་བཅུད་རྟེན་འབྲེལ་མ་འཆོལ་བའི་ངོ་མཚར་ལྟད་མོ་རང་བྱུང་སྒོས་ངོམས་ཡོད་ལ། སྒོས་སུ་ལོ་རེའི་མཚོ་སྔོན་པོའི་གསེར་ཉ་རྒྱུད་སྤེལ་ཕྱིར་རྒྱུའི་རྣམ་པ་ནི་མིའི་བློ་ཡིད་ལ་ཤོང་དཀའ་བའི་ངོ་མཚར་ཡུལ་ལྗོངས་ཤིག་ཏུ་རང་ཤུགས་སུ་གྲུབ་པས་ན། རྐང་ཚ་རྫོང་སྲིད་གཞུང་གིས་ཀྱང་ངོ་མཚར་གྱིས་ཕྱུག་པའི་རང་བྱུང་ལྟད་མོ་དེ་ཡུལ་སྐོར་མགྲོན་འགུག་གི་ལས་གཞི་གལ་ཆེན་ཞིག་དུ་བཀོད་ནས་གཞི་ཁྱོན་ཅན་གྱིས་དར་སྤེལ་བྱེད་བཞིན་ཡོད་ལ། “༢༠༡༡ལོའི་མཚོ་སྔོན་པོའི་ཙོང་ཉ་སྐྱོང་སྤེལ་ཆུར་གཏོང་དང་ཉ་གཟིགས་སྲོག་བླུའི་དུས་ཆེན”དེ་ནི་རྐང་ཚའི་མཚོ་མོའི་རིག་གནས་ཡུལ་སྐོར་གྱི་ནང་ཚན་གཙོ་བོར་བགྲང་ཡོད།  “༢༠༡༡ལོའི་མཚོ་སྔོན་པོའི་ཙོང་ཉ་སྐྱོང་སྤེལ་ཆུར་གཏོང་དང་ཉ་གཟིགས་སྲོག་བླུའི་དུས་ཆེན”གྱི་མཛད་སྒོ་༢༠༡༡ལོའི་ཟླ༧པའི་ཚེས༤ཉིན་གཡུ་གཞུང་རྐང་ཚའི་འོམ་ཆུ་འགྲམ་གྱི་གསེར་ཉའི་གཞིས་ཁྱིམ་དུ་འབོད་པའི་ཚོམས་རར་གཟབ་རྒྱས་ངང་བསྡུས། མཚོ་བྱང་བོད་རིགས་རང་སྐྱོང་ཁུལ་མི་དམངས་སྲིད་གཞུང་གི་ཁུལ་དཔོན་ཉི་མ་སྒྲོལ་མས་མཛད་སྒོར་འགོ་འཛིན་བྱས།
