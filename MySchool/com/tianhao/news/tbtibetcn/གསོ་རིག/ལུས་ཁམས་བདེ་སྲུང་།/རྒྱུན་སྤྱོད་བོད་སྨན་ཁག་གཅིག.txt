རྒྱུན་སྤྱོད་བོད་སྨན་ཁག་གཅིག
ཟླ་ཤེལ། ཟླ་ཤེལ་གྱི་ཕན་ནུས་ནི་ཕོ་བ་ན་བ་དང་། ཚ་ཤར་བརྒྱབ་པ། དུག་དང་སྨུག་པོའི་ནད། ཁྲག་སྐྱུགས་པ།ཁྲག་བཤལ་བ། གཉན་ཚད་དང་ཚད་རྙིང་། འགྲམས་འཁྲུགས། མ་ཞུ་བ།གླང་འཐབ། སྲིན་ནད། ཆུ་སེར། སྐྲན་ནད་ལྡན་པ་དང་། འདུས་པའི་ནད་རྣམས་ཀུན་ལ་ཕན་གྱི་ཡོད་པས་ཉིན་རེར་ཐེངས་གཅིག་ནས་གཉིས་བར་དང་ཐེངས་རེར་རིལ་བུ་གསུམ་བཏང་ན་འགྲིག 
སྨུག་པོ་གཡུ་རྒྱལ། 
  གཡུ་རྒྱལ་ནི་ཅུང་བསིལ་ཞིང་བཟི་མེད། ཕན་ནུས་ནི་བྲང་ཚ་བ། ཆུ་སྐྱུར་སྐྱུགས་པ། མཁྲིས་པ་སྐྱུགས་པ། རྒྱ་བཤལ་སྐྱུགས་པ། ཕོ་མཆིན་རྒྱབ་མདུན་སྦྲགས་ནས་ན་བ། སྟོད་ན་བ། ཁྲག་མཁྲིས་ཕོ་བར་ལྷུང་བ། སྨུག་པོ་རྒྱས་པ་དང་སྨུག་པོ་བྱེར་བ། འདུས་ནད་སྨུག་པོའི་ནད་ལ་ཕན་པས་གཏོང་ཐབས་ཉིན་རེར་ཐེངས་གཅིག་དང་ཐེངས་རེར་རིལ་བུ་གཅིག་བཏང་ན་ཆོག 
བདེ་སྨུག 
  བདེ་སྨུག་ནི་བསིལ་དྲོད་སྙོམས་པ་དང་། བཞི་མེད། ཕན་ནུས་ཁྲག་མཁྲིས་ཕོ་བར་ལྷུང་བ་དང་། སྨུག་པོ་ཁྲག་སྐྲན་འདྲི་བ། ཕོ་བའི་མེ་དྲོད་ཉམ་པ། མ་ཞུ་གསར་རྙིང་གི་ནད་དང་ཕོ་ནད་ཚ་གྲང་ཀུན་ལ་ཕན་པས་གཏོང་ཐབས་ཉིན་རེར་ཐེངས་གཅིག་ནས་གཉིས་བར་དང་ཐེངས་རེར་རིལ་བུ་གསུམ་གཏོང་དགོས། 
གྲང་སྦྱོར། 
  གྲང་སྦྱོར་ནི་ཕོ་བར་ཚ་ཤར་བརྒྱབ་པ། ཁྲག་ནག་དུད་ཁུ་སྐྱུགས་པ། བཤལ་བ། འདུས་ནད་སྨུག་པོའི་ནད།གཅོང་ནད། ཁྲག་མཁྲིས་ཀྱི་ནད། བད་རླུང་གི་ནད། མགོའི་རྨ། བྲང་ཁོག་གི་རྨ། ཡན་ལག་གི་རྨ། མཛེ་འབྲས། ཚད་རྙིང་། རིམས་ནད། རྒྱུ་གཟེར། ཁྱི་དུག་སོགས་ལ་ཕན་པས་ཉིན་རེར་ཞོགས་པ་ལྟོ་ཆས་མ་བཟས་གོང་རིལ་བུ་གཅིག་བཟས་ན་འགྲིག 
ཐང་ཆེན་ཉེར་ལྔ། 
  ཐང་ཆེན་ཉེར་ལྔ་ནི་དུག་དང་སྨུག་པོའི་ནད། ཚད་རྙིང་བྱེར་བ། འདུ་བ་ཚ་གྲང་སྒོམ་པ། རླུང་མི་སྐྱེ་བ། བད་ཀན་སེལ་བ། མཁྲིས་པ་སེལ་བ། ལུས་ལྕི་བ། ཁ་སྐམ་པ། དང་ག་འགག་པ་སོགས་ལ་ཕན་པས་གཏོང་ཐབས་ནི་ཉིན་རེར་ཐེངས་གཅིག་ནས་གཉིས་བར་དང་ཐེངས་རེར་རིལ་བུ་གསུམ་བཏང་ན་ཆོག 
གཡུ་ཁྱུང་། 
  གཡུ་ཁྱུང་ནི་ཕོ་བ་དང་རྒྱུ་མར་ན་ཟུག་ཡང་ཡང་གཏོང་བ་དང་། ཁྲག་མཁྲིས་ཕོ་བར་ལྷུང་བ། སྨུག་པོ་གླང་འཐབ་ཀྱི་ནད་དང་གཉན་གཟེར་གཅོག་པ་སོགས་ལ་ཕན་པས་གཏོང་ཐབས་ནི་ཉིན་རེར་ཐེངས་གཅིག་དང་ཐེངས་རེར་གཡུ་དྲིལ་བཅུ་གསུམ་རིལ་བུ་བཞི་དང་།ཁྱུང་ལྔ་རིལ་བུ་གསུམ་མཉམ་དུ་གཏོང་དགོས།
