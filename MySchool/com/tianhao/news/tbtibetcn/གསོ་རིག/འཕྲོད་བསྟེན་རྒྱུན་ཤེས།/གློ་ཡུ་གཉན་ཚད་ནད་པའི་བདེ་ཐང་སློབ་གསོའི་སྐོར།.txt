གློ་ཡུ་གཉན་ཚད་ནད་པའི་བདེ་ཐང་སློབ་གསོའི་སྐོར།
དལ་གཤིས་གློ་ཡུའི་གཉན་ཚད།（chronic bronchitis）ནད་འདིའི་སློང་རྐྱེན་ནི་སྲིན་རིགས་འགོས་པ་ལས་བྱུང་བའི་གློ་ཡུ་དང་། གློ་ཡུའི་ནང་སྐྱི་དང་། དེའི་མཐའ་འཁོར་དུ་བྱུང་བའི་དལ་གཤིས་རང་བཞིན་ཅན་གྱི་གཉན་ཚད་བཅས་ཡིན།དེའི་ན་ལུགས་ཀྱི་ཁྱད་ཆོས་ནི་གློ་ཡུའི་རྨེན་ཆེར་རྒྱས་པ་དང་། གཤེར་ཁུ་ཟགས་ཐོན་མང་བ་ཞིག་ཡིན།ནད་རྟགས་ལོ་གཉིས་ཡན་བསྟུད་མར་བྱུང་བ་དང་། གློ་ལུ་བ་དང་ལུད་པ་འབྱིན་པའམ་དབུགས་འཚང་བ་སོགས་ཀྱི་ནད་རྟགས་ཡོད། ལོ་རེར་ཟླ་བ་གསུམ་ལ་བསྟུད་མར་བྱུང་ལ། སྔ་མའི་དུས་སུ་ནད་རྟགས་ཅུང་ཡང་བ་དང་། མང་ཆེ་བ་དགུན་དུས་སུ་ལྡང་བ། དཔྱིད་དུས་དྲོད་བབས་རྗེས་ནད་རྟགས་ཞི་བ། ནད་ཡུན་འགོར་བའི་རྗེས་གཉན་ཚད་ཇེ་ལྕིར་སོང་ནས་ནད་རྟགས་དུས་རྒྱུན་དུ་འབྱུང་བ་དང་དུས་ཚིགས་ལ་ཡང་ཁྱད་པར་མེད། གསོ་བཅོས་མ་གནང་བར་འགོར་འགྱངས་བྱས་ན་གློ་ནད་ཐེས་པོ་དང་། གློ་ནད་ལས་སློང་བའི་སྙིང་གི་ནད་སོགས་བསྐྱེད་ངེས།
            ནད་རྟགས། གློ་ལུ་བ་དང་། ལུད་པ་འབྱིན་པ་དང་། དབུགས་འཚང་བ། བྲང་ཁོག་ན་བ། སྣ་གཤོག་འདར་བ། ཉལ་མི་ཐུབ་པ་བཅས་མངོན།
            ནད་སློང་བའི་རྒྱུ་རྐྱེན། མཁའ་དབུགས་མི་གཙང་བ་དང་ཐ་མག་འཐེན་པ་དང་། འབུ་ཕྲའི་རིགས་འགོས་པ། ཚོར་ལྷག་ལོག་པའི་རྒྱུ་རྐྱེན། དབུགས་ལམ་གྱི་ནད་འགོག་བྱེད་ནུས་ཉམས་པ། འཚོ་བཅུད་འཕེལ་ཟད་ཆེ་བའི་རྒྱུ་རྐྱེན་སོགས་ཀྱིས་སློང་གིན་ཡོད།
            བཅོས་ཚུལ། རེད་ནད་འགོག་པར་བྱེད་པ་དང་། ལུད་པ་འབྱིན་ནས་གློ་མི་ལུ་བར་བྱེད་པ། དབུགས་བདེ་རུ་ཕྱིན་པ། རང་དགར་གཉན་སྲིན་སྨན་ཡུན་རིང་དུ་བསྟེན་མི་རུང་བ།
            བཟའ་བཏུང་གི་མཛུབ་ཁྲིད། ལན་ཚྭ་ཆེ་བའི་ཟས་མི་བསྟེན་པ། སྣུམ་སྲེག་བྱས་པའི་ཟས་དང་དབུགས་འབྱུང་སླ་བའི་ཟས་རིགས་བསྟེན་མི་རུང་། སྤྲི་དཀར་དང་། དྲོད་ཆེ་བ་དང་། བཅུད་ཆེ་བ་དང་། ཚིལ་ཉུང་བ་དང་། འཇུ་སླ་བའི་ཟས་རིགས་མང་དུ་བསྟེན་དགོས། དཔེར་ན། ཤ་དང་། སྒོ་ང་། འོ་མ། ཉ་ཤ སྔོ་ཚོད། སིལ་ཏོག་རིགས་སོགས་མང་དུ་བསྟེན་དགོས། སྐོམ་ཆུའི་རིགས་མང་དུ་འཐུང་དགོས་ལ་ཉིན་རེར་ཆུ་ཧའོ་ཧྲེང1500ཡན་འཐུང་དགོས། དེས་ལུད་པ་སླ་མོར་འགྱུར་བ་དང་ཕྱིར་འབྱིན་ཐུབ་པར་ཕན་ཆེ། དགོང་སྔ་ཉལ་དང་ནངས་སྔ་ལངས་ཀྱི་ལུགས་སྲོལ་བཟང་པོ་དེ་གོམས་དགོས།
དབུགས་འབྱིན་རྔུབ་ཤ་གྲིམ་གྱི་སྦྱོང་བརྡར།
            འབྱིན་རྔུབ་ཤ་གྲིམ་གྱི་སྦྱོང་བརྡར་གྱིས་འབྱིན་རྔུབ་ཤ་གྲིམ་དང་ལྷག་པར་མཆིན་དྲི་ཤ་གྲིམ་ལ་སྟོབས་ནུས་ལྡན་པར་གཏོང་ཐུབ་པ་དང་།དབུགས་འབྱིན་རྔུབ་ཀྱི་བྱེད་ནུས་ཡང་ཇེ་བཟང་དུ་གཏོང་ཐུབ། དེའི་ཞོར་དུ་ལུད་པ་ཡང་ཕྱིར་འབྱིན་ཐུབ་པ་དང་། གློ་བ་དང་གློ་ཡུའི་རེད་ནད་དང་གཉན་ཚད་འཇོམས་པར་ཕན་ཆེ། དེ་བས་ནད་པས་དུས་རྒྱུན་དུ་དབུགས་འབྱིན་རྔུབ་ཀྱི་སྦྱོང་བརྡར་ཡང་ཡང་གནང་ན་ཕན་ཆེ།

            གློ་ཡུའི་གཉན་ཚད་ཪྙིང་པ་ཅན་གྱིས་གློ་བའི་གཉན་ཚད་ཡང་བསྐྱར་འཕར་དུ་བཅུག་སྟེ་ནད་བབ་ཇེ་སྡུག་དུ་འགྱུར་བ་དང་། གློ་བའི་ཚད་པ་འབྱུང་དུ་འཇུག་པས།སྔོན་འགོག་དང་བླ་གཉན་ནད་མ་བྱུང་བར་དུ་གློ་བའི་གཉན་ཚད་གསོ་བཅོས་བྱེད་དགོས། དེ་དང་ཆབས་ཅིག་གྲང་ངར་དུས་སུ་སྦྱོང་བརྡར་བྱེད་པ། དཔེར་ན། ཆུ་འཁྱགས་ཀྱིས་ངོ་གདོང་བཀྲུ་བ་དང་། གནོད་ཤུགས་ཆེ་བ་དང་གྲང་ངར་ཆེ་བའི་མཁའ་དབུགས་ཀྱིས་གློ་སྦུབས་ལ་གནོད་པ་གཏོང་བ་དང་། དུ་ཆང་གིས་གློ་སྦུབས་ཀྱི་སྟེང་སྐྱིར་གནོད་པ་བཟོས་ཏེ་གློ་ལུ་བའི་ནད་རྟགས་འབྱུང་བས། ངེས་པར་ཆང་གཅོད་དགོས། ཁང་པའི་སྒོ་དང་དྲ་མ་ཕྱེས་ནས་མཁའ་དབུགས་གསར་པ་རྒྱུ་རུ་འཇུག་དགོས། མི་གཙང་བའི་མཁའ་དབུགས་དང་ཐལ་རྡུལ། ན་བུན། གནོད་ཡོད་མཁའ་དབུགས་སོགས་འབྱིན་མི་རུང་ངོ་།   །
