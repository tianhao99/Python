གལ་སྲིད་ཁྱོད་ཀྱི་སྲུང་ཤུབས་ཀྱི་མདོག་འགྱུར་བ་ཡིན་ན། དོན་དག་ཅིག་འབྱུང་ཉེན་ཡོད།
ལོ་བཅུ་བཞི་ཡས་མས་ཀྱི་ན་གཞོན་འགས་ཉེ་ཆར་མདོག་འགྱུར་བའི་སྲུང་ཤུབས་ཤིག་གསར་གཏོད་བྱས། ཡིན་ཡང་མདོག་འགྱུར་བ་ནི་མཚན་མོར་བྱ་སྤྱོད་གསར་བ་སྤེལ་ཕྱིར་མ་ཡིན་པར། ཁྱེད་རང་ལ་བདེ་འཇགས་བསྐྲུན་ཕྱིར་ཡིན།
          ཁོ་ཚོས་གསར་གཏོད་གནང་པའི་སྲུང་ཤུབས་རིགས་འདིས་མཚན་མའི་ནད་བརྟག་ཐུབ། གལ་སྲིད་ཆུ་བུར་དང་རེག་དུག་སོགས་མཚན་བརྒྱུད་ནད་དུག་དང་འབུ་ཕྲར་འཕྲད་སྐབས། སྲུང་ཤུབས་དེ་མདོག་འགྱུར་འགྲོ་བ་དང་། ལམ་ཐོག་གི་གློག་དམར་གསལ་བ་དང་མཚུངས་པས། ཁྱོད་རང་སྨན་ཁང་དུ་བརྟག་དཔྱད་ལ་འགྲོ་དགོས་པའི་གསལ་བརྡ་གཏང་སྲིད།
          གལ་སྲིད་མཚན་མའི་ནད་དུག་དང་འབུ་ཕྲར་འཕྲད་སྐབས། "S.T.EYE"སྲུང་ཤུབས་རང་ཤུགས་ཀྱིས་མདོག་འགྱུར་འགྲོ།
          ལོན་ཏོན་ཨ་ས་ཁི་ཉིའུ་ཏུན་སློབ་གླིང（Isaac Newton Academy）ནས་ཡོང་བའི་སློབ་བུ Muaz Nawaz（ལོ ༡༣）དང་Daanyaal Ali（ལོ ༡༤）Chirag Shah（ལོ ༡༤）བཅས་ཀྱིས་མཚན་མའི་ནད་འགོས་ཁྱབ་ཏུ་འགྲོ་བ་ནི་དབྱིན་ཇིའི་སྤྱི་ཚོགས་ཡོངས་ཀྱི་གནད་དོན་ཚབས་ཆེན་ཞིག་ཏུ་མཐོང་ནས། སྲུང་ཤུབས་འདི་འདྲ་ཞིག་གསར་གཏོད་བྱེད་པའི་འདུན་པ་སྐྱེས་པ་རེད།
          སྲུང་ཤུབས་འདི་ནི་ནད་དུག་དང་འབུ་ཕྲ་རིགས་མི་འདྲ་བར་འཕྲད་དུས། ཁ་དོག་ཀྱང་སོ་སོར་མི་འདྲ་བ་མངོན། དེར་དབྱི་མ་གཟུགས(衣原体)འགོས་ཡོད་ཚེ་ལྗང་མདོག་དང་། ཆུ་བུར་འགོས་ཡོད་ཚེ་སེར་པོ། གལ་སྲིད་ནུ་མགོའི་སྐྲན་ནད་འགོས་པ་ཡིན་ན་སྨུག་མདོག་མངོན། སྔོན་པོ་ཡིན་ན་རེག་དུག་འགོས་ཡོད་པའོ། སྐྱེས་པ་དང་བུད་མེད་གང་ཡིན་རུང་འཁྲིག་པ་སྤྱོད་པའི་སྐབས་སྲུང་ཤུབས་འདི་བརྒྱུད་མཚན་མའི་ནད་བརྟག་ཐུབ།
          འཛམ་གླིང་འཕྲོད་བསྟེན་རྩ་འཛུགས་ཀྱི་གཞི་གྲངས་ལྟར་ན། འཛམ་གླིང་ཧྲིལ་བོར་ལོ་རེར་མི་གྲངས་ས་ཡ་གཅིག་ལ་མཚན་མའི་ནད་འགོས་ཁྱབ་བྱུང་ཡོད་པ་དང་། ལོ་ཆུང་གཞོན་ནུ་ཚོས་བལྟས་ན"S.T.EYE"སྲུང་ཤུབས་ཀྱིས་མི་རྣམས་ལ་སྔར་བས་གཟབ་ནན་གྱིས་གནད་དོན་འདིར་མདོང་གཏོད་དུ་འཇུག་ཐུབ་པར་འདོད། "རྒྱུན་ལྡན་དུ་མི་རྣམས་མཚན་མའི་ནད་འདི་རིགས་སྨན་ཁང་དུ་བརྟག་དཔྱད་ལ་འགྲོ་བར་ངོ་ཚ་བ་དང་། སྲུང་ཤུབས་འདིའི་རིགས་སྤྱོད་པ་བརྒྱུད་བརྟག་ཐུབ་ན་ཁོ་ཚོའི་གསང་བ་སྲུང་ཐུབ"ཅེསAliཡིས་བཤད།
          གཞོན་ནུ་འདི་གསུམ་པོས་མཚན་མའི་ནད་བརྟག་ཐུབ་པའི་སྲུང་ཤུབས"S.T.EYE"གསར་གཏོད་གནང་།
Nawaz、Aliདང་Shahཡིས་འགྲན་བསྡུར་ཡུལ་དངོས་སུ་སྲུང་ཤུབས་ངོ་སྤྲོད་བྱེད་བཞིན་པ།
          གསར་གཏོད་འདིས་ལོ་ཆུང་གཞོན་ནུ་གསུམ་ལ་ན་གཞོན་ཚན་རྩལ་འགྲན་བསྡུརTeenTech Awardsབདེ་ཐང་རིགས་ཀྱི་བྱ་དགའ་ཨང་དང་པོ་ཐོབ། གཞན་ཁོ་ཚོར་བྱ་དགའི་དངུལ་དབྱིན་སྒོར༡༠༠༠ ཐོབ་པ་མ་ཟད་བྷ་ཁེན་ཧཱ་སྨི་ཕོ་བྲང་(Buckingham Palace)དུ་ལྟ་སྐོར་ལ་འགྲོ་བའི་གོ་སྐབས་ཀྱང་ཐོབ།
     ཁ་བྱང་ཕྱི་མོ། གལ་སྲིད་ཁྱོད་ཀྱི་སྲུང་ཤུབས་ཀྱི་མདོག་འགྱུར་བ་ཡིན་ན། དོན་དག་ཅིག་འབྱུང་ཉེན་ཡོད།
