བོད་དུ་ཚད་མ་གསར་མའི་དབུ་རྩོམ་མཁན་རྔོག་བློ་ལྡན་ཤེས་རབ་ཀྱི་ཚད་མའི་དགོངས་པའི་སྐོར་གླེང་བ།
[ནང་དོན་གནད་བསྡུས།] དཔྱད་རྩོམ་འདིར་བོད་དུ་《ཚད་མ་རྣམ་ངེས་》འཆད་ཉན་གྱི་སྲོལ་འབྱེད་པ་སྟེ་རྔོག་བློ་ལྡན་ཤེས་རབ་ཀྱི་ཚད་མའི་དགོངས་པ་དང་དེའི་ཤན་ཞུགས་སྐོར་གླེང་བ་སྟེ། ཐོག་མར་སྔོན་གྱི་མཁས་པའི་གསུང་དང་འབྲེལ་ཏེ་ཁོང་གི་མི་ཚེའི་ལོ་རྒྱུས་ལ་དཔྱད་འཇོག་དང་། དེ་རྗེས་གཙོ་བོ་《ཚད་མ་རྣམ་ངེས་ཀྱི་དཀའ་གནས་རྣམ་བཤད་》གཞིར་བྱས་ཏེ་མཁས་དབང་འདིས་གཟུང་ཡུལ་དང་འཇུག་ཡུལ་གཉིས་བཏོན་པའི་སྒོ་ནས་ཚད་མ་གཏན་ལ་འབེབས་པ་དང་། ཚད་མིན་གྱི་བློའི་བཞེད་ཚུལ་སོགས་ཀྱིས་མཚོན་ཁོང་གི་ཚད་མའི་དགོངས་པ་བརྗོད་པ་དང་། མཐར་བསྟན་བཅོས་འདིའི་ནང་གི་ཐལ་འགྱུར་དང་མུ་བཞིའི་རིགས་པས་ཕྱིས་ཀྱི་མཁས་པའི་ཚད་མའི་གཞུང་ལ་ཤུགས་རྐྱེན་ག་འདྲ་ཐེབས་ཚུལ་ལ་དཔྱད་པ་བཏང་ཡོད།
[བརྡ་ཆད་གཙོ་བོ།] རྔོག་བློ་ལྡན་ཤེས་རབ། ཚད་མ་རྣམ་ངེས་ཀྱི་དཀའ་གནས་རྣམ་བཤད། ཚད་མའི་དགོངས་པ། ཤུགས་རྐྱེན།
[ཀྲུང་གོའི་དཔེ་དེབ་དབྱེ་རྟགས།] B946.6
[ཡིག་ཚང་མཚོན་བྱང་།] A
[རྩོམ་གྱི་གོ་རིམ་ཨང་སྒྲིག] 1002-9060-2020-01-0094-10
        གནའ་བོའི་རྒྱ་གར་གྱི་ནང་པའི་ཚད་གཞུང་རྣམས་ལས་བོད་དུ་ཆེས་ཐོག་མར་བསྒྱུར་བ་ནི། སློབ་དཔོན་ཞི་བ་འཚོ་དང་བོད་ཀྱི་མཁས་པ་ཆོས་ཀྱི་སྣང་བ་གཉིས་ཀྱིས་མཛད་པའི་《ཕྱོགས་ཆོས་ཅན་གཏན་ཚིགས་དགུའི་འཁོར་ལོ་》ཞེས་བྱ་བ་རྒྱུ་མཚན་ནམ་གཏན་ཚིགས་ཀྱི་ཆ་གཙོ་བོར་སྟོན་པའི་གཞུང་དེ་ཡིན། དེའི་རྗེས་དུལ་བ་ལྷས་མཛད་པའི་《དམིགས་པ་བརྟག་པའི་རྩ་འགྲེལ་》དང་། ཞི་བ་འཚོའི་《ཚད་མ་དེ་ཁོ་ན་ཉིད་བསྡུས་པ་》དང་དེའི་དཀའ་འགྲེལ་སློབ་དཔོན་པདྨའི་ངང་ཚུལ་གྱི་《སྟོང་ཕྲག་བཅོ་བརྒྱད་པ》 སྡེ་བདུན་གྱི་ཕྱི་མ་བཞི་པོའི་འགྲེལ་པ་དང་བཅས་པ་ཕལ་ཆེར་བསྒྱུར་བ་དག་ལས་མང་ཤོས་ནི་རྗེས་དཔག་གཙོ་བོར་སྟོན་པའི་གཞུང་དག་ཡིན་ཞིང་། དུས་སྐབས་དེར་ད་དུང་རིགས་པའི་དབང་ཕྱུག་རྣམ་གཉིས་ཀྱི་དགོངས་པ་ཡོངས་སུ་རྫོགས་པར་མཚོན་པའི་ཚད་གཞུང་《ཀུན་ལས་བཏུས་པ་》དང་། 《རྣམ་འགྲེལ་》གཉིསབོད་ཡིག་ཏུ་བསྒྱུར་མེད། དུས་རབས་བཅུ་གཅིག་པར་དགེ་སློང་རྨ་དགེ་བའི་བློ་གྲོས་ཞུ་བས་《ཚད་མ་རྣམ་འགྲེལ་》ཡོངས་སུ་རྫོགས་པ་ཐོག་མར་བོད་སྐད་དུ་བསྒྱུར་ཅིང་དེའི་འཆད་ཉན་སྤེལ་ཀྱང་ཆེར་དར་མ་ནུས་པ་དེར་ལོ་རྒྱུས་ཐོག་ཚད་མ་རྙིང་མ་ཞེས་ཟེར། སྤྱི་ལོ་1073ལ་བོད་ཀྱི་འཆད་ཉན་གྱི་གདན་ས་དང་པོར་གྱུར་པ་གསང་ཕུ་ཕྱག་བཏབ་རྗེས་གྲུབ་མཐའ་རིས་མེད་ཀྱི་དགེ་བའི་བཤེས་གཉེན་རྣམས་གནས་དེར་འདུས་ཏེ་གཞུང་ལུགས་སྤྱི་དང་ཁྱད་པར་དུ་ཕར་ཕྱིན་དང་ཚད་མའི་བགྲོ་གླེང་གཙོ་བོར་གནང་། རྔོག་བློ་ལྡན་ཤེས་རབ་ཀྱིས་གནས་དེར་《ཚད་མ་རྣམ་འགྲེལ་》ལ་འགྱུར་བཅོས་མཛད་ཅིང་། སྐབས་དེ་ཤེད་དུ་《ཚད་མ་རྣམ་ངེས་ཀྱི་དཀའ་གནས་རྣམ་བཤད་》﹙གཤམ་དུ་《རྣམ་བཤད་ཆེན་པོ་》ཞེས་སུ་འགོད་རྒྱུ﹚མཛད། དེ་ནི་བོད་རང་ལ་མཚོན་ན་དཔལ་ཆོས་ཀྱི་གྲགས་པའི་སྡེ་བདུན་གྱི་ཡ་གྱལ་《ཚད་མ་རྣམ་པར་ངེས་པ་》ཡི་གཞུང་གི་འགྲེལ་པ་དང་པོ་ཉིད་ཡིན་པར་མ་ཟད། ཚད་མའི་གཞུང་དེའི་འཆད་ཚུལ་ཐུན་མོང་མ་ཡིན་པ་ཞིག་ཡིན་པས་བོད་ཀྱི་ལོ་རྒྱུས་ཐོག་ཚད་མ་གསར་མ་ཞེས་འབོད་ཅིང་། ཁོང་ཉིད་《རྣམ་ངེས་》འཆད་ཉན་གྱི་སྲོལ་འབྱེད་པ་རུ་གྱུར་པ་ལགས། བོད་དུ་ཚད་མ་གསར་རྙིང་གི་ཁྱད་པར་འབྱེད་དོན་ནི། སྔ་མའི་དུས་སུ་ཚད་མའི་འཆད་ཉན་བྱུང་ཡང་འཆད་ཚུལ་ངོ་གདོང་ཙམ་དང་ཐར་ཐོར་ཙམ་ལས་གཏིང་ཟབ་ཅིང་གོ་སླ་བར་མ་བྱས་པའི་དབང་གིས་རྙིང་མར་ཆགས་ལ། ཕྱི་མ་སྟེ་རྔོག་ལོའི་འཆད་ཚུལ་ནི་དགག་སྒྲུབ་ཀྱི་རིགས་པའི་ལམ་ནས་རྣམ་པར་བཞག་པ་ཡིན་པས་ཚད་མ་གསར་མ་ཞེས་སུ་བརྗོད། ཚད་གཞུང་《རྣམ་བཤད་ཆེན་པོ་》རུ་གཟུང་ཡུལ་དང་འཇུག་ཡུལ་གཉིས་ཀྱི་སྒོ་ནས་ཚད་མའི་རྣམ་གཞག་གཏན་ལ་ཕབ་པས། བརྩམས་ཆོས་འདི་ནི་ཚད་མའི་བསྟན་བཅོས་མང་པོའི་ནང་ཁྱད་དུ་འཕགས་ཤིང་། རྔོག་བློ་ལྡན་ཤེས་རབ་ཀྱི་ཚད་མའི་འགྲེལ་སྟངས་གསར་པ་དང་ཚད་མའི་དགོངས་པས་ཕྱིས་ཀྱི་སློབ་དཔོན་ཕྱྭ་བ་ཆོས་སེང་དང་དེའི་རྗེས་འབྲངས་སོགས་ཀྱི་ཚད་མའི་གཞུང་ལུགས་ལ་ཐད་ཀར་ཤན་ཞུགས་ཡོད་དོ། །དེ་ཡང་ཤེས་རྟོགས་ཀྱི་རྣམ་གཞག་ལྡན་པ་དང་། སེམས་ཁམས་རིག་པའི་ཆ། གཏན་ཚིགས་རིག་པ་དང་རྩོད་པའི་རིག་པ་བཅས་ཚང་བའི་བོད་རང་གི་ཐུན་མོང་མ་ཡིན་པའི་ཚད་གཞུང་ནི་བསྡུས་གྲྭ་ཉིད་ཡིན་པས། ཀུན་གྱིས་བསྡུས་གཞུང་ལ་གསར་གཏོད་ཀྱི་ཁྱད་ཆོས་ལྡན་པར་ཡང་ཡང་བསྔགས་པ་བདེན་མོད། ཁོ་བོའི་ལྟ་ཚུལ་ལ་བསྡུས་གཞུང་དུ་རྒྱུན་པར་མཐོང་བའི་ཐལ་འགྱུར་གྱི་རིགས་པ་དང་། མུ་བཞི་སོགས་རིགས་པ་གཙོ་བོ་དེ་དག་ནི་གློ་བུར་གསར་དུ་བྱུང་བ་ཞིག་མིན་པར། དེའི་ཡར་སྔོན་དུ་བརྩམས་པའི་《རྣམ་བཤད་ཆེན་པོ་》རུ་གསལ་པོར་བསྟན་ཡོན་ཅིང་། དེ་ཉིད་ཕྱིས་སུ་ཕྱྭ་བ་ཆོས་སེང་གིས་རྒྱུན་འཛིན་དང་གོང་འཕེལ་དུ་བཏང་བ་ཡིན་པར་བསམ་མོ། །
དང་པོ། 《རྣམ་བཤད་ཆེན་པོ་》དང་འབྲེལ་ཏེ་རྔོག་བློ་ལྡན་ཤེས་རབ་ཀྱི་མི་ཚེའི་ལོ་རྒྱུས་ཀྱི་དཔྱད་འཇོག
        རྔོག་ལོ་ཙཱ་བ་བློ་ལྡན་ཤེས་རབ་ནི་རབ་བྱུང་དང་པོའི་ས་ཕག་(སྤྱི་ལོ་1059)ལ་ཆོས་རྒྱལ་ཁྲི་སྲོང་ལྡེ་བཙན་གྱི་བློན་པོ་རྔོག་ཆེན་པོའི་རྒྱུད་པ་ཡར་འབྲོག་དོའི་ཕྱི་རོལ་གློ་བོ་ཞེས་པའི་མཚོ་འགྲམ་དུ་གནས་གཞི་ཆགས་པ་སྔ་འགྱུར་རྙིང་མ་པའི་དགེ་བཤེས་རྡོ་རྗེ་གཞོན་ནུའི་སྲས་ཆོས་སྐྱབས་དང་ཡུམ་དཔལ་མོ་གཉིས་ཀྱི་སྲས་སུ་འཁྲུངས། ཆུང་དུས་སྐུ་ངལ་ཤིན་ཏུ་ཆེ་བས་དགུང་ལོ་བཅུ་བདུན་པར་ཁྱིམ་ནས་ཁུ་བོ་ལེགས་པའི་ཤེས་རབ་ཀྱི་དྲུང་དུ་བྲོས་བྱོལ་དུ་ཕེབས། ཁུ་བོ་དང་གཟུས་ཀྱི་སློབ་མ་སྦོ་ཆུང་བ་ཚུལ་ཁྲིམས་ཤེས་རབ་གཉིས་ལས་རབ་ཏུ་བྱུང་ནས་མཚན་ལ་རྔོག་བློ་ལྡན་ཤེས་རབ་ཅེས་གསོལ། དགུང་ལོ་བཅུ་བདུན་པར་ཁུ་བོས་མཐུན་རྐྱེན་གནང་སྟེ་རྒྱ་གར་དུ་ཕེབས། ལམ་གྲོགས་སུ་ལོ་ཙཱ་བ་བཙན་ཁ་བོ་ཆེ་དང་། ཁྱུང་པོ་ཆོས་བརྩོན། རྭ་ལོ་ཙཱ་བ། གཉན་ལོ་ཙཱ་བ་རྣམས་མཉམ་དུ་བྱུང་། རྒྱ་གར་ཡུལ་དབུས་དང་ཁ་ཆེ་བཅས་སུ་ལོ་བཅུ་བདུན་རིང་པཎྜི་ཏ་སཛྫ་ན་དང་། པ་ར་ཧི་ད། མཱ་ཧཱ་ར་ཛ། སུ་མ་ཏི་ཀཱིརྟི་སོགས་དགེ་བའི་བཤེས་གཉེན་མང་པོ་བསྟེན། ཚད་མ་ལ་མཆོག་ཏུ་མཁས་པའི་སློབ་དཔོན་དུལ་བ་ལྷའི་སློབ་མ་པཎྜི་ཏ་གཞན་ཕན་བཟང་པོ་ལས་《ཚད་མ་རྣམ་འགྲེལ་》སོགས་ལེགས་པར་གསན་ཅིང་སྦྱངས་པ་མཐར་ཕྱིན་པར་མཛད། ཚད་མའི་འགྱུར་རྙིང་པ་ལ་དག་བཅོས་མཛད་ཅིང་། 《རྣམ་ངེས་》དང་《རིགས་ཐིགས་》སོགས་ཚད་མ་རིག་པ་དང་ཕར་ཕྱིན་གྱི་གཞུང་མང་པོ་རིགས་མི་འདྲ་བ་སྟོང་ཕྲག་དུ་མ་རྒྱ་གར་སྐད་ནས་བོད་སྐད་དུ་བསྒྱུར་བར་མཛད། བོད་ཀྱི་ལོ་ཙཱ་བའི་ནང་ནས་གཞུང་བསྒྱུར་མང་ཤོས་དང་མཁས་ཤོས་ཡིན་པར་གྲགས། དགུང་ལོ་སོ་ལྔ་པ་རབ་བྱུང་དང་པོའི་ཆུ་སྤྲེལ་(སྤྱི་ལོ་1092)ལོར་ཕྱིར་བོད་དུ་ཕེབས་ཤིང་། ཁུ་བོ་ལེགས་པའི་ཤེས་རབ་དང་མཁར་ཕུ་བ་ལས་ཇོ་བོ་བཀའ་གདམས་པའི་གདམས་ངག་རྫོགས་པར་གསན། གསང་ཕུ་ཆོས་གྲྭའི་གདན་ས་པ་གནང་སྟེ་《རྣམ་ངེས་》དང་《ཚད་མ་ཀུན་བཏུས》 དབུ་མ་རིགས་ཚོགས་དང་བྱམས་ཆོས་སྡེ་ལྔ། རྒྱན་སྣང་བདེན་གསུམ་སོགས་འཆད་ཉན་རྒྱ་ཆེར་སྤེལ།གཞན་ལམ་རིམ་གྱི་བསྟན་བཅོས་ཀྱང་མཛད་པ་སོགས་མཐར་དགུང་ལོ་ལྔ་བཅུ་རྩ་གཅིག་པ་རབ་བྱུང་གཉིས་པའི་ས་མོ་གླང་(སྤྱི་ལོ་1109)ལོར་སྐུ་གཤེགས།
        དེ་ཡང་རྔོག་བློ་ལྡན་ཤེས་རབ་འཚོ་བཞུགས་ཀྱི་དུས་སྐབས་དེ་ནི་ཏག་ཏག་བཙན་པོ་དར་མས་བསྟན་པ་ལ་བཀག་དམ་ཆེ་བས་ཆོས་ལ་རང་དབང་དུ་དད་པ་བྱེད་དུ་མི་འཇུག་པར་ཡུན་རིང་སོང་ཞིང་། བསྟན་པ་སླར་གསོས་པ་ན་ཆོས་དང་ཆོས་མིན་གསལ་པོར་འབྱེད་མ་ཐུབ་པར་གཞན་གྱི་གཡམ་ལ་རྒྱུག་མཁན་མང་དུ་བྱུང་བ་དཔེར་ན། སྐྱེ་བོ་རྣམས་ཡུན་རིང་ཟས་བཀག་སྟེ་བཀྲེས་པའི་སྡུག་བསྔལ་གྱིས་མནར་བའི་ཚེ། ཟས་དང་ཕྲད་པ་ན་བཟང་ངན་མ་ཕྱེ་བར་ལོངས་སྤྱད་དེ་ནད་དུ་གྱུར་པ་བཞིན་ནོ་ཞེས་རང་རེའི་ལོ་རྒྱུས་དང་ཆོས་འབྱུང་སོགས་སུ་བཀོད་པའི་སྤྱི་ཚོགས་ཀྱི་གནས་བབ་དེ་ལྟ་བུར་ཕྲད་ཅིང་། 《རྣམ་ངེས་》ཀྱི་མཆོད་བརྗོད་དུ། “གང་ལ་འཕགས་པ་ཉིད་བྱོན་རྗེས་གཟུང་དེ་ནི་དཔལ་ལྡན་དྲི་མེད་བློ་མངའ་སྟེ། །འཇིག་རྟེན་བློ་གྲོས་བླུན་པོ་འདི་ནི་ཚིག་བརླིང་བ་གསལ་བར་མི་ཤེས་པས། །ཉེར་བསྟན་འཇིག་རྟེན་རྒྱས་མཛད་དེ་ལ་རྨོངས་པས་ཅུང་ཟད་སྨད་བྱས་ནའང་། །དོན་མིན་སྐྱེ་བར་འགྱུར་བ་ཡིན་པས་བརྩེ་བས་དེ་ཡི་ལུགས་ནི་གསལ་བར་བྱ། །” ①ཞེས་པ་ནི་ནང་གི་རྒྱུ་ཡིན་ཏེ། 《རྣམ་ངེས་》རྩོམ་པའི་རྒྱུ་གཙོ་བོ་ནི་སློབ་དཔོན་གྱིས་རྟོག་གེ་མི་ཤེས་སོ་ཞེས་སྨོད་པ་ལ་སྙིང་རྗེས་མ་བཟོད་པར་བསྟན་བཅོས་འདི་མཛད་པ་དང་། རིགས་པའི་བསྟན་བཅོས་འདིའི་ཁ་གཏད་ས་ཡང་ཕྱི་རོལ་རྟོག་གེ་སྨྲ་བ་དེ་དག་ཡིན་པར་གསུངས་པ་སོགས་ཀྱིས་མཚོན་རྒྱུ་རྐྱེན་དེ་གཉིས་ཚོགས་པ་ལ་བརྟེན་ནས་རྔོག་ལོ་ཆེན་པོས་གཏན་ཚིགས་ཀྱི་ཆ་གཙོ་བོར་སྟོན་པའི་གཞུང་《རྣམ་བཤད་ཆེན་པོ་》འདི་བརྩམས་པ་རེད་ལ། རིག་གནས་ཀྱི་གདམ་ག་དེ་ལྟར་བྱས་པ་ནི་སྐབས་དེའི་སྤྱི་ཚོགས་ལ་མཚོན་ན་རིགས་པའི་གཞུང་ལུགས་ཀྱི་མཛུབ་སྟོན་ཡང་དག་ཅིག་ཀྱང་རེད། བོད་ཀྱི་རིག་གནས་བསམ་བློའི་ལོ་རྒྱུས་ཐོག་ཏུ། “སྤྱིར་བོད་ཀྱི་མཁན་པོས་བསྟན་བཅོས་མཛད་པ་ཆ་ཤས་ཙམ་ལས་མེད་ན་ཡང་བདག་ཉིད་ཆེན་པོ་འདི་ནས་བསྟན་བཅོས་རྒྱས་པ་མཛད་པའི་དབུ་ཚུགས་ནས་རྣམ་པར་ངེས་པའི་དཀའ་འགྲེལ་སོགས་ཆོས་འབྱུང་གི་དཀར་ཆག་ཏུ་ཚུད་པའི་རྟོག་གེའི་ཆོས་ཕྲན་དུ་མ་དག་མཛད།” ② ཅེས་བརྗོད་པ་ལས། ཁོང་ལ་མཁྲེགས་འཛིན་མི་མངའ་ཞིང་སྔར་མེད་སྒྱུར་བཅོས་ཀྱི་བསམ་བློ་ལྡན་པ་མཚོན་ནུས། ཆོས་གསུང་བ་ལ་ཤིན་ཏུ་མཁས་པས་ལྷ་ས་དང་བསམ་ཡས་སོགས་སུ་སློབ་མ་ཁྲི་ཕྲག་མང་པོ་འདུས་མྱོང་བ་སྟེ། “ཆོས་གཞི་བསྐོར་བའི་གནས་ཀྱང་། །ལྷ་ས་བསམ་ཡས་སྒང་ཐོག་དང་། ལྷ་ཡངས་པའམ་སྨྱུག་གུ་སྣར། །འཁོར་གྱི་གྲངས་ནི་རིམ་པ་བཞིན། །ཁྲི་དང་སུམ་སྟོང་བདུན་བརྒྱ་དང་། །གཉིས་སུ་སྟོང་ཚོ་བཅུ་གསུམ་ཚོགས། །ཁྲི་ཚོ་གཉིས་དང་ཁྲི་ཚོ་གཅིག །རྣལ་འབྱོར་ཕོ་མོ་ཁྲི་ཚོ་གཉིས། །ཚོགས་སུ་ལུང་རིགས་སྨྲ་བ་ཡི། །ཞར་ཆོས་སྟོང་དང་བརྒྱད་བརྒྱ་དང་། །བརྒྱད་ཅུ་རྩ་ལྔ་ཡོད་པ་ལ། །རྒྱན་དང་ཆོས་མཆོག་ལྔ་བཅུ་ལྔ། །རྣམ་ངེས་ཉིས་བརྒྱ་ལྔ་བཅུ་ལྔ། །ལུང་ཆོས་སྨྲ་བ་སྟོང་དང་ནི། །ལྔ་བརྒྱ་བདུན་ཅུ་རྩ་ལྔ་ཡོད། །” ③ཅེས་བཀོད་ཅིང་། དེ་ལས་སྐབས་དེར་གདན་ས་གསང་ཕུར་བཀའ་གདམས་པའི་མཁས་པས་ཚད་མའི་ཡིག་ཆའམ་སློབ་དེབ་གཙོ་བོ་ནི་《རྣམ་ངེས་》གཅེས་སུ་བཟུང་བ་ཞར་ལ་ཤེས་ཐུབ། ཡང་། “ལོ་ཆེན་མ་བྱོན་པའི་གོང་ན་འདུལ་བ་དང་ཆོས་མངོན་པའི་བཤད་ཉན་དར་སོ་ཡོད་ཀྱང་།རིགས་པའི་ལམ་ནས་དགག་སྒྲུབ་ཀྱི་རྣམ་པར་བཞག་པ་ཆོས་ཀྱི་གྲགས་པའི་གཞུང་དང་མཐུན་པར་འཆད་ཤེས་པ་མ་བྱུང་ངོ་། །” ④ཞེས་གསུངས་ཏེ། ཁོང་གི་འཆད་ཚུལ་དེ་གཞུང་ལུགས་རྐྱང་རྐྱང་ངོ་སྤྲོད་ཙམ་དང་བསྐྱར་ཟློས་ཙམ་མ་ཡིན་པར་རིགས་པའི་བསྟན་བཅོས་འཆད་པའི་ཆོས་ཉིད་དང་མཐུན་པར་བྱས་པས་ཡིན་ནོ། །འདིའི་ཡར་སྔོན་དུ་《ཚད་མ་རྣམ་འགྲེལ་》དང་《རྣམ་ངེས་》འཆད་མཁན་མེད་པ་མིན་ཡང་། དེ་དག་གི་སྐབས་སུ་ནི་ཆོས་དང་རིག་གནས་གང་ཅིའི་ཐད་ནས་ཀྱང་རྒྱས་པར་འཆད་ཤེས་མཁན་མ་བྱུང་བས་གཞི་བརྟན་པོ་འདིང་མ་ཐུབ་པར་མངོན། ཚུལ་དེ་ནི་པཎ་ཆེན་ཤཱཀྱ་མཆོག་ལྡན་གྱིས་“ལྷ་བཙན་པོ་ཁྲི་སྲོང་ལྡེའུ་བཙན་གྱི་དུས་བྱང་ཆུབ་སེམས་དཔའ་ཞི་བ་འཚོས་ཚད་མའི་དེ་ཁོ་ན་ཉིད་བསྡུས་པ་དང་། ཀ་མ་ལ་ཤཱི་ལས་དེའི་འགྲེལ་པ་སྟོང་ཕྲག་བཅོ་བརྒྱད་དང་། སྡེ་བདུན་ཕྱི་མའི་རྩ་འགྲེལ་གྱི་ཆ་ཤས་ཡེ་ཤེས་སྡེ་སོགས་ཀྱི་འགྱུར་ལ་བརྟེན་ནས་ལྷ་ས་དང་བསམ་ཡས་སུ་ཚད་མའི་བསྟན་བཅོས་ཀྱི་བཤད་ཉན་ཆ་ཤས་ཙམ་བྱུང་བར་གྲགས་ཀྱང་རྐང་ཚུགས་པ་དང་རྒྱུན་ཆགས་སུ་མ་བྱུང་།” ⑤ ཞེས་གསུངས་པ་ལྟར་ཡིན་ནོ། །
གཞན་ཡང་། “རྣམ་འགྲེལ་མཛད་པའི་སྐབས་འདིར་ཡང་དོན་དམ་བདེན་པའི་ངོས་འཛིན་ཡང་ཐ་སྙད་དུ་བློ་ཡི་ཡུལ་ལས་འདས་པར་བཞེད་དོ། །གལ་ཏེ་འོ་ན་ལོ་ཆེན་གྱིས་རྣམ་འགྲེལ་དབུ་མར་བཀྲལ་ན་ཐལ་རང་དུ་གྲགས་པ་གཉིས་སམ། རྣམ་བརྫུན་དབུ་མར་གྲགས་པ་དག་ལས་གང་དུ་འགྲེལ་ཞེ་ན། རྣམ་འགྲེལ་མཛད་པའི་རིགས་པ་ཉིད་ལ་བརྟེན་པས་ན་ཐལ་རང་གི་ལུགས་ལྟར་ཡང་མི་འགྲེལ་ལ།སྟོང་ཉིད་ཀྱི་ངོས་འཛིན་བྱམས་ཆོས་བསྒོམ་ལུགས་པའི་འཆད་ཚུལ་ལྟར་ཡེ་ཤེས་ལའང་མི་འཆད་པས་ན་ལུགས་ཕྱི་མ་དེའང་མ་ཡིན་ནོ། །ལུགས་དེ་དག་གང་ཡང་མ་ཡིན་ན་དབུ་མའི་འགྲེལ་ཚུལ་གཞན་མེད་དོ་སྙམ་ན། འགྲེལ་ཚུལ་འདི་ནི་སྲོལ་འབྱེད་སྔ་མ་དེ་དག་ལ་རག་མ་ལས་པའི་ཤིང་རྟའི་སྲོལ་ཆེན་པོ་གཞན་ཞིག་སྟེ། ཇི་སྐད་དུ་སྤྲིངས་ཡིག་བདུད་རྩི་ཐིགས་པ་ལས། ཆོས་རྣམས་རང་བཞིན་མེད་པའི་ཚུལ་ལ་འཇུག་པའི་སྒོ། །ཡང་དག་རིགས་མཆོག་ཀླུ་སྒྲུབ་ཞབས་ལས་འབྱུང་བ་གང་། །རིགས་པའི་མཐར་ཐུག་རྣམ་འགྲེལ་མཛད་པའི་གཞུང་བཟང་ལས། །གསལ་བར་རྟོགས་ནས་ལུགས་ངན་གཞན་ཀུན་རྩྭ་བཞིན་དོར། །ཞེས་གསུངས་སོ། །འདིས་རྣམ་འགྲེལ་གྱི་རྒྱན་ལ་དཔྱད་པ་མཛད་པ་ནི་ཆ་ཤས་ཙམ་ཡང་མ་མཐོང་ལ། ཆོས་མཆོག་གི་བཤད་པ་རྣམས་ཐུགས་ཡུལ་དུ་འཐད་པ་རྣམས་ཐད་སོར་བཞག་ནས་མི་འཐད་པར་དགག་པ་མང་དུ་མཛད་དོ། །” ⑥ཞེས་བརྗོད་པ་ལྟར། བཀའ་གདམས་པའི་མཁས་དབང་འདི་ནི་གཞན་གྱི་ལུང་ཙམ་གྱི་རྗེས་སུ་མི་འབྲང་ཞིང་། གཞན་གྱི་ངོ་བསྲུང་བ་ལ་མི་དགའ་བར་རིགས་པའི་རྗེས་སུ་འབྲང་བར་སྤྲོ་ཞིང་། གསར་གཏོད་ཀྱི་བློ་གྲོས་དང་ལྡན་པ་ཞིག་ཡིན་པ་ཤེས་ཐུབ། འདིར་ཚད་མ་ལས་གཞན་པའི་ཕར་ཕྱིན་གཙོ་བྱས་པའི་མཛད་རྗེས་གཞན་དག་ཀྱང་མཐའ་ཡས་པ་ཞིག་མཆིས་ནའང་། ཁོང་གིས་ཚད་མའི་འཕྲིན་ལས་གཅིག་པུས་ཀྱང་གདུལ་བྱའི་རེ་བ་ཚིམ་པར་བྱས་ཤིང་ཤུགས་རྐྱེན་ཡུན་རིང་ཐེབས་མྱོང་ཡོད་པ་སྟེ། “བདག་ཉིད་ཆེན་པོ་དེ་ལ། ལྷ་མངའ་བདག་རྩེ་ལྡེ་ཞེས་བྱ་བས་འགྱུར་གྱི་སྦྱིན་བདག་མཛད་ནས་རྣམ་འགྲེལ་རྒྱན་དང་བཅས་པ་དང་། རྣམ་ངེས་ཆོས་མཆོག་དང་བཅས་པ་བསྒྱུར་ཞིང་འཆད་ཉན་གྱིས་གཏན་ལ་ཕབ་པས་དིང་སང་གི་བར་དུ་ལོ་ཕྱེད་དང་བཅས་པའི་ལྔ་བརྒྱ་ཙམ་དུ་ཉན་བཤད་ཀྱི་རྒྱུན་མ་ཆད་པར་བྱོན་པ་འདི་ཡིན་ལ།” ⑦ ཞེས་བྱ་བ་སོགས་ཀྱིས་མཚོན་པ་ཡིན་ནོ། །
གཉིས་པ། 《རྣམ་བཤད་ཆེན་པོ་》གཞིར་བྱས་ཏེ་རྔོག་བློ་ལྡན་ཤེས་རབ་ཀྱི་ཚད་མའི་དགོངས་པ་གླེང་བ།
        《ཚད་མ་ཀུན་ལས་བཏུས་》ལས་བསྟན་པའི་ཡང་དག་པའི་ཤེས་པ་འཁོར་དང་བཅས་《རྣམ་ངེས་》ཀྱི་གཞུང་འདི་ལ་བརྟེན་ནས་རྟོགས་པས་ན། 《རྣམ་འགྲེལ་》ལྟ་བུར་མངོན་སུམ་གྱི་ལེའུ་ལས་རྣལ་འབྱོར་མངོན་སུམ་དེ་ཐམས་ཅད་མཁྱེན་པར་བསྒྲུབ་པ་ནི་དཀའ་བའི་གནས་དང་གཙོ་བོའི་དོན་དུ་གཟིགས་ཤིང་། སྟོན་པ་ཚད་མར་དངོས་པོ་སྟོབས་ཞུགས་ཀྱི་རིགས་པས་རྒྱས་པར་གཏན་ལ་འབེབས་པའི་ལེའུ་ཞིག་ཟུར་དུ་མཛད་པ་དེ་དང་ཡང་ཁྱད་པར་ལྡན་ནོ། །འདིར་གླེང་བའི་《ཚད་མ་རྣམ་ངེས་ཀྱི་དཀའ་གནས་རྣམ་བཤད་》ནི་བོད་ལ་མཚོན་ན་《རྣམ་ངེས་》ཀྱི་གཞུང་གི་འགྲེལ་པ་ཐོག་མ་ཉིད་ཡིན་པ་དང་། རིགས་པའི་བསྟན་བཅོས་འདི་རྩ་བའི་ཆ་ནས་སློབ་དཔོན་ཕྱོགས་གླང་གིས་སྐུ་ཚེ་སྨད་དུ་བརྩམས་པའི་《ཚད་མ་ཀུན་ལས་བཏུས་པ་》ཡི་དགོངས་པ་དང་མཐུན་ཞིང་། བྱེ་བྲག་ཏུ་དཔལ་ཆོས་ཀྱི་གྲགས་པའི་《རྣམ་ངེས་》ཀྱི་དགོངས་པ་རྒྱུན་འཛིན་གོང་འཕེལ་དུ་བཏང་བའི་བརྩམས་ཆོས་ཤིག་ཡིན་པས་ན། 《རྣམ་བཤད་ཆེན་པོ་》རུ། ཡང་དག་པའི་ཤེས་པ་འཁོར་དང་བཅས་རྟོགས་པ་དེའི་དོན་དུ་སྤྱིའི་ཆ་ནས་ཚད་མར་གྱུར་པ་དང་། གཞལ་བྱ་གཉིས་སུ་ངེས་པའི་རྒྱུ་མཚན་གྱིས་ཚད་མ་གཉིས་སུ་ངེས་ཚུལ།རང་རིག་མངོན་སུམ་དང་མངོན་སུམ་ལྟར་སྣང་བདུན་གྱི་རྣམ་གཞག གཞན་སེལ་གྱི་རྣམ་པར་གཞག་པ་བཅས་ལ་འགྲེལ་བརྗོད་གནང་བ་དང་ཆབས་ཅིག ཁྱད་པར་གྱི་དབང་དུ་བྱས་པ་ལས་རང་གི་ལྟ་ཚུལ་དང་བཅས་ཏེ། གཞལ་བྱ་ཡུལ་གཉིས་བཏོན་པའི་སྒོ་ནས་ཚད་མའི་རྣམ་གཞག་བརྗོད་པ་དང་། མངོན་སུམ་ཐུན་མོང་གི་མཚན་ཉིད་ཀྱི་ཟུར་དུ་མ་འཁྲུལ་བའི་ཚིག་གསལ་བར་མཛད་ཚུལ། ཚད་མིན་གྱི་བློའི་བཞེད་ཚུལ་སོགས་ཞིབ་ཕྲར་གསུངས།བསྟན་བཅོས་ཀྱི་ལེའུ་ཡི་གོ་རིམ་སྒྲིག་སྟངས་ཀྱང་། དང་པོ་མངོན་སུམ་གྱི་ལེའུའི་དཀའ་གནས་རྣམ་པར་བཤད་པ་ཞེས་པ་དང་། ས་བཅད་གཉིས་པར་རང་དོན་རྗེས་དཔག་གི་དཀའ་གནས་རྣམ་པར་བཤད་པ། ས་བཅད་གསུམ་པར་གཞན་དོན་རྗེས་དཔག་གི་ལེའུའི་དཀའ་གནས་རྣམ་པར་བཤད་པ་བཅས་གསུམ་གྱིས་གྲུབ་ཅིང་། རྗོད་བྱེད་ཚིག་ལ་བསྟན་བཅོས་འདིའི་མགོ་ནས་མཇུག་བར་བརྡ་རྙིང་གིས་བཅིངས་པ་རེ་འགའ་ཡོད་མོད། དེ་ནི་དོན་གོ་བ་ལ་གེགས་སུ་མི་འགྲོ་སྟེ་དཔེར་ན། མི་ལ་མྱི་དང་། སོགས་ལ་སྩོགས་ཞེས་པ་ལྟ་བུ་ཙམ་མོ། །
        《རྣམ་བཤད་ཆེན་པོ་》འདིའི་རྩོམ་པ་པོ་ཉིད་གྲུབ་མཐའ་གང་ལ་གནས་ཤིང་ལྷག་པར་གྲུབ་མཐའ་དེའི་བསམ་བློའི་ཁྱད་ཆོས་རགས་ཙམ་ཞིག་བརྗོད་ན། དེ་ཡང་“མདོ་སྡེ་སྣ་ཚོགས་པའི་ཐེག་ཆེན་ནི་མདོ་མང་འདི་རྣམས་རེ་རེ་བཞིན་ཉམས་སུ་ལེན་པའི་གདམས་པ་དང་བཅས་པ་གསན་ཞིང་མཁྱེན། ཤེར་ཕྱིན་གྱི་སྒོར་ཞུགས་པའི་ཐེག་ཆེན་ནི་ལྟ་བ་སེམས་ཙམ་དུ་བཀྲོལ་བ་ཤཱནྟི་བའི་ལུགས་དང་དབུ་མར་བཀྲོལ་བ་སེང་བཟང་གི་ལུགས་དང་མདོར་ན་ཐེག་ཆེན་གྱི་སྡེ་སྣོད་བསྟན་བཅོས་ཐམས་ཅད་བྲི་ཀ་མ་ལར་བཞུགས་པ་དེ་ཐམས་ཅད་རྫོགས་པར་མཁྱེན། གསང་སྔགས་རྟོག་པའི་རྒྱུད་བྱ་བ་སྤྱོད་པ་གཉིས་ཀ། རྣམ་འབྱོར། རྣལ་འབྱོར་ཆེན་པོ་རྣལ་འབྱོར་བླ་མེད་དང་བདུན་དུ་བཞེད་པའི་རེ་རེའི་ཡང་རྒྱུད་སྡེ་མཐའ་ཡས་པ་གསན་ཞིང་མཁྱེན།” ⑧ ཞེས་བཀོད་ཅིང་། མདོ་སྔགས་ལེགས་བཤད་ཀྱི་འབྱུང་གནས་ཇོ་བོ་རྗེའི་ཡོན་ཏན་བརྗོད་པ་ལས་རྔོག་ལོ་ཆེན་པོའི་བསམ་བློའི་ཉིང་བཅུད་དེ་ཡང་ཆུ་ལ་ཆུ་བཞག་པ་ལྟར་གསལ་ལེར་འཆར་འོང་ཞིང་། དེ་བརྒྱུད་ནས་ཀྱང་ཁོང་གི་ཚད་མའི་བསམ་བློར་རྒྱུས་རགས་ཙམ་ལོན་ཐུབ། ཡིན་ནའང་རང་ཉིད་གྲུབ་མཐའ་གང་ལ་གནས་ན་བརྩམས་པའི་བསྟན་བཅོས་ཀྱི་དགོངས་པ་ཡང་ངེས་པར་གྲུབ་མཐའི་བསམ་བློ་དེར་ཡོངས་སུ་མཐུན་པའི་ངེས་པ་མེད་དེ། རྔོག་ལོ་ཆེན་པོའི་དགོངས་པ་མཐར་ཐུག་དབུ་མའི་ལྟ་བ་ཡིན་མོད། རིགས་པའི་བསྟན་བཅོས་《རྣམ་བཤད་ཆེན་པོ་》བརྩམས་པའི་ལྟ་གྲུབ་ཀྱི་གཞོལ་ས་དེ་གཙོ་བོར་ཕྱི་རོལ་དོན་གྲུབ་ཁས་ལེན་པ་ནས་རིམ་གྱིས་རང་རིག་དང་རྣལ་འབྱོར་མངོན་སུམ་བཅས་ཀྱིས་མཚོན་མདོ་སེམས་གཉིས་ཀའི་ཐུན་མོང་གི་ལུགས་སུ་བཞེད་པའི་ཕྱིར།ནང་པའི་ཚད་མ་རིག་པར་མཚོན་ན་བརྗོད་བྱ་གཟུགས་ནས་རྣམ་མཁྱེན་གྱི་བར་ཏེ་ཡུལ་དང་ཡུལ་ཅན་གྱིས་བསྡུས་པའི་ཤེས་བྱ་དང་གཞལ་བྱ་རྣམས་མི་བསླུ་བའི་ཤེས་པས་རྟོགས་པར་བྱེད་པའི་བྱེད་པ་ཡིན་ཞིང་། དེ་བས་“ཚད་མ་བསླུ་མེད་ཅན་ཤེས་པ། །དོན་བྱེད་ནུས་པར་གནས་པ་ནི། ། མི་བསླུ།” ⑨ ཞེས་དང་། “མ་ཤེས་དོན་གྱི་གསལ་བྱེད་ཀྱང་། །” ⑩ཞེས་སྐྱེས་བུའི་དོན་འགྲུབ་པ་ལ་ཡང་དག་པའི་ཤེས་པ་ནི་སྔོན་དུ་འགྲོ་བ་ཅན་ཡིན་པ་དེའི་ཆེད་དུ་ཚད་མ་སྤྱིའི་མཚན་ཉིད་བསྟན་ལ། ཚུལ་འདིར་སྤྱིར་བཏང་མདོ་སྡེ་བ་ནས་དབུ་མ་རང་རྒྱུད་པའི་བར་དགོངས་པ་མཐུན་པར་གནས་པ་ཞིག་ཡིན་མོད། དེར་ཡང་དཔལ་ལྡན་ཟླ་བ་གྲགས་པས་ཕྱོགས་གླང་གི་ཚད་མའི་མཚན་ཉིད་དང་མངོན་སུམ་འདོད་ཚུལ་སོགས་རང་མཚན་གྱི་ཆོས་གཞིར་བྱས་པའི་བཤད་པ་ཡིན་པ་སོགས་ཀྱི་སྐྱོན་བརྗོད་པ་དེ་བཀག་ཅིང་། རང་ལུགས་ལ་འཇིག་རྟེན་ན་མི་བསླུ་བ་ཙམ་ལ་ཚད་མར་འཇོག་པས། མི་བསླུ་བ་ཙམ་གྱིས་ཚད་མའི་དོན་རྫོགས་པ་ཡིན་པ་དང་། ཚད་མ་ལ་གསར་རྟོགས་ཀྱི་ཁྱད་པར་སྦྱོར་མི་དགོས་ཞེས་པའི་བཞེད་ཚུལ་བཏོན་ཡོད།
        རྔོག་བློ་ལྡན་ཤེས་རབ་ཀྱིས་《རྣམ་བཤད་ཆེན་པོ་》རུ་རང་གི་ཚད་མའི་བཞེད་ཚུལ་ཇི་ལྟར་གསུངས་ཤེ་ན། “ཅི་དངོས་པོའི་རང་བཞིན་ལ་ཕྱིན་ཅི་མ་ལོག་པ་ཡང་དག་པའི་ཤེས་པར་འདོད་དམ། འོན་ཏ་མྱི་བསླུ་བ་ཡང་དག་པའི་ཤེས་པ་ཡིན། དང་པོ་ལྟར་ན་རྗེས་སུ་དཔག་པ་ལ་མ་ཁྱབ་བོ། ། གཉིས་པ་ལ་ཡང་རྣམ་པར་རྟོག་པ་གཉིས་ཏེ། མྱི་བསླུ་བ་ནི་ཐོབ་པ་ཡིན་ལ། དེ་བྱེད་པ་མཚན་ཉིད་ཡིན་ན། ཅི་བཟུང་བའམ་མ་བཟུང་བ་ཐོབ་པར་བྱེད་པ་ཡིན། དང་པོ་ལྟར་ན་མྱི་སྲིད་དེ། མངོན་སུམ་ལ་སྣང་བ་ནི་སྐད་ཅིག་མ་ཡིན་པ་དང་། རྗེས་སུ་དཔག་པ་ལ་སྣང་བ་ཡང་སྒྲོ་བཏགས་པ་ཡིན་པས། ཐོབ་པར་མྱི་ནུས་པའི་ཕྱིར་རོ། །གཉིས་པ་ལྟར་ན་ཧ་ཅང་ཁྱབ་པ་ཡིན་ཏེ། མ་བཟུང་བ་ལ་ནི་ལོག་པའི་ཤེས་པ་ལ་སོགས་པས་ཀྱང་ཐོབ་པའི་ཕྱིར་རོ། །” ⑪ཞེས་བཀོད་པ་སྟེ། ནང་པའི་ཚད་མ་ལྟར་མཐར་ཐུག་དངོས་པོ་ཙམ་ཐོབ་པ་ཡང་མ་ཡིན་ཞིང་། དངོས་པོའི་དོན་ཐོབ་པར་བྱེད་པ་ཡིན་པའི་ཚད་མའི་ལྟ་གྲུབ་ཀྱི་རྗེས་སུ་འབྲངས་ཤིང་། ཁོང་གིས་གཞལ་བྱ་ཡུལ་གཉིས་བཏོན་པའི་སྒོ་ནས་ཚད་མ་གཏན་ལ་འབེབས་པའི་ཚུལ་ཉིད་ཡིན་ལ། “དེ་ལ་དོན་བྱ་བ་ལ་སླུ་བ་མྱེད་པའི་ཕྱིར་རོ། ། ཞེས་བྱ་བས་ནི་དངོས་པོ་འཛིན་པ་ཉིད་སྤངས་ཏེ། མྱི་སླུ་བར་བྱེད་པ་ཁས་བླངས་པས། འདོད་པའི་མཚན་ཉིད་གཟུང་བའོ། ། འདི་ལྟར་མྱི་སླུ་བར་བྱེད་པ་ནི་ཐོབ་པར་བྱེད་པའི་ནུས་པ་ཡོངས་སུ་བཅད་པའི་དོན་གྱི་མཐར་ཐུག་པ། ཤེས་པར་བྱེད་པའི་བྱེད་པ་དང་འབྲེལ་བ་སྟེ།ངོ་བོའི་ཁྱད་པར་གཉིས་ཀྱིས་ཡོངས་སུ་ཟིན་པའོ། །དེས་ཀྱང་དངོས་པོ་རྟོགས་པ་ཙམ་ཚད་མ་ཡིན་པ་དང་། མ་བཅད་པའི་དོན་ཐོབ་པའི་ནུས་པ་ཚད་མ་ཡིན་པ་དང་། འཇལ་བྱེད་ཀྱི་བྱེད་པ་མྱེད་པ་ལ་ཡོད་པའི་ནུས་པ་ཚད་མ་ཡིན་པ་རྣམས་རིམ་བཞིན་བསལ་ལོ། །དེས་ན་མྱི་སླུ་བར་བྱེད་པའི་སྒྲའི་ཉིད་ཀྱིས་དོན་གསུམ་ལས་ཐ་དད་པ་འཕངས་ཀྱང་དེ་ཉིད་གསལ་བར་བྱ་བའི་ཕྱིར་ཡུལ་དང་བྱེད་པ་བསྟན་ཏོ། །དེ་ལ་འཇུག་པ་ན་ཞེས་བྱ་བས་ནི་གཟུང་ཡུལ་ཚད་མའི་གཞལ་བྱ་ཡིན་པ་བཀག་ལ། ཡོངས་སུ་བཅད་ནས་ཞེས་བྱ་བས་ཀྱང་རྟོགས་པའི་བྱེད་པ་ཅན་ཚད་མ་ཡིན་པ་དང་།རྟོགས་པའི་ཡུལ་ཚད་མའི་ཡུལ་དུ་བསྒྲུབས་པས་ན་ཤུགས་ཀྱིས་བཟུང་བ་ལས་གཞན་རྟོགས་པའི་ཡུལ་ཐོབ་བྱར་བསྟན་ཏེ། དེས་ན་ཡོངས་སུ་བཅད་པའི་དོན་གྱི་མཐར་ཐུག་པ་དང་། ཤེས་བྱེད་ཀྱི་བྱེད་པ་ལ་གནས་པའི་ནུས་པ་ཚད་མ་ཡིན་པར་ཚིག་གཉི་ག་ཕན་ཚུན་ལྟོས་པས་འགྲུབ་བོ། །” ⑫ཞེས་གསུངས་པ་ནི། སྤྱིར་《རྣམ་ངེས་》སུ། འདི་དག་གིས་དོན་ཡོངས་སུ་བཅད་ནས་འཇུག་པ་ན་དོན་བྱ་བ་ལ་མི་བསླུ་བའི་ཕྱིར་རོ། །ཞེས་འདོད་པའི་ཚད་མའི་མཚན་ཉིད་དེར་དགོངས་པ་སྟེ། ཡོངས་སུ་བཅད་ནས་ཞེས་པ་དང་། འཇུག་པ་ན་ཞེས་པ་དང་། དོན་བྱ་བ་ལ་མི་བསླུ་བ་ཞེས་པའི་ཚིག་གསུམ་ཡོད་པ་ལས། ཚིག་ཐ་མས་ནི་འདོད་པའི་མཚན་ཉིད་བཟུང་བ། བར་མས་ནི་གཟུང་ཡུལ་ཙམ་ཚད་མ་ཡིན་པ་བཀག་ལ། ཚིག་དང་པོས་ནི་ཧ་ཅང་ཁྱབ་ཆེ་བ་སྤངས་པ་བཅས་མདོར་ན་དངོས་པོའི་དོན་ལ་མི་བསླུ་བ་ཙམ་ནི་ཡང་དག་པའི་ཤེས་པའམ་ཚད་མ་ཡིན་པར་རང་གི་མཚན་ཉིད་རིག་པའི་བསམ་བློ་དེ་བསྟན་པ་ཡིན། ཡང་དག་པའི་ཤེས་པ་དང་ཕྱིན་ཅི་ལོག་པའི་ཤེས་པ་བཅས་ཤེས་པའི་རྣམ་གཞག་ནི་ཚད་མ་རིག་པའི་གཞུང་ལུགས་ཀྱི་འགོ་རྩོམ་ས་དང་མཇུག་སྒྲིལ་ས་དེ་ཡིན་ལ། རྔོག་བློ་ལྡན་ཤེས་རབ་ཀྱི་《རྣམ་བཤད་ཆེན་པོ་》འདིར་སྔར་ནས་བློ་རིག་བདུན་དུ་ཕྱེ་བ་དེ་ལས་ཚད་མིན་གྱི་བློའི་བཞེད་ཚུལ་བཏོན་པ་འདི་ལྟ་སྟེ། “དེས་ན་ཚད་མ་མ་ཡིན་པའི་བློ་ལྔའོ། སྣང་ལ་མ་ངེས་པ་དང་། བཅད་པའི་ཡུལ་ཅན་དང་། ལོག་པའི་ཤེས་པ་དང་། ཡིད་དཔྱད་དང་། ཐེ་ཚོམ་རྣམས་ནི་་་་་་།” ⑬ཞེས་བགྲངས་པ་དང་། ཚད་མིན་གྱི་བློ་ལྔ་ལས། “དང་པོ་གཉིས་ལ་ནི་དོན་དམ་པར་བྱེད་པའི་རིག་པ་ཡོད་ཀྱང་། སྔར་མ་རྟོགས་པ་ངེས་པ་མྱེད་པ་ཡོངས་སུ་གཅོད་བྱེད་ཀྱི་ཐ་སྙད་དུ་བྱ་བ་མ་ཡིན་ལ། གསུམ་པ་ནི་ཡོངས་སུ་གཅོད་བྱེད་ཙམ་ཡིན་ཡང་། མི་སླུ་བ་མྱེད་པས་ལོག་པར་ཡོངས་སུ་གཅོད་པའི་ཕྱིར་དང་། བཞི་པ་ཡང་དེའི་རང་བཞིན་ཅན་ཡིན་པའི་ཕྱིར་ཏེ། དཔེར་ན་དུ་བ་དང་དུས་མཉམ་དུ་སྐྱེས་པའི་རླངས་པ་ལ་སོགས་པ་མྱེ་དང་ལྷན་ཅིག་ཡོད་ཀྱང་མྱེའི་འབྲས་བུ་མ་ཡིན་པའི་རང་བཞིན་མྱི་སྲིད་པ་ཡིན་ཏེ་དེ་དང་ལྡོག་པ་མ་གྲུབ་པའི་ཕྱིར་རོ། །དེ་བཞིན་དུ་བློ་འདི་དོན་དང་ལྷན་ཅིག་ཡོད་ཀྱང་ལོག་པའི་ཤེས་པ་དོན་མྱེད་པ་ཉིད་དུ་མྱི་སྤང་སྟེ། དོན་མྱེད་ཀྱང་འབྱུང་བའི་ཕྱིར་རོ། །ལྔ་པ་ནི་དོན་འགའ་ཡང་ཡོངས་སུ་གཅོད་པ་མ་ཡིན་ཏེ། གང་ཡང་རྣམ་པར་མྱི་གཅོད་པའི་ཕྱིར་རོ། །” ⑭ཞེས་བློའི་རྣམ་གཞག་ལ་རང་གི་ཐུན་མོང་མ་ཡིན་པའི་དགོངས་པ་བཞག་ཡོད། 
        ཕྱོགས་གླང་གིས་《ཀུན་བཏུས་》སུ། མིང་དང་རིས་སོགས་སྦྱོར་བ་ཡི། །རྟོག་པ་དང་བྲལ་མངོན་སུམ་མོ། །ཞེས་མངོན་སུམ་གྱི་མཚན་ཉིད་བསྟན། ཡིན་ནའང་རྟོག་བྲལ་ཡིན་ན་མངོན་སུམ་ཡིན་པས་མ་ཁྱབ་སྟེ། རྟོག་མེད་འཁྲུལ་ཤེས་རྨི་ལམ་ལ་སོགས་པའང་དེར་ཐལ་བའི་རིགས་པ་དང་འགལ་བས། འགའ་ཞིག་གིས་དེར་དགག་པ་མང་དུ་བྱས་མོད། སློབ་དཔོན་གྱིས་མངོན་སུམ་གྱི་མཚན་ཉིད་ཀྱི་ཟུར་དུ་མ་འཁྲུལ་བའི་ཚིག་དངོས་སུ་མ་སྨྲས་ཀྱང་དོན་གྱིས་བསྟན་པའི་ཕྱིར་ཏེ། ཁོང་གིས་མངོན་སུམ་ལྟར་སྣང་བཀོད་པའི་ཚེ་ན། འཁྲུལ་དང་ཀུན་རྫོབ་ཤེས་པ་དང་། །རྗེས་དཔག་རྗེས་སུ་དཔག་ལས་བྱུང་། །དྲན་དང་མངོན་འདོད་ཅེས་བྱ་བའི། །མངོན་སུམ་ལྟར་སྣང་རབ་རིབ་བཅས། །ཞེས་བརྗོད་པའོ། །《རྣམ་བཤད་ཆེན་པོ་》རུ་མངོན་སུམ་ཐད་ཀྱི་ལྟ་ཚུལ་ནི།“མཚན་ཉིད་ཀྱི་ངོ་བོ་ལ། རྟོག་པ་དང་བྲལ་བའི་རང་བཞིན་ནི་འཇིག་རྟེན་ན་གྲགས་པས་སྟོང་བ་སྟེ་མྱི་མཐུན་པ་མྱེད་དོ། །མ་འཁྲུལ་བའི་ངོ་བོ་ནི། སློབ་དཔོན་ཆོས་མཆོག་ལྟར་ན། གཟུང་བའི་དོན་ལ་ཕྱིན་ཅི་མ་ལོག་པའོ། །འདིར་ཐོབ་པར་བྱ་བའི་དོན་ལ་ཕྱིན་ཅི་ལོག་པ་ཙམ་ནི་མ་ཡིན་ཏེ།འཁྲུལ་བ་ནི་ལོག་པར་སྣང་བའོ་ཞེས་བཤད་པས་དེ་མྱེད་པ་མ་འཁྲུལ་བར་གྲུབ་པའི་ཕྱིར་རོ། །གཞན་ཡང་རྣམ་རྟོག་དངོས་མྱི་སྣང་ཕྱིར་ཞེས་མངོན་སུམ་ལྟར་སྣང་བར་བཤད་པའི་ཕྱིར་ཏེ། གཟུང་ཡུལ་འཁྲུལ་བ་མངོན་སུམ་ཡིན་ན་ནི།དངོས་པོ་མྱི་སྣང་བས་མངོན་སུམ་ལྟར་སྣང་བ་མྱི་འཐད་དོ། །དེས་ན་དངོས་པོ་མྱི་སྣང་བ་ལས་བཟློག་པས་གཟུང་ཡུལ་མ་འཁྲུལ་བར་གྲུབ་བོ། ། རྗེས་དཔག་གཟུང་བའི་ཡུལ་ལ་འཁྲུལ་བར་བཤད་པ་ན། མངོན་སུམ་དེ་ལས་བཟློག་པར་བཞེད་ན། གཟུང་བ་ལ་མ་འཁྲུལ་བ་གྲུབ་བོ། །དེས་སྟེ་མྱི་བཞེད་ན་རྗེས་དཔག་དམྱིགས་ཀྱིས་བརྗོད་པའི་རྒྱུ་ཅི་ཡོད། གཞན་ཡང་ཡང་དག་པའི་ཤེས་པ་མྱི་སླུ་བར་གྲུབ་བཞིན་དུ་ཡང་འདིར་མ་འཁྲུལ་བས་མྱི་སླུ་བ་དེ་ཉིད་བརྗོད་ན། མྱི་སླུ་བ་ནི་རྟོག་པ་དང་བྲལ་ཞིང་མྱི་སླུ་བའོ་ཞེས་བརྗོད་པ་མ་འབྲེལ་པར་ཡང་འགྱུར་རོ། །དེས་ན་གཟུང་བའི་དོན་ལ་མ་འཁྲུལ་བ་ཉིད་མཚན་ཉིད་དུ་འདོད་པར་བྱའོ་ཞེས་སོ། །འདི་ནི་མྱི་བདེན་ཏེ། ལོག་པར་སྣང་བ་ཞེས་བྱ་བས་འཁྲུལ་བའི་སྒྲའི་བརྗོད་བྱ་ཀུན་གྱི་མཚན་ཉིད་བཤད་པ་ནི་མ་ཡིན་ཏེ། གཟུང་བའི་ཡུལ་འཁྲུལ་བ་ཙམ་གྱི་མཚན་ཉིད་སྟོན་པ་ཡིན་ནོ། ། དེ་ལྟ་མ་ཡིན་ན། འཁྲུལ་བ་ཀུན་རྫོབ་ཤེས་པ་དང་། །ཞེས་བྱ་བ་ལ་ཡང་འཁྲུལ་བའི་དོན་ལོག་པར་སྣང་བ་ཙམ་དུ་འགྱུར་རོ། །དེ་ལྟར་ན་ཀུན་རྫོབ་ཤེས་པ་ལ་སོགས་པ་དེར་འདུས་པས་ལོགས་སུ་བརྗོད་པ་མྱི་འཐད་དོ། །” ⑮ཞེས་བཀོད་པར། མངོན་གྱུར་རམ་མངོན་སུམ་ནི་འཇིག་རྟེན་ཐུན་མོང་བའི་ཤེས་པ་ཞིག་ཡིན་པ་དང་། མངོན་སུམ་གྱི་རིག་གཞུང་ཡིན་ན་ནི་འཇིག་རྟེན་ཐུན་མོང་བའི་མཚན་ཉིད་རིག་པ་ཡིན་དགོས་ཏེ།མ་འཁྲུལ་བ་ནི་ཕྱི་རོལ་གྱི་དངོས་པོའི་གནས་ཚོད་དང་མཐུན་པའི་ཤེས་པ་དེ་ཡིན་པར་བརྗོད་ཅིང་། ད་དུང་རྗེས་དཔག་ཚད་མ་སྒྲུབ་བྱེད་ཀྱི་གཏན་ཚིགས་སྐོར་གྱི་བཞེད་པ་ནི་འདི་ལྟར། “དེ་ལ་སྤྱིར་གཏན་ཚིགས་དང་གཏན་ཚིགས་ལྟར་སྣང་བའི་རྣམ་པར་དབྱེ་བ་མྱི་ཤེས་པ་ལ་གང་ཞིག་ཆོས་འདི་རྣམས་དང་ལྡན་པ་ནི་གཏན་ཚིགས་ཡིན་ནོ་ཞེས་མཚན་ཉིད་དང་མཚོན་བྱའི་འབྲེལ་བའི་ཐ་སྙད་ངེས་པ་ནི་མཚན་ཉིད་གཏན་ལ་འབེབས་པ་ཞེས་བྱ་སྟེ། དངོས་པོའི་རིག་པ་ཙམ་ནི་མ་ཡིན་ནོ། །མཚན་ཉིད་གཏན་ལ་འབེབས་པ་ནི་ཐམས་ཅད་དུ་ཐ་སྙད་ངེས་པ་ལ་བྱའོ། །དངོས་པོ་རིག་པ་ཙམ་ནི་མཚན་ཉིད་དང་མཚོན་བྱའི་ཐ་སྙད་མྱི་ཤེས་པ་བ་ལང་ལ་སོགས་པ་ལའང་ཡོད་དོ། །དེས་ན་དངོས་པོ་རིག་པ་ཡིན་ན་ཡང་མཚན་ཉིད་དང་མཚོན་བྱའི་ཐ་སྙད་ཀྱི་འབྲེལ་བ་མྱི་ཤེས་པ་ནི་ཚད་མའི་མཚན་ཉིད་ལ་སོགས་པར་ཤེས་པར་ཐོབ་པ་མྱེད་དོ། །” ⑯ཞེས་རྔོག་ལོ་ཆེན་པོས་བཞེད་དོ། །གཞན་ཡང་《རྣམ་ངེས་》ཀྱི་གཞུང་དང་དེའི་འགྲེལ་པའི་བསྟན་བཅོས་ཡིན་ན་ད་དུང་གཞན་དོན་གྱི་ལེའུ་རྒྱས་པར་སྟོན་པ་སྟེ། ཚུལ་གསུམ་གྱི་རིག་པའི་རྨང་གཞིར་གྱུར་པའི་ཕྱོགས་ཆོས་ཅན་གཏན་ཚིགས་དགུའི་གཞུང་ལུགས་ཞིབ་ཕྲ་ལྡན་པ་དང་དེའི་སྐོར་གྱི་འགྲེལ་བརྗོད་གནང་བ་སོགས་ཡོད་དོ། ། 
གསུམ་པ། 《རྣམ་བཤད་ཆེན་པོ་》ཡིས་ཕྱིས་ཀྱི་མཁས་པའི་ཚད་མའི་དགོངས་པར་ཤུགས་རྐྱེན་ཐེབས་ཚུལ།
        སྤྱི་ལོ་1073ལ་རྔོག་ལེགས་པའི་ཤེས་རབ་ཀྱིས་གསང་ཕུ་ནེའུ་ཐོག་བཏབ་པ་ནས་བཟུང་བོད་ཀྱི་གྲུབ་མཐའ་རིས་མེད་ཀྱི་མཁས་པ་རྣམས་དེར་འདུས་ཤིང་། གཞུང་ལུགས་སྤྱི་དང་ཁྱད་པར་དུ་ཚད་མ་རིག་པའི་འཆད་ཉན་ཆེར་དར། གནས་དེར་ཚད་མའི་སློབ་རྟེན་ནི་《ཚད་མ་རྣམ་ངེས་》གཙོ་བོར་བྱས་ཤིང་། རྔོག་བློ་ལྡན་ཤེས་རབ་ཀྱིས་ཀྱང་དུས་སྐབས་དེར་《རྣམ་བཤད་ཆེན་པོ་》འདི་བརྩམས་པ་ཡིན་ལ། བརྩམས་ཆོས་དེ་གཞིར་བྱས་པའི་ཁོང་གི་ཚད་མའི་དགོངས་པས་ཕྱིས་ཀྱི་མཁས་པའི་ཚད་མའི་བསྟན་བཅོས་དག་ལ་ཤུགས་རྐྱེན་ཕྲན་བུ་རེ་མ་ཐེབས་པ་མེད་སྙམ། དེ་དག་གི་ཁྲོད་ནས་ཀྱང་ཚད་མའི་སློབ་དཔོན་ཕྱྭ་བ་ཆོས་ཀྱི་སེང་གེའི་ཚད་མའི་གཞུང་ལུགས་ལ་རྔོག་ལོའི་ཚད་མའི་དགོངས་པས་ཤུགས་རྐྱེན་མངོན་གསལ་ཐེབས་ཡོད། དེ་ཡང་བསྡུས་གྲྭ་རིག་པ་ནི་བོད་ཀྱི་ཐུན་མོང་མ་ཡིན་པའི་ཚད་མའི་གཞུང་ལུགས་ཤིག་ཡིན་ཞིང་། དེ་ལས་ང་ཚོས་རྒྱུན་དུ་མཐོང་བའི་རིགས་ལམ་ནི་ཐལ་འགྱུར་གྱི་རིགས་པ་དང་མུ་བཞིའི་རིགས་པ་སོགས་ཡིན། ཆོས་གཉིས་ཀྱི་བར་ལ་ཁྱད་པར་གང་ཡོད་མིན་མཐའ་གཅོད་པའི་ཐབས་སྟབས་བདེ་ཞིང་རྨད་དུ་བྱུང་བའི་རིགས་ལམ་འདི་ཉིད་ཕཱ་བ་ཆོས་ཀྱི་སེང་གེའི་དུས་དེར་གློ་བུར་བྱུང་བ་ཞིག་ནི་མིན་ཏེ། རྔོག་ལོའི་《རྣམ་བཤད་ཆེན་པོ་》རུ། “དེས་ན་མུ་བཞིར་འགྱུར་ཏེ། གཟུང་བ་ཡིན་ལ་འཇུག་ཡུལ་མ་ཡིན་པ་ནི་དཔེར་ན་སྒྲོ་བཏགས་པའམ། རང་གི་རྣམ་པ་ལྟ་བུའམ། ཡང་ན་རྣམ་པར་འཇོག་པའི་སྐད་ཅིག་མ་ངེས་པས་ཡུལ་དུ་མ་བྱས་པ་ལྟ་བུའོ།།འཇུག་ཡུལ་ཡིན་ལ་གཟུང་ཡུལ་མ་ཡིན་པ་ནི་དཔེར་ན་མྱིག་གི་བློ་ལ་རེག་པ་ལ་སྩོགས་ལྟ་བུའོ། །གཉི་ག་ཡིན་པ་ནི་རྣམ་པར་འཇོག་པའི་རྒྱུ་ངེས་པས་ཡུལ་དུ་བྱས་པའམ། འཛིན་པའི་རང་གི་ངོ་བོ་བཞིན་ནོ། །གཉི་ག་མ་ཡིན་པ་ནི་དུ་བ་འཛིན་པའི་བློ་ལ་མེ་ལྟ་བུའོ། །” ⑰ཞེས་གསུངས་པ་སྟེ། རྔོག་ལོས་སྔར་ནས་གསུངས་ཟིན་པའི་རྨང་གཞིའི་སྟེང་སློབ་དཔོན་ཕྱྭ་བས།“གཟུང་ཡུལ་ཡིན་ལ་འཇུག་ཡུལ་མ་ཡིན་པའང་ཡོད་དེ་སྒྲ་མི་རྟག་པར་དཔོག་པའི་རྗེས་དཔག་ལ་མི་རྟག་པའི་དོན་སྤྱི་སྣང་བས་དེ་གཟུང་ཡུལ་ཡིན་ཡང་དོན་སྤྱི་ལ་རྟག་པར་འཛིན་པའི་སྒྲོ་འདོགས་བཅད་པ་མེད་པས་འཇུག་ཡུལ་མ་ཡིན་པའམ་སྣང་ལ་མ་ངེས་པ་ལ་སྐད་ཅིག་མ་སྣང་བས་དེ་གཟུང་ཡུལ་ཡིན་ཡང་རྟག་པའི་སྒྲོ་འདོགས་མ་བསལ་བས་འཇུག་ཡུལ་མ་ཡིན་པ་དང་། ཡིད་དཔྱོད་ལ་དོན་སྤྱི་སྣང་བ་དང་ལོག་ཤེས་དང་དཔྱད་པའི་ཡུལ་ཅན་དང་ཐེ་ཚོམ་ལ་དོན་སྤྱི་སྣང་བ་དང་རྟོག་མེད་འཁྲུལ་བ་ཟླ་བ་གཉིས་ལ་སོགས་པར་སྣང་བ་དང་དཔྱད་པའི་ཡུལ་ཅན་གྱིས་མངོན་སུམ་ལ་སྔོན་པོ་སྣང་བ་རྣམས་བློ་དེ་དག་ལ་སྣང་བས་དེ་དག་གི་གཟུང་ཡུལ་ཡིན་ཡང་བློ་དེ་རྣམས་ཀྱིས་ཡུལ་དེ་རྣམས་ལ་སྒྲོ་འདོགས་བཅད་པ་མེད་པས་འཇུག་ཡུལ་མ་ཡིན་པ་ལྟ་བུའོ། །འཇུག་ཡུལ་ཡིན་ལ་གཟུང་ཡུལ་མ་ཡིན་པའང་ཡོད་དེ་སྒྲའི་མི་རྟག་པ་ལ་རྗེས་དཔག་གིས་རྟག་པའི་སྒྲོ་འདོགས་བསལ་བས་དེ་རྗེས་དཔག་གིས་འཇུག་པའི་ཡུལ་ཡིན་ཡང་མི་རྟག་པའི་དོན་རང་གི་མཚན་ཉིད་རྗེས་དཔག་ལ་མི་སྣང་བས་གཟུང་བའི་ཡུལ་མ་ཡིན་པ་དང་། རང་རིག་པས་དོན་སྤྱི་དང་ཟླ་བ་གཉིས་ལ་སོགས་པ་ལ་སྣང་ཚད་དུ་མེད་པའི་སྒྲོ་འདོགས་བཅད་པས་དེ་དག་རང་རིག་པའི་འཇུག་པའི་ཡུལ་ཡིན་ཡང་རང་རིག་ལ་དེ་དག་མི་སྣང་བས་རང་རིག་པའི་གཟུང་བ་མ་ཡིན་པ་ལ་སོགས་པ་མངོན་སུམ་གྱི་ཤུགས་ཀྱི་ཡུལ་རྣམས་ལྟ་བུའོ། །གཟུང་བའི་ཡུལ་དང་འཇུག་པའི་ཡུལ་གཉི་ག་ཡིན་པའང་ཡོད་དོ། །སྔོན་པོ་དེ་མངོན་སུམ་ལ་སྣང་བས་མངོན་སུམ་གྱི་གཟུང་ཡུལ་ཡང་ཡིན་ལ་མངོན་སུམ་གྱིས་དེ་ལ་སྔོན་པོ་མ་ཡིན་པའི་སྒྲོ་འདོགས་བསལ་བས་འཇུག་ཡུལ་ཡང་ཡིན་པས་དང་། དོན་སྤྱི་ཆོས་ཅན་དུ་བྱས་ཏེ་དཔྱད་མི་བཟོད་པས་ཀུན་རྫོབ་ཀྱི་བདེན་པའི་ཐ་སྙད་བསྒྲུབ་པ་ན་རྗེས་དཔག་དེ་ལ་དོན་སྤྱི་ཙམ་སྣང་བས་གཟུང་ཡུལ་ཡང་ཡིན་ལ་དོན་སྤྱི་དེ་ལ་ཀུན་རྫོབ་ཀྱི་བདེན་པ་མ་ཡིན་པའི་སྒྲོ་འདོགས་བཅད་པས་རྗེས་དཔག་དེའི་འཇུག་ཡུལ་ཡང་ཡིན་པ་ལྟ་བུའོ། །གཉི་ག་མ་ཡིན་པའང་ཡོད་དེ་མིག་གི་རྣམ་ཤེས་ལ་ལྟོས་ནས་སྒྲ་ལྟ་བུའོ། །” ⑱ཞེས་བརྗོད་དེ། རྔོག་ལོའི་ཚད་མའི་ཁྲོད་ཀྱི་མུ་བཞིའི་རིགས་པ་ནི་རྗེས་སུ་སློབ་དཔོན་ཕྱྭ་བས་རྒྱུན་བསྐྱངས་ཏེ་གོང་འཕེལ་བཏང་བ་ཡིན་པ་ཤེས། སློབ་དཔོན་ཐར་པའི་འབྱུང་གནས་སྦས་པས། “ཐལ་བར་འགྱུར་བ་ཞེས་བྱ་བ་འདི་ཅི་ཞེ་ན། ཚད་མས་གྲུབ་པའི་ཁྱད་པར་ཅན་གྱི་དོན་གྱིས་ཕ་རོལ་པོའི་མི་འདོད་པ་འབྱུང་བའི་ཆེད་དུ་ཐལ་བར་བྱེད་པ་ནི་ཐལ་བར་འགྱུར་བ་སྟེ།” ⑲ཞེས་གསུངས་པའི་ཐལ་འགྱུར་གྱི་རིགས་པ་དེ་ཡང་མང་པོ་ཞིག་གིས་དུས་རབས་བཅུ་གཉིས་པའི་སློབ་དཔོན་ཕྱྭ་བ་ཆོས་སེང་གི་དུས་སྐབས་དེ་ནས་བཟུང་འགོ་བརྩམས་པར་བརྗོད་མོད། དེའི་ཡར་སྔོན་གྱི་བརྩམས་ཆོས་《རྣམ་བཤད་ཆེན་པོ་》རུ། དང་པོ་མངོན་སུམ་དང་གཉིས་པ་རང་དོན། གསུམ་པ་རྗེས་དཔག་བཅས་ལེའུ་གསུམ་ལྡན་པའི་ནང་དུ་ཐལ་འགྱུར་གྱི་རིགས་པ་དེས་གང་སར་ཁྱབ་ནས་ཡོད་ཅིང་། འདིར་དེ་རྣམས་ལས་དཔེ་མཚོན་ཙམ་ཞུ་ན། “ཡུལ་ནི་མཚན་ཉིད་ཀྱི་གཞི་ལ་འདོད་དོ། །དེ་ཡང་ཁ་ཅིག་ན་རེ། ལའི་ཁྱད་པར་ཅན་གྱི་དུ་བ་ལྟ་བུ་ཁྱད་པར་ཅན་ཡིན་ནོ་ཞེའོ། །འདི་ལ་ཉེས་པ་གསུམ་སྟེ། མཚན་ཉིད་དང་མཚན་གཞི་ཐ་མྱི་དད་པར་ཐལ་བ་དང་། ཁྱབ་པ་དང་བསྒྲུབ་བྱ་རྟོགས་པ་ཐ་མྱི་དད་པར་ཐལ་བ་དང་། ཐུན་མོང་བ་མྱེད་པར་ཐལ་བའོ། །” ⑳ཞེས་དང་། “འོ་ན་རྟགས་མ་ཡིན་པ་རྗོད་པ་ཇི་ལྟར་སྒྲུབ་བྱེད་ཡིན་ཏེ། ཧ་ཅང་ཐལ་བའི་ཕྱིར་རོ་ཞེ་ན། ཐལ་བར་འགྱུར་བརྗོད་པ་ནི་གཉིས་ཏེ། སྒྲུབ་པ་དང་སུན་འབྱིན་པའོ། །དེ་ལ་སུན་འབྱིན་པ་ནི་གཏན་ཚིགས་ཀྱི་སྐྱོན་ཙམ་རྗོད་པ་ཡིན་པས་སྒྲུབ་བྱེད་མ་ཡིན་པར་འདོད་པ་ཁོ་ནའོ། །” (21)ཞེས་པ་སྟེ། རྔོག་ལོས་ཐལ་འགྱུར་གྱི་རིགས་པ་དེ་དུས་རབས་བཅུ་གཉིས་པར་སྔར་ནས་བེད་སྤྱོད་བྱས་ཡོད་པས་མཚོན་འདི་ཙམ་གྱིས་ཆོག་པར་བྱས་སོ། །ཤེས་བྱེད་བློའི་སྐོར་ནས་ཀྱང་སློབ་དཔོན་ཕྱྭ་བས་བློ་རིག་བདུན་དུ་ཕྱེ་བར་གྲགས་པ་དེ་ཡང་རྔོག་ལོས་སྔར་ནས་ཚད་མའི་བློ་མངོན་རྗེས་ཚད་མ་གཉིས་དང་ཚད་མ་མ་ཡིན་པའི་བློ་ལྔ་བཅས་བསྡོམས་པས་བདུན་དུ་དབྱེ་བའི་རྗེས་སུ་འབྲངས་པར་འདོད་ལ། དེ་ཡང་ཚད་མིན་གྱི་བློ་ལྔ་ལས་ཡིད་དཔྱོད་ཅེས་པའི་ཐ་སྙད་ནི་སློབ་དཔོན་ཕྱོགས་གླང་དང་དཔལ་ལྡན་ཆོས་གྲགས་ཀྱི་གཞུང་དུ་དངོས་སུ་བསྟན་མེད་ན་ཡང་།མཁན་ཆེན་ཞི་བ་འཚོའི་《ཚད་མའི་དེ་ཁོ་ན་ཉིད་བསྡུས་པ་》ལས། “གཞན་ནི་སྒྲ་ལས་བྱུང་བ་ཡི། །ཤེས་པས་དོན་རྟོགས་སྒྲ་ལས་བྱུང་། །ཡང་ན་ཡིད་ཆེས་ཚིག་ཡིན་ཟེར། །འདི་ནི་ལྐོག་ཏུ་གྱུར་པ་ཡི། །ཡུལ་ཅན་ཡིན་གྱི་མངོན་སུམ་མིན། །རྗེས་དཔག་ཏུའང་མི་འཐད་དེ། །དེ་ཡི་མཚན་ཉིད་བྲལ་ཕྱིར་རོ། །” (22)ཞེས་གསུངས་པས་ཡིད་དཔྱོད་དོན་གྱི་སྒོ་ནས་བསྟན་ལ། རྗེས་སུ་ཕྱྭ་བས་རྔོག་ལོའི་བཞེད་ཚུལ་ལྟར་གྲངས་ངེས་ལ་དེར་ཡིད་དཔྱོད་བཞག་པས། ས་པཎ་གྱི་《རིགས་གཏེར་》དུ། “ཡིད་དཔྱོད་རྟགས་ལ་ལྟོས་མ་ལྟོས། །དམ་བཅའ་ཙམ་ཡིན་ཐེ་ཚོམ་འགྱུར། །” (23)ཞེས་དང་། “རྟགས་ལ་ལྟོས་ན་ཡང་དག་གམ། །ལྟར་སྣང་གསུམ་ལས་འདའ་བ་མེད། །” (24)ཅེས་དགག་པའང་གནང་འདུག་མོད། གང་ལྟར་དེ་ནི་སྤྱིར་གྲངས་ཅན་པའི་གཞུང་དུ་སྒྲ་བྱུང་ཚད་མ་ཞེས་པ་ཞིག་འདོད་ཅིང་། ནང་པའི་གཞུང་དུའང་སྒྲ་ལ་བརྟེན་ནས་ལྐོག་གྱུར་གྱི་དོན་འཛིན་པའི་ཤེས་པ་ཞིག་ཡོད་ལ་དེ་ནི་ཡིད་དཔྱོད་ལས་གཞན་མ་ཡིན་པས་ན་རྒྱུ་མཚན་དང་ལྡན་པ་དེའི་རྐྱེན་གྱིས། རྔོག་ལོ་ཆེན་པོ་ནས་བརྒྱུད་ཅིང་ཕྱྭ་བ་ཆོས་སེང་གིས་འདོད་པའི་ཚད་མ་མ་ཡིན་པའི་བློ་ལྔར་གྲངས་ངེས་པ་དེ་བཀའ་བརྒྱུད་པ་དང་ཕྱིས་ཀྱི་དགེ་ལུགས་པ་དང་བཅས་པས་ཞལ་གྱིས་བཞེས་ཏེ་ད་ལྟའི་བར་མ་ཉམས་པར་གནས་པ་འདིའོ། །
        མཇུག་བསྡུའི་གཏམ།
        དུས་རབས་བརྒྱད་པ་ནས་བཟུང་བོད་དུ་ཚད་མའི་བསྟན་བཅོས་སྒྱུར་བའི་འགོ་བརྩམས་ཤིང་། དུས་རབས་བཅུ་གཅིག་པར་《རྣམ་བཤད་ཆེན་པོ་》ཞེས་པའི་《རྣམ་ངེས་》ཀྱི་འགྲེལ་པ་འདིའང་ཐོག་མར་བརྩམས་ཤིང་དགག་སྒྲུབ་ཀྱི་རིགས་པའི་ལམ་ནས་འཆད་པས་བོད་དུ་ཚད་མ་གསར་མའི་སྲོལ་ཕྱེས་པ་སོགས་ཀྱིས་མཚོན་དུས་སྐབས་ངེས་ཅན་ཞིག་གི་ནང་བོད་དུ་གཏན་ཚིགས་ཀྱི་ཆར་གཞོལ་བའི་《རྣམ་ངེས་》ཀྱི་འཕྲིན་ལས་རྒྱ་ཆེར་དར་མྱོང་བར་མངོན། རྗེས་སུ་དུས་རབས་བཅུ་གསུམ་པའི་སྐབས་ཏེ་ས་སྐྱ་བའི་དུས་དང་། ལྷག་པར་དུས་རབས་བཅུ་བཞི་པ་དང་བཅོ་ལྔ་པ་སྟེ་དགེ་ལྡན་པའི་དུས་སུ་གནའ་བོའི་རྒྱ་གར་གྱི་སྟབས་བདེ་དངོས་གཙོ་རིང་ལུགས་ཀྱི་ལྟ་བ་དེ་དང་། སངས་རྒྱས་པའི་བདག་མེད་རྟོགས་པའི་ཤེས་རབ་ཀྱིས་མཚོན་པའི་ཆོས་ལུགས་མཚན་ཉིད་རིག་པ་གཉིས་ཟུང་དུ་འབྲེལ་བའི་ཚད་གཞུང་གི་འཆད་ཉན་སྤེལ་བ་ལས། བོད་ཀྱི་ཡུལ་གྲུ་ཀུན་ཏུ་ད་ལྟའི་བར་གཙོ་བོ་《རྣམ་འགྲེལ་》གྱི་བསམ་བློས་མ་ཁྱབ་པ་མེད་པར་གྱུར། དེ་བས་འགའ་ཞིག་གིས་རྔོག་ལོ་དང་ཕྱྭ་བ་སོགས་ཀྱི་ཚད་མའི་ལྟ་གྲུབ་ལ་བོད་སྔ་རབས་པའི་ལྟ་བ་ཞེས་རྗོད་ཀྱི་ཡོད། ཡིན་ཡང་ང་ཚོས་རིག་གནས་བྱུང་འཕེལ་རིག་པའམ་གྲུབ་ཚུལ་རིག་པའི་ངོས་ནས་བལྟས་ཚེ། རྔོག་བློ་ལྡན་ཤེས་རབ་ཀྱི་《ཚད་མ་རྣམ་ངེས་ཀྱི་དཀའ་གནས་རྣམ་བཤད་》ཅེས་པ་ནི་བོད་རང་གི་མཁས་པས་རྒྱས་པར་མཛད་པའི་རྣམ་ངེས་ཀྱི་བསྟན་བཅོས་ཐོག་མ་ཉིད་ཡིན་པ་དང་། དུས་རབས་བཅུ་གཉིས་པའི་སྐབས་དེ་ནི་བོད་དུ་རིག་གནས་བསམ་བློ་སྣ་མང་འབྱུང་བའི་དུས་ཤིག་ལ་ཁེལ་བས། ཁོང་གི་བརྩམས་ཆོས་དང་ཚད་མའི་དགོངས་པར་ཡང་སྒྱུར་བཅོས་ཀྱི་བསམ་བློས་ཕྱུག་ལ། རིགས་པའི་བསྟན་བཅོས་འདི་བཀའ་གདམས་པའི་ཚད་མའི་བསྟན་བཅོས་ཀྱི་ཚབ་མཚོན་རང་བཞིན་ལྡན་པ་ཞིག་ཀྱང་ཡིན་པ་སོགས་ལ་བརྟེན་འདི་ཉིད་སློབ་སྦྱོང་དང་ཞིབ་འཇུག་གི་དོན་སྙིང་ཆེ་བར་སེམས་སོ། །
