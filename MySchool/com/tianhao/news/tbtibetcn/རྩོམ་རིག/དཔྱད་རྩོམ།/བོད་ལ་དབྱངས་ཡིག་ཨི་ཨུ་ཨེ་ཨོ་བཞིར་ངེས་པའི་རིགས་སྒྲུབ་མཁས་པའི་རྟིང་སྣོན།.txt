བོད་ལ་དབྱངས་ཡིག་ཨི་ཨུ་ཨེ་ཨོ་བཞིར་ངེས་པའི་རིགས་སྒྲུབ་མཁས་པའི་རྟིང་སྣོན།
[ནང་དོན་གནད་བསྡུས།]དཔྱད་རྩོམ་འདིའི་ནང་མི་རིགས་སོ་སོའི་སྒྲའི་བརྗོད་པ་སྣ་ཚོགས་གསལ་བར་བྱེད་པའི་སྐད་ཀྱི་གདངས་སུ་དབྱངས་ཨ་འདི་ལྷན་སྐྱེས་སུ་ཡོད་ཅིང་། དེ་ལས་ཡི་གེའི་རྩོམ་གཞིར་གྱུར་བའི་སྐད་ཀྱི་གདངས་སྣ་ཚོགས་བྱུང་བས་སྐད་ཀྱི་གདངས་ལ་ཡི་གེ་ཞེས་བརྗོད། མི་རིགས་དང་སྐད་རིགས་མི་འདྲ་བ་རྣམས་ཀྱིས་རང་རང་གི་ཐུན་མིན་སྐད་གདངས་དེ་མཚོན་བྱེད་དུ་བཟོས་པའི་ཡི་གེར་ཐུན་མིན་གྱི་ཁྱད་ཆོས་ཡོད་དེ། ལེགས་སྦྱར་གྱི་སྒྲ་གཟུགས་ཡི་གེར་ཨ་དང་བཅས་པའི་དབྱངས་བཅུ་དྲུག་གང་རུང་གསལ་བྱེད་ཀྱི་སྲོག་ཏུ་མ་ཞུགས་པར་ངག་ནས་བརྗོད་དུ་མེད་ཅིང་། བོད་ཀྱི་ཡི་གེ་ནི་དབྱངས་ཡིག་གསལ་བྱེད་ཀྱི་སྲོག་ཏུ་ཞུགས་ཚུལ་མེད་པར་གསལ་བྱེད་གང་ཞིག་ལ་སྒྲ་གདངས་ཇི་ལྟར་ལྡན་པའི་ཚུལ་གྱིས་བཀླག་པའི་ཡི་གེ་ཡིན། དེའི་ཕྱིར། མི་རིགས་ཀུན་གྱི་སྐད་གདངས་སུ་ཡོད་པའི་ཨ་དེ་སྤྱི་ཁྱབ་ཐུན་མོང་པའི་ཨ་ཡིན་ལ། དེ་འདྲའི་གདངས་དེ་མཚོན་བྱེད་དུ་བཟོས་པའི་ཨ་དེ་མི་རིགས་སོ་སོའི་ཡི་གེར་ངེས་པའི་ཨ་ཡིག་བྱེ་བྲག་པ་ཡིན་པར་ངོས་འཛིན་དགོས་ཚུལ་དང་། ཨ་ཡིག་བྱེ་བྲག་པ་སྤྱོད་སྟངས་ཐད་ལའང་ཁྱད་པར་ཡོད་དེ། ལེགས་སྦྱར་གྱི་ཡི་གེ་ནི་ཨ་ཡིག་གསལ་བྱེད་ཀྱི་སྲོག་ཏུ་ཞུགས་པར་ནུས་པའི་འཕྲལ་སྦྱོར་ཡི་གེར་གྱུར་ལ། བོད་པས་འཕྲལ་སྦྱོར་བྱེད་ཚུལ་མེད་པའི་གསལ་བྱེད་དུ་སྤྱོད་པ། དེ་བཞིན་མི་རིགས་གཞན་དང་གཞན་ལའང་རང་རང་གི་ཁྱད་ཆོས་སམ་སྤྱོད་སྟངས་མི་འདྲ་བའི་ཨ་ཡིག་བྱེ་བྲག་པ་སྣ་ཚོགས་ཡོད་ཚུལ། བོད་ཀྱི་སྔོན་གྱི་ཡིག་རྙིང་ཁག་ཏུ་འ་ཆུང་རེ་རྗེས་སུ་འདོགས་པ་དེ་དབྱངས་སྲོག་ཞུགས་ཀྱི་རྟགས་མིན་ལ། བོད་ཀྱི་དབྱངས་ཨ་ངེས་ཀྱི་སྙན་ངག་འབྲི་སྲོལ་དེ་ཡང་རྒྱ་གར་རྩོམ་རིག་དཔེར་བྱས་པའི་རྩོམ་ཚུལ་ཙམ་ལས་བོད་ལ་དབྱངས་སྲོག་ཞུགས་ཡོད་པ་མཚོན་མི་ཐུབ་ཚུལ་སོགས་གླེང་ཡོད།

        [ཀྲུང་གོའི་དཔེ་དེབ་དབྱེ་རྟགས།] H214

        དང་པོ། བོད་ལ་དབྱངས་ཡིག་བཞིར་འདོད་པའི་སྔོན་བྱོན་མཁས་པའི་ལྟ་ཚུལ།
        རྗེ་ས་པཎ་(1182 - 1251) གྱིས་མཛད་པའི་《བསྟན་བཅོས་མཁས་པའི་ཁ་རྒྱན་》དུ། “དེ་ལ་དང་པོ་དབྱངས་རྣམས་ནི་ལྔའོ (ཨ་ཨི་ཨུ་ཨེ་ཨོ་)  ཨ་ཡིག་བོར་བ་རྣམས་(ཨི་ཨུ་ཨེ་ཨོ་)ནི་ཡན་ལག་གོ ཀ་ལ་སོགས་པ་(ཉི་ཤུ་དགུ་)རྣམས་ནི་གསལ་བྱེད་དོ། བཞི་བཞི་པའི་སྡེ་པ་ནི་བདུན་ནོ།ཧ་ནི་མཐར་གནས་པའོ་”①ཞེས་བོད་ལ་དབྱངས་ཡིག་ལྔ་ཡོད་པར་གསུངས་པ་དང། ཡང་ཁོང་གི་རྩོམ་གཞན་《དག་པར་བྱེད་པའི་སྡེབ་སྦྱོར་རིན་ཆེན་རྒྱ་མཚོ》། “སྤྱིར་ནི་བོད་ཀྱི་ཡི་གེ་ལ། །དབྱངས་དང་གསལ་བྱེད་གཉིས་སུ་དབྱེ། །དབྱངས་ཡིག་ཨི་ཨུ་ཨེ་ཨོ་བཞི། །གསལ་བྱེད་ཀ་སོགས་སུམ་ཅུ་ཡིན། །”ཞེས་གསུངས་པས། རྗེ་ས་པཎ་ནི་བོད་ལ་དབྱངས་ཡིག་ལྔར་འདོད་མཁན་གྱི་གྲས་སུ་མཐའ་གཅིག་ཏུ་བགྲངས་ན་མི་འོས་པའི་གནས་ལུགས་ཤིག་མཐོང་ཐུབ་མོད། རྗེས་བྱུང་བོད་ཀྱི་བརྡ་སྤྲོད་སྨྲ་བ་ཁ་ཅིག་གིས་རྗེ་ས་པཎ་ནི་བོད་ལ་དབྱངས་ལྔར་འདོན་མཁན་དུ་ངོས་འཛིན་བྱེད། རྗེ་ས་པཎ་བཞུགས་དུས་སུ་བོད་ལ་དབྱངས་ལྔར་བཞེད་པའི་ལྟ་ཚུལ་དེར་ཆེད་དུ་དགག་མཁན་མ་བྱུང་ཡང། རྗེ་ས་པཎ་གྱི་སྐུ་ཚེའི་སྨད་དུ་འཁྲུངས་པའི་བཅོམ་ལྡན་རིག་པའི་རལ་གྲིས།(1228 -1305) “སུམ་ཅུ་པོ་དེ་ཇི་ལྟར་བསྟུ་(བསྡུ་)ཞེ་ན། སང་སྐྲི་ཏ་ལྟར་སྙུང་(བསྡུ་) ན་ནི་ཨ་དབྱངས་ཡིག་དང་གཞན་རྣམས་གསལ་བྱེད་ཡིན་ནོ་”②ཞེས་ཨ་ཡིག་འདི་སཾ་སྐྲྀ་ཏ་ལྟར་མ་བསྡུས་པར་བོད་ལྟར་ན་གསལ་བྱེད་ཀྱི་གྲས་ཡིན་པ་དོན་གྱིས་ཁས་བཞེས་ཤིང། དེའི་ཤུགས་ནས་ལེགས་སྦྱར་ལ་དབྱངས་ལྔར་བསྡུ་ཚུལ་ཡོད་ལ། བོད་ལ་དེ་ལྟར་བསྡུ་ཚུལ་མེད་པ་བསྟན་ཡོད། དེ་རྗེས་ཞྭ་ལུ་ལོ་ཙཱ་བ་ཆོས་སྐྱོང་བཟང་པོའི་(1441 -1527)《སུམ་ཅུ་པའི་རྣམ་འགྲེལ་》དུ། “བོད་ཀྱི་ཡི་གེ་ཨཱ་ལི་གསལ་བར་མཚོན་པར་བྱེད་པ་ནི་ཨི་ཡིག་དང་སོགས་ཀྱི་སྒྲས་བསྟན་པ་ཨི་ཨུ་ཨེ་ཨོ་རྣམས་ཏེ་བཞིར་ངེས་ཞེས་པའོ། གང་ཡང་མཁས་པའི་ཁ་རྒྱན་དུ། བོད་ཀྱི་དབང་དུ་མཛད་དེ། དབྱངས་རྣམས་ནི་ལྔའོ། གསལ་བྱེད་རྣམས་ནི་ཉི་ཤུ་རྩ་དགུའོ་ཞེས་གསུངས་པ་ནི་གཞུང་འདི་དང་དངོས་སུ་འགལ་བ་ཁོ་ནར་ཟད་དོ། བོད་ཀྱི་སྐད་ལ་ཀཱ་ལི་སྟེ། གསལ་བྱེད་ནི་གྲངས་སུམ་ཅུ་ཐམ་པ་ཉིད་དུ་ངེས་ཏེ་”③ཞེས་གསུངས་ཏེ་རྗེ་ས་པཎ་གྱི་ལྟ་བའམ་བོད་ལ་དབྱངས་ལྔ་འདོད་པར་དགག་པ་དངོས་སུ་མཛད་ཡོད། 《ཐོན་མིའི་ཞལ་ལུང་》དུ།  “ས་པཎ་གྱིས་མཛད་པའི་མཁས་པའི་ཁ་རྒྱན་དུ།   བོད་ཀྱི་དབང་དུ་མཛད་དེ་དབྱངས་རྣམས་ནི་ལྔའོ། གསལ་བྱེད་རྣམས་ནི་ཉི་ཤུ་རྩ་དགུའོ། ཞེས་དང། དེའི་རྗེས་སུ་འབྲངས་ནས་དཔལ་ཁང་ལོ་ཙཱ་བའི་དག་ཡིག་ངག་སྒྲོན་དང་། ས་སྐྱ་ཨ་མྱེས་(ཨ་མྱེས་ཞབས་ཀུན་དགའ་རབ་གསལ།537 -?)ཀྱི་མིག་དབང་རབ་གསལ་རྣམས་སུ་ཡང། དབྱངས་ལྔ་ཡ་ར་ལ་ལྦའི་འདོགས། ཞེས་པའི་བརྡ་ཆད་འདོན་མཁན་ཡོད་ནའང། འདིར་སུམ་ཅུ་པའི་གཞུང་དང་བསྟུན་ན་དབྱངས་བཞི་དང་གསལ་བྱེད་སུམ་ཅུར་ངེས་པས་ན་བརྡ་ཆད་དེ་དག་ལ་ཁས་བླངས་བར་མི་བྱའོ། ཞེས་ཞྭ་ལུ་ལོ་ཙཱ་བ་སོགས་ཀྱིས་བཀག་པ་ནི་དོན་ལ་གནས་པ་ཡིན་ནོ་”④ཞེས་དང། “འདིར་བོད་རང་ལ་མཁོ་བའི་ཡི་གེ་མང་མི་དགོས་ཉུང་ན་མི་འདུ་བའི་དབྱངས་བཞི་དང་གསལ་བྱེད་སུམ་ཅུའི་གྲངས་ངེས་པ་མཛད་ནས་དབྱེ་བསྡུ་བྱས་སོ་”⑤ཞེས་གསུངས་པ་རྣམས་ལས་མཁས་དབང་ཚེ་ཏན་ཞབས་དྲུང་ཁོང་ཉིད་ནི་བོད་ལ་དབྱངས་བཞི་དང་གསལ་བྱེད་སུམ་ཅུར་འདོད་མཁན་གྱི་གྲས་ཤིག་ཡིན་པ་གསལ་བར་བསྟན་པ་དང་།  ཡང་ཁོང་གིས།“ལེགས་སྦྱར་སྒྲ་རིག་པའི་གཞུང་ཡོད་ཚད་དུ་ཨ་ཡིག་དབྱངས་ཀྱི་གྲས་སུ་ཡོད་པ་ལས་གསལ་བྱེད་ཀྱི་གྲས་སུ་བཞག་པ་གཅིག་ཀྱང་མེད་ལ། ཨ་ཡིག་གསལ་བྱེད་ཀྱི་གྲས་སུ་བཞག་པ་ནི་བོད་ཡིག་གི་རྣམ་གཞག་ཁོ་ན་ལས་མི་འབྱུང་”⑥ཞེས་གསུངས་ཏེ་ཨ་ཡིག་དེར་ལེགས་སྦྱར་དང་བོད་གཉིས་ལ་ངོས་འཛིན་ནམ་བཀོལ་སྤྱོད་བྱེད་སྟངས་ཐད་ཁྱད་པར་ཡོད་ཚུལ་གསལ་བར་བསྟན་ཡོད། མཁས་དབང་རྡོ་རྗེ་རྒྱལ་པོས། “ཨ་ཡིག་གོང་དུ་བཤད་པ་ལྟར་དགོས་དབང་གིས་གསལ་བྱེད་ཀྱི་གྲས་སུ་བཞག་པས་དབྱངས་ཡིག་བཞིར་གྱུར་ལ་”⑦ཞེས་གསུངས་པ་ཡང་གོང་གི་འདོད་ཚུལ་དང་མཐུན་པ་ཡིན་ལ། རྗེ་དམུ་དགེས།“བོད་སྐད་ཀྱི་གདངས་ལ་ཀ་མད་སུམ་ཅུ་དང་དབྱངས་བཞི་ཡོད་པ་དང་” ⑧ཞེས་གསུངས་པ་དེས་ཁོང་ཡང་བོད་ལ་དབྱངས་བཞི་དང་གསལ་བྱེད་སུམ་ཅུར་འདོད་མཁན་གྱི་གྲས་ཡིན་པ་གསལ་བར་རྟོགས་ཐུབ། (ཁོང་གི་གསུང་གཞན་དུ་བོད་ཡིག་ལ་དབྱངས་ཀྱི་ང་རོ་ལྔ་ཡོད་པར་གསུངས་པ་ནི་སྐྱེས་བུ་སུ་ཞིག་གི་ངག་ནས་སྒྲ་འབྱིན་པའི་ཚུལ་དྲང་པ་བསྒྲེང་སོགས་ཀྱི་ང་རོ་ལྔ་ཡོད་པར་དགོངས་པའི་གསུང་ཡིན།)ད་ལྟ་སྤྱིར་བཏང་གི་ཀློག་དེབ་རྣམས་སུ། “རྒྱ་ཡིག་ལྔ་བཅུ་ཐམ་པ་དེ་བོད་ཡིག་སུམ་ཅུར་གཏན་ལ་ཕབ་”ཅེས་དང། 《སུམ་ཅུ་པ་》རུ།  “ཡི་གེ་ཨ་ལི་ཀ་ལི་གཉིས། ཨ་ལི་གསལ་བྱེད་ཨི་སོགས་བཞི ཀ་ལི་སུམ་ཅུ་ཐམ་པའོ་”ཞེས་པ་སོགས་ཀྱི་ངོས་འཛིན་དེ་དག་ནི་རྒྱུན་གཏན་ཅན་དུ་གྱུར་ཡོད།   
        འདིར་ཁ་ཅིག་གིས། ཡི་གེ་ཨ་ལི་ཀ་ལི་གཉིས་ཞེས་པའི་ཨ་ལིས་དབྱངས་ཡིག་ཨ་བསྟན་ཚུལ་གྱི་རིགས་པ་འགོད་པར་བྱེད་ཀྱང་པྲ་སྟི་རིན་པོ་ཆེས། “བོད་ཀྱི་དབྱངས་ལ་ཨ་ཡིག་མེད་པས། ཨ་ཕྲེང་གི་དོན་དངོས་སུ་མེད་ཀྱང་། རྒྱ་གར་གྱི་དབྱངས་ལ་ཨ་ཡིག་ལ་སོགས་པའི་ཕྲེང་བ་དངོས་སུ་ཡོད་པས་དེ་ལ་ཨ་ཕྲེང་ཞེས་བཏགས་པ་བཞིན་དུ།  བོད་ཀྱི་ཨི་ལ་སོགས་པ་ཡང་དེའི་ནང་ནས་ཕྱུངས་ཤིང་དེ་དང་ཆ་མཐུན་པས་ཨཱ་ལི་ཞེས་གདགས་སུ་རུང་བར་མཚོན་ཞིང་གོ་ནུས་པའི་སྒོ་ནས་སོ། དེའི་གྲངས་ངེས་བསྟན་པའི་ཕྱིར།  ཨི་སོགས་བཞི ཞེས་སྨྲས་ཏེ། ཨི་ལ་སོགས་པ་བཞི་ནི་ཨུ་ཨི་ཨེ་ཨོ་བཞི་ཡོད་པས། དེར་ངེས་པའོ་”⑨ཞེས་དང། “ཨ་ཡིག་ནི་གང་ལ་འདོགས་པའི་གཞི་བརྗོད་དོན་ཙམ་མ་གཏོགས་གསལ་བྱེད་ཡིན་པའི་ཕྱིར་རོ་”⑩ཞེས་གསུངས་པ་དེ་དག་གིས། “ཡི་གེ་ཨ་ལི་ཀ་ལི་གཉིས། ཨ་ལི་གསལ་བྱེད་ཨི་སོགས་བཞི་”ཞེས་པའི་དོན་ལ་གོ་བ་ཇི་ལྟར་ལེན་དགོས་ཚུལ་བསྟན་པས། ཡི་གེ་ཨ་ལི་ཀ་ལི་གཉིས་ཞེས་པ་ལེགས་སྦྱར་ལ་དཔེ་བལྟས་པའི་མིང་འདོགས་ཙམ་ཡིན་ནོ།   
        གཉིས་པ།  སྐད་རིགས་ཀུན་གྱི་གདངས་སུ་ཡོད་པའི་ཨ་དང་འཕྲལ་སྦྱོར་གྱི་ཨ། གསལ་བྱེད་དུ་ངོས་འཛིན་པའི་ཨ་རྣམས་ཀྱི་ཁྱད་པར་ལས་བོད་ལ་དབྱངས་བཞིར་ངེས་ཚུལ།
        སྔོན་དང་ད་ལྟའི་བརྡ་སྤྲོད་མཁན་པོ་རྣམས་ཀྱིས་ཡིག་རིགས་ཀུན་གྱི་འབྱུང་གནས་སུ་གྱུར་བའི་སྐད་རིགས་ཀུན་གྱི་གདངས་སུ་དབྱངས་ཨ་འདི་ཉིད་གནས་ཤིང་དེ་ཉིད་མེད་ན་སྒྲ་ཐམས་ཅད་དབྱུང་དུ་མེད་པའི་རིགས་པས་ཨ་ཡིག་འདིའི་ནུས་པར་ནན་བཤད་གནང་བ་མང་སྟེ། ས་ཆེན་བསོད་ནམས་རྩེ་མོས། “ཡི་གེའི་མ་ནི་ཨ་ཡིན་ནོ། ཨ་ནི་རང་ཉིད་བརྗོད་རུང་བས། སེང་གེ་ལྟ་བུའི་ཨ་ཞེས་བྱ་”ཞེས་དང་། “ཨ་ནི་ཡི་གེ་ཐམས་ཅད་ཀྱི་ཕྱི་མོ་ཡིན་ནོ་”⑾ཞེས་དང་། རྗེ་དམུ་དགེ་བསམ་གཏན་གྱིས། “མི་རྣམས་ཀྱང་མའི་མངལ་ནས་སྐྱེ་དུས་ཐོག་མར་ཨ་ཞེས་པའི་སྒྲ་འབྱིན་པར་བྱེད་དོ་”⑿ཞེས་སོགས་གསུངས་པ་ལྟར། སྐད་གདངས་ཀྱི་རྣམ་པའི་ཐོག་མ་ཨ་ཡིག་དང་དེ་ལས་གདངས་སྣ་ཚོགས་སྐྱེས་པའི་ཆོས་ཉིད་ལས། ཡི་གེ་དངོས་ནི་སྐད་ཀྱི་གདངས་ཡིན་པའི་ངོས་འཛིན་བྱུང་ཞིང། ཡི་གེའི་མ་ནི་ཨ་ཡིན་ནོ། ཞེས་པའི་གནས་ལུགས་ཀྱང་དེར་ཐུག་ཡོད། སྐད་ཀྱི་གདངས་དེ་མེད་ན་ཡི་གེའི་རྩོམ་གཞི་ཟེར་བའང་མེད། ཡིན་ན་ཡང་སྐད་ཀྱི་གདངས་དེ་རྩོམ་གཞིར་བྱས་པའི་ཡི་གེ་སྟེ། བལྟ་བའི་གཟུགས་རིས་ཡི་གེ་རྣམས་ནི་མི་རིགས་རང་རང་གི་དགའ་མོས་སམ་ལག་ལེན་ཁྲོད་ནས་བྱུང་བས། དེ་ལ་མི་རིགས་གང་དང་ལོ་རྒྱུས་གང་དེའི་རྟགས་དམ་འཁེལ་ཡོད་པ་ཞིག་རེད།   
        ལེགས་སྦྱར་གྱི་ཡི་གེ་ནི་ཐོག་མ་ཉིད་ནས་རང་ཚུགས་ཀྱི་ཚུལ་དུ་བྱུང་ཞིང་། དེར་ཁྱད་ཆོས་ཐུན་མོང་མ་ཡིན་པ་ཞིག་ཡོད་པ་ནི་ཡི་གེར་སྲོག་ལྡན་དང་སྲོག་མེད་ཀྱི་ཁྱད་པར་འབྱེད་ཚུལ་དང། དབྱངས་སྲོག་ཏུ་མ་ཞུགས་པར་ངག་ནས་སྒྲ་བརྗོད་དུ་མེད་པའི་ཁྱད་ཆོས། མིང་ལས་ཚིག་སྒྲུབ་པ་ལ་དབྱངས་སྲོག་ཞུགས་འཕྲལ་ནས་མཚམས་སྦྱོར་དགོས་པ་སོགས་ལེགས་སྦྱར་གྱི་ཡི་གེ་ཁོ་ནར་ངེས་པའི་ཁྱད་ཆོས་དུ་མ་ལྡན་ཞིང། ཁོང་ཆོའི་གསལ་བྱེད་ལ་དབྱངས་སྲོག་ཞུགས་ཡོད་མེད་ཀྱི་རྟགས་ཧ་ལན་ཏའི་མཚན་བྱ་བ་སྟེ། བོད་ཀྱི་གསལ་བྱེད་ཀ་ལ་ར་བཏགས་པ་༼ཀྲ༽ལྟ་བུ་འདི་ཡོད་མེད་གལ་ཆེ་ཞིང། ཡི་གེ་གང་ཞིག་ལ་དབྱངས་སྲོག་ཞུགས་ཡོད་མེད་ཀྱི་ཁྱད་པར་ཡང་དེའི་སྟེང་ནས་འབྱེད་པ་དང། ཁོང་ཚོའི་ཡི་གེ་དེ་ཡིག་རིགས་གཞན་དང་མི་འདྲ་བའི་ཁྱད་ཆོས་གཙོ་བོའང་དེ་ལས་མངོན་བཞིན་ཡོད། བོད་ཀྱི་ཡི་གེ་སོགས་ཡིག་རིགས་གཞན་དག་ལ་གསལ་བྱེད་ཀྱི་སྲོག་ཏུ་དབྱངས་ཞུགས་ཚུལ་དང་སྲོག་མེད་ཀྱི་རྟགས་འཇོག་སྲོལ་མེད་པས། བོད་ཡིག་ནི་ལེགས་སྦྱར་གྱི་ཡི་གེ་ལྟར་དབྱངས་ཡིག་ཨ་འདི་སྲོག་ཏུ་ཞུགས་ཚུལ་མེད་པའི་ཡི་གེ་ཡིན། 《ཐོན་མིའི་ཞལ་ལུང་》དུ། “སྲོག་མེད་ཀྱི་ཡི་གེ་ནི་ལེགས་སྦྱར་ལྟར་ན་ངག་གིས་བརྗོད་དུ་མེད་པས་ན་ཚུལ་དེ་ནི་བོད་ཡིག་གི་རྣམ་གཞག་དང་གཏན་ནས་མི་མཐུན་པ་ཞིག་ཡིན་ནོ་”⒀ཞེས་གསུངས་པ་ཡང་གནད་འདིར་ཐུག་ཡོད། དེ་ལ་བརྟགས་ན། ཡི་གེ་དངོས་ནི་མི་རིགས་ཀུན་གྱི་སྐད་གདངས་མི་འདྲ་བའི་སྒྲའི་བརྗོད་པ་སྣ་ཚོགས་ཡིན་ལ། སྒྲའི་བརྗོད་པ་སྣ་ཚོགས་ཀྱི་གདངས་ཅི་འདྲ་བ་མཚོན་བྱེད་དུ་བཟོས་པའི་ཡི་གེ་ཡོད་དོ་ཅོག་ནི་སྐད་གདངས་མི་འདྲ་བ་རྣམས་གདགས་གཞིར་བྱས་ནས་བཟོས་པའི་ཡི་གེའམ་རི་མོའི་ཡི་གེ་བྱ་བ་དེ་ཡིན། དཔེར་ན། མདུན་ཅོག་ཅེས་པའི་སྒྲའམ་ཐ་སྙད་དེ་འཇུག་པའི་འཇུག་གཞི་དངོས་པོ་ཞིག་དང་། སྒྲ་དེ་མཚོན་བྱེད་ཀྱི་སྒྲ་གཟུགས་ཡི་གེ་སྟེ་རྣམ་པ་གཉིས་ཡོད་ལ། དེ་ལྟའི་གཟུགས་སུ་བྱས་པའི་ཡི་གེར་དབྱངས་སྲོག་ཞུགས་ཀྱི་ཁྱད་ཆོས་ཡོད་མེད་ཅི་རིགས་ཏེ། ལེགས་སྦྱར་གྱི་ཡི་གེ་ལྟ་བུར་དབྱངས་སྲོག་ཞུགས་ཀྱི་ཁྱད་ཆོས་བསྟན་དུ་ཡོད་ཅིང། བོད་དང་རྒྱ་ནག་གི་ཡི་གེ་ནི་ལེགས་སྦྱར་ལྟར་དབྱངས་སྲོག་ཞུགས་ཀྱི་ཁྱད་ཆོས་སམ་འཕྲལ་སྦྱོར་བྱེད་སྲོལ་མེད་པའི་ཡི་གེ་ཡིན། དེས་ན་ཡི་གེ་ཀུན་གྱི་སྲོག་ཏུ་དབྱངས་ཨ་འདི་ཞུགས་པར་བཤད་པ་ནི་ཡིག་རིགས་ཀུན་གྱི་རྩོམ་གཞིར་གྱུར་བའི་སྐད་ཀྱི་གདངས་དེ་ལ་དམིགས་ནས་བཤད་པ་སྟེ། དེང་སང་སྒྲ་རིག་སྨྲ་བ་རྣམས་ཀྱིས་བཟློག་མེད་དུ་ཁས་ལེན་པའི་གནད་འདི་ཡིན། དེ་ལྟར་སྤྱིའི་ཆ་ནས་བཤད་ན། སྐད་རིགས་གང་དང་གང་དེའི་གདངས་སུ་ཨ་ཞེས་པའི་ང་རོ་འདི་རང་ཆས་སུ་ཡོད་པ་མཚུངས་ཀྱང། བྱེ་བྲག་ཏུ་མི་རིགས་སོ་སོའི་སྐད་རིགས་དང་སྐད་གདངས་དེའི་རྩོམ་གཞིར་གྱུར་པའི་ཐུན་མིན་གྱི་ཡི་གེ་སོ་སོའི་ཁྱད་ཆོས་མི་མཚུངས་ཏེ། དཔེར་ན། ཡི་གེའི་སྲོག་ཏུ་ཞུགས་པའམ་འཕྲལ་སྦྱོར་བྱེད་དུ་ཡོད་པའི་ལེགས་སྦྱར་གྱི་དབྱངས་ཡིག་ཨ་ལྟ་བུ་དང། འཕྲལ་སྦྱོར་སོགས་བྱེད་དུ་མེད་པའི་བོད་པའི་གསལ་བྱེད་གྲས་ཀྱི་ཨ་ལྟ་བུ་སྟེ། ཨ་ཡིག་འདིའི་གདངས་དང། འབྲི་ཚུལ་ལམ་གཟུགས་ཀྱི་ཆ་ནས་སཾ་བོད་གཉིས་ལ་ཁྱད་པར་ཆེར་མེད་ཀྱང་སྤྱོད་སྟངས་ཀྱི་ཆ་ནས་ཁྱད་པར་ཡོད། དེ་བཞིན་རྒྱའི་ཡི་གེ་阿དང་啊ལྟ་བུ་ཡང་སྐད་ཀྱི་གདངས་སུ་རང་ཆས་སུ་བྱུང་བའི་ཨ་དེ་ཉིད་རི་མོའི་གཟུགས་མི་འདྲ་བར་བཀོད་པ་མཚོན་པའི་ཡི་གེ་ཡིན། དེ་ལྟར་སྐད་གདངས་དང་གཟུགས་རིས་ཡི་གེ་གཉིས་ཀྱི་ངོས་ནས་བརྟགས་ན། ཨ་ཡིག་ལ་སྤྱི་བྱེ་བྲག་གཉིས་ཡོད་དེ། དཔེར་ན། བུམ་པ་དེ་གསེར་བུམ་སོགས་ཀྱི་སྤྱི་ཡིན་པ་བཞིན་དུ་སྐད་རིགས་ཀུན་གྱི་གདངས་སུ་རང་ཆས་སུ་ཡོད་པའི་ང་རོའམ་དབྱངས་སུ་བཤད་པའི་ཨ་དེ་སྤྱི་བྱེ་བྲག་གཉིས་སུ་ཕྱེས་བའི་སྤྱི་ཡིན་ལ། བུམ་པར་དབྱེ་ན་གསེར་བུམ་དང་ཟངས་བུམ་སོགས་བྱེ་བྲག་ཡོད་པ་བཞིན་དུ་ཨ་ལ་འཕྲལ་སྦྱོར་གྱི་ཨ་དང་། གསལ་བྱེད་ཀྱི་ཨ། དེ་བཞིན་མི་རིགས་དང་སྐད་རིགས་མི་འདྲ་བ་རྣམས་ཀྱི་སྐད་གདངས་རྩོམ་གཞིར་བྱས་པའི་གཟུགས་དང་རྣམ་པ་མི་འདྲ་བ་མཚོན་པའི་ཨ་ཡིག་བྱེ་བྲག་པ་སྣ་ཚོགས་ཡོད། རྟགས་གྲུབ་སྟེ།  གསེར་བུམ་བུམ་པ་ཡིན།  གསེར་བུམ་བུམ་པ་དང་བདག་གཅིག་ཏུ་འབྲེལ། གསེར་བུམ་མ་ཡིན་ཞིང་བུམ་པ་ཡིན་པའི་གཞི་མཐུན་པ་དུ་མ་གྲུབ་པ་བཞིན། འཕྲལ་སྦྱོར་གྱི་ཨ་ཨ་ཡིན། འཕྲལ་སྦྱོར་གྱི་ཨ་ཨ་དང་བདག་གཅིག་ཏུ་འབྲེལ། འཕྲལ་སྦྱོར་གྱི་ཨ་མ་ཡིན་ཞིང་ཨ་ཡིན་པའི་གཞི་མཐུན་པ་དུ་མ་གྲུབ་པའི་ཕྱིར་ཏེ་གསལ་བྱེད་ཀྱི་ཨ་སོགས་འཇོག་ཆོག་པ་བཞིན་ནོ། དེ་ལ་བརྟགས་ན། ཨ་ལ་ཤའི་《རྟགས་འཇུག་འགྲེལ་པ་》 དང་《ཡིག་བཤད་མཁས་པའི་ཁ་རྒྱན་》གཉིས་ཀར། “ཨ་ནི་ཡི་གེ་ཀུན་གྱི་སྲོག་ཏུ་ཞུགས་”ཞེས་བཤད་པ་དེ་བཟོས་པའི་ཡི་གེ་སྟེ། ལེགས་སྦྱར་ཡི་གེ་ལྟ་བུ་ཨ་ཡིག་སྲོག་ཞུགས་འཕྲལ་སྦྱོར་ནུས་པའི་ཡི་གེར་གོ་དགོས་པ་ལས། བོད་ཡིག་ལྟ་བུ་སྲོག་ཞུགས་འཕྲལ་སྦྱོར་བྱེད་ཚུལ་མེད་པའི་རིགས་སུ་གོ་ཐབས་མེད་དེ། བོད་ཀྱི་གསལ་བྱེད་གྲས་ཀྱི་ཨ་ཡིག་དེ་རང་ཉིད་གསལ་བྱེད་ཡིན་ཞིང་། བོད་ལ་དེ་ལས་གཞན་པའི་འཕྲལ་སྦྱོར་གྱི་ཨ་ཞིག་ལོགས་སུ་ངོས་འཛིན་རྒྱུ་མེད་པས་སོ། མཁས་དབང་ཚེ་ཏན་ཞབས་དྲུང་གིས།“ཨ་ལ་ཤའི་རྟགས་འཇུག་འགྲེལ་པ་དང། ཡིག་བཤད་མཁས་པའི་ཁ་རྒྱན་གཉིས་ཀར། ཨ་ནི་ཡི་གེ་ཀུན་གྱི་སྲོག་ཏུ་ཞུགས། ཞེས་བཤད་པའང་བོད་ཡིག་གཞིར་བཞག་པ་ལ་ཐ་སྙད་དེ་སྦྱོར་རུང་མི་རུང་ལ་བརྟག་རྒྱུ་ཡོད་དོ་”⒁ཞེས་དོགས་སློང་མཛད་པའང་རྒྱུ་མཚན་འདི་འདྲར་ཐུག་པས་ཡིན། རྒྱུ་མཚན་དེའི་ཕྱིར་ན། མི་རིགས་ཤིག་གི་ཡི་གེར་དབྱངས་ཡིག་དུ་ཙམ་ཡོད་པ་ངོས་འཛིན་པའི་དུས་སུ་མི་རིགས་སོ་སོའི་ཐུན་མིན་གྱི་ཡི་གེར་གསལ་བའི་དབྱངས་ཡིག་བྱེ་བྲག་པ་དེ་ཅི་ཡོད་ཀྱི་ངོས་ནས་བརྟག་དགོས་པ་ལས་སྐད་རིགས་ཀུན་གྱི་གདངས་སུ་རང་ཆས་སུ་ཡོད་པའི་ཨ་དེ་མི་རིགས་གང་ཞིག་གིས་བཟོས་པའི་དབྱངས་ཡིག་གི་གྲས་སུ་ངེས་པར་དུ་ཡོད་དགོས་པའི་ངེས་པ་མེད་དེ། གང་ཡིན་ཟེར་ན། ལེགས་སྦྱར་ཡི་གེ་ཕུད་པའི་བོད་ཡིག་ལྟ་བུ་ནི་དབྱངས་སྲོག་ཞུགས་ཀྱི་རྟགས་སམ་འཕྲལ་སྦྱོར་བྱེད་སྲོལ་མེད་པར་ངག་ནས་བརྗོད་པའི་ཡི་གེའམ་སྐད་ཀྱི་གདངས་དེ་མཚོན་བྱེད་དུ་བཟོས་པའི་ཡི་གེ་ཡིན་ཞིང། བོད་ཀྱི་བསྟན་བཅོས་ཇི་སྙེད་པ་རྣམས་ཀློག་པའི་དུས་སུ་ངག་གི་གདངས་དེ་གཟུགས་རིས་ཡི་གེ་དེ་དང་དེར་ལྡན་པའི་ཚུལ་གྱིས་ཀློག་པས། ཡི་གེ་དངོས་ནི་ཀ་སོགས་ཀློག་བཞིན་པའི་དུས་ཀྱི་རང་རང་གི་སྐད་གདངས་མི་འདྲ་བའི་བརྗོད་པ་དེ་དག་ཡིན་པ་ལས་ཤོག་བུའི་ངོས་སུ་བཀོད་པའི་རི་མོའི་ཡི་གེ་དེ་མིན་པ་ཤེས་སླ་བ་ཞིག་རེད། ཡིན་ན་ཡང།  མི་རྣམས་ཀྱིས་རྟག་ཏུ་གདགས་གཞི་སྐད་གདངས་ཇི་ལྟ་བ་མཚོན་བྱེད་ཀྱི་རི་མོ་དེར་ཡི་གེ་ཟེར་བ་དང་། དེ་འདྲའི་ཡི་གེའི་སྲོག་ཏུ་དབྱངས་ཨ་མ་ཞུགས་པར་ངག་ནས་བརྗོད་དུ་མེད་ཚུལ་བཤད་པ་ནི་སྐད་ཀྱི་གདངས་དང་རི་མོའི་ཡི་གེའི་དབར་ལ་ཁྱད་པར་ཅི་ཡོད་པ་དང་འབྲེལ་བ་ཇི་ལྟར་འབྱུང་བཞིན་པའི་རྟེན་གནས་ཀྱི་ཚུལ་གསལ་པོར་ངེས་མ་ཐུབ་པས་ལན་པ་ཡིན། དེར་བརྟེན། ང་ཚོས་ཡི་གེ་ཞིག་ཀློག་པའི་དུས་སུ་ཡི་གེ་དེ་དང་དེའི་གདངས་ཇི་ལྟ་བ་བཞིན་ཀློག་བཞིན་ཡོད་པའི་ངོས་འཛིན་བས་ཀྱང་ཀློག་བཞིན་པའི་སྐད་ཀྱི་གདངས་དེ་ནི་མི་རིགས་དེ་དང་སྐད་རིགས་དེར་ངེས་པའི་ཐུན་མིན་གྱི་ཡི་གེ་སོ་སོ་བ་སྟེ། རང་རང་གི་སྐད་གདངས་ཀྱི་ངོས་ནས་ཐོན་པའི་བརྗོད་པ་མི་འདྲ་བ་ཇི་སྙེད་པ་རྣམས་རི་མོའི་ཡི་གེར་འཁོད་པ་རྣམས་ལ་སྦྱར་བའི་ཚུལ་གྱིས་ཀློག་པ་ལས་རི་མོའི་ཡི་གེ་ལས་སྐད་གདངས་དེ་རྣམས་ཚུར་སྣང་བ་ཞིག་མིན། དཔེར་ན། ལེགས་སྦྱར་གྱི་ཥ་ཡིག་འདི་ཁ་དང་ཧྲ་ཞེས་གང་བདེར་ཀློག་ཆོག་པ་དང་། དེ་བཞིན་ཛྙཱ་ཡིག་ལའང་། ཛ་ལ་ཉཱ་བཏགས་འགྱ་འམ་གཉའ་ཞེས་པ་བཞིན་རང་རང་གི་བློའི་ངོར་ཇི་ལྟར་ཡོད་པ་ལྟར་ཀློག་པ་དང་།  བོད་ཡུལ་དུ་ཨ་སྔོན་དགུང་གི་ཉི་མ་ལྟ་བུ་ལ་ཡུལ་འགའ་ཞིག་ཏུ་ནུ་མར་ཀློག་བཞིན་པ་བཅས། དེ་འདྲ་བའི་ངོས་ནས་བརྟགས་ན། རི་མོའི་ཡི་གེའི་ངོས་ནས་སྒྲའི་ཀློག་ཚུལ་དེ་འདྲ་སོ་སོར་བྱུང་བ་མ་ཡིན་པར་ཀློག་མཁན་དེས་ཐོག་མར་ཀློག་སློབ་པའི་དུས་སུ་རི་མོའི་ཡི་གེ་དང་དེའི་ཀློག་ཚུལ་འདི་དང་འདི་ལྟར་ཡིན་པའི་བསླབ་རིམ་རྣམས་རང་གི་ཡིད་ངོ་ན་ཇི་ལྟར་ཡོད་པ་བཞིན་ཀློག་པ་སྟེ། རི་མོའི་ཡི་གེ་རང་ངོས་ནས་སྒྲ་འདི་ཞེས་པ་ཞིག་བསྟན་དུ་མེད་པས་ཡི་གེ་རེ་འགར་ཀློག་སྟངས་མི་འདྲ་བ་རེ་འབྱུང་བཞིན་པའི་རྒྱུ་མཚན་ཡང་འདི་རེད། ཏུན་ཧོང་གཏེར་ཡིག་སོགས་སྔོན་གྱི་ཡི་གེ་རྣམས་སུ་སྐྱེ་གནས་མཚུངས་པའི་ཡི་གེ་མང་པོ་ཞིག་ཕན་ཚུན་བརྗེས་ནས་བྲིས་པའི་གནས་ཚུལ་དེ་ཡང་གཟུགས་ཀྱི་ཡི་གེར་སྤྱི་སྲོལ་གྱི་སྒྲ་ངེས་ཅན་ཞིག་གཏན་འཁེལ་མ་བྱུང་བའི་དུས་ཀྱི་རྣམ་པ་ཞིག་རེད། ཀློག་དག་པོའམ་ཀློག་ཚུལ་ཇི་ལྟ་བ་བསླབ་རྗེས་གཞི་ནས་ཐུན་མིན་གྱི་དབྱངས་དང་གསལ་བྱེད་རྣམས་ཐོགས་པ་མེད་པར་ཀློག་ཐུབ་པའི་ཡོན་ཏན་འབྱོར་ཐུབ། བདག་ལྟ་བུས་དཔལ་སྣར་ཐང་བའི་ལུགས་བཞིན་ཥ་ཡིག་དེ་བོད་ཀྱི་ཁ་ཡིག་ལྟ་བུར་ཀློག་བཞིན་ཡོད་པ་དང་། སྔར་ཉི་མ་ལ་ནུ་མ་ཟེར་མཁན་དེས་ཀྱང་སྔར་གྱི་ཀློག་སྟངས་བསྒྱུར་ནས་ཉི་མ་ཞེས་ཀློག་པར་འགྱུར།  དེའི་ངོས་ནས་བརྟགས་ན། གདགས་གཞིའི་དངོས་པོའི་མིང་གི་སྒྲ་རྣམས་ཐོག་མར་མི་རིགས་དང་སྐད་རིགས་མི་གཅིག་པ་རྣམས་ཀྱིས་རང་རང་གི་སྐད་ཀྱི་མངའ་ཁོངས་ནས་ཕན་ཚུན་གོ་བརྡ་སྤྲོད་བྱེད་དུ་བཟོས་པ་དང་དེ་འདྲའི་ངོས་འཛིན་གཅིག་གྱུར་བྱུང་བ་རྨང་གཞིར་བས་ནས་བཟོས་པའི་ཡི་གེ་དེ་དང་དེའི་སྒྲ་གདངས་འདི་དང་འདི་འདྲ་ཡིན་པ་གཏན་འབེབས་བྱས།  མི་རིགས་སོ་སོའི་སྐད་ཀྱི་གདངས་དེ་མཚོན་བྱེད་ཀྱི་ཡི་གེ་བྱུང་བ་ནས་བཟུང་ཡི་གེ་དེ་དང་དེར་སྒྲ་གདངས་དེ་དང་དེ་ལྟ་བུ་ལྡན་པའི་ཚུལ་གྱིས་ཀློག་བཞིན་ཡོད་ན་ཡང། དོན་ངོ་མར་ཀློག་མཁན་དེ་ལ་རང་རང་གི་སྐད་གདངས་ཀྱི་ངོས་ནས་ཡི་གེ་དེ་དག་ཀློག་ཐུབ་པའི་བསླབ་པ་སྔོན་སོང་གི་ཡོན་ཏན་ཞིག་འབྱོར་དུ་བཅུག་པ་ལྟར་ཀློག་བཞིན་ཡོད། འཕགས་ཡུལ་གྱི་ལེགས་སྦྱར་ཡི་གེ་བཟོ་བའི་དུས་སུ་སྐད་གདངས་ཐུན་མོང་ལ་ཡོད་པའི་དབྱངས་ཨ་དེ་ཉིད་བཟོས་པའི་ཡི་གེའམ་རི་མོའི་ཡི་གེ་དེར་ཞུགས་ཚུལ་ཡོད་པའི་ཚུལ་གྱིས་བཟོས་ཤིང། དབྱངས་རིང་ཐུང་གི་ངོས་ནས་སྒྲ་རིང་ཐུང་གི་རྣམ་གཞག་འབྱེད་པས། འཕགས་ཡིག་ནི་དབྱངས་ཨ་དེ་ཉིད་སྲོག་ཏུ་མ་ཞུགས་པར་བརྗོད་དུ་མེད་པའི་ཡི་གེར་གྱུར་བ་མ་ཟད། ཁོང་ཆོའི་དབྱངས་ཡིག་ལ་ཨ་དང་བཅས་པའི་དབྱངས་ཡིག་བཅུ་དྲུག་ངོས་འཛིན་དུ་ཡོད་པའི་རྒྱུ་མཚན་ཡང་དེ་ཡིན། དེ་ལས་གཞན་པའི་ཡི་གེ་སྟེ་བོད་ཡིག་ལྟ་བུར་ཨ་འདི་སྲོག་ཏུ་ཞུགས་ཚུལ་མེད་པས། མཁས་དབང་སི་ཏུས། “བསྟན་བཅོས་འདི་ཡི་ལུགས་ཀྱིས་དབྱངས་ཡིག་སྲོག་ཞུགས་མི་བཞེད་” ⒂ཅེས་གསུངས་སོ།   
        ཁ་ཅིག་གིས་དེ་འདྲའི་ཁྱད་ཆོས་མི་འདྲ་བའི་ཚུལ་ལ་མ་བརྟགས་པར་“ཨ་ཡིག་དབྱངས་སུ་བཞག་ཀྱང་བོད་ཡིག་ལ་སྲོག་ལྡན་མིན་གྱི་རྣམ་གཞག་མི་དགོས་ཤིང་ལེགས་སྦྱར་བཞིན་སྲོག་ཞུགས་ཀྱི་ཐ་སྙད་ཀྱང་མི་འཇུག་པས། ཀ་ནས་ཧ་བར་གྱི་གསལ་བྱེད་ཉེར་དགུ་པོ་ཚང་མར་བརྗོད་པའི་གནས་སྐབས་སུ་དབྱངས་ཨ་རེ་མི་མངོན་པའི་ཚུལ་གྱིས་ཞུགས་ཡོད་ཅེས་བརྗོད་ན་སྐྱོན་མེད་ལ་”⒃ཞེས་བོད་ལ་དབྱངས་ལྔ་ཡོད་ཚུལ་བཤད་ཡོད་ཀྱང། ཁྱོད་ཀྱིས་“བོད་ཡིག་ལ་སྲོག་ལྡན་མིན་གྱི་རྣམ་གཞག་མི་དགོས་ཤིང་ལེགས་སྦྱར་བཞིན་སྲོག་ཞུགས་ཀྱི་ཐ་སྙད་ཀྱང་མི་འཇུག་”ཚུལ་བཤད་པ་དེས་བོད་དང་ལེགས་སྦྱར་གཉིས་ཀྱི་ཡི་གེར་ཁྱད་པར་ཡོད་ཚུལ་ཁས་ལེན་པ་ཡིན་ལ། བོད་ཡིག་ལ་“ཀ་ནས་ཧ་བར་གྱི་གསལ་བྱེད་ཉེར་དགུ་པོ་ཚང་མར་བརྗོད་པའི་གནས་སྐབས་སུ་དབྱངས་ཨ་རེ་མི་མངོན་པའི་ཚུལ་གྱིས་ཞུགས་”ཞེས་པའི་ཚིག་དེ་ནི་ཁོ་བོས་གོང་དུ་བསྟན་པའི་ཨ་ལ་སྤྱི་བྱེ་བྲག་གཉིས་སུ་ཕྱེ་བའི་ནང་ནས་སྤྱི་ལ་ཁྱབ་པའི་ཨ་དེར་ངོས་འཛིན་པའི་ཚིག་ཏུ་གྱུར།  དེ་ལྟར་མིན་ཟེར་ན་བོད་ཡིག་ལ་“བརྗོད་པའི་གནས་སྐབས་སུ་དབྱངས་ཨ་རེ་མི་མངོན་པའི་ཚུལ་གྱིས་”ཇི་ལྟར་བྱུང་བ་རེད། སྐབས་དེའི་ཨ་དེ་ལེགས་སྦྱར་ལྟར་འཕྲལ་སྦྱོར་ཆོག་པའི་ཨ་ལ་ཟེར་བ་ཡིན་ནམ། ཡང་ན་བོད་ཀྱི་གསལ་བྱེད་སུམ་ཅུའི་གྲས་ཀྱི་ཨ་དེ་ཉིད་བོད་ཀྱི་ཡི་གེ་ཀུན་གྱི་སྲོག་ཏུ་ཞུགས་ཡོད་ཟེར་བ་ཡིན། དང་པོ་ལྟར་དུ་ངོས་འཛིན་ཐབས་མེད་དེ། བོད་ལ་ལེགས་སྦྱར་བཞིན་འཕྲལ་སྦྱོར་ཆོག་པའི་ཨ་ཞིག་མེད་པའི་ཁས་ལེན་སྔོན་དུ་སོང་ཟིན་ལ། གཉིས་པ་ལྟར་དུའང་ངོས་འཛིན་མི་ཐུབ་སྟེ། བོད་ཀྱི་གསལ་བྱེད་ཀྱི་ཨ་དེ་རང་ཉིད་གསལ་བྱེད་ཡིན་བཞིན་དུ་གཞན་པའི་གསལ་བྱེད་ཉེར་དགུའི་སྲོག་ཏུ་ཞུགས་ཐབས་མེད་པ་མཁས་དབང་ཚེཏན་ཞབས་དྲུང་གིས་གསུངས་པ་ལྟར་ལགས་པས། ཐབས་ཚུལ་དེ་གཉིས་སྤང་པའི་རྒྱུ་མེད་རྐྱེན་མེད་ཀྱི་ཨ་ཞིག་“བརྗོད་པའི་གནས་སྐབས་སུ་”ཇི་ལྟར་བྱུང་བ་རེད། དེ་ལྟར་བརྟགས་ན། བོད་ཡིག་ལ་“བརྗོད་པའི་གནས་སྐབས་སུ་དབྱངས་ཨ་རེ་མི་མངོན་པའི་ཚུལ་”གྱིས་ཡོད་པར་བཤད་པའི་ཨ་དེ་དོན་དངོས་སུ་མི་རིགས་གང་དང་གང་ཞིག་གི་སྐད་གདངས་མཚོན་བྱེད་དུ་བཟོས་པའི་རི་མོའི་ཡི་གེ་རྣམས་ཀློག་དུས་ཀྱི་མི་རིགས་གང་དེའི་སྐད་གདངས་མི་འདྲ་བ་རྣམས་ཡི་གེ་དང་དེར་ལྡན་པའི་ཚུལ་གྱིས་ཀློག་པའི་ཀློག་མཁན་གྱི་ངག་གི་གདངས་དེ་ཡིན་པ་ཁྱོད་ཀྱིས་མི་འདོད་བཞིན་དུ་ཁས་ལེན་དགོས་པ་མ་ཡིན་ནམ། དེ་ལྟར་བརྟགས་ན། བོད་ལ་ལེགས་སྦྱར་བཞིན་ཨ་དབྱངས་ཡིག་ཏུ་ངོས་འཛིན་ཆོག་པ་ཞིག་མེད་པ་མ་ཟད། གསལ་བྱེད་སུམ་ཅུའི་གྲས་ཀྱི་ཨ་དེ་ཡང་དོན་མེད་དུ་ཕྱིར་ཕུད་མི་རུང་བའི་གནས་ལུགས་དེ་གསལ་པོར་ཤེས་ཐུབ་པ་མ་ཡིན་ནམ། བོད་ལ་དབྱངས་ལྔ་ཡོད་པར་འདོད་པའི་བརྡ་སྤྲོད་སྨྲ་བ་མང་པོས་གནད་འདི་དག་ལ་དཔྱད་པ་མེད་པར་ལེགས་སྦྱར་ལྟར་བཤད་ཆོག་པའི་ལྟ་ཚུལ་འཛིན་གྱི་ཡོད་ཀྱང། འཕགས་བོད་ཀྱི་ཡི་གེར་ཐུན་མིན་གྱི་ཁྱད་ཆོས་མངོན་གསལ་ཡོད་བཞིན་དུ་སྐྱོན་མེད་པའི་ཚུལ་བཤད་ནས་ཐུན་མིན་གྱི་ཁྱད་ཆོས་འགེབས་ཐབས་བྱས་པ་ནི་སྐྱོན་ཆེན་པོ་སྟེ། གནས་ལུགས་མེད་པར་འདྲ་བའམ་སྙམ་པ་གཞིར་བྱས་ནས་མིན་པ་ཡིན་པར་བཤད་པ་དེ་སྐྱོན་མིན་ན་ཅི་ཡིན། 《ཐོན་མིའི་ཞལ་ལུང་》དུ། “བོད་ཡིག་གི་ཕྱི་མོ་ལེགས་སྦྱར་གྱི་ཡི་གེ་ཡིན་ཡང་བོད་མིའི་སྐད་ཀྱི་དབང་གིས་དེ་གཉིས་ཆ་ཐམས་ཅད་ནས་འདྲ་མཚུངས་སུ་བཟོ་མི་ཐུབ་” ⒄ཅེས་གསུངས་པ་དེས་ཡིག་རིགས་དེ་གཉིས་ཀྱི་མི་འདྲ་བའི་ཁྱད་པར་གྱི་རང་བཞིན་བསྟན་གནང་བ་ཡིན་པས། ང་ཚོས་གནས་ལུགས་དེ་འདྲ་གཞིར་བྱས་ནས་འཕགས་བོད་ཀྱི་ཡི་གེའི་འདྲ་ས་གང་དང་མི་འདྲ་ས་གང་ཡིན་གྱི་ཁྱད་པར་བརྟག་དགོས་པ་ལས་ཁྱད་པར་ཤེས་བཞིན་དུ་སྐྱོན་མེད་ཚུལ་བཤད་ནས་ཕྱོགས་འཐེན་བྱས་ན་ག་ལ་འགྲིག དཀར་ལེབ་དྲུང་ཡིག་པད་མ་རྡོ་རྗེའི་(1858 -1933)སུམ་རྟགས་སུ “ལེགས་པར་སྦྱར་བའི་སྒྲའི་རྐྱེན་གྱིས། །བོད་ཀྱི་སྐད་སྒྲུབ་མི་ནུས་ལ། །བོད་ཀྱི་སྐད་དང་བརྡ་ཉིད་ཀྱིས། །རང་གི་ཐ་སྙད་གོ་བའི་ཕྱིར། ། སཾ་ཀྲྀ་ཏ་ཡིས་སྒྲུབ་མི་དགོས། །”ཞེས་རང་གི་ཁྱད་ཆོས་ལ་མི་སེམས་པ་དང་ལེགས་སྦྱར་ལ་ཁུངས་གཏུགས་ན་འོས་མིན་ཡང་མི་བརྟག་པའི་ཨུ་ཚུགས་ཅན་རྣམས་ལ་བཀའ་བཀྱོན་དེ་ལྟར་གནང་བར་རྒྱུ་མཚན་མེད་པ་མིན། བྱེ་བྲག་མི་འདྲ་བའི་ཁྱད་པར་དེ་སྐབས་འགར་གྲངས་ཀྱི་ངོས་ནས་ངོས་འཛིན་བྱེད་དགོས་ཏེ། བོད་ལ་དབྱངས་བཞི་དང་ལེགས་སྦྱར་ལ་དབྱངས་བཅུ་དྲུག་ཡོད་པའི་ནང་གསེས་ཁག་གཅིག་འདྲ་ཡང་མང་ཉུང་གི་ངོས་ནས་དེ་གཉིས་མི་འདྲ་བའི་ཚུལ་བཤད་ན་དོན་ལ་གནས་པ་ཞིག་རེད། དེ་སྔ་བོད་ལ་དབྱངས་ལྔ་ཡོད་པར་འདོད་པ་རྣམས་ཀྱིས་དོན་ལ་གནས་མིན་ལ་མ་བརྟག་པར་ལེགས་སྦྱར་ལ་དཔེ་བྱས་པའི་སྔོན་གྱིམཁས་པ་མང་པོའི་གསུང་ཚུལ་རྣམས་མཐུན་ཕྱོགས་སུ་བྱས་པའི་ལུང་རིགས་དཔེ་གསུམ་གང་མང་དྲངས་ཞིང་ལྡོག་ཕྱོགས་ཀྱི་དཔྱད་གཞི་མང་པོར་དཔྱད་བརྗོད་ཙམ་ཡང་མེད་པར་རྣ་བ་བཀབ་ནས་ཅོང་རྐུ་བའི་རྣམ་ཐར་གྱི་ཚུལ་འཛིན་པ་དང་། སྐད་རིགས་སྤྱིའི་གདངས་སུ་ཡོད་པའི་ཨ་དེ་ཡང་བོད་ཀྱི་སྒེར་ཆོས་སུ་བྱས་པའི་བཤད་ཚུལ་དེ་དག་ནི་སྤྱི་བྱེ་བྲག་དང་ཐུན་མིན་གྱི་ཆོས་ཉིད་དང་ཁྱད་པར་ལ་བསམ་གཞིག་མ་བྱས་པའི་བཤད་ཚུལ་ཞིག་རེད། གལ་ཏེ་བོད་ཀྱི་གསལ་བྱེད་ཀྱི་ཨ་དེ་དབྱངས་ཡིག་གི་གྲས་སུ་འཇོག་དགོས་ཟེར་ན། ཨ་དེ་ཙམ་མ་ཟད་ཧ་ཡིག་དང་འ་ཡིག་གཉིས་ཀྱང་བོད་ཀྱི་དབྱངས་ཡིག་གི་གྲས་སུ་མི་འཇོག་ཐུ་མེད་ཡིན་ཏེ། ཡི་གེ་དེ་གསུམ་ལ་ང་རོའི་རྩ་བ་གསུམ་ཞེས་སྒྲ་རིག་སྨྲ་བ་ཨ་ལག་ཤ་ངག་དབང་བསྟན་དར་གྱི་གསུངས་ཤིང་། ང་རོ་དེ་གསུམ་གྱི་བྱེད་ནུས་ཀྱི་ཆ་ནས་བོད་ཀྱི་ཡི་གེའི་སྒྲ་གདངས་དྲག་ཞན་ཕོ་མོ་མ་ནིང་ལ་སོགས་དབྱེ་བ་ཡིན་ལ། སྔོན་གྱི་ཡི་གེར་དེ་གསུམ་ཇི་ལྟར་སྦྱོར་ཚུལ་གཟུགས་ཅན་དུ་བྱུང་ཚུལ་ནི། གསར་རྙིང་བརྡའ་དག་ཁོག་བུབས་མ་ཡིན་ལ་(ཨ་ལ་ཤའི་ཚིག་)ཞེས་བརྡ་ལ་མཐའ་རྟེན་འ་བྱུང་བ་དང་། ཏུན་ཧོང་ཡི་གེ་བཟང་པོ་སྤྱོད་པའི་སྨོན་ལམ་བ “དེ་དག་རྙྷཱེད་པ་རབ་རྙྷཱེད་བདེ་བར་མཚོ”ཞེས་རྙེད་ལ་ཧཱ་ཡིག་ཨ་རིང་ཅན་བྱུང་བ་དང་། ཞང་ཤུ་ཡི་བསྒྱུར་ཡིག་དུམ་བུའི་ནང་དུ། “ཁེར་གྱི་སྲིད་བལྟོག་པར་”ཞེས་རླ་ཡིག་ལ་ཧ་བཏགས་པ་དང། དེ་བཞིན་“རྫོགས་སྷོ་”ཞེས་པའི་ཧ་རྣམས་དང་། གཞན་ཡང། ང་རོ་དྲང་པོ་མཚོན་བྱེད་ཀྱི་ཨ་ཡིག་སྟེ་ང་རོ་དེ་གསུམ་གྱི་བྱེད་ནུས་ཀྱི་ཆ་ནས་ཧ་ལམ་མཚུངས་ཚུལ་གྱི་རིགས་པའི་སྒྲུབ་བྱེད་འདིར་ཕྱོགས་ཙམ་བཀོད་པ་སྟེ། དེ་ལྟར་ཁས་བླངས་ན་བོད་ལ་གསལ་བྱེད་ཉེར་བདུན་ལས་མེད་པ་ཡང་ཁྱོད་ཀྱིས་ཁས་ལེན་དགོས་ཏེ། ང་རོ་རྩ་བ་གསུམ་ལས་ཨ་ཡིག་དབྱངས་ཀྱི་ཡི་གེ་ཡིན་པ་ཁྱོད་ཀྱི་ཁས་ལེན་ཡིན་པའི་ཕྱིར། ཡང་ཁྱོད་ཀྱིས་བོད་ཀྱི་ང་རོའི་རྩ་བ་གསུམ་ལས་ཨ་ཡིག་མ་གཏོགས་གཞན་ཧ་དང་འ་གཉིས་དབྱངས་མ་ཡིན་ཞེ་ན། དེ་ཡང་མི་འཐད་དེ། ང་རོའི་རྩ་བ་དེ་གསུམ་ལས་ཁོག་པ་ནས་ཤུགས་བསྐྱེད་དེ་འདོན་པའི་ཡི་གེའི་རྩ་བ་དེ་ནི་ཧ་ཡིག་དང། མགྲིན་པ་བཙུམ་ཞིང་དོག་པར་བྱས་ནས་སྟོད་དུ་ཤུགས་བསྐྱེད་དེ་འདོན་པའི་ཡི་གེའི་རྩ་བ་ནི་ཨ་ཡིག་དང། མགྲིན་པ་ཕྱེ་ཞིང་གློད་དེ་ཡངས་པར་བྱས་ནས་སྨད་དུ་ཤུགས་བསྐྱེད་དེ་འདོན་པའི་ཡི་གེའི་རྩ་བ་ནི་འ་ཡིག་ཡིན་པས། ཧ་ཨ་འ་གསུམ་ལས་ཐོག་མ་ཧ་ཡིག་ཡིན་པ་དང་། དེའི་གདངས་ཀྱི་མགྲིན་པ་གྲིམ་ལྷོད་ཀྱིས་ཨ་དང་འའི་སྒྲ་འབྱུང་བས། བོད་ལྟར་ན་ཧ་ཡིག་འདི་ཐོག་མ་དང་དེ་ནས་ཨ་དང་འའི་སྒྲ་འབྱིན་པ་སུམ་རྟགས་ཀྱི་གཞུང་དུ་བཤད་པ་དང་སྔོན་གྱི་ཡི་གེར་དེ་དག་སྲོག་ཏུ་ཞུགས་པ་ལྟ་བུའི་འབྲི་ཚུལ་ཡང་ཡོད་པ་གོང་དུ་འཁོད་ཟིན་པས། དེ་དག་དབྱངས་ཡིན་ན་ཡིན་ཁྱབ་མཉམ་དང་གསལ་བྱེད་ཡིན་ན་ཡིན་ཁྱབ་མཉམ་དུ་འགྲོ་བ་ལས། རང་གི་འདོད་པ་ལྟར་འདོར་ལེན་གང་འདོད་དུ་བྱེད་ཆོག་པ་ཞིག་མིན། ཧ་ཨ་ཕན་ཚུན་བརྗེ་བའི་འབྲི་ཚུལ་ཏུན་ཧོང་ཡིག་རྙིང་P.  T315ནང་དུ་བྱུང་ལ།    P.T. 1291 -2 -1ནང་དུ། “འགུའྀ་རྒྱལ་པོ་ཧན་ལྀ་འྭང་རྒོ་རིང་ལ་”ཞེས་པའི་ཧན་ལྀ་འྭང་ནི་རྒྱ་རྗེ་安厘王ཞེས་པའི་འབྲི་ཚུལ་ཡིན་པས། ཨ་དང་ཧ་ཕན་ཚུན་བརྗེས་པའི་དཔེ་ཡིན་པ་དང་། དེ་ལ་བརྟགས་ན། ང་རོ་ལྔ་ཞེས་པ་ནི་མི་རིགས་ཀུན་གྱི་སྐད་ཀྱི་གདངས་ལ་ཡོད་པའི་དྲང་པོ་བསྒྲེང་དང་བཀུག་པ་སོགས་ཀྱི་གནས་ལུགས་ངོས་འཛིན་པའི་མིང་ཡིན་ལ། རྗེས་མའི་རྩ་བའི་ང་རོ་གསུམ་ཞེས་པ་ནི་བོད་ཀྱི་གསལ་བྱེད་ཕོ་མོ་མ་ནིང་གི་དབྱེ་བ་ཇི་ལྟར་འབྱེད་ཚུལ་གྱི་ཁུངས་སྟོན་བྱེད་ཀྱི་མིང་ཡིན་པས། སྔོན་གྱི་མཁས་པས་དེ་ལྟའི་ཁྱད་པར་མཚོན་བྱེད་དུ་ང་རོ་ལྔ་དང་རྩ་བའི་ང་རོ་གསུམ་ཞེས་མིང་གིས་དབྱེ་བ་མཛད་དོ། ཐོན་མིས། “བསླབ་ལ་བརྩོན་པའི་གང་ཟག་གིས། །དང་པོ་ང་རོ་རྣམས་ལ་སྦྱང་། །”ཞེས་དང། སྨྲ་སྒོར་དབྱངས་བཅུ་དྲུག་ལ་ང་རོ་བཅུ་དྲུག་ཅེས་མིང་འདོགས་མཛད་པའི་ང་རོ་ནི་ཡི་གེ་རྐྱང་པ་རྣམས་ཀྱི་རང་རང་གི་སྐྱེས་གནས་དང་མཐུན་པ་དང་། དབྱངས་ལྡན་རྣམས་ཀྱང་སོ་སོའི་ང་རོའམ་སྒྲ་གདངས་འབྱུང་ཚུལ་ཇི་ལྟ་བ་བཞིན་དུ་ཀློག་དགོས་ཚུལ་བསྟན་པ་སྤྱིར་བཏང་དུ་ང་རོ་དང་དབྱངས། གདངས། ཀློག་ཚུལ་རྣམས་དོན་གཅིག་མིང་གི་རྣམ་གྲངས་སུ་ངེས་ཀྱང་། སྐབས་བབ་ཀྱི་དོན་སྟོན་པར་རྩ་བའི་ང་རོ་གསུམ་དང་ལྔ་སོགས་ཀྱི་མིང་འདོགས་མཛད་ཡོད།   
        གསུམ་པ། སྔོན་གྱི་ཡི་གེར་འ་ཡིག་མཐའ་རྟེན་དུ་བྱུང་བ་ནི་དབྱངས་སྲོག་ཞུགས་ཀྱི་རྟགས་མིན།   
        བོད་ཀྱི་ཡི་གེར་དབྱངས་སྲོག་ཞུགས་ཡོད་མེད་ལ་དཔྱད་པའི་དུས་སུ སྔོན་གྱི་ཡིག་རྙིང་ཁག་དང་རྡོ་རིང་། ཏུན་ཧོང་ནས་ཐོན་པའི་ཤོག་དྲིལ། བསྟན་པ་ཕྱི་དར་དུས་ཀྱི་མཁས་པའི་རྩོམ་སོགས་ན་མིང་གང་ཡང་རུང་བ་དང་ཡི་གེ་རྐྱང་པའི་མཐའ་རྟེན་དུ་འ་ཡིག་ཞུགས་པའི་འབྲི་ཚུལ་བྱུང་ཞིང་། རྣམ་པ་དེ་ཁ་ཅིག་གིས་དབྱངས་སྲོག་ཞུགས་ཀྱི་རྟགས་སུ་ངོས་འཛིན་པ་བཞིན། ཁོ་བོས་ཀྱང་དེའི་རྗེས་སུ་ཞུགས་ནས་ཡུན་རིང་འགོར།  ཉེ་བའི་ཆར་《བསྟན་འགྱུར་》ཁྲོད་བཞུགས་པའི་《སྒྲའི་རྣམ་པར་དབྱེ་བ་བསྟན་པ་》ཞེས་པའི་ནང་། “གཞན་ཡང་མིང་གང་ཡང་རུང་བའི་ཐ་མ་ན་འས་བརྟེན་ལ། དེ་གཉིས་པ་ལས་སུ་སྦྱོར་ན་རུ་ཁོ་ནར་སྦྱོར་ཏེ། མིང་གི་མཐའ་མ་འ་ཡིས་བརྟེན་ཞེས་པའང་། མིང་གང་ཡང་རུང་པའི་ཐ་མ་ན་ན་རོ་གུ་གུ་ལ་སོགས་པའི་ཡན་ལག་གང་འདུག་ཀྱང་རུང་སྟེ། ཡི་གེ་གཞན་གང་གིས་ཀྱང་མ་བརྟེན་པ་དེ་ནི་འ་ཁོ་ནས་བརྟེན་པ་ཡིན་ཏེ། དཔེར་ན་མ་གདྷ་རུ་འགྲོ་ཞེས་པ་ལ་སོགས་པའི་ལས་རྣམས་རུང་སྟེ་འའི་རྗེས་འཇུག་སྡུད་པར་བལྟའོ་”⒅ཞེས་གོང་གསལ་གནད་དོན་དང་འབྲེལ་པའི་རྣམ་བཤད་དུམ་བུ་ཞིག་རྙེད་པ་དེ་ནི་བོད་བཙན་པོའི་དུས་ཀྱི་ཡི་གེའི་འབྲི་སྲོལ་སྟོན་པའི་བརྡ་སྤྲོད་ཀྱི་ཁུངས་སུ་གྱུར་ཞིང་། རྣམ་བཤད་དེ་ལས་ཅི་ཞིག་ཤེས་ཐུབ་ཅེ་ན། མིང་གང་ཡང་རུང་བའི་ཐ་མ་ན་ན་རོ་གུ་གུ་ལ་སོགས་པའི་དབྱངས་ཡིག་སྟེ་ཡན་ལག་གང་འདུག་ཀྱང་རུང་སྟེ། རྗེས་འཇུག་ཡི་གེ་གཞན་གང་གིས་ཀྱང་མ་བརྟེན་པའི་ཡི་གེ་རྣམས་ལ་འ་ཁོ་ནས་བརྟེན་པ་དང་། མ་གདྷའ། ཞེས་པའི་མིང་དེ་བྱ་བའི་ཡུལ་དུ་གྱུར་པའི་སྐབས་སུ་འ་ཡིག་མཐའ་རྟེན་སྤངས་ནས་དེའི་དོད་དུ་ལ་དོན་གྱི་ཕྲད་རུ་འདི་འབྲི་དགོས་པ་སྟེ། དཔེར་ན་མ་གདྷ་རུ་འགྲོ། ཞེས་འབྲི་དགོས་ཚུལ་བསྟན་པ་དེ་ནི་སྔོན་གྱི་ཡིག་རྙིང་ཁྲོད་འ་ཡིག་མཐའ་རྟེན་མང་དུ་བྱུང་བ་རྣམས་དབྱངས་སྲོག་ཞུགས་ཀྱི་རྟགས་མིན་པར་བསྟན་ལ། འཆད་ཚུལ་དེ་དོན་ལ་གནས་པ་དཔེར་ན། ཞོལ་རྡོ་རིང་ངོས་སུ། “ཟླ་གོང་གི་བུ་ཚའ་རྒྱུད་འཕེལད་”ཅེས་དང། “ཡི་གྼེ་བྲིས་པའ་”ཞེས་དང། “དཀུ་རྒྱལ་གྱི་ཡི་གེའ་”“དཔྱའ་འཇལ་”“མི་སྤོའ་”སོགས་ཡི་གེ་རྐྱང་པ་དང་དབྱངས་ཡིག་ཡོད་པའི་རྗེས་སུ་འ་ཆུང་ཐོན་པའི་འབྲི་ཚུལ་བྱུང་བ་དང་། ཏུན་ཧོང་ནས་ཐོན་པའི་ཤོག་དྲིལ་P.T.93པའི་ནང་དུ།   དགེའ་འདུན།  གཏིའ་མུག འཇམ་དཔལ་གྱྀས་གསོལ་པའ། བཀུར་བསྟིའ།  སེམས་ཆན་ཁམས་སུ་བལྟའ་མྱྀ་བའ། བློ་གྲོས་ཤེས་བགྱིའ་བ་ཞིག་བྱུང་ཞེས་པ་ལྟ་བུ་བྱུང་བ་ལས། འ་ཡིག་མཐའ་རྟེན་དུ་བྱུང་བའི་སྔོན་གྱི་འབྲི་ཚུལ་དེ་དག་གང་འདོད་དུ་བྲིས་པའམ། དབྱངས་སྲོག་ཞུགས་ཀྱི་ཁྱད་ཆོས་སུ་བྱས་པ་མིན་པའི་གནས་ལུགས་ཤེས་ཐུབ། དེ་ལྟ་བུའི་འབྲི་ཚུལ་གྱི་རྣམ་པ་དེ་བོད་རྒྱལ་སྲོང་བཙན་གྱི་དུས་ནས་ཡོད་པར་ངོས་འཛིན་དགོས་ཀྱང་དུས་སྐབས་དེའི་དཔང་པོར་རུང་བ་ཞིག་བཙལ་ནས་མ་རྙེད་པས། འདིར་ཁྲི་སྲོང་ལྡེ་བཙན་དུས་ཀྱི་ཞོལ་རྡོ་རིང་ཡི་གེ་དཔེར་བཀོད་པ་ཡིན།  
        གཙུག་ལག་ཁང་མདུན་གྱི་བོད་ཐང་རྡོ་རིང་དེའི་ནུབ་ངོས་ཀྱི་ཐིག་ཕྲེང25ལ། “མཇལ་དུམ་ཆེན་པོའ་”ཞེས་ན་རོའི་རྗེས་སུ་འ་ཡིག་གཅིག་བྱུང་བ་དང། ལྷོ་ངོས་ཀྱི་ཐིག་ཕྲེང21ལ། “དའེ་པུའ་”ཞེས་ཞབས་ཀྱུའི་རྗེས་སུ་འ་ཡིག་མཐའ་རྟེན་ཅན་བྱུང་ཞིང། སྔ་མ་དེ་ཕབ་ནོར་དུ་ཤོར་བ་མིན་ན་སྔོན་གྱི་འབྲི་ཚུལ་གྱིས་ཤན་ཤོར་བའི་ཡི་གེར་གྱུར་ལ། ཕྱི་མ་དེ་མི་རིགས་གཞན་གྱི་སྒྲ་འཁོད་པའི་ཡི་གེར་གྱུར་བས་བོད་ཡིག་གི་སྔོན་གྱི་འབྲི་ཚུལ་ལ་འབྲེལ་བ་མེད། ཚུལ་དེ་ལ་དཔྱད་ན། གཙུག་ལག་ཁང་མདུན་གྱི་བོད་ཐང་རྡོ་རིང་ནི་《སྒྲའི་རྣམ་པར་དབྱེ་བ་བསྟན་པ་》ཡི་གཞུང་ལུགས་ཀྱིས་ཤུགས་རྐྱེན་ཅུང་ཙམ་ཡང་ཐེབས་མེད་པའི་རྡོ་རིང་ཡིན་པར་ངོས་འཛིན་བྱེད་ཆོག རྡོ་རིང་དེ་སྤྱི་ལོ་823ལོར་བོད་རྒྱལ་ཁྲི་རལ་གྱི་དུས་སུ་བསླངས་ཤིང་། དུས་སྐབས་དེར་བཀས་བཅད་ཐེངས་གཉིས་པ་བྱེད་ཚུལ་ནི་བཅོམ་ལྡན་རིག་རལ་( 1228 - 1305) གྱིས།“དེང་སང་བོད་ལ་གྲགས་པའི་མིང་། བཀའ་བཅས་ཡུལ་སྐད་འཕྲལ་སྐད་གསུམ། བཀས་བཅད་ལའ་དང་པོ་དང་། བར་དང་ཐ་མ་གསུམ་མི་མཐུན། ཆུས་བསྒྱུར་བ་ན་རྒྱལ་བློན་དང་།  ལོ་པཎ་དང་བརྡའ་ལ་མཁས་པ་རྣམས་འདུས་ཏེ་ཡུལ་ཐ་དད་ན་མིང་དུ་མ་ཡོད་ཀྱང་འདིའི་མིང་ནི་འདིར་འཐད་དོ་ཅེས་བཀས་བཅད་པ་ནི་བཀས་བཅད་ཏེ་དེ་ཡང་གསུམ་གསུམ་མོ། དེ་ལ་འགོན་མི་སཾ་བ་ར་(ཐོན་མིའི་མཚན་)དང་། བཙན་པོ་ཁྲི་སྲོང་ལྡེ་བཙན་གྱི་དུས་ཀྱི་དང་པོར་བྱས་པའི་འགྱུར་རྣམས་ནི་དང་པོ་བྱས་པའི། །བཀས་བཅས་ཀྱིས་བསྒྱུར་སྟེ་སངས་རྒྱས་ཕལ་པོ་ཆེ་དང་ལུང་སྡེ་བཞི་དང་། མདོ་སྡེ་ཁག་ཅིག་དང་ཤེར་ཕྱིན་གྱི་མདོ་སྡེ་ཁག་ཅིག་སྐད་གསར་བཅད་ཀྱིས་གཏན་ལ་ཕབ་པ་རྣམས་སོ། དེ་ལ་ནི་གང་ལ་གག་དང། ཅིའི་ཕྱིར་ལ་ཇི་རོབ་དང་། ཇི་སྙམ་ལ་ཇི་ལྟར་དང་། སྡུག་བསྔལ་བ་ལ་སྙོན་རྨོངས་པ་དང་། རྩེ་མོ་ལ་སྤྱི་བོ་དང་། ཤིན་ཏུ་ལ་ཏེ་ཕོར་དང་། རྐྱེན་སྨན་ལ་རྐྱེན་རྩི་དང་། ལྷག་པ་ལ་གླགས་པ་དང་སྤོས་དང་སྤོས་ཕོར་ལ་པོག་པ་པོག་ཕོར་དང་། བསལ་བ་ལ་བརྩལ་བ་དང་གཡོགས་པ་ལ་ཡོགས་པ་དང་། གོ་ཆ་ལ་ག་ཅ་དང་། བྲག་ཆ་ལ་བྲག་ཅ་དང་། རིན་པོ་ཆེ་ལ་རིན་པོ་ཅེན་དང་། ངག་འཁྱལ་པ་ལ་ངག་ཁྱལ་བ་དང་། ཁྱིམ་ལ་ཏུམ་བར་དང་། དབུས་ལ་དབུང་དང་། འགྲོ་བ་ལ་གོམ་པ་དང་། གྲུབ་པ་ལ་རེད་པ་དང་། འགྲོལ་བ་ལ་འགྲེལ་བར་བསྒྱུར་བ་ལ་སོགས་པ་ཤིན་ཏུ་མང་པོ་ཡོད་དེ་རྒྱས་པ་ནི་གསུང་རབས་རྣམས་སུ་བལྟའོ། གཉིས་པ་སྐད་གསར་བཅད་ལ་ནི་དེ་དག་གི་ཟླ་བོ་ཇི་སྐད་བཤད་པ་རྣམས་དང་གཞན་ཡང་དེང་སང་གསུང་རབས་ལ་གྲགས་པའི་མིང་ཕལ་ཆེ་བ་རྣམས་ནི་ཡིན་ནོ།  བཀས་བཅད་གཉིས་པོ་དེ་ནི་མངའ་བདག་ཁྲི་རལ་པ་ཅན་ཡན་ཆད་དུ་ཀ་ཅོག་ཞང་གསུམ་ལ་སོགས་པས་བྱས་པ་ཡིན་ནོ་”⒆ཞེས་བཀོད་ཡོད། འོན་ཀྱང་བཅོམ་ལྡན་རིག་རལ་གྱིས་བསྟན་གནང་བའི་བཀས་བཅད་གཉིས་ཀྱི་ཁྲོད། གོང་གསལ་ཞོལ་རྡོ་རིང་ལྟ་བུ་དང་ཏུན་ཧོང་ནས་ཐོན་པའི་ཤོག་དྲིལ་ཡི་གེར་འ་མཐའ་འགོད་ཚུལ་བྱུང་བ་རྣམས་ལ་བཅོས་སྒྱུར་བྱེད་ཚུལ་གླེང་སློང་གནང་མེད་ལ། དུས་སྐབས་དེར་གཙུག་ལག་ཁང་མདུན་གྱི་རྡོ་རིང་ཡི་གེར་འའི་མཐའ་རྟེན་བྱུང་མེད་པའི་འབྲི་སྟངས་དམིགས་བསལ་ཅན་གྱི་ཐད་ལའང་གླེང་བརྗོད་གནང་མེད། རྗེས་ཀྱི་མཁས་པ་ཨཱ་ཀྱ་ཡོངས་འཛིན་གྱིས། “མིང་རྗེས་འཇུག་མེད་པའི་མཐར་འ་ཡིག་འཐོབ་ཀྱང་སྐད་གསར་བཅད་ལ་ཡིག་ཚོགས་བསྐྱངས་པའི་ཆེད་མ་བཞག་པ་ཡིན་ནོ་”⒇ཞེས་གསུངས་པ་ལྟར་ཡིན། དེ་ནི་ཁྲི་རལ་པ་ཅན་གྱི་དུས་སུ་མིང་གང་ཡང་རུང་བའི་ཐ་མ་ན་ན་རོ་གུ་གུ་ལ་སོགས་པའི་ཡན་ལག་གང་འདུག་ཀྱང་རུང་སྟེ། ཡི་གེ་གཞན་གང་གིས་ཀྱང་མ་བརྟེན་པ་དེ་ནི་འ་ཁོ་ནས་བརྟེན་དགོས་པར་བཤད་པའི་སྔོན་གྱི་འབྲི་ཚུལ་དེ་རིགས་བཀས་བཅད་དམ་སྐད་གསར་བཅད་ཀྱི་ཚུལ་དུ་བཅོས་སྒྱུར་བས་པར་སྟོན་པའི་ཆི།ག་ཏུ་གྱུར་ཞིང་། བོད་ཐང་རྡོ་རིང་ནི་དེ་ལྟའི་བཅོས་སྒྱུར་རམ་བཀས་བཅད་གཉིས་པ་དེའི་དཔང་པོར་འཛིན་རུང་བའི་རྡོ་རིང་ཞིག་རེད།   
        ཁྲི་རལ་དུས་ཀྱི་འའི་མཐའ་རྟེན་ཐད་བཅོས་སྒྱུར་གནང་བའི་གཞུང་ལུགས་ཀྱི་རྒྱབ་རྟེན་ཞིག་ནི་ཕྱིས་སུ་རྔོག་བློ་ལྡན་ཤེས་རབ་ཀྱིས་མཛད་པའི་《དག་ཡིག་ཉེར་མཁོ་བསྡུས་པ་》ལས།“རྐྱང་པ་འཕུལ་བ་འ་མཐའ་དགོས། བརྩེགས་དང་བུ་དྲུག་ཅན་ལ་སྤང་།  དེ་ཕྱིར་འ་ནི་མཐའ་རྟེན་ཕྱེད་”（21）ཅེས་བྱུང་བ་འདི་ཡིན་སྙམ་སྟེ། ཨཱ་ཀྱ་ཡོངས་འཛིན་གྱིས་དེར་འགྲེལ་མཛད་པའི་དུས་སུ “རྐྱང་པ་འཕུལ་ལ་འ་མཐའ་དགོས། བརྩེགས་པ་བུ་དྲུག་ཅན་ལ་སྤང་”ཞེས་འཕུལ་བ་ཞེས་པ་འཕུལ་ལ་ཞེས་པ་དང་། བརྩེགས་དང་ཞེས་པ་བརྩེགས་པ་ཞེས་པར་བཅོས་ཤིང་དེའི་དོན་ནི། ར་ལ་སའི་མགོ་དང་ཡ་ར་ལྦའི་འདོགས་སོགསགང་ཡང་མེད་པའི་མིང་གཞིའི་ཡི་གེ་རྐྱང་པ་འཕུལ་ཏེ་སྔོན་འཇུག་ཅན་ལ་འ་ཡིག་མཐའ་རྟེན་ངེས་པར་དགོས་ཏེ།   གཞན་དུ་ན་མིང་གཞིའི་ཡི་གེ་དེ་རྫེས་མཇུག་ཏུ་འཁྲུལ་འབྱུང་བའི་ཕྱིར། དཔེར་ན།  དགའ། མདའ། འབའ་ལྟ་བུ་ལ། དག མད། འབ། ཞེས་མཐའ་རྟེན་འ་ཡིག་དངོས་སུ་མེད་ན་དེ་འདྲའི་ག་ཡིག་དང། ད་ཡིག་སོགས་མིང་གཞིར་གོ་མི་ནུས་ཤིང་རྗེས་འཇུག་ཏུ་འཁྲུལ་པའི་ཕྱིར་རོ།   ཡི་གེ་བརྩེགས་པ་སྒ་དང། བརྡ་ལྟ་བུ་དང་། བུ་ཡིག་གུ་གུ་དང། ཞབས་ཀྱུ། འགྲེང་གྲུ ན་རོ་བཞིན་དང་། ཡ་བཏགས་དང་། ར་བཏགས་ཏེ་དྲུག་ལ་བུ་དྲུག་ཅེས་སྔོན་མ་རྣམས་ལ་གྲགས་ཤིང་། དེ་འདྲའི་བུ་དྲུག་ཅན་ནི། དཔེར་ན། འདི། དབུ།  ཆེ།  ཞོ །  བྱ། ཕྲ། ཞེས་པ་ལྟ་བུ་རྣམས་ལ་འ་མཐའ་དངོས་སུ་སྤངས་པར་བྱ་སྟེ། དེ་འདྲ་ལ་འཁྲུལ་གཞི་མེད་ཅིང་དེ་ལ་མཐའ་རྟེན་འ་ཡིག་དངོས་སུ་འཇོག་པ་སྔོན་གྱི་བརྡ་རྙིང་ལ་བྱུང་མོད། འོན་ཀྱང་ཡིག་ཚོགས་ཆེ་བ་ལས་དགོས་པ་གཞན་ཆེར་མི་འགྲུབ་པའི་ཕྱིར་སྐད་གསར་བཅད་ལ་དོར་བ་ཡིན་ནོ་（22）ཞེས་གསུངས་པ་དེ་དག་ནི། ཏུན་ཧོང་ཡིག་རྙིང་སོགས་སུ་མཐའ་རྟེན་འ་ཡིག་དངོས་སུ་འཇོག་པའི་རྒྱུ་མཚན་དང་། བོད་ཐང་རྡོ་རིང་གི་ངོས་སུ་དེ་ལྟའི་འབྲི་ཚུལ་མ་བྱུང་བའི་གནས་ལུགས་ཤེས་བྱེད་དུ་གྱུར་ཡོད། དེ་ཡང་རྐྱང་པ་འཕུལ་ལ་འ་མཐའ་དགོས། ཞེས་རྐྱང་པ་དེ་མིང་གཞི་རྐྱང་པ་འཕུལ་ཅན་ལས་གཞན་ཏེ། “མཆུ་གཉིས་གས་ཤིང་ལྕི། ཁའ་ཡང་མི་བདེ། མཆིལ་མའང་ཀོར་ཀོར་པོར་འདུག་”( P. T.1057-52-53)ཅེས་པའི་འཕུལ་མེད་ཀྱི་མིང་གཞི་རྐྱང་པ་ཁའ་ལྟ་བུ་དང་མིང་མཐའི་ཡི་གེ་རྐྱང་པ་སྟེ། “ཡིད་ཆེས་པའི་གནས་ཡིན་པའ་”ཞེས་པའི་པ་ལྟ་བུར་འ་མཐའ་རྟེན་དུ་བྱུང་བ་ཁྲི་རལ་ཡར་སྔོན་གྱི་འབྲི་སྲོལ་དེ་དག་ལ་གསུངས་པ་ཡིན་ནམ་སྙམ་ཡང་། རྔོག་ལོའི་《དག་ཡིག་ཉེར་མཁོ་》དང་《ཚད་མ་རྣམ་ངེས་》སོགས་དེང་དུས་མཐོང་དུ་ཡོད་པ་དག་ལ་རྗེས་སུ་བཟོ་བཅོས་བརྒྱབ་པ་མིན་ན་དེ་ལྟའི་འབྲི་ཚུལ་བྱུང་མེད་པས། ཨཱ་ཀྱ་ཡོངས་འཛིན་གྱིས་གསུངས་པ་ལྟར་གོ་བར་བྱ་ལ། གལ་ཏེ་མིང་མཐའི་ཡི་གེ་རྐྱང་པར་འ་ཡིག་མཐའ་རྟེན་ཡོད་པར་བཤད་པ་ཡིན་ན་འདིའི་ཐད་ནས་ཁྲི་རལ་གྱིས་བཅོས་སྒྱུར་གནང་བའི་འ་མཐའ་སྤྱོད་སྟངས་ལ་ཡོངས་སུ་བསྟུན་མེད་པའི་གཞུང་ལུགས་སུ་གྱུར་ཡོད་དོ་སྙམ་པས་རྗེས་སུ་དཔྱད་པ་བྱའོ།   
        སྤྱིར་བཤད་ན། ཁྲི་རལ་གྱི་དུས་སུ་འ་ཡིག་མཐའ་རྟེན་དུ་བྱུང་བའི་འབྲི་སྲོལ་ལ་བཅོས་སྒྱུར་བྱེད་ཚུལ་ལྷ་སའི་རྡོ་རིང་ལ་གསལ་བ་ལྟར་ལུགས་སྲོལ་ངེས་ཅན་དེའི་ལམ་དུ་སྐྱོད་དགོས་མོད། རང་རང་གི་འདོད་པ་གཙོ་བོར་བྱེད་པའི་རང་དབང་ལ་དགའ་བ་རྣམས་ཀྱིས་རྟག་ཏུ་རང་ཉིད་ལ་གང་ཞིག་ལོབ་པའི་ལུགས་དེ་རྒྱུན་འཛིན་བྱེད་པའི་གནས་ཚུལ་ཆགས་ཡོད་པ་དཔེར་ན། ཨོཾ་ལ་འ་རིང་བཏགས་པའི་ཨོཾ་གྱི་འབྲི་སྲོལ་དེ་བཙན་པོའི་དུས་སྐབས་སུ་བྱུང་བ་ནས་བཀའ་གདམས་པའི་དགེ་བཤེས་རྣམས་ཀྱིས་སྤྱོད་པའི་ཡི་གེར་གྱུར་བ་དང་།  ས་པཎ་གྱི་《སྒྲ་ཉེར་བར་བསྡུས་པ་》ལས། “རྒྱ་དཔེ་ཨོ་ལ་རིང་ཆ་མེད། །འ་(ཨོྃ) བཏགས་ནོར་ཅེས་བླ་མས་(པཎ་ཆེན་ཤཱཀྱ་ཤྲཱིས་)གསུངས། །”ཞེས་ཨོཾ་ལ་རིང་ཆ་བྱེད་པའི་ལུགས་དེ་བསྒྱུར་དགོས་པར་གསུངས་ཤིང་། སྣར་ཐང་ལོ་ཙཱ་བས། “འགྲེང་བུ་ན་རོ་ཡོད་གྱུར་ན་རིང་པོ་ཉིད་ཕྱིར་འ་འདོགས་འཁྲུལ་”ཞེས་གསུངས། སྒྲ་རིག་གཞུང་ལུགས་སུ་ཁུངས་གཏུགས་པའི་གསུང་དེ་དག་ཡོད་བཞིན་དུ་འབྲུག་པ་པདྨ་དཀར་པོས།(1527 -1596) “དེ་ལ་བོད་རྣམས་རྙིང་མ་ཡང་དག་པ་མིན་ཟེར་བ་ལ་སྔགས་མ་དག་ཟེར་ཏེ། རྒྱ་སྐད་ཨོཾ་ལ་རིང་ཆ་མེད་ཟེར་ནས་ཨོཾ་ལ་འ་བཏགས་པ་སྐྱོན་འདོགས་ཏེ། རྒྱ་སྐད་ཨོཾ་རིང་པོ་ཡིན་པས་རིང་པོ་མེད་སྨྲ་བ་རང་འཁྲུལ། དེ་རིང་པོ་ཡིན་པ་ཤེས་པར་བྱ་བ་ལ། འ་བཏགས་པས་སྐྱོན་ཅིའང་མེད། རྒྱ་སྐད་སོར་བཞགས་ན་ཨོ་སོགས་ཞུགས་པ་ལ་སྣ་རོ་ཉིས་བརྩེགས་ལས་མི་འབྱུང་རྒྱུལ། དེ་བོད་དུ་ཀློག་པ་ལ་གོའུ་ཏ་མའམ། ཀོའུ་ཤི་ཀ་སོགས་སྔོན་གྱི་ལོ་ཙཱ་བ་མཁས་པ་དག་གིས་བཀོད་པ་ལྟ་བུའོ་”（23）ཞེས་གསུངས་ཏེ་ས་པཎ་གྱི་ལྟ་ཚུལ་བཀག་ཀྱང་ཨོཾ་དང་གོའུ་ཏ་མའམ། ཀོའུ་ཤི་ཀ་ལྟ་བུ་སྒྲ་རིང་པོ་དང་ཤིན་ཏུ་རིང་པོའི་ཁྱད་ཡིན་པ་ལས་འ་རིང་ཡོད་མེད་ཀྱི་ཁྱད་མིན་པ་ཤེས་སླ།  ས་ཁུལ་དང་གྲུབ་མཐའ་ཁ་ཅིག་གིས་ད་ལྟ་ད་དུང་སྒྲ་རིག་གཞུང་ལུགས་སུ་འགལ་བའི་ཨོཾ་དང་ཨོཾ་ལྟ་བུའི་འབྲི་ཚུལ་དེ་དག་འདོར་ཐུབ་མེད་པ་ལྟར།  དུས་རབས་བཅུ་གཅིག་པའི་ནང་ཙམ་ལ་བརྩམས་པར་གྲགས་པའི་《རོང་ཟོམ་བཀའ་འབུམ་》དུ།  “བྱ་བ་འཕྲའ་མོ་”ཞེས་དང་། “འདྀའ་ནྀ་འགྲོའ་བ་རིས་དྲུག་”（24）ཅེས་དང་“གཡོའ་བ་དང་མི་གཡོའ་བ་”（25）ཞེས་པ་སོགས་ཁྲི་རལ་སྔོན་གྱི་འབྲི་ཚུལ་ལྟར་འ་ཡིག་མཐའ་རྟེན་དུ་བྲིས་ཤིང་།  དེ་བཞིན་དུས་རབས་བཅུ་གཉིས་པར་འཚོ་བའི་བཅོམ་ལྡན་རིག་རལ་གྱི་《བསྟན་པ་རྒྱས་པ་རྒྱན་གྱི་ཉི་འོད་》ཀྱི་ཤོག་ངོས་དང་པོར། “ཤཀྱའི་གཙོའ་ལ་”ཞེས་དང། “རྩོད་དུས་ཀྱི་མགོའ་འཛིན་པའི་ཚེ”ཞེས་དང།  “ཡིད་ཆེས་པའི་གནས་ཡིན་པའ་”ཞེས་པ། “ཡང་ན་རྒྱ་དཔེའ་ལས་”（26）ཞེས་པ་སོགས་འ་ཡིག་མཐའ་རྟེན་དུ་བྱུང་བའི་འབྲི་ཚུལ་མང་ཞིང་། ད་ལྟའང་འགའ་རེས་བཅོམ་ལྡན་རིག་རལ་གྱི་ལག་རྒྱུན་ལྟར། བརྡའ་སྤྲོད་པ་ཞེས་པའི་བརྡ་ལ་འ་མཐའ་འབྲི་བཞིན་ཡོད་པ་དེ་དག་ལས་ཁྲི་རལ་གྱི་དུས་སུ་བཅོས་སྒྱུར་གནང་བའི་བཀས་བཅད་དེར་དུས་རབས་བཅུ་པའི་རྗེས་སུ་དོ་སྣང་མེད་པར་གྱུར་ཞིང་། བཅོམ་ལྡན་རིག་རལ་གྱིས། “བཀས་བཅད་གསུམ་པ་ནི་ལྷ་བླ་མ་ཡེ་ཤེས་འོད་ཀྱི་དུས་ཀྱི་སྒྲ་བསྒྱུར་རིན་ཆེན་བཟང་པོ་ནས་བཟུང་སྟེ་ཁོ་བོའི་བླ་མ་ཆག་ལོ་ཙཱ་བ་དགེ་སློང་ཆོས་རྗེ་དཔལ་(1197 -1265)ཡན་ཆན་གྱིས་བྱས་པ་ཡིན་ནོ་”（27）ཞེས་གསུངས་པ་ལྟར་བསྟན་པ་ཕྱི་དར་གྱི་མཁས་པ་རིན་ཆེན་བཟང་པོ་ནས་འགོ་ཚུགས་ཏེ་རིམ་བྱོན་མཁས་པ་མང་པོས་འབད་བརྩོན་བྱས་པ་ལས་ད་ལྟའི་བརྡ་དག་གི་ལམ་དུ་ཞུགས་ཐུབ་ཡོད་ཀྱང་། དེ་ལྟའི་འགྱུར་ལྡོག་དེ་བཀས་བཅད་གསུམ་པར་ངོས་འཛིན་ཆོག་མིན་ད་དུང་དཔྱད་བར་ལུས་ཡོད།   
        མདོར་ན། སྔོན་དུས་འ་ཡིག་མཐའ་རྟེན་དུ་སྤྱོད་སྲོལ་བྱུང་བ་དེ་དབྱངས་ཡིག་ཨི་ཨུ་དང་ཨེ་ཨོ་སོགས་བརྩེགས་པ་བུ་དྲུག་ཅན་ལའང་བྱུང་ཞིང་།  དབྱངས་བཞི་གང་རུང་ཡོད་བཞིན་དུ་སླར་ཡང་འ་ཆུང་མཐའ་རྟེན་དུ་བྲིས་པ་སོགས་སྒྲ་མདོ་དང་རིགས་པའི་ལམ་སྲོལ་དུ་མི་གནས་པའི་འབྲི་ཚུལ་བྱུང་བ་དེ་ལས། འབྲི་སྲོལ་དེ་ནི་བོད་ཀྱི་གསལ་བྱེད་ལ་དབྱངས་སྲོག་ཞུགས་ཀྱི་རྟགས་སུ་ངོས་འཛིན་མི་རུང་བ་གསལ་པོར་ཤེས་ཐུབ་བོ།   
        ཁ་ཅིག་གིས་བོད་བཙན་པོའི་དུས་སུ་རྗེས་འཇུག་འ་ཡིག་སྦྱོར་བ་སོགས་ལ་དགོས་པ་ཁྱད་པར་ཅན་འགའ་ཞིག་ཡོད་པར་བཤད་དེ། ཞོལ་རྡོ་རིང་ངོས་སུ། “དཀུ་རྒྱལ་གཙིགས་གནང་བའི་མདོ་རྡོ་རིངས་ལ་ཡིག་གྲུ་བྲིས་པའ་”ཞེས་པའི་འ་ཡིག་ལྷག་བཅས་དང་ཀོང་པོ་དེ་མོ་ཁུལ་གྱི་བྲག་བརྐོས་ཡི་གེར།“གཙྀགས་འཕྲ་མེན་སྒྲོམ་བུ་སྩལད་བའྀ་དཔེར་བྲྀས་པའི་”（28）ཞེས་པའི་འ་ཡིག་རྫོགས་ཚིག་ཏུ་བཤད་པ་སོགས་ད་དུང་མང་དུ་སྤྲོས་འདུག་པ་དེ་དག་ནི་རྗོད་ཚིག་དུམ་བུ་རེ་རེའི་དོན་ལ་སེམས་པའི་བཤད་ཚུལ་ཏེ། དཔེ་དེ་དག་གིས་འ་ཡིག་དེར་བརྡ་སྤྲོད་ཀྱི་ནུས་པའམ་ཁྱད་ཆོས་ཤིག་ཡོད་པའི་རྟགས་སམ་རྒྱུ་མཚན་གྱི་གནད་ཅིག་གཏན་ནས་བསྟན་མེད་དེ། གལ་ཏེ་འ་ཡིག་མཐའ་རྟེན་དུ་བྱུང་བར་བརྡ་སྤྲོད་ཀྱི་ཁྱད་ཆོས་ཡོད་ཟེར་ན། 《ལི་ཤིའི་གུར་ཁང་》གུ “དེའ་ལའ་ནིའ་ཞེས་པ་འ་ཡིག་མཐའ་རྟེན་སྦྱར་བ་ནི་སུམ་རྟགས་ཀྱི་དགོངས་པ་ཡིན་ཀྱང་དོན་ལ་ཐོབ་པ་གཞིར་བཅས་ནས་ཡི་གེ་ཉུང་བའི་སླད་དུ་བཞག་(བཞག་ནི་མཐའ་རྟེན་གྱི་འ་དེ་དག་དོར་པའི་དོན་ཡིན་)”（29）ཅེས་པའི་ཚིག་དུམ་དེའི་ནང་གི་དེའ་ལའ་ནིའ་ཞེས་པ་གསུམ་གྱི་འ་ཡིག་གིས་ལྷག་བཅས་དང་སྡུད་པ་དང་གཞན་འདྲེན་སོགས་ཀྱི་དོན་ཅི་ཞིག་བསྟན་ཡོད་ཚུལ་ཁྱོད་ཀྱིས་ལན་ཐོབ་ཅེས་དྲི་བ་བཏོན་ན་ཁྱོད་ལ་ལན་མེད་པར་འགྱུར་ཏེ། དེ་དག་གི་ངོས་ནས་འ་ཡིག་གིས་བརྡ་སྤྲོད་ཚུལ་ཞིག་ཁྱོད་ཀྱི་ཀུན་རྟོག་གི་ཡུལ་དུ་ནམ་ཡང་ཤར་རྒྱུ་མེད་པས་སོ། ཡང་། “ནམ་དུ་མྱྀ་གཏང་མའ་ཞིག་པར་བགྱྀའོ་”ཞེས་པའི་འ་ཡིག་ནན་གྱིས་བརྗོད་པའི་སྒྲ་དང་།“སྡེ་ཆ་གུདུ་མྱྀ་སྤོའ་མྱྀ་བསྒྱུར་བར་གནང་ངོ་”ཞེས་པའི་འ་ཡིག་སྡུད་པའི་དོན་ཡིན་པར་བཤད་ལ། འ་ཡིག་ཞིབ་འཇུག་པ་ཁ་ཅིག་གིས་འཆད་ཚུལ་དེར་ལེགས་པའི་ངོ་སོ་བྱེད་བཞིན་ཡོད་མོད། ཞིབ་ཏུ་བརྟགས་ན། གོང་གསལ་ཚིག་གྲུབ་དེ་གཉིས་ལ་བརྗོད་སྟངས་ཀྱི་ཆ་ནས་ཁྱད་པར་མེད་པའི་བརྗོད་ཚིག་གཉིས་ཏེ། “མྱྀ་གཏང་མའ་ཞིག་”ཅེས་པ་ནི་ནམ་ཡང་མི་གཏོང་བ་དང་མ་ཞིག་པར་འཇོག་དགོས་པའི་དོན་དང་། “གུདུ་མྱྀ་སྤོའ་མྱྀ་བསྒྱུར་”ཞེས་པ་ནི་གཞན་དུ་མི་སྤོ་བ་དང་གཞན་དུ་མི་བསྒྱུར་བར་སྔར་བཞིན་འཇོག་དགོས་པའི་དོན་ཡིན་པས། སྔ་ཕྱི་གཉིས་ཀ་ནན་ཏན་དུ་བརྗོད་པའི་ཚིག་ཡིན་པ་ཚིག་དེའི་བབ་ལས་ཤེས་པ་ལས་འ་ཡིག་གི་ནུས་པས་ཅི་ཞིག་ཤེས་སུ་བཅུག  གཞི་གཅིག་ལ་ངོས་འཛིན་དགོས་པའི་གནད་ཀྱང་བརྗོད་བྱའམ་བརྗོད་གཞི་གཅིག་དང་དོན་གཅིག་ལས་མེད་པ་དེའི་ནུས་པ་ཡིན་པ་ལས་འ་ཡིག་གི་ནུས་པས་དོན་དེ་འདྲ་ཤེས་སུ་བཅུག་པ་ག་ལ་ཡིན། སྐབས་དེའི་འ་ཡིག་ལ་བརྡ་སྤྲོད་ཀྱི་ནུས་པ་ཡོད་པར་ཁས་ལེན་རྒྱུ་ཞིག་ཡོད་ན་འ་ཡིག་དེས་སྐབས་དེར་ནུས་པ་ཇི་ལྟར་ཐོན་སྟངས་ཀྱི་ཆོས་ཉིད་དམ་བརྡ་སྤྲོད་ཚུལ་གང་ཡིན་པ་དེ་བཤད་དགོས་ཏེ། དེ་འདྲའི་ཆོས་ཉིད་དང་བརྡ་སྤྲོད་ཚུལ་བཤད་རྒྱུ་མེད་པར་ཚིག་གི་དོན་ཅི་ལ་བབ་ཡོད་པའི་དོན་དེ་དག་མཐའ་རྟེན་འ་ཡིག་གི་ནུས་པས་ཡིན་པར་བཤད་པ་ནི་བབ་འབྲེལ་མེད་པའི་གཏམ་ཁོ་ནའོ། བོད་ཀྱི་ཡིག་རྙིང་ཁག་ཏུ་འ་ཆུང་རྗེས་འཇུག་གི་ཚུལ་དུ་བྱུང་བ་དཔེར་ན། “ཁྲོམ་གྱི་བྲང་མ་ལགས་པའ་བྲང་གཞན་ནས་རྐུས་པའྀ་ཁྲིམས་” ( P.  T.1075 -25)ཞེས་པའི་དོན་ནི་ཁྲོམ་གྱི་བྲང་ངམ་སྐར་ནང་དུ་མ་ཡིན་པར་སྐར་གཞན་ནས་རྐུས་པའི་ཁྲིམས། ཞེས་པ་ཡིན་པས། ཁྱོད་ལྟར་ན་སྐབས་དེའི་“པའ་”ཡི་འ་དེར་ར་སྒྲ་ལྟར་གོ་བ་ལེན་དགོས་པར་འགྱུར་མོད། དུས་སྐབས་དེ་དང་དེང་སང་གི་ཁ་སྐད་ཀུན་ཏུ། རང་གི་སྒར་མ་ཡིན་པ་སྒར་གཞན་ནས།  ཞེས་“པ་”རྐྱང་ལྟར་བརྗོད་ཀྱང་དོན་ལ་འགྱུར་བ་མེད་པའི་ཚིག་དུམ་ཞིག་ཡིན། དེ་བཞིན། “ལྡེག་རེན་པའ：མནོན：སྣང：གྲགས：དང་། ཁེ་རྒད་མདོ་སྣང་ལས་སྩོགས་པ་ལོག་སྟེ་” (1. 0. 750 - 100) ཞེས་པའི་“པའ་”དེ་ཡང་བདག་པོ་ངོས་བཟུང་གི་སྒྲ་འདི་ལས་གོ་རྒྱུ་གཞན་མེད་པ་དང།“ཟླ་གོང་གི་བུ་ཚའ་རྒྱུད་འཕེལད་གྱྀ་ལག་ན་བྲན་ཞིང་འབྲོག་སོག་ཚལ་ལས་སྩོགས་ཏེ་དབང་ངོ་ཅོག་”(ཞོལ་རྡོ་རིང་གི་བྱང་ངོས་ཡིག་ཕྲེང50)ཞེས་པའི་“ཚའ་”དེ་ཡང་བུ་རབས་ཚ་རྒྱུད་ཀྱི་དོན་ཏེ། སྐབས་དེར་བུ་ཚའ་རྒྱུད་འཕེལད་ཞེས་པའི་ཚིག་གི་འབྲི་སྲོལ་འདི་ལས་འ་ཡིག་ཡོད་མེད་དང་འབྲེལ་བའི་དོན་སྟོན་ཚུལ་གཞན་ཅི་ཡང་མེད། ཁྱོད་ཀྱིས་ཡོད་ཟེར་ན། རྡོ་རིང་དེར། “ཟླ་གོང་གི་བུ་ཚ་རྒྱུད་འཕེལད་གྱི་རྣམས་”(ཞོལ་རྡོ་རིང་གི་བྱང་ངོས་ཡིག་ཕྲེང་39) ཞེས་དང་།  “བུ་ཚ་རྒྱུད་པེལད་ལས་”(ཞོལ་རྡོ་རིང་གི་བྱང་ངོས་ཡིག་ཕྲེང44)ཞེས་པའི་ཚ་ཡིག་དེ་གཉིས་ཀྱི་རྗེས་སུ་སྔ་མ་བཞིན་འ་ཡིག་མེད་པའི་རྒྱུ་མཚན་ཁྱོད་ཀྱིས་བཤད་དགོས་ཏེ། དེ་ལྟའི་ཚུལ་ལ་དཔྱད་པ་མེད་པར་རང་གི་འདོད་ཕྱོགས་ཁོ་ན་བཤད་དེ་མི་མཐུན་ཕྱོགས་སུ་དོ་སྣང་ཙམ་ཡང་མེད་པར་དཔྱད་རྩོམ་སྤེལ་བ་དེ་ནི་ངེས་པར་དུ་དེང་སང་གི་ལུགས་སུ་བྱས་ན་འོས་ཀྱང། གཞན་གྱིས་ཅི་ཟེར་ལ་མ་ཉན་པར་རྣ་བཀབ་ནས་སྡོད་ས་མེད་པས་གཟབ་གཟབ་བྱས་ན་ལེགས། འདིའི་ཐད་《ཐོན་མིའི་ཞལ་ལུང་》དུ། “བོད་ཀྱི་སྐད་ཕལ་ཆེ་བའི་གདངས་ནི་རིང་ཆ་ཅན་ཡིན་པའི་ཕྱིར་ན་མཐའ་རྟེན་གཞན་མེད་པའི་རྗེས་སུ་འ་མཐའ་སྦྱོར་བ་ཡང་བོད་སྐད་ཀྱི་གདངས་དང་མཐུན་པར་མཛད་པ་ཡིན་ཏེ་”ཞེས་དང།“འོན་ཀྱང་ད་ལྟའང་འགའ་རེའི་ལག་རྒྱུན་དུ། བརྡའ་སྤྲོད་པ་ཞེས་པའི་བརྡ་ལ་འ་མཐའ་འབྲི་བ་མ་དག་པར་བརྗོད་མི་རུང་”（30）ཞེས་མཐའ་རྟེན་གཞན་མེད་པའི་རྗེས་སུ་འ་མཐའ་སྦྱོར་བ་ནི་གདངས་རིང་ཆ་ཅན་མཚོན་བྱེད་དང་། བརྡའ་ལྟ་བུའི་མཐའ་རྟེན་འ་དེ་གོམས་སྲོལ་དུ་ངོས་འཛིན་དགོས་ཚུལ་གསུངས་པ་ཙམ་ལས་དེ་ལྟའི་འ་ལ་བརྡ་སྤྲོད་ཀྱི་ནུས་པ་ཡོད་ཚུལ་གསུངས་མེད་ཅིང་། བརྡ་སྤྲོད་སྨྲ་བ་སི་ཏུས། “ཡི་གེ་གཅིག་ཡིན་ཡང་བཤད་མ་ཐག་པ་ལྟར་རྗེས་འཇུག་དངོས་སུ་མེད་ཀྱང་དོན་གྱིས་ཐོབ་པས་དའ་ཞེས་ཡི་གེ་གཉིས་སུ་འགྱུར་བ་དེ་ལྟ་བུ་ནི་ཡི་གེ་གཉིས་ཚོགས་པའི་མིང་གི་མཚན་གཞི་དང་”（31）ཞེས་ཡི་གེ་རྐྱང་པའི་རྗེས་སུ་མཐའ་རྟེན་འ་བྱུང་བ་ནི་ཡི་གེ་དུ་མ་འདུས་པ་ལས་མིང་གྲུབ་པའི་མཚན་གཞིར་གསུངས་པ་དང་། མཁས་དབང་པྲ་སྟིས། “དེས་ན་རྗེས་འཇུག་གཞན་དགུ་པོ་གང་རུང་མེད་པའི་མཐར་འ་ཡིག་རེ་ངེས་པར་བཞག་པ་སྒྲ་བསྒྱུར་སྔ་མ་རྣམས་ཀྱིས་མཛད་ལ། དཔེར་ན། ཀ་ཞེས་བརྗོད་པའི་ཚེ་རྗེས་འབྲེལ་དུ་ཨ་ཡིག་མི་གསལ་པའི་ཚུལ་དུ་ཡོད་པའི་ཕྱིར་རོ་”（32）ཞེས་སྔོན་གྱི་ཡི་གེ་རྐྱང་པའི་རྗེས་སུ་འ་མཐའ་རྟེན་གྱི་ཚུལ་དུ་བྱུང་བ་དང་།  དེང་སང་ད་དུང་མི་མངོན་པའི་ཚུལ་གྱིས་ཡོད་པར་བཤད་པ་ནི།“རྗེས་མཇུག་བཅུ་པོ་མ་ཞུགས་ན། མིང་གཞན་སྦྱོར་བ་ཡོད་མི་སྲིད་”ཅེས་པའི་གཞུང་ཚིག་དང་། རྔོག་ལོ་ཆེན་པོས། “རྐྱང་པ་འཕུལ་ལ་འ་མཐའ་དགོས། ......དེ་ཕྱིར་འ་ནི་མཐའ་རྟེན་ཕྱེད་”ཅེས་གསུངས་པ་དེ་དག་གི་དགོངས་པར་བསྟུན་པའི་འགྲེལ་མཛད་ཚུལ་ཡིན་པ་ལས་དབྱངས་སྲོག་ཞུགས་ཀྱི་ཁྱད་ཆོས་ཡིན་ཚུལ་གཏན་ནས་གསུངས་མེད། ཁ་ཅིག་གིས།“ཁའ་ཞེས་པ་འདིར་འ་མཐའ་སྦྱར་བ་བསྡུས་པའི་ཁ་ཡིག་གིས་སྨྲ་བྱེད་ཁ་ཡི་དོན་སྟོན་ལ། འ་མཐའ་གཏན་ནས་མ་སྦྱར་བའི་ཁ་ཡིག་གིས་སྨྲ་བྱེད་ཁ་ཡི་ཁམས་ཙམ་ལས་དོན་དངོས་སྟོན་དུ་མེད་དེ་”（33）ཞེས་སོགས་སྨྲ་བར་བྱེད་མོད། ཁ་ཡིག་གི་མཐའ་རྟེན་དུ་འ་ཡིག་བྱུང་བ་སྔོན་གྱི་འབྲི་ཚུལ་ཙམ་ཡིན་པ་ནི་ཏུན་ཧོང་ནས་ཐོན་པའི་ཡི་གེར། “ཁའ་ཡང་མི་བདེ་”（34）ཞེས་བྱུང་བ་ལྟར་ཡིན་ལ། གལ་ཏེ་ཁ་ཡིག་ལ་མཐའ་རྟེན་འ་བྱུང་མིན་གྱིས་དོན་སྟོན་པ་ལ་ཁྱད་པར་ཡོད་ཟེར་ན། ཀའ་དང་ཀ་ལ་སོགས་པ་མཐའ་ཡས་པའི་ཡི་གེ་རྣམས་ལ་དེ་ལྟར་ཁྱད་པར་འབྱེད་དགོས་པར་འགྱུར་བ་སོགས་ཧ་ཅང་ཐལ་བར་འགྱུར་ལ། དེ་འདྲ་བ་སྔོན་དང་ཕྱིས་ཀྱི་གཞུང་དུ་མ་བྱུང་བས། དེ་ནི་དོན་ལ་སྦྱོར་དུ་མེད་པའི་འཆད་སྟངས་ཙམ་ཡིན་པ་ཤེས་པར་གྱིས།   
        བཞི་པ། དབྱངས་ཨ་ངེས་ཀྱི་སྙན་ངག་འབྲི་སྲོལ་དེ་རྒྱ་གར་རྩོམ་རིག་ལ་དཔེ་བལྟས་པའི་རྩོམ་ཚུལ་ཞིག་ཡིན།   
        བོད་ལ་དབྱངས་ཡིག་ལྔ་ཡོད་པར་འདོད་པ་ཁ་ཅིག་གིས་ཐོན་མིའི་ཡིག་ཕུད་དུ་གྲགས་པ། “ཞལ་རས་གསལ་ལ་ངང་མདངས་གང་བ་བཟང་། ་་་་”ཞེས་པ་དང། 《སྙན་ངག་དབྱངས་ཅན་དགྱེས་གླུ་》ལས། “དཔལ་ལྡན་བླ་མ་དམ་པ་གང་། །གདན་ས་འགྲན་ཟླ་བྲལ་བ་ན།  །ངག་དབང་འཇམ་དཔལ་དབྱངས་ཞལ་ནས།  ། དཀའ་གནད་གཞལ་བ་རབ་འབྱམས་གྲགས། །”（35）ཞེས་པ་སོགས་དབྱངས་ཨ་ངེས་ཀྱི་སྙན་ངག་རྩོམ་སྲོལ་བོད་ལ་བྱུང་བ་རྣམས་དཔེར་བྱས་ཏེ། རྩོམ་སྟངས་དེ་རིགས་ནི་བོད་ཀྱི་ཡི་གེར་ཨ་སྲོག་ཞུགས་ཡོད་པའི་དཔང་པོར་བྱེད་མོད། ཞིབ་ཏུ་བརྟགས་ན། རྩོམ་སྟངས་དེ་རིགས་ནི་ལེགས་སྦྱར་གྱི་དབྱངས་ཨ་ངེས་ཀྱི་འབྲི་ཚུལ་ལམ་རྩོམ་ཚུལ་ཇི་ལྟ་བ་བཞིན་བོད་ཀྱི་ཡི་གེའི་ལམ་ནས་བརྩམས་པའི་དཔེ་ཡིན་ཏེ། མཁས་པའི་དབང་པོ་དམུ་དགེ་བསམ་གཏན་གྱིས། འདིར་བོད་ཀྱི་དབྱངས་ངེས་ཀྱི་དཔེར་བརྗོད་འཇོག་ཚེ་དབྱངས་བཞི་བོ་གང་ཡང་མེད་པའི་ཡི་གེ་དང་མིང་ཚིག་ལ་འཕགས་ཡུལ་གྱི་ཨ་སྲོག་ཞུགས་ཀྱིས་ཚབ་བྱས་ཏེ། དབྱངས་ཨ་གཅིག་པུར་ངེས་པ་ཞེས་དཔེར་བརྗོད་སྔ་མ་དག་གིས་བཞག་པ་ལྟར་བ་དཀའ་བའི་སྐབས་དེ་ལྟར་བས་ཆོག་གོ་”（36）ཞེས་བོད་ཀྱི་གསལ་བྱེད་སུམ་ཅུ་ལ་དབྱངས་བཞི་མ་རྦུར་བའི་ཆི།ག་བཅད་རྩོམ་ཚུལ་དེ་ནི་ཐོག་མ་ཉིད་ནས་ལེགས་སྦྱར་གྱི་ཨ་སོགས་དབྱངས་ངེས་ཀྱི་རྩོམ་སྟངས་དཔེར་བྱས་ནས་བརྩམས་པ་ལས་བོད་རང་གི་ཡི་གེའི་ཁྱད་ཆོས་ལྟར་མིན་པ་གསལ་པོར་གསུངས་ཡོད། དེ་ལྟའི་དཔེ་དང་རང་གི་ཡི་གེའི་ཁྱད་ཆོས་གཉིས་ཀྱི་ཁྱད་པར་མ་ཕྱེད་པར་རྒྱ་གར་གྱི་དབྱངས་ཨ་ངེས་ཀྱི་རྩོམ་སྟངས་དེ་བོད་ལ་བྱུང་བར་གཞིགས་ནས་བོད་ཀྱི་ཡི་གེའི་སྲོག་ཏུ་དབྱངས་ཨ་ཞུགས་པར་བཤད་པ་ནི་ཆུ་ནང་གི་ཟླ་བ་དེ་ནམ་མཁའི་ཟླ་བ་དངོས་ལ་འཁྲུལ་བ་དང་ཁྱད་པར་མེད། འདྲ་བ་རྒྱུ་མཚན་དུ་བྱས་ནས་མིན་པ་ཡིན་པར་བཤད་པ་དེ་བསྡུས་གྲྭར་བྱང་བའི་ལུགས་སུ་མཛེས་ཀྱང་དངོས་དོན་དཔྱོད་པར་ཤིན་ཏུ་མི་འོས་པས། དཔེར་བལྟས་པ་ཡིན་མིན་དང་རང་རང་གི་ཡི་གེའི་ཁྱད་ཆོས་ལྟར་བྱས་པ་ཡིན་མིན་གྱི་ཁྱད་པར་འབྱེད་ཤེས་དགོས། ཁ་ཅིག་གིས་ཐོན་མིའི་ཡིག་ཕུད་དུ་གྲགས་པ་དེ་བཙན་པོའི་དུས་སྐབས་ཀྱི་སྙན་རྩོམ་ཡིན་པའི་སྒྲུབ་བྱེད་དུ། 《རྭ་ལོ་ཙཱ་བའི་རྣམ་ཐར་》དུ། “གྱེར་དུ་ཐོན་མི་སམྦྷོ་ཊའི་རྡོ་ལ་བརྐོས་པ་ཅིག་མཇལ་ཚེ་ཡི་གེ་ལས་བདུད་རྩི་བབས་”（37）ཞེས་བྱུང་བ་དེ་བཀོད་ཡོད་ཀྱང། ཞིབ་ཏུ་བསམས་ན། གྱེར་ཞེས་པའི་ས་ཆ་དེ་ད་ལྟའི་བཙུན་མོའི་ཚལ་དགོན་དུ་ངོས་འཛིན་ཆོག་མིན་དང་། ཐོན་མི་སམྦྷོ་ཊའི་རྡོ་ལ་བརྐོས་པའི་ཡི་གེ་ཞེས་པ་དང་། ད་ལྟའི་ཐོན་མིའི་ཡིག་ཕུད་དུ་གྲགས་པ་དེ་གཉིས་བསྟན་ཡུལ་སོ་སོ་བ་ཡིན་ནམ་མིན་གྱི་ཐེ་ཚོམ་དེ་དག་ཐག་གཅོད་མ་ཐུབ་པར་རྭ་ལོ་ནི་ད་ལྟའི་ཐོན་མིའི་ཡིག་ཕུད་དུ་གྲགས་པ་དེ་མཐོང་མཁན་ཡིན་པར་བརྗོད་མི་ནུས། གཞན་ཡང་རྣམ་ཐར་དེར། “ཤོང་རྡོ་རྗེ་རྒྱལ་མཚན་ལ་སོགས་སློབ་མ་སུམ་སྟོང་ཙམ་འདུས་”（38）ཞེས་འཁོད་པའི་ཤོང་རྡོ་རྗེ་རྒྱལ་མཚན་ནི་སྤྱི་ལོ་1240ལོའི་ཡས་མས་སུ་འཁྲུངས་པའི་ཤོང་སྟོན་རྡོ་རྗེ་རྒྱལ་མཚན་ཏེ་དེབ་ཐེར་གཞན་དུ།  “གཤོང་རྡོ་རྗེ་རྒྱལ་མཚན་དང་གཤོང་བློ་བརྟན་སོགས་”（39）ཞེས་དང་“གཤོང་སྟོན་ནི།  ......མཚན་རྡོ་རྗེ་རྒྱལ་མཚན་”（40）ཞེས་པའི་འབྲི་ཚུལ་བྱུང་ཡོད་པ་དེ་ཡིན། དེ་ལྟར་ན། རྭ་ལོ་ཙཱ་བ་ནི་སྤྱིལོ་1016ལོར་འཁྲུངས་པར་གྲགས་པ་ནས་ཤོང་སྟོན་འཁྲུངས་དུས་སུ་མི་ལོ་ཉིས་བརྒྱ་ལྷག་འགོར་ཞིང་དེའི་རྗེས་སུ་ད་དུང་འཚོ་བཞིན་པའི་ལོ་ཙཱ་བ་ཞིག་ཡིན་ཞིང་། སྤྲུལ་པའི་ལོ་ཙཱ་བ་མི་ལོ་དེ་ཙམ་བཞུགས་པར་བཤད་ཀྱང་བོད་པས་ཁས་ལེན་མི་བཏུབ་པ་ཞིག་མིན་མོད། དེ་ལྟར་ཁས་བླངས་ཚེ་རྭ་ལོའི་ཚ་བོ་རྭ་ཆོས་རབ་ནི་ཤོང་སྟོན་རྡོ་རྗེ་རྒྱལ་མཚན་གྱི་དུས་སྐབས་སམ་དེའི་རྗེས་སུ་འཁྲུངས་པར་ཁས་ལེན་དགོས་ཏེ། ཁོང་ནི་རྭ་ལོ་འདས་ནས་ཡུན་རིང་སོང་དུས་ཀྱི་རྣམ་ཐར་རྩོམ་པ་པོ་ཐོག་མ་དེ་ཡིན་པས་སོ། དེའི་སྲས་རྭ་ཡེ་ཤེས་སེང་གེ་ནི་རྭ་ལོའི་རྣམ་ཐར་གྱི་མཇུག་སྐྱོང་མཁན་ཡིན་ཞིང་། ཁོང་འཚོ་བཞུགས་ཀྱི་དུས་གསལ་པོ་མི་ཤེས་ཀྱང་ཧ་ལམ་དུས་རབས་14ནས་15པའི་ནང་ཙམ་ལ་ཡིན་པར་ངོས་འཛིན་བྱས་ཀྱང་འགལ་བ་མེད་པར་འགྱུར།  དུས་སྐབས་དེར་བོད་ལ་《སྙན་ངག་མེ་ལོང་》དར་ནས་ཡུན་རིང་སོང་ཞིང་། རྗེ་རིན་པོ་ཆེ་ལྟ་བུའི་བྱ་དཀའི་སྙན་ངག་རྩོམ་པ་ལ་ཕུལ་དུ་ཕྱིན་པའི་མཁས་པ་མང་པོ་བཞུགས་དུས་ཤིག་ཡིན། གང་ལྟར་《རྭ་ལོའི་རྣམ་ཐར་》ནི་དེ་འདྲའི་དོགས་གཞི་མང་པོ་ཡོད་པ་ཞིག་ཡིན་ཞིང་།  ལྷག་པར་དུ་“གྱེར་དུ་ཐོན་མི་སམྦྷོ་ཊའི་རྡོ་ལ་བརྐོས་པ་ཅིག་མཇལ་ཚེ་ཡི་གེ་ལས་བདུད་རྩི་བབས་”ཞེས་འཁོད་པ་དེ་ཐོན་མིའི་ཡིག་ཕུད་དུ་བསྟན་པ་ཡིན་མིན་མི་ཤེས་བཞིན་ཐོན་མིའི་ཡིག་ཕུད་དུ་གོ་དགོས་པར་བཤད་པ་ནི་ཨུ་ཚུགས་ཀྱི་ཚུལ་ལས་མ་འདས་ཤིང་། དེ་འདྲ་རྒྱུ་མཚན་དུ་བཀོད་ནས་ཡིག་ཕུད་སྙན་ངག་དེ་བཙན་པོའི་དུས་སྐབས་ཀྱི་ཡིན་པར་བཤད་པ་ནི་ལོང་བས་གླང་ཆེན་ལ་རེག་པའི་དཔེ་ལྟར་གདེང་ཚོད་མེད་པའི་ལུང་འདྲེན་དུ་གྱུར། ཡིག་ཕུད་སྙན་ངག་རང་ངོས་ནས་དབྱངས་ཨ་ངེས་ཀྱི་རྩོམ་ཉམས་ཡོངས་སུ་རྫོགས་ཤིང་བཙན་པོའི་དུས་ཀྱི་ཡིག་རྙིང་གི་རྣམ་པ་ཏོག་ཙམ་ཡང་མེད་པས། དེ་ནི་ཕྱིས་དུས་ཀྱི་རྩོམ་ཡིན་པའི་རྟགས་སུ་གསལ་ལ། གཞན་ཡང་སྔོན་གྱི་ཟེར་སྒྲོས་ཡིན་པར་བཤད་ནས་བརྟག་དཔྱད་མེད་པར་དེའི་རྗེས་སུ་འཇུག་མི་རུང་བ་ནི་རྟོན་བཞི་རིགས་སྨྲ་བ་རྣམས་ཀྱི་ཁྱད་ཆོས་ཏེ། བརྟགས་ན་དཔྱད་གཟོད་མེད་པའི་བཤད་རྒྱུན་གང་མང་ད་ལྟ་ད་དུང་དམངས་ཁྲོད་དུ་ལུས་ཡོད་པ་རྣམས་ཀྱིས་གནས་ལུགས་དེ་འདྲ་ཤེས་པར་འགྱུར། གལ་ཏེ་ཡིག་ཕུད་སྙན་ངག་དེ་བཙན་པོའི་དུས་སྐབས་ཀྱི་རྩོམ་ཡིན་པར་བས་ཀྱང་། དཔེར་བལྟས་པའི་རྩོམ་ཚུལ་ལས་མ་འདས་ཤིང་། དཔེར་བལྟས་པའི་རྩོམ་སྟངས་དེ་རིགས་བོད་ལ་ཇི་འདྲ་བྱུང་ཡང་། དེས་བོད་ཀྱི་གསལ་བྱེད་ཀྱི་སྲོག་ཏུ་དབྱངས་ཨ་འདི་ཞུགས་ཡོད་པའམ་བོད་ལ་དབྱངས་ལྔ་ཡོད་ཚུལ་བསྒྲུབ་མི་ཐུབ།    
        ལྔ་པ།  མཇུག་བསྡུ་བ།   
        དེ་ལྟར་བོད་ལ་དབྱངས་བཞི་གསལ་བྱེད་སུམ་ཅུར་ངེས་ཤིང་། བོད་ཀྱི་བརྡ་སྤྲོད་ཁག་ཏུ་དབྱངས་ཨ་འདི་ཡི་གེ་ཀུན་གྱི་འབྱུང་གནས་ཡིན་པ་སོགས་མང་དུ་གླེང་བ་རྣམས་ནི་བོད་ལ་དབྱངས་ཀྱི་ཡི་གེ་ལྔ་ཡོད་ཚུལ་བཤད་པ་མིན་པར་ཡི་གེ་རྣམས་ཀྱི་འབྱུང་གནས་སྐད་རིགས་ཀུན་གྱི་གདངས་ལ་དབྱངས་ཨས་མ་ཁྱབ་པ་མེད་པའི་སྒྲ་རིག་སྨྲ་བ་རྣམས་ཀྱི་བཞེད་ཚུལ་གྱི་དོན་སྤྱི་ལས་བརྩམས་ཏེ་བོད་ཀྱི་ཡི་གེ་སྟེ། མི་རིགས་འདིའི་སྐད་ཀྱི་གདངས་ལ་དེ་ལྟའི་ཁྱད་ཆོས་ཐུན་མོང་བ་ལྡན་ཚུལ་བཤད་པ་རེད། རྗེ་ས་པཎ་ནི་བོད་ལ་དབྱངས་ལྔ་དང་བཞི་གཉིས་ཀ་གསུང་མཁན་དུ་གྱུར་ཞིང་། དེ་བཞིན་བོད་ཀྱི་བརྡ་སྤྲོད་སྨྲ་བ་མི་ཉུང་བ་ཞིག་གིས་བོད་ལ་དབྱངས་ཡིག་དུ་ཡོད་ཐད་ལེགས་སྦྱར་གྱི་ཡི་གེའི་ཁྱད་ཆོས་སུ་སྦྱར་ནས་བཤད་ཆོག་པའི་རིགས་ལམ་བསྟན་ཡོད་ཀྱང་། འདིར་སུམ་ཅུ་པ་མཛད་པོས་ཀ་ལི་སུམ་ཅུ་དང་ཨི་སོགས་དབྱངས་བཞིའི་གྲངས་གསལ་བར་བསྟན་གནང་བ་དང་། ཞྭ་ལུ་ལོ་ཆེན་དང་མཁས་དབང་སི་ཏུ་དང་པྲ་སྟི་རིན་པོ་ཆེ། དཔལ་ཚེ་ཏན་ཞབས་དྲུང་སོགས་སྔོན་གྱི་མཁས་དབང་དེ་རྣམས་ཀྱིས་བོད་ལ་དབྱངས་བཞི་བཞེད་པའི་དགོངས་པའི་ཚུལ་ཇི་ལྟར་ཡིན་པ་ལས་བརྩམས་ཏེ། ཡི་གེའི་རྩོམ་གཞིར་གྱུར་བའི་སྐད་ཀྱི་གདངས་དང་བཟོས་པའི་ཡི་གེ་གཉིས་ཀྱི་རྟེན་གནས་ཀྱི་འབྲེལ་བ་སྟེ། སྔ་མ་ལ་མ་བརྟེན་པར་ཕྱི་མ་མེད་པ་དང་དབྱངས་ཡིག་ངོས་ནས་བཤད་ན་སྐད་རིགས་ཀུན་གྱི་གདངས་སུ་ཡོད་པའི་ཨ་དང་མི་རིགས་སོ་སོས་བཟོས་པའི་ཡི་གེར་བྱུང་བའི་ཨ་སོགས་ངོ་བོ་གཅིག་ཀྱང་སྤྱོད་སྟངས་དང་ནུས་པའི་ཐད་ཁྱད་པར་ཡོད་ཚུལ་དང་། སྤྱི་དང་བྱེ་བྲག་དང་ཐུན་མིན་གྱི་ཁྱད་ཆོས་ལྡན་པའི་ངོས་ནས་གོ་བ་ལེན་དགོས་པའི་ཚུལ་སོགས་རགས་ཙམ་གླེང་བ་ཡིན།
