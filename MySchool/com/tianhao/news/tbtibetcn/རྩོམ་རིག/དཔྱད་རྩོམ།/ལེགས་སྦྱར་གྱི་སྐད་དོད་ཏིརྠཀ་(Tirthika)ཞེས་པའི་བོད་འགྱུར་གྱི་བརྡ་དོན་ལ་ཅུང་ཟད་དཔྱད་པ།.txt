ལེགས་སྦྱར་གྱི་སྐད་དོད་ཏིརྠཀ་(Tirthika)ཞེས་པའི་བོད་འགྱུར་གྱི་བརྡ་དོན་ལ་ཅུང་ཟད་དཔྱད་པ།
གནད་བསྡུས། དཔྱད་རྩོམ་འདིའི་ནང་མིང་དང་། བརྡ། ཐ་སྙད་ཀྱི་ཁྱད་འབྲེལ་གླེང་བ་བརྒྱུད་ཐ་སྙད་ལ་ངོས་ཇི་ལྟར་བཟུང་ཚུལ་དང་ཐ་སྙད་ཞིབ་འཇུག་གི་དགོས་པ་འཆད་པ་དང་། ལུང་ཁུངས་ཁག་གཅིག་ལ་གཞི་བཅོལ་ཏེ་མུ་སྟེགས་ཞེས་པའི་བོད་འགྱུར་ཐ་སྙད་ཀྱི་གོ་དོན་བཀྲལ་ཚུལ་མི་འདྲ་བ་དག་ལ་དབྱེ་ཞིབ་ཀྱིས། དེ་ཉིད་ཐོག་མར་ཕྱི་རོལ་པའི་གྲུབ་མཐའ་བྱེ་བྲག་གི་མིང་སྤྱི་ལ་ཐོགས་པ་ཞིག་དང་། གདོད་མའི་གོ་དོན་ལ་དམའ་འབེབས་ཀྱི་རང་བཞིན་མེད་ཚུལ་བསྒྲུབས་པར་བྱས་འདུག །
          གནད་ཚིག ཐ་སྙད། མུ་སྟེགས། གོ་དོན། དམའ་འབེབས།
          [ཀྲུང་གོའི་དཔེ་དེབ་དབྱེ་རྟགས།] H711 [ཡིག་ཚང་མཚོན་བྱང་།] A [རྩོམ་གྱི་གོ་རིམ་ཨང་སྒྲིག] 1005-5746(2019)04-096-13
          གངས་ཅན་སཱ་ལའི་སྨན་ལྗོངས་ཀྱི་ཞིང་ཁམས་འདིར་ཕ་སྤྲེལ་རྒན་བྱང་ཆུབ་སེམས་དཔའ་དང་མ་བྲག་སྲིན་གཉིས་ལས་མཆེད་པར་གྲགས་པའི་སྐྱེ་བོ་རང་གཤིས་བཟང་ངན་དང་འཇམ་རྩུབ་ཅི་རིགས་པ་རྣམས་ཀྱིས་རང་གི་བློ་གྲོས་ཀྱི་རྩལ་སྤྲུགས་ཏེ་འཚོ་གནས་ཀྱི་ཞིང་ཁམས་འདི་དང་འབྲེལ་བའི་ཤེས་རིག་སྣ་ཚོགས་བཏོད་ཡོད། ལྷག་པར་དུ་བཙན་པོ་ལྷ་ཐོ་ཐོ་རིའི་སྐུ་རིང་ལ་བོད་ལ་ལེགས་སྦྱར་གྱི་བསྟན་བཅོས་ཐོག་མར་བྱུང་ཞིང་།བཙན་པོ་སྲོང་བཙན་སྒམ་པོའི་སྐབས་སུ་སློབ་དཔོན་ཐོན་མིས་བོད་སྤྱི་ལ་བཀོལ་བའི་ཡི་གེ་གསར་བཅོས་མཛད་རྗེས་ཁོང་དང་མཆན་བུ་དྷརྨ་ཀོ་ཥ་དང་ལྷ་ལུང་རྡོ་རྗེ་དཔལ་གྱིས་《སྤྱན་རས་གཟིགས་ཀྱི་མདོ་རྒྱུད་ཉེར་གཅིག》སོགས་ཀྱིས་ཐོག་དྲངས་ཏེ་ནང་ཆོས་བོད་སྐད་དུ་སྒྱུར་བའི་དབུ་བརྩམས་པ་དང་། བཙན་པོ་ཁྲི་སྲོང་ལྡེ་བཙན་གྱི་སྐབས་སུ་སྒྲ་སྒྱུར་རྒྱ་གར་གླིང་བཙུགས་པས་པ་གོར་བཻ་རོ་ཙ་ན་དང་སྐ་ཅོག་ཞང་གསུམ་སོགས་ལོ་ཙཱ་བ་རབ་དགུ་བྱོན་པ་དང་། བསྟན་པ་ཕྱི་དར་གྱི་སྐབས་སུ་ལོ་ཆེན་རིན་ཆེན་བཟང་པོ་དང་རྔོག་ལོ་ཡབ་སྲས་སོགས་བྱོན་པས། མདོ་རྒྱུད་བསྟན་བཅོས་དུ་མ་ལེགས་སྦྱར་སཾསྐྲྀ་ཏའི་（sanskrit）སྐད་ལས་བོད་སྐད་དུ་བསྒྱུར་བར་མཛད་ཡོད། སྤྱིར་སངས་རྒྱས་ཤཱཀྱ་ཐུབ་པ་སྐུ་འཚོ་བཞུགས་ཀྱི་སྐབས་སུ་ཡུལ་མ་གྷ་དྷ་རུ་འཚོ་གནས་བྱེད་མཁན་དམངས་ཀྱི་སྐད་ནི་ཡུལ་མི་རང་བཞིན་པའི་སྐད་དེ། པཱ་ལིའི་འམ་པྲཀྲྀ་ཏིའི་（prakrit）སྐད་ཡིན་ཞིང་། སངས་རྒྱས་ཀྱིས་སྐད་དེའི་ཐོག་ནས་དམ་ཆོས་གསུངས་མྱོང་བ་དང་། རྗེས་འཇུག་རྣམས་ཀྱིས་ཀྱང་སྐད་དེ་ལ་བརྟེན་ནས་བཀའ་བསྡུ་དང་གསུང་རབ་ཉར་ཚགས་གནང་མོད། སྐད་དེ་ཡོངས་ཁྱབ་དུ་བཀོལ་མ་ཐུབ་པར་གྱུར་ཡོད་ལ། རང་རེ་བོད་ཀྱི་སྐད་དུ་བསྒྱུར་བའི་བཀའ་བསྟན་ཕལ་ཆེ་བ་ནི་ཡུལ་མི་རང་བཞིན་པའི་པྲཀྲྀ་ཏིའི་སྐད་དང་རྒྱུད་ཁོངས་འདྲ་བའི་སྐད་ཡིག་སྟེ། ལེགས་སྦྱར་སཾསྐྲྀ་ཏའི་སྐད་ལས་བསྒྱུར་བའོ། །དེའི་ཕྱིར་བོད་དུ་ནང་བསྟན་དང་འབྲེལ་བའི་ལེགས་སྦྱར་གྱི་ཐ་སྐད་སྒྲ་སྒྱུར་དང་དོན་སྒྱུར་ཅི་རིགས་ཀྱིས་ཁྱབ་གདལ་དུ་སོང་ཡོད་པ་དང་། ཐ་སྙད་དུ་མ་ཞིག་དོན་སྒྱུར་ཡིན་པ་གང་ཞིག་བཀས་བཅད་ཀྱིས་གཏན་ལ་ཕབ་པ་ཤ་སྟག་ཡིན་པས། སྤྱིར་ན་ལོ་ངོ་ཆིག་སྟོང་སྔོན་གྱི་ཆོས་ཚིག་ཐ་དག་མ་ཡིག་གི་དོན་ལ་གནས་པ་གང་ཞིག་ད་ལྟའི་ཆར་ཇི་བཞིན་དུ་རྟོགས་ཐུབ་པའི་ཁྱད་ནུས་ཐོན་ཡོད། དེ་ལྟར་ན་ཡང་རང་རེ་བོད་དུ་རྒྱལ་བའི་བཀའ་བསྟན་གྱི་དགོངས་འགྲེལ་དང་ཡང་འགྲེལ་རྣམས་སུ་ཐ་སྙད་ཀྱི་དོན་རྣལ་མ་བསྟན་སླད་བཀྲལ་ལུགས་མ་གཅིག་པ་འགའ་བྱུང་ཡོད་པ་དག་ལ་ཞིབ་པར་བརྩད་རྒྱུ་ནི་ལོ་ཙཱའི་རིགས་ལམ་གྱི་ཞིབ་འཇུག་དེ་བས་འཐུས་ཚང་ཡོང་བ་སོགས་ཕྱོགས་མང་པོར་ཕན་དགེ་ངེས་ཅན་ཐོན་བཞིན་པ་ལགས་ན། འདིར་བཀའ་བསྟན་བོད་འགྱུར་རྣམས་སུ་ཅུང་ཟད་ཐོན་མང་བའི་ཐ་སྙད་སཾསྐྲྀ་ཏའི་སྐད་དོད་ཏིརྠི་ཀ་（Tirthika）[1]དོན་སྒྱུར་གྱིས་བརྡར་ཆགས་པ་མུ་སྟེགས་ཅེས་པའི་ཐོག་མའི་གོ་དོན་ཇི་ལྟ་བུ་ཞིག་ཡིན་ཚུལ་ཐད་དཔྱད་གླེང་ཅུང་ཟད་ཞུ་བར་འདོད།
          དང་པོ། མིང་དང་། བརྡ། ཐ་སྙད་ཀྱི་སྐོར་ལ་དབྱེ་ཞིབ་དང་འབྲེལ་ཏེ་ཐ་སྙད་ཞིབ་འཇུག་གི་དགོས་པ་བཤད་པ། 
          མིང་དང་། བརྡ། ཐ་སྙད་ཀྱི་སྐོར་ལ་《སྨྲ་སྒོ་མཚོན་ཆ》ལས། “ཇི་ལྟར་རྡུལ་ལས་གོང་བུ་བཞིན། ཡི་གེ་འདུས་པར་གྱུར་པ་ན་མིང་ནི་རབ་ཏུ་འགྲུབ་པར་འགྱུར་”[2]ཞེས་དང་། ཆོས་རྗེ་ས་པཎ་གྱི་《མཁས་པ་འཇུག་པའི་སྒོ》ལས། “ཡི་གེ་འདུས་པའི་བདག་ཉིད་ཅན། །དོན་གྱི་ངོ་བོ་བརྡ་སྤྲོད་པ། །དེའི་ཁྱད་པར་དག །སྟོན་པས་ཚིག་ཅེས་རབ་ཏུ་བཤད། །བརྗོད་འདོད་གསལ་བར་སྟོན་ཏེ་ངག །དེ་བསགས་པ་ལས་རབ་བྱེད་སོགས། །ཆོས་ཀྱི་ཕུང་པོའི་བར་དུ་འགྲུབ། །མིང་ཚིག་དེ་དག་རྒན་པོ་ཡིས། །ཐོག་མར་སྦྱར་བ་བརྡ་ཡིན་ལ། །ཕྱི་ནས་དཔྱོད་པ་ཐ་སྙད་ཡིན། །”[3]ཞེས་ཡི་གེ་འདུས་པ་ཙམ་ལ་མིང་ཞེས་མི་བྱ་སྟེ། མི་དང་། བ་གླང་། ཀ་བ། བུམ་པ་ལྟ་བུ་ཆོས་སོ་སོའི་ངོ་བོའམ་རང་བཞིན་གསལ་མཚོན་ནུས་པའམ་བརྗོད་འདོད་དོན་གྱི་ངོ་བོར་བརྡ་སྤྲོད་ནུས་པ་ལ་མིང་ཞེས་ཟེར་བ་དང་། མི་སྤྱང་པོ་དང་བ་གླང་གླེན་པ། ཀ་བ་རིང་པོ། བུམ་པ་ལྟོ་ལྡིར་ལྟ་བུ་ནི་དོན་གྱི་ཁྱད་པར་སྟོན་པའི་ཚིག་ཡིན་ཞིང་། སྔར་རྒན་པོ་དག་གིས་མིང་དང་ཚིག་ཏུ་བརྗོད་འདོད་ཁོ་ནའི་དབང་གིས་སྦྱར་བ་ནི་བརྡ་ཡིན་ལ། ཕྱིས་བརྡ་དོན་ལ་དཔྱོད་པ་ནི་ཐ་སྙད་ཡིན་པར་གསུངས་སོ། །
          སུམ་རྟགས་ཀྱི་གཞུང་དུ། “མིང་ཚིག་བརྗོད་པ་ཀུན་གྱི་གཞི། །ཡི་གེའི་སྦྱོར་བ་བཤད་པར་བྱ། །”ཞེས་དང་། “ཡི་གེའི་ཁོངས་ནས་མིང་དབྱུང་སྟེ། །མིང་གི་ཁོངས་ནས་ཚིག་ཕྱུང་ནས། །ཚིག་གིས་དོན་རྣམས་སྟོན་པར་བྱེད། །”ཅེས་སོགས་གསལ་ཡང་། མིང་དང་ཚིག་གི་གོ་དོན་དང་དབྱེ་བ་སོགས་ལ་ཞིབ་ཆ་མེད་པར་བལྟོས་ཏེ་སློབ་དཔོན་གྱིས་འབྲེལ་ཡོད་བསྟན་བཅོས་ཟུར་པ་ཞིག་མཛད་ཡོད་ངེས་པར་དེང་གི་མཁས་པས་བཞེད། དགེ་རྒན་ཆེན་མོ་ལྕགས་ཐར་རྒྱལ་གྱིས། “ད་ཡོད་སུམ་རྟགས་གཉིས་སུ་མིང་ཚིག་ཡི་གེའི་མཚན་ཉིད་ཙམ་ཡང་གསལ་བར་མ་བསྟན་པའི་ཁར། རྒྱ་གར་དུ་བརྡ་སྤྲོད་པ་དང་མངོན་པའི་ལུགས་ཀྱི་ཚིག་ལ་ངེས་ཚིག་ཐ་དད་པ་གཉིས་ཡོད་པས། ད་ལྟའི་བར་རང་རེ་རྣམས་ལ་ཚིག་གི་ངེས་ཚིག་ཀྱང་བམ་བཤད་ཙམ་དུ་ལུས་ཏེ།ཚིག་གི་རྣམ་གཞག་གསལ་ཞིང་རྒྱས་པར་བཤད་པ་གང་ཡང་མ་མཆིས་”[4]ཞེས་གསུངས་ཤིང་། ཏུན་ཧོང་གཏེར་ཡིག་ཏུ་ཡང་། “བརྡ་སྐད་མྱིང་དུ་གདགས་པའྀ་ཚིག་གྀས་བརྗོད་ཀྱྀས་མྱྀ་ལང་བ་”[5]ཞེས་དང་། “གུས་པར་རྨས་པའི་ཚིགས་སྙན་པ་དང་།”[6] ཞེས་སོགས་བསྒྱུར་ཡིག་ཏུ་ཚིག་ཅེས་པ་མང་དུ་བྱུང་བ་གང་ཞིག་ཚིགས་དང་ཚིག་གཉིས་སྐབས་རེ་འགར་བརྗེས་པའི་རྣམ་པ་ཡང་མཆིས་པ་ལས། ཚིག་ཅེས་པ་སློབ་དཔོན་ཐོན་མིའི་རྗེས་ནས་རྒྱ་ཁྱབ་ཏུ་བཀོལ་བ་ཞིག་དང་། དེའི་གདོད་མའི་དོན་ནི་སྐད་ཆ་མཚམས་བཅད་པའི་ཆ་ལ་ཚིག་ཅེས་བརྗོད་པར་སྣང་སྟེ། ཚིགས་ནི་སྨྱུག་ཚིགས་དང་རུས་ཚིགས་ལྟ་བུའི་མཚམས་ཚིགས་ཀྱི་དོན་ཡིན་པས། སྐད་ཆའི་མཚམས་ཚིགས་གསལ་བྱེད་ལ་ཚིག་ཅེས་བརྡ་བྱས་པར་སེམས། ཕྱིས་སུ་དོན་གྱི་ཁྱད་པར་སྟོན་པར་ཚིག་ཅེས་པ་ནི་སྒྲ་ཚད་ཀྱི་གཞུང་དུ་བྱུང་ལ། ལེགས་སྦྱར་བརྡ་སྤྲོད་ལ་གཏུགས་ནས་བོད་ཀྱི་བརྡ་སྤྲོད་དུ་གཞུང་བཤད་བྱས་པ་ཞིག་གོ །བརྡ་སྤྲོད་པའི་བསྟན་བཅོས་སོགས་ཐུན་མོང་དང་ཐུན་མིན་རིག་གནས་ཀྱི་བསྟན་བཅོས་དུ་མའི་ནང་དུ་མིང་དང་ཚིག་གཉིས་བསྡུས་ནས་མིང་ཚིག་ཅེས་མང་དུ་བྱུང་བ་དང་། དེ་བཞིན་མིང་གི་ཁོངས་སུ་གཏོགས་པའི་བྱ་ཚིག་དང་བྱེད་ཚིག་[7]སོགས་ཀྱང་བྱུང་ཁར། རྒྱ་ཡིག་གི་ཐ་སྙད་“名词”ཞེས་པའི་ཤུགས་རྐྱེན་ཐེབས་པ་སོགས་རྐྱེན་དུ་མའི་དབང་གིས། དེང་གི་བོད་སྐད་ཡིག་ཏུ་དཔེར་ན་ཀ་བ་ལྟ་བུར་མིང་ཚིག་ཅེས་ཕལ་མོ་ཆེས་སྤྱོད་པར་བྱེད་ལ། མིང་མཛོད་དང་ཚིག་མཛོད་ཀྱི་དབྱེ་བ་ཕལ་ཆེར་ལོགས་སུ་མི་བྱེད་པར། མིང་དང་ཚིག་གཉིས་ཀ་བསྟན་པ་ལ་ཚིག་མཛོད་ཅེས་སྤྱོད་པ་ནི་《བོད་ཀྱི་ཚིག་མཛོད་ཆེན་མོ》དང་《བོད་རྒྱ་ཚིག་མཛོད་ཆེན་མོ》ཞེས་པ་ལྟ་བུའོ། །
          《མཁས་འཇུག》ཏུ་བསྟན་པའི་བརྡ་དང་ཐ་སྙད་ནི་མིང་གི་ནང་གསེས་སུ་གྱུར་ལ། གཅིག་ནི་མིང་རང་ག་བ་དང་ཅིག་ཤོས་ནི་སྤྱི་ཁྱབ་དང་ཚད་ལྡན་གྱི་མིང་ཡིན་པའི་ཆ་ནས་དེ་གཉིས་ལ་ཞིབ་ཕྲའི་དབྱེ་བ་ཡང་བསྟན་མོད། དུས་ཕྱིས་བོད་རྩོམ་རྣམས་སུ་བརྡ་དང་ཐ་སྙད། མིང་བཅས་ལ་རེ་རེའི་དབྱེ་བ་གསལ་པོར་དགར་ནས་བེད་སྤྱོད་པ་ཕལ་ཆེར་མི་སྣང་། དཔེར་ན། དབུས་པ་བློ་གསལ་རྩོད་པའི་སེང་གེ་ཡིས་མཛད་པའི་《བརྡ་གསར་རྙིང་གི་རྣམ་པར་དབྱེ་བ》ཞེས་པའི་ལྗགས་རྩོམ་དུ་ཡང་ཞིབ་པར་བརྟགས་ན་བརྡ་དང་ཐ་སྙད་གཉིས་ཀའི་ཆ་ཡོད་ཚུལ་ནི་སྔོན་འགྲོའི་ཚིགས་བཅད་དུ། “འགྱུར་བྱེད་སྔ་མ་མཁས་པ་ཡི། ། སྐད་རྙིང་འགའ་ཞིག་གོ་དཀའ་བས། །གསར་བཅད་ཀྱིས་ནི་བཅོས་པའི་ཚུལ། །ཁྱེད་ལ་ཅུང་ཟད་གསལ་བར་བྱ། །”[8]ཞེས་གསལ་བའི་འགྱུར་བྱེད་སྔ་མ་ཞེས་པ་ལས་བརྡ་རྙིང་པ་དེ་དག་སྔ་འགྱུར་ལོ་ཙཱ་བའི་འགྱུར་ལས་བྱུང་བ་རྟོགས་ནུས་ལ། བརྡ་རྙིང་དེ་དག་ལ་བཀས་བཅད་པའི་ཐ་སྙད་ཀྱི་རང་བཞིན་ཆེར་མེད་པར་སྔར་གྱི་རྒན་པོའམ་མཁས་པ་འགས་བརྗོད་འདོད་ཁོ་ནའི་དབང་གིས་བརྡར་བྱས་པའི་རང་བཞིན་དང་ལྡན་པ་རྟོགས་ནུས། བོད་སྐད་ལ་སྔོན་གྱི་མཁས་པ་རྣམས་ཀྱིས་མཛད་པའི་བརྡ་གསར་རྙིང་གི་རྣམ་གཞག་ཤིན་ཏུ་མང་པོ་མཆིས་པ་དག་ལ་ཞིབ་པར་བརྩད་ན་བརྡ་དང་ཐ་སྙད་གཉིས་ཀའི་ཆ་འདུས་ན་ཡང་། བརྡ་གསར་རྙིང་ཞེས་པ་ལས་ཐ་སྙད་གསར་རྙིང་ཞེས་པ་མཚན་བྱང་དུ་བྱས་པ་མཇལ་རྒྱུ་དཀོན། དེང་གི་ཆར་བརྡ་བཏང་བ་དང་བརྡ་ལན་བྱས་པ་ཞེས་སོགས་བརྡ་ཞེས་པ་སྤྱིར་གོ་བརྡ་ཞེས་པའི་དོན་ལ་གཙོ་བོར་བཀོལ་བར་སྣང་ལ། ཡིག་རྩོམ་དུ་བཀོལ་མཁན་ཡང་མང་པོ་མི་སྣང་ཞིང་། གསར་དུ་བསྒྱུར་བའི་མིང་ལ་སྤྱིར་ཐ་སྙད་ཅེས་བཀོལ་བར་བྱེད་པ་ནི། “བོད་རང་སྐྱོང་ལྗོངས་བོད་ཡིག་ཐ་སྙད་བཀས་བཅད་ཨུ་ཡོན་ལྷན་ཁང་”ཞེས་པ་དང་། “བོད་ཡིག་ཐ་སྙད་བཀས་བཅད་སྤྱི་བསྒྲགས་”[9]ཞེས་པ་ལྟ་བུའོ། །
          བསྟན་བཅོས་ཀྱི་མཚན་དོན་སོགས་གླེང་བའི་སྐབས་སུ་《ལངྐར་གཤེགས་པའི་མདོ》ལས། “མིང་དུ་གདགས་པར་མ་མཛད་ན། །འཇིག་རྟེན་ཐམས་ཅད་རྨོངས་པར་འགྱུར། །དེ་བས་རློངས་པ་སེལ་བའི་ཕྱིར། །མགོན་པོས་མིང་དུ་གདགས་པར་མཛད། །”[10]ཅེས་གསུངས་པ་ལུང་དུ་དྲངས་ནས། ཆོས་རྣམས་ལ་རང་རང་གི་མིང་དང་ལྡན་པ་ཡིས་འཇིག་རྟེན་གྱི་རྨོངས་མུན་ཐོག་མ་སེལ་བར་བྱེད་པ་ཡིན་ཞིང་། མིང་དུ་མ་བྱས་ཚེ་བྱ་བ་དང་དངོས་པོ་ཐ་དག་སྐྱེ་རྒུའི་དབང་པོའི་འདུ་བྱེད་དང་འདུ་ཤེས་དག་གིས་ཤན་འབྱེད་མ་ཐུབ་ལ། ཕན་ཚུན་བརྡ་དང་ཐ་སྙད་ཀྱང་བྱེད་མ་ཐུབ་པས། འཇིག་རྟེན་ཧྲིལ་པོ་སྨག་གི་ཐུམ་བུ་ཞིག་ཏུ་གྱུར་ཏེ། ཤེས་རབ་དང་དཔལ་ཡོན་ཟེར་བ་ཡང་རྩ་བ་ནས་ཆད་འགྲོ་བར་བསྟན། དེ་བཞིན་དུ་《དབུ་མ་རྩ་བ་ཤེས་རབ》ལས། “ཐ་སྙད་ལ་ནི་མ་བརྟེན་པར། །དམ་པའི་དོན་ནི་རྟོགས་མི་འགྱུར། །དམ་པའི་དོན་ནི་མ་རྟོགས་པར། །མྱ་ངན་འདས་པ་ཐོབ་མི་འགྱུར། །”[11]ཞེས་བརྡ་དང་ཐ་སྙད་ལ་མ་བརྟེན་ན་འཇིག་རྟེན་དང་འཇིག་རྟེན་ལས་འདས་པའི་ཤེས་བྱའི་གནད་རྣམས་རྟོགས་པར་མི་འགྱུར་ལ། འདི་ཕྱིའི་ལེགས་ཚོགས་ཀྱི་རྒྱུ་གསོག་པའི་སྤང་བླང་གི་བགྱི་བ་ཇི་བཞིན་དུ་མ་ནུས་པར་ལམ་མ་ཡིན་པ་ལམ་དུ་བྱེད་པ་དང་། ཆོས་ལ་ཆོས་སུ་འདུ་ཤེས་མ་ནུས་པ། ལེགས་ཉེས་འབྱེད་པའི་བློ་གྲོས་ཀྱི་སྒོ་འགག་པ་སོགས་རྐྱེན་ངན་པ་དུ་མ་བབས་ནས་ཐབས་དང་ཤེས་རབ་ཀྱི་ནུས་པ་རྫོགས་མི་ཐུབ་ལ། རྣམ་པ་ཐམས་ཅད་མཁྱེན་པ་དང་ཞི་བ་མྱང་འདས་ཀྱི་ས་ལ་བགྲོད་མི་ནུས་པ་བསྟན་ཏོ། །
          རིག་གཞུང་སློབ་གཉེར་དང་ཞིབ་འཇུག་གང་གི་ཆ་ལའང་ཆེས་སྔ་མོ་ནས་དར་བའི་ཐ་སྙད་ལ་གོ་བ་ཡང་དག་ལེན་ཐབས་བྱེད་པ་ནི་ཚིག་དོན་གྱི་འཁྲུལ་བ་སྤོང་བའི་བྱ་བ་ཆེས་སྔ་མ་ཡིན་ཞིང་། རྩོད་གཞི་དང་འཁྲུལ་གཞི་ལྡན་པའི་ཐ་སྙད་ལ་ཞིབ་འཇུག་མ་བྱས་ཚེ་དེ་འབྲེལ་གྱི་རིག་གཞུང་སློབ་གཉེར་དང་ཞིབ་འཇུག་གི་ལག་ལེན་ཁྲོད་དུ་དོན་གྱི་འཁྲུལ་བ་ཐོན་ཏེ། ཞིབ་འཇུག་གི་མཐའ་མ་རྫོགས་པའམ་དོན་དངོས་དང་མི་འཚམས་པའི་བཤད་པར་བརྟེན་ནས་ཤེས་བྱའི་གནས་ལུགས་ཕྱིན་ཅི་ལོག་ཏུ་བཀྲལ་བའི་ཉེན་ཁ་ཡོད་ལ། འབྲེལ་ཡོད་ཤེས་བྱ་དོན་དུ་གཉེར་མཁན་གྱི་མི་རབས་རེ་རེ་བཞིན་ལམ་ལོག་པར་འཁྲིད་པའི་ཉེན་ཁ་ཡང་མཆིས།
          གཉིས་པ། མུ་སྟེགས་ཞེས་པའི་གོ་དོན་ཇི་ལྟར་བཀྲལ་ཚུལ་བཤད་པ།
          སཾསྐྲྀ་ཏའི་སྐད་དོད་ཏིརྠི་ཀ་ཞེས་པ་དོན་སྒྱུར་བྱས་པའི་ཐ་སྙད་མུ་སྟེགས་ཞེས་པས་མཚོན་པར་བྱ་བའི་་དོན་ཅི་ཡིན་ཐད་ལ་བོད་གཞུང་རྣམས་སུ་སྤྱིར་ན་“རྟག་ཆད་ཀྱི་ལྟ་བ་ངན་པའི་མུའམ་མཐའ་ལ་བརྟེན་ཞིང་གྲུབ་མཐའ་ལོག་པའི་ལམ་ལ་སྟེགས་འཆའ་བས་ན་མུ་སྟེགས་པ་”[12]ཞེས་པར་བཀྲལ་བར་བྱེད་དེ། “ལྟ་བ་ངན་པའི་མུར་སྟེགས་འཆའ་བས་ན་མུ་སྟེགས་ཞེས་ཀུན་མཁྱེན་འཇིགས་མེད་དབང་པོ་དང་། ཐུའུ་བཀྭན་བློ་བཟང་ཆོས་ཀྱི་ཉི་མས་གསུང་”[13]ལ། མཁས་དབང་དུང་དཀར་རིན་པོ་ཆེས་ཀྱང་། “གོ་དོན་ཆ་ཚང་ནི་ལྟ་བ་ངན་པའི་མུ་ལ་སྟེགས་སུ་བྱས་ནས་ལྟ་བ་ངན་པ་དེ་ཐར་པའི་ལམ་ལ་འཁྲུལ་བས་ན་མུ་སྟེགས་པ་ཞེས་བརྗོད་”[14]ཅེས་གསུངས་སོ། །འོན་ཏེ་ཀུན་མཁྱེན་འཇམ་དབྱངས་བཞད་པའི་《གྲུབ་མཐའ་ཆེན་མོར》“མུ་སྟེགས་ཀྱི་སྐད་དོད་ཏཱིར་ཐི་ཀ་ནི། སྒྲ་སྦྱོར་བམ་གཉིས་ལས། ཏཱིར་ཐི་ཀ་ནི་ཏཱིར་ཐི་ཡཱིདྱ་ཏེ་ཏཱིརྠ་ཀཿ་ཞེས་བྱ་སྟེ་མཚོའི་འགྲམ་ནས་མཚོར་འཇུག་པས་སྟེགས་འཆའ་བ་དང་། ཞེས་སོགས་དང་། ཤེར་སྒྲོན་འགྲེལ་བཤད་ལས། མུ་སྟེགས་བྱེད་ཅེས་བྱ་བ་ནི། གང་དག་ཆོས་ཀྱི་ཕྱིར་འཇུག་ངོགས་སུ་འཇུག་པར་བྱེད་པ་སྟེ། ནོར་ལྷ་དང་ལྷ་ཆེན་པོ་དང་ཚངས་པ་དང་སེར་སྐྱ་དང་གཟེགས་ཟན་དང་། རྐང་མིག་དང་། འཕེལ་བ་དང་། རྒྱལ་དཔོགས་སོགས་ཀྱི་བུ་ལ་སོགས་པའོ་ཞེས་གསུངས་པ་དང་། ཁྱད་པར་འཕགས་བསྟོད་འགྲེལ་པ་ལས། མུ་སྟེགས་ནི་ཐར་པ་དང་མཐོ་རིས་ཀྱི་ལམ་སྟེ། དོན་སྟོན་པར་བྱེད་པའི་བསྟན་བཅོས་ལ་མུ་སྟེགས་ཞེས་བྱ་ལ། དེ་བྱེད་པས་ན་མུ་སྟེགས་བྱེད་དེ་ཞེས་དང་། གཞི་འགྲེལ་ལས། གང་དག་ལ་ཆུ་ཀླུང་དུ་འཇུག་པའི་ཐར་ལམ་ཡོད་པ་འདི་དག་ནི་མུ་སྟེགས་ཅན་ནོ། །ཞེས་གསུངས་པ་ལྟར་འཇུག་ངོགས་སུ་ཁྲུས་དང་སྟེགས་བུ་བྱས་ཏེ། དེར་ཞུགས་པ་ཐར་པ་སྟེ་བདེ་འགྲོའི་རྒྱུར་སྨྲ་བར་བྱེད་པས་མུ་སྟེགས་བྱེད་ཅེས་བྱ་”[15]ཞེས་ལྟ་བ་ངན་པའི་མུའམ་མཐའ་ལ་ཞེས་སོགས་ཟུར་ཙམ་བཀྲལ་མེད་ལ། རོང་ཟོམ་པཎྜི་ཏས་ཀྱང་། “མུ་སྟེགས་ཞེས་བྱ་བ་ནི། །ཏྀརྠི་ཏེརྠེ་ན་ཏིརྠི་ཀ་ཞེས་བྱ་སྟེ། མྱ་ངན་ལས་འདས་པའི་རྒྱ་མཚོར་འབབ་པའི་ལམ་གྱི་ཆུ་རྒྱུན་གྱི་མུ་ལ་སྟེགས་བཅས་ཤིང་བརྟེན་པ་སྐད་དུ་འཆད་པས། མུ་སྟེགས་སུ་བཏགས་ཏེ། འདི་ཡང་དབང་ཕྱུག་རྒྱུར་ལྟ་བ་དང་། བྱེད་པ་པོ་རྒྱུར་ལྟ་བ་ལྟ་བུ་ཐརད་པ་ཚོལ་བའི་མུ་ལ་རྟེན་པར་འདོད་པ་རྣམས་ལ་འཇུག་གི །དུས་རྒྱུར་ལྟ་བ་དང་། ཆད་པར་ལྟ་བ་ལྟ་བུ་ཐར་པའི་ལམ་ལ་མི་རྟེན་པ་རྣམས་ལ་མི་འཇུག་མོད་ཀྱི། ཡན་ལག་གི་མིང་སྤྱི་ལ་འཇུག་པའི་མཚན་མར་བྱས་པ་སྟེ་”[16]ཞེས་སོགས་བཀྲལ་བར་མུ་སྟེགས་ཞེས་པའི་ཐ་སྙད་ཀྱི་ཐོག་མའི་གོ་དོན་ལ་བསམ་གཞིག་གི་རྒྱ་ཞིག་བཀྲམ་འདུག ། 
          《གྲུབ་མཐའ་ཆེན་མོར》《སྒྲ་སྦྱོར་བམ་གཉིས》ཀྱི་ཚིག་ལུང་དུ་དྲངས་པར་“མཚོའི་འགྲམ་ནས་མཚོར་འཇུག་པས་སྟེགས་འཆའ་བ་”ཞེས་དང་། ལུང་ཚིག་དག་གི་སྡོམ་དུ་“འཇུག་ངོགས་སུ་ཁྲུས་དང་སྟེགས་བུ་བྱས་ཏེ། དེར་ཞུགས་པ་”ཞེས་གསུངས་ཤིང་། རོང་ཟོམ་པཎྜི་ཏས་“ཆུ་རྒྱུན་གྱི་མུ་ལ་སྟེགས་བཅས་ཤིང་བརྟེན་པ་སྐད་དུ་འཆད་པ་”ཞེས་གསུངས་པར། ཐུན་མོང་དུ་ཆུར་འཇུག་པ་དང་འཇུག་ངོགས་སུ་འདུག་པ་ཞེས་པའི་དོན་ཡ་མ་བྲལ་བར་གསུངས་ཡོད། དེ་ནི་ཆོས་ལུགས་ཀྱི་བྱེད་སྒོ་སྤེལ་ཡུལ་ཆུ་བོ་གངྒཱཱ་ལྟེ་བར་འཛིན་པའི་གནའ་བོའི་རྒྱ་གར་གྱི་མུ་སྟེགས་གཅེར་བུ་བའི་ལུགས་སུ་“ལུས་ཀྱི་གསང་བ་མེ་ལྔ་ལ་བསྟེན་པ་དང་གཾ་གཱར་འཇུག་པ་”[17]སོགས་ཀྱིས་སྡིག་གྲིབ་དག་པར་འདོད་པ་དང་འབྲེལ་བ་ངེས་ཅན་ལྡན་པ་ཞིག་ཡིན་ཏེ། མཚོའམ་ཆུར་ཞུགས་ནས་ཁྲུས་བྱེད་པའམ་ཆུ་འགྲམ་ནས་ཆུར་འཇུག་པ་ཡང་དུས་རིང་པོར་མ་ཐོགས་པར་ཆུའི་འགྲམ་ངོགས་སྟེགས་སུ་བྱས་ནས་ངལ་གསོས་ཏེ་འདུག་དགོས་ཕྱིར། ཆུར་ཁྲུས་བྱེད་པའི་སྐབས་སྐབས་ལ་ཆུ་བོའི་མུའམ་མཐའ་ཡི་འཇུག་ངོགས་སམ་རབ་ཁར་ངལ་གསོས་ནས་འདུག་མཁན་གྱི་གྲུབ་མཐའ་འཛིན་པའི་རིགས་ལ་ཆེས་ཐོག་མར་མུ་སྟེགས་ཞེས་བཏགས་སམ་སྙམ། དེ་ལ་གྲུབ་བྱེད་གཞན་ཞིག་དང་འབྲེལ་ནས་བསམ་གཞིག་བྱ་རྒྱུ་མཆིས་ཏེ། གཤམ་གྱི་པར་རིས་ལས་གསལ་བ་བཞིན་ཏུན་ཧོང་P.T1257དུ་རྒྱ་ཡིག་གི་“外道”ཞེས་པའི་འགྱུར་ལ་མུར་འདུག་ཅེས་གསལ་བ་དང་། ངག་དབང་མཐུ་སྟོབས་ཚུལ་ཁྲིམས་བསྟན་འཛིན་གྱི་《བོད་ཀྱི་བརྡ་དག་ལ་བརྟེན་ནས་ཤེས་བྱ་ཀུན་ལ་འཇུག་པའི་སྒོ་རྣམ་པར་བཀྲལ་བ་ཀེ་ཏ་ཀའི་ཕྲེང་བ》ཞེས་བྱ་བར། “མུར་ཐུག་ནི། མཐར་ཐུག་གི་བརྡ། མུར་འདུག་ནི། མུ་སྟེགས་ཀྱི་བརྡ་རྙིང་”[18]ཞེས་གསལ་བ་ལས། མུ་སྟེགས་ཞེས་པའི་བརྡ་རྙིང་ལ་མུར་འདུག་ཅེས་ཟེར་བ་རྟོགས་ནུས་ལ། དམིགས་བསལ་མུར་འདུག་ཅེས་པའི་ཐ་སྙད་ཀྱི་ལ་དོན་གྱི་ཕྲད་“ར་”དང་བྱ་ཚིག་“འདུག་”གཉིས་ཀྱིས་མུའམ་མཐའ་ལ་སྡོད་པ་ཞེས་པ་ལྷག་པར་གསལ་འོང་སྟེ། དགེ་བཤེས་ཆོས་ཀྱི་གྲགས་པའི་《བརྡ་དག་མིང་ཚིག་གསལ་བ་བཞུགས་སོ》ཞེས་པར། “མུར། མཐའ་རུ།……མུར་འཇུག་པ། ཆུའི་མཐའ་རུ་འཇུག་པའམ་འཇུག་ངོགས་སུ་འཇུག་པ། མུར་ཐུག་པ། མཐའ་ལ་ཐུག་པ། མུར་འདུག་པ། མཐའ་རུ་འདུག་པའམ་སྡོད་པ། མུ་སྟེགས་པ་”[19]ཞེས་བཀྲལ་བ་ལས་ཀྱང་གསལ་ལོ། །
          ཡང་《གྲུབ་མཐའ་ཆེན་མོར》 “ཤེར་སྒྲོན་འགྲེལ་བཤད་ལས། མུ་སྟེགས་བྱེད་ཅེས་བྱ་བ་ནི། གང་དག་ཆོས་ཀྱི་ཕྱིར་འཇུག་ངོགས་སུ་འཇུག་པར་བྱེད་པ་”ཞེས་དང་། “གཞི་འགྲེལ་ལས། གང་དག་ལ་ཆུ་ཀླུང་དུ་འཇུག་པའི་ཐར་ལམ་ཡོད་པ་”ཞེས་པ། རོང་ཟོམ་པཎྜི་ཏས་“མྱ་ངན་ལས་འདས་པའི་རྒྱ་མཚོར་”ཞེས་པའི་ཚིག་དག་ལས། ཕྱི་རོལ་གྱི་གྲུབ་མཐའ་འཛིན་པའི་སྐྱེ་རྒུ་གང་དག་ཆུ་ཀླུང་དུ་འཇུག་པར་བྱ་བའི་དགོས་པ་གསལ་པོར་བསྟན་ཡོད། ཆོས་ཀྱི་ཕྱིར་ཆུ་ཀླུང་དུ་འཇུག་པ་དེ་ཡང་མཁས་དབང་དགེ་འདུན་ཆོས་འཕེལ་གྱི་《ལེགས་སྦྱར་བང་མཛོད》དུ། “ཁྲུས་ཀྱི་སྡིག་པ་དག་པ་ཡང་ཆུ་ཙམ་ལ་སྡིག་པ་འཁྲུ་པའི་ནུས་པ་ཡོད་ཅེས་མི་སྨྲ་ཡི། ཆུའི་རྣམ་པར་ཤར་བའི་ལྷ་མོའི་མཐུ་དང་། སྔོན་གྱི་ཐུབ་པ་རྣམས་ཀྱི་སྨོན་ལམ་གྱི་དབང་གི་སོགས། མ་ཎི་བུམ་ཆུའི་ཕན་ཡོན་གྱི་ཚིག་ཐམས་ཅད་ཁོང་ཅག་ལ་ཡང་རྒྱུགས་སྤྲོད་ངེས་པ་རེ་སྨྲ་རྒྱུ་འདུག་”[20]ཅེས་སོགས་གསུངས་ཡོད་པ་བཞིན། ཕྱི་རོལ་པའི་གྲུབ་མཐའ་འཛིན་པ་དེ་རིགས་ཀྱིས་བསྒོམ་བྱའི་ལྷ་སེམས་སུ་སྒོམ་ནས་ཆུ་ལ་འཁྲུས་བྱས་པ་ཡིས་སྡིག་པ་དག་པར་འདོད་པ་དང་། ཆུ་ལ་འཇུག་པར་བྱས་པ་ཡིས་མྱ་ངན་ལས་འདས་པའི་ཐར་པའི་རྒྱ་མཚོར་བགྲོད་ཐུབ་པར་འདོད་པ་ཡིན་ལ། འཁོར་བ་དང་མྱང་འདས་ཀྱི་དཔེར་ཆུ་དང་རྒྱ་མཚོ་བྱེད་པ་ཡང་མཆིས་སོ། །
          གོང་ཞུས་ལས་མུ་སྟེགས་ཞེས་པའི་ཐ་སྙད་ཀྱི་སྒྲ་བཤད་ལ་ཆོས་ཕྱིར་འཇུག་པའི་ཆུ་ཀླུང་གི་མུ་ལ་སྟེགས་འཆའ་བ་ཞེས་པ་ཞིག་མཆིས་པ་མཐོང་ནུས། ཡང་སྒྲ་བཤད་ཙམ་མ་ཡིན་པའི་གོ་དོན་བཀྲལ་ཚུལ་ཁ་གསལ་བ་ཞིག་ནི་སློབ་དཔོན་ཤེས་རབ་གོ་ཆས་མཛད་པའི་མཐོ་བཙུན་གྲུབ་རྗེའི་《ཁྱད་པར་འཕགས་བསྟོད་ཀྱི་འགྲེལ་པ》ལས། “མུ་སྟེགས་ནི་ཐར་པ་དང་མཐོ་རིས་ཀྱི་ལམ་སྟེ། དོན་སྟོན་པར་བྱེད་པའི་བསྟན་བཅོས་ལ་མུ་སྟེགས་ཞེས་བྱ་ལ། དེ་བྱེད་པས་ན་མུ་སྟེགས་བྱེད་དེ་”ཞེས་པ་དེ་ཡིན་ཞིང་། དབྱངས་ཅན་དགའ་བློའི་《ལམ་རིམ་བརྡ་བཀྲོལ》ལས། “མུ་ནི་མཐའ་ལ་འཇུག་པས་འཁོར་བའི་མཐའ་ཐར་པ་དང་། སྟེགས་ནི་ཐར་པ་དེར་འཛེག་པའི་རྐང་སྟེགས་ལྟ་བུ་ཐབས་སམ་ལམ་སྟེ། རང་ངོར་ཐར་པ་དང་མཐོ་རིས་འཐོབ་པའི་ཐབས་སུ་འདུ་ཤེས་པའི་ལམ་ལ་མུ་སྟེགས་ཞེས་བྱ་ལ། དེ་སྟོན་པའི་བསྟན་བཅོས་འཛིན་ཅིང་སྤེལ་བར་བྱེད་པས་ནས་མུ་སྟེགས་བྱེད་དམ་མུ་སྟེགས་ཅན་ཟེར་”[21]ཞེས་གསུངས་ལ། རྗེ་བཙུན་བློ་བཟང་ནོར་བུས་ཀྱང་། “མུ་ནི་མཐའ་ལ་འཇུག་པས་འཁོར་བའི་མཐའ་ཐར་པ་དང་། སྟེགས་ནི་ཐར་པ་དེར་འཛེག་པའི་རྐང་སྟེགས་ལྟ་བུ་ཐབས་སམ་ལམ་ལ་མུ་སྟེགས་ཞེས་བྱ་”[22]ཞེས་གསུངས་སོ། །གསུངས་སྟངས་དེ་རིགས་ནི་Tirthika ཞེས་པའི་ལེགས་སྦྱར་སྐད་ཀྱི་སྔ་མོའི་འགྲེལ་བཤད་ཅིག་ཡིན་ངེས་ལ། མུ་སྟེགས་གྲུབ་མཐའ་འཛིན་པ་དག་གིས་རང་གི་གྲུབ་མཐའི་མིང་བཀྲལ་ཚུལ་ཞིག་ཡིན་ཡང་ངེས།
          གསུམ་པ། མཐའ་དཔྱད།
          སྤྱིར་བོད་གཞུང་རྣམས་སུ་“ཕྱི་རོལ་མུ་སྟེགས་པ་”ཞེས་མུ་སྟེགས་པ་ནི་ཕྱི་རོལ་པའི་མིང་གི་རྣམ་གྲངས་སུ་བརྗོད་པར་བྱེད་པ་དང་། ཚིག་མཛོད་དུ་ཡང་དེ་ལྟར་བརྗོད་པར་བྱེད་དེ། དཔེར་ན་《བོད་རྒྱ་ཚིག་མཛོད་ཆེན་མོར》“མུ་སྟེགས། དཀོན་མཆོག་གསུམ་སྐྱབས་གནས་སུ་མི་འཛིན་ཞིང་ལྟ་བ་བཀར་བཏགས་ཀྱི་ཕྱག་རྒྱ་བཞི་ཁས་མི་ལེན་པའི་ཆོས་ལུགས་མཁན་”ཞེས་དང་“མུ་སྟེགས་ཅན། ཕྱི་རོལ་པ་”[23]ཞེས་པ་ལྟ་བུའོ། ། འོན་ཏེ་བོད་ལྗོངས་སློབ་གྲྭ་ཆེན་མོའི་བསླབ་གཞི་རྒན་མཁྱེན་རབ་དབང་ཕྱུག་གི་《གྲུབ་མཐའ་རིན་ཆེན་ཕྲེང་བའི་ཚིག་འགྲེལ་ཐོར་བུར》“རྟག་ཆད་ཀྱི་ལྟ་བ་ངན་པའི་མུའམ་མཐའ་ལ་བརྟེན་ཞིང་གྲུབ་མཐའ་ལོག་པའི་ལམ་ལ་སྟེགས་འཆའ་བས་ན་མུ་སྟེགས་པ་”ཞེས་གསུངས་པ་དང་། དུང་དཀར་རིན་པོ་ཆེས་“གོ་དོན་ཆ་ཚང་ནི་ལྟ་བ་ངན་པའི་མུ་ལ་སྟེགས་སུ་བྱས་ནས་ལྟ་བ་ངན་པ་དེ་ཐར་པའི་ལམ་ལ་འཁྲུལ་བས་ན་མུ་སྟེགས་པ་ཞེས་བརྗོད་”ཅེས་གསུངས་པ་བཞིན་སྔ་མོའི་བརྡ་དག་གི་གཞུང་དུ་ཡང་“མུ་སྟེགས་ལོག་པར་ལྟ་བ་”[24]ཞེས་པ་དང་། “ཕྱི་རོལ་པ་སྟེ་གྲུབ་མཐའ་ངན་པའི་མུ་ལ་སྟེགས་འཆའ་བའོ། །”[25]ཞེས་སོགས་བྱུང་ལ། 《བོན་གྱི་ཐ་སྙད་རིག་གནས་བསྟན་བཅོས་ཕྱོགས་སྒྲིག》ཏུའང་། “མུ་སྟེགས་（ཟབ་མོ་བོན་གྱི་ལྟ་སྤྱོད་སོགས་རྒྱུ་འབྲས་ལ་ལོག་ལྟའི་ཕྱི་བ་སྟེ་མུར་འདུག་ཀྱང་ཟེར་）”[26]ཞེས་བྱུང་། དེ་བཞིན། རྗེ་བཙུན་བློ་བཟང་ནོར་བུས། “ལས་ཉོན་གཉིས་ཐར་པར་བགྲོད་པའི་ཐབས་མིན་ཡང་སྤང་བྱ་དེ་དག་ཤེས་པར་བྱས་ནས་ཐར་ལམ་ལ་འཇུག་དགོས་པ་ལྟར། ཕྱི་རོལ་པའི་ལྟ་བ་རྣམས་རང་གི་ངོ་བོའི་སྒོ་ནས་ཐར་པའི་སྟེགས་མ་ཡིན་ཀྱང་། ལྟ་བའི་གོལ་ས་དེ་དག་ཤེས་པར་བྱས་ནས་སྤངས་ཏེ་ཐར་པའི་ལམ་ཡང་དག་པའི་ལྟ་བ་འཚོལ་དགོས་པའི་དོན་ཡིན་པའི་ཕྱིར། དེས་ན་སྒྲ་བཤད་པའི་ཚེ་དེ་ལྟར་བཤད་ཀྱང་དོན་ངོ་བོ་ལ་དེ་ལྟར་མ་ཡིན་པ་དུ་མ་བཤད་དོ། །ཡང་ན། ལྟ་བ་ངན་པའི་མུ་ལ་སྟེགས་འཆའ་བས་ན་མུ་སྟེགས་པ་ཞེས་བྱའོ། །”[27]ཞེས་འཁོར་བའི་མུ་ལས་གྲོལ་བའི་ལམ་སྟེགས་སུ་བཤད་པའི་གོ་དོན་ཁེགས་པར་བྱས་ཏེ། དོན་གྱི་བབས་ས་ལྟ་བ་ངན་པའི་མུ་ལ་སྟེགས་འཆའ་བ་ཡིན་པས་མུ་སྟེགས་ཞེས་བཏགས་པར་གསུངས་འདུག །
          དུང་དཀར་རིན་པོ་ཆེས་“མུ་སྟེགས་ཞེས་པའི་གོ་དོན་ནི་མུ་ཞེས་པ་མཐའ་དང་། སྟེགས་ནི་སྡོད་སའི་སྟེགས་བུའི་དོན་ཡིན་པས། ལྟ་བ་ངན་པའི་མུ་ལ་སྡོད་པའི་སྟེགས་བྱེད་པ་ཞེས་ནང་པ་སངས་རྒྱས་པ་རྣམས་ཀྱིས་ཟུར་ཟ་བྱས་པའི་མིང་ཞིག་ཡིན་”[28]ཞེས་དང་། “མུ་སྟེགས་པ་ཟེར་བ་ནི་སངས་རྒྱས་ཆོས་ལུགས་པས་དམའ་འབེབས་ཀྱི་ཐོག་ནས་བཏགས་པའི་མིང་ཞིག་ཡིན་”[29]ཞེས་ཆོད་སྡོམ་ལྟ་བུའི་བཀའ་མོལ་གསུངས་གནང་བའི་གནད་འགག་གཙོ་བོ་ནི་ཟུར་ཟ་དང་དམའ་འབེབས་ཞེས་པའི་ཐ་སྙད་གཉིས་པོ་ཡིན་ཞིང་། ཐ་སྙད་དེ་གཉིས་ལས་ནང་པ་དག་གིས་མུ་སྟེགས་ཀྱི་རྣལ་མའི་གོ་དོན་དེ་དེ་བཞིན་དུ་མ་བཞག་པར་གྲུབ་མཐའ་ཕྱི་རོལ་པར་དམའ་འཇོག་གི་ཐོག་ནས་བཀྲལ་ཚུལ་གྱི་དངོས་དོན་བསྟན་ཡོད་དེ། ནང་པ་སངས་རྒྱས་པ་རྣམས་ཀྱིས་ཞེས་པ་དང་སངས་རྒྱས་ཆོས་ལུགས་པས་ཞེས་པའི་ཚིག་དེ་གཉིས་ཀྱིས་ཀྱང་སྒྲུབ་བྱེད་ཕར་མ་བྱེད་ནུས། སྤྱིར་ན་གཞན་སྡེ་ཞེས་པ་དང་ཕྱི་པ་ཞེས་པ་ནི་ནང་པའི་ཆོས་ལུགས་འཛིན་མཁན་གྱིས་ཕར་བཏགས་པ་ཞིག་ཡིན་པ་སྟེ།དཔེར་ན་ངག་རྒྱུན་དུ་ནང་མི་དང་ཕྱི་མི་ཞེས་སུ་འབོད་ཚུལ་ཡོད་པ་བཞིན་རང་སྡེ་དང་ནང་པ་ཞེས་པ་ལས་ལྡོག་སྟེ་བཏགས་པ་ཞིག་དང་། ནང་པ་དང་རང་སྡེ་ཞེས་པའི་ཐ་སྙད་གཉིས་པོ་ཡང་ཧ་ལམ་དོན་གཅིག་ལ་གནས་པ་ཞིག་ཡིན་ཞིང་། གཞན་སྡེ་དང་ཕྱི་པ་ཞེས་འབོད་པ་ལ་དམའ་འཇོག་གི་དོན་རྩ་བ་ནས་མེད། ཁ་ཅིག་གིས་ནང་གི་སྤང་བྱ་མ་རིག་པ་སྤངས་ཤིང་གཉེན་པོ་བདག་མེད་རྟོགས་པའི་ཐབས་སྟོན་པའི་ཤེས་རབ་བརྗོད་པའི་བསྟན་བཅོས་ལ་ནང་རིག་གི་བསྟན་བཅོས་ཞེས་ནང་པའི་བསྟན་པ་ནི་ནང་སེམས་ཀྱི་སྤང་གཉེན་བསྟན་པས་ནང་པར་བཏགས་སོ་སྙམ་དུ་བསམས་ནས་སྨྲ་བར་བྱེད་ཀྱང་། དོན་དངོས་སུ་ཕྱི་རིག་པ་ཞེས་པའི་བརྗོད་སྲོལ་མེད་པས། རིག་གནས་ཆེ་བ་ལྔའི་ཡ་གྱལ་ནང་རིག་པ་ཞེས་པའི་བརྡ་དོན་བཀྲལ་དུས་ནང་སེམས་ཁམས་ཀྱི་སྤང་གཉེན་སྟོན་པའི་ཆ་ནས་སྨྲས་ན་འོས་པ་ལས། ཕྱི་པ་དང་ནང་པ་ཞེས་པའི་ཐ་སྙད་བཀྲལ་དུས་ནང་སེམས་ཁམས་ཀྱི་སྤང་བྱ་དང་གཉེན་པོ་སྟོན་པའི་ཆ་ནས་སྨྲས་ན་མི་འོས་པར་སེམས། འོན་ཏེ། མུ་སྟེགས་ཞེས་པ་ནི་རང་རེའི་སྔོན་བྱོན་མཁས་པ་ཁག་གཅིག་གིས་ཧ་ལམ་མགྲིན་གཅིག་ཏུ་ལོག་པའི་ལྟ་བའི་མུ་ལ་སྟེགས་འཆའ་བ་ཞེས་འགྲེལ་རྐྱང་བྱེད་པས། དེ་ལྟའི་འགྲེལ་ཚུལ་ནི་དམའ་འཇོག་གི་དོན་དང་བཅས་པ་ཞིག་ཡིན་པ་ལྟ་ཅི་སྨོས། དོན་དུ་གནའ་བོའི་རྒྱ་གར་དུ་ཕྱི་རོལ་པའི་གྲུབ་མཐའ་ཤིན་ཏུ་མང་པོ་ནང་པའི་གྲུབ་མཐའ་དང་ལྷན་ཅིག་ཏུ་དར་བའི་དུས་དེར། གྲུབ་མཐའ་གང་རུང་ཞིག་ལྟ་ངན་འཛིན་པ་ཡིན་ཚུལ་གསལ་བའི་མིང་བྱས་པ་ཡོང་མི་སྲིད་དེ། གྲུབ་མཐའ་བྱེ་བྲག་པ་ཕལ་མོ་ཆེ་ནི་གྲུབ་མཐའ་དེའི་སྟོན་པའི་མཚན་དང་ལྟ་བའི་ཁྱད་ཆོས་གཞིར་བྱས་ཏེ་མིང་ཐོགས་པ་ཡིན་ཚུལ་བོད་ཀྱི་གྲུབ་མཐའི་བསྟན་བཅོས་རྣམས་སུ་གསལ་བ་སོགས་ཕྱོགས་མང་པོ་ལས། གནའ་བོའི་རྒྱ་གར་དུ་གྲུབ་མཐའ་ཤིན་ཏུ་མང་པོ་མཆིས་པ་དེ་དག་ནི་རང་ལུགས་གཅེས་སུ་འཛིན་པའམ་རང་ལྟ་མཆོག་འཛིན་དང་ཕན་ཚུན་འགྲན་པའི་ཚུལ་དུ་གནས་པ་ཞིག་ཡིན་པས། གྲུབ་མཐའ་ཅི་དང་གང་གིས་ཀྱང་རང་གིས་རང་ལ་དམའ་འབེབས་ཀྱི་བརྡ་བྱེད་པ་ནི་ནམ་ཞར་ལའང་མི་སྲིད་པ་ཞིག་ལགས་ཕྱིར། དུང་དཀར་རིན་པོ་ཆེས་ནང་པ་སངས་རྒྱས་པས་ཟུར་ཟ་དང་དམའ་འབེབས་སུ་བཏགས་པར་གསུངས་པ་ཡང་གཏམ་གོ་རྒྱུ་ཅན་ཞིག་ཏུ་མད་སྙམ།
          དོན་དམ་མུ་སྟེགས་ཞེས་པའི་གདོད་མའི་འགྱུར་ནི་མུར་འདུག་ཡིན་ཏེ། ལྕང་སྐྱ་རིན་པོ་ཆེས་“མུར་འདུག་ནི་མུ་སྟེགས་” [30]ཞེས་གསུངས་པ་བཞིན་གོང་དུ་ལུང་གི་སྒྲུབ་བྱེད་དུ་མ་སོང་ཟིན་པ་ལས་གསལ་ལ། དེ་ལྟར་ཡིན་པ་གང་ཞིག་དེ་ནི་ལེགས་སྦྱར་གྱི་སྐད་དོད་དང་ཐག་ཉེ་བ་ཡང་ཡིན་ཏེ། བཙན་པོ་ཁྲི་ལྡེ་སྲོང་བཙན་གྱི་སྐབས་སུ་བོད་དང་རྒྱ་གར་གྱི་ལོ་པཎ་རྣམས་ཀྱིས་སྔ་འགྱུར་ལ་དག་བཅོས་དང་ཕྱི་འགྱུར་རྣམས་བོད་སྐད་དང་མཐུན་པ་གཞིར་བཞག་དང་སྒྲ་གཞུང་དང་ཡང་མི་འགལ་བར་བསྒྱུར་སླད་གཏན་ལ་ཕབ་པའི་བོད་ཀྱི་སྒྲ་སྒྱུར་རིག་པའི་བསྟན་བཅོས་《སྒྲ་སྦྱོར་བམ་གཉིས》ལས། “ཏཱིར་ཐི་ཀ་ནི་ཏཱིར་ཐི་ཡཱིདྱ་ཏེ་ཏཱིརྠ་ཀཿ་ཞེས་བྱ་སྟེ་མཚོའི་འགྲམ་ནས་མཚོར་འཇུག་པས་སྟེགས་འཆའ་བ་”ཞེས་གསུངས་པ་ལས་རྟོགས་ནུས། འོན་ཏང་དེ་ལ་ལྕང་སྐྱ་རིན་པོ་ཆེའི་《གྲུབ་མཐའ་ཐུབ་བསྟན་ལྷུན་པོའི་མཛེས་རྒྱན》དུ། “ཏིརྠ་ཀའི་སྒྲ་འཇུག་ངོགས་སུ་འཇུག་པར་སྒྲ་སྦྱོར་བམ་གཉིས་ལས་བཤད་མོད་ཀྱང་དེ་ཙམ་གྱི་སྒྲ་བཤད་པ་ལ་གོ་དོན་ཅི་ཡང་མེད་དོ། །”[31]ཞེས་གསུངས་པ་ནི་ཅུང་ཟད་ཕྱོགས་སུ་ལྷུང་བའི་གསུང་ཡིན་ནམ་སྙམ་སྟེ། 《སྒྲ་སྦྱོར་བམ་གཉིས》ནི་ཨཱ་ཙཱརྱ་ཛི་ན་མི་ཏྲ་དང་། སུ་རེནྡྲ་བོ་ངྷི། ཤཱི་ལེནྡྲ་བོ་ངྷི། དཱ་ན་ཤཱི་ལ། བོ་ངྷི་མི་ཏྲ། ཨཱ་ཙཱརྱ་བོ་དྷི་ས་ཏྭ་སོགས་རྒྱ་བལ་གྱི་མཁས་པས་གཏན་ལ་ཕབ་པ་ཞིག་ཡིན་པས་ཁུངས་བཙུན་པ་དང་། བཟང་སྤྱོད་ཀྱི་འགྲེལ་པ་ལོ་ཆེན་ཡེ་ཤེས་སྡེས་མཛད་པ་ལས་ཀྱང་“མུ་སྟེགས་ནི། དཔེར་ན་ཆུའི་རྫིང་བུའམ་མཚེའུར་འཇུག་པའི་མུར་ཐེམ་པ་བྱས་པ་ལྟ་བུའམ་རབ་བརྒལ་བའི་རབ་ཁ་བདེ་བ་ལ་མུ་སྟེགས་ཞེས་བྱའོ། ། དེ་བཞིན་མྱ་ངན་འདས་པའི་མཚོར་འཇུག་པའི་མུ་སྟེགས་དང་། འཁོར་བའི་ཆུ་བོ་ལས་བརྒལ་བའི་ཐབས་ལམ་གྱི་རབ་ཁ་ལ་མུ་སྟེགས་ཞེས་བྱ་སྟེ། གང་སུ་དག་ལ་ལམ་དེ་ཡོད་ཅིང་དེ་སྤྱོད་ན་དེ་དག་ནི་མུ་སྟེགས་ཅན་ནོ། །”[32]ཞེས་གསུངས་ཤིང་། སྐ་ཅོག་ཞང་གསུམ་གྱི་ཡ་གྱལ་ལོ་ཆེན་ཞང་སྣ་ནམ་ཡེ་ཤེས་སྡེ་ནི་སྤྱི་ལོའི་དུས་རབས་བརྒྱད་པའི་ནང་བྱོན་པའི་སྒྲ་སྒྱུར་ལ་ཤིན་ཏུ་མཁས་པ་ཞིག་ཡིན་པས་ཆེས་ཁུངས་བཙུན་པའི་གཏམ་མོ་སྙམ། ཡང་དེ་ལ་ཁ་ཅིག་གིས་ལོ་ཆེན་ཡེ་ཤེས་སྡེས་བཙན་པོའི་སྐབས་སུ་བསྒྱུར་བ་ལ་མུ་སྟེགས་ཞེས་པའི་ཐ་སྙད་ལས་མུར་འདུག་ཅེས་པ་མ་མཆིས་སོ་སྙམ་ངེས་མོད། དེ་ཡང་མ་ངེས་ཏེ། བཀའ་འགྱུར་དུ་བཞུགས་པའི་《འཕགས་པ་བཟང་པོ་སྤྱོད་པའི་སྨོན་ལམ་གྱི་རྒྱལ་པོར》ཡང་“རྒྱ་གར་གྱི་མཁན་པོ་པྲཛྙཱ་བར་མ་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙྪ་བ་བན་དེ་ཡེ་ཤེས་སྡེ་ལ་སོགས་པས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་”[33]ཅེས་གསལ་ན་ཡང་གཞུང་ཚིག་ལ་ཕྱིས་འགྱུར་བཅོས་ཞུས་པ་ཚིག་རིས་ལས་རྟོགས་ནུས་པས། ལོ་ཆེན་ཡེ་ཤེས་སྡེས་ཐོག་མར་སྒྱུར་དུས་སུ་ཡི་གེར་བཏབ་པའི་བྲིས་མ་མ་མཐོང་ཕྱིན་ཐོག་མའི་འགྱུར་གྱི་ཐ་སྙད་མུ་སྟེགས་དེ་རང་ཡིན་ངེས་པར་ཁ་ཚོན་གཅོད་མི་ནུས། དེས་ན། མུར་འདུག་གམ་མུ་སྟེགས་ཞེས་བྱའི་ཐ་སྙད་དེ་རོང་ཟོམ་ཆོས་བཟང་གིས་“ཡན་ལག་གི་མིང་སྤྱི་ལ་འཇུག་པའི་མཚན་མར་བྱས་པ་”ཞེས་གསུངས་པ་བཞིན། བྱེ་བྲག་“གང་དག་ཆོས་ཀྱི་ཕྱིར་འཇུག་ངོགས་སུ་འཇུག་པར་བྱེད་”པ་སྟེ་ཚིག་ངོས་ཇི་བཞིན་ལུས་ལ་ཁྲུས་བྱེད་པ་ཐར་པའི་གྲོགས་སུ་འཛིན་པའམ། ཡང་ན་“མྱ་ངན་འདས་པའི་མཚོར་འཇུག་པའི་མུ་སྟེགས་དང་། འཁོར་བའི་ཆུ་བོ་ལས་བརྒལ་བའི་ཐབས་ལམ་གྱི་རབ་ཁ་ལ་མུ་སྟེགས་ཞེས་བྱ་སྟེ། གང་སུ་དག་ལ་ལམ་དེ་ཡོད་ཅིང་དེ་སྤྱོད་ན་དེ་དག་ནི་མུ་སྟེགས་ཅན་”ཞེས་པ་ལྟར་དཔེ་ཅན་མྱང་འདས་དང་འཁོར་བ་ལ་དཔེ་མཚོ་དང་ཆུར་བྱེད་ཅིང་། ངོ་བོ་ཐར་འདོད་ལ་གཞོལ་བའི་གྲུབ་མཐའ་འཛིན་པ་བྱེ་བྲག་གི་མིང་ཕྱི་རོལ་པ་སྤྱི་ལ་ཐོགས་པ་ཞིག་དང་།  དོན་ལ་“ཐར་པ་དང་མཐོ་རིས་ཀྱི་ལམ་སྟེ། དོན་སྟོན་པར་བྱེད་པའི་བསྟན་བཅོས་ལ་མུ་སྟེགས་”ཞེས་སོགས་བཀྲལ་བར་སྣང་སྟེ། 《སྒོམ་སྡེ་ཚིག་མཛོད་ཆེན་མོར》ཡང་། “མུ་སྟེགས། ཕྱི་རོལ་པ་རང་ལུགས་ཀྱི་མཐོ་རིས་དང་ཐར་པའི་ལམ་དང་། དེ་སྟོན་པའི་བསྟན་བཅོས་ཀྱི་མིང་ཡིན།”[34]ཞེས་གསལ་བ་སོགས་ཕྱོགས་དུ་མ་ལས། ཐ་སྙད་དེའི་གདོད་མའི་དོན་ལ་དམའ་འབེབས་ཀྱི་ཚིག་དང་བྲལ་ཅིང་། འགྱུར་ཡང་མུར་འདུག་ཅེས་པ་ལེགས་སྦྱར་སྐད་ཀྱི་མ་ཡིག་དང་ཉེ་བར་མཐོང་ལ། ཕྱིས་ནང་པའི་མཁས་པ་ཁ་ཅིག་གིས་དོན་གྱི་ཆ་ནས་ཕྱི་རོལ་པ་ནི་ལྟ་བ་ངན་པའི་ལམ་ལ་འཕྱན་པར་དགོངས་ཏེ། རྟག་ཆད་ཀྱི་ལྟ་བ་ངན་པའི་མུའམ་མཐའ་ལ་བརྟེན་ཞིང་གྲུབ་མཐའ་ལོག་པའི་ལམ་ལ་སྟེགས་འཆའ་བ་སོགས་དམའ་འབེབས་ཀྱི་རང་བཞིན་ལྡན་པའི་དོན་བཀྲལ་ཚུལ་དག་བྱུང་བ་དང་། བཀྲལ་ཚུལ་དེ་བཞིན་མ་དག་རྒྱུན་འབྱམས་སུ་ཡུན་དུ་གནས་པར་སྣང་།
          སྐབས་བབས་གླེང་བྱའི་དོན་འདི་བཞིན་བོད་ཀྱི་རིག་གནས་གཏིང་མཐའ་བྲལ་བའི་ནང་གི་ཆེས་ཆུང་བའི་ཉག་མ་ཞིག་ཡིན་ཕྱིར། གལ་འགངས་ཆེན་པོ་ཞིག་ཏུ་བཟུང་ནས་སྣང་བ་འཇོག་རིན་ཆེར་མེད་པ་ལྟ་བུ་བློ་ལྡན་སུ་དང་གང་གི་དྲན་པའི་ངོགས་སུ་འཆར་ངེས་ན་ཡང་། 《བཞི་འགྲེལ》ལས། “དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱི་ཆོས་ནི་མདོར་བསྡུ་ན་མི་འཚེ་བ་དེ་ཉིད་ཡིན་ནོ་ཞེས་བསྟན་ཏོ། །”[35]ཞེས་གསུངས་པ་ལྟར་དེ་བཞིན་གཤེགས་པས་ནང་པའི་ཆོས་ནི་མདོར་ན་མི་འཚེ་བར་གསུངས་ཡོད་པས། མུ་སྟེགས་ཞེས་བྱའི་ཐ་སྙད་ཀྱི་གོ་དོན་དམའ་འབེབས་ཀྱི་བརྡའི་ཐོག་ནས་བཀྲལ་བའམ་ནང་པས་ཕྱི་རོལ་པར་དམའ་འབེབས་ཀྱི་བརྡ་བྱས་པར་བརྗོད་ཚེ། ངག་གི་ཆ་ནས་གཞན་ལ་འཚེ་བ་བྱས་པ་གསལ་ནུས་ཕྱིར། དེ་ལྟར་སྨྲས་ན་དོན་དམ་ངག་རབ་ཏུ་མི་དགེ་བར་གྱུར་པས་ནང་ཆོས་ཀྱི་ངོ་བོ་དང་འགལ་བར་མཐོང་། རིགས་པས་དེད་ན་ཐུབ་བསྟན་ཐོག་མར་དར་དུས་དང་ཁྱད་པར་སྟོན་པ་བཅོམ་ལྡན་འདས་ཀྱིས་གདུལ་བྱར་ཆོས་གསུངས་པའི་ཚེ་གྲུབ་མཐའ་གཞན་ལ་དམའ་འབེབས་ཀྱི་བརྡ་བྱེད་མི་སྲིད་དེ། དམའ་འབེབས་ཀྱི་བརྡ་བྱས་ན་ཐུབ་བསྟན་གྱི་དགོངས་པ་དང་འགལ་བའི་ཕྱིར། དེས་ན་ནང་བསྟན་ཐོག་མར་དར་བའི་སྐབས་ཉན་ཐོས་དགྲ་བཅོམ་པ་སོགས་ནས་བརྒྱུད་པའི་པཎ་གྲུབ་མང་དུ་བྱོན་པ་དག་གིས་མདོ་ལ་གཏུགས་ནས་བསྟན་བཅོས་རྣམས་སུ་མུ་སྟེགས་པའི་ལྟ་ངན་ལ་སུན་འབྱིན་པའི་སྐབས་རིགས་པའི་ཁུངས་མི་འཕེར་བའི་ལྟ་བ་ལོག་པ་དང་ལྟ་བ་ངན་པ་ཞེས་ལན་གྲངས་མང་དུ་སྨྲོས་པ་དང་། རིམ་གྱིས་ཕྱི་རོལ་པའི་གྲུབ་མཐའི་ངོ་བོར་བལྟོས་ནས་མུ་སྟེགས་ཞེས་པའི་གོ་དོན་ལ་ཡང་ལྟ་བ་ལོག་པའི་མཐའ་དང་གྲུབ་མཐའ་ངན་པའི་མཐའ་སོགས་ལོག་པར་བཀྲལ་བ་དང་སྒྲོ་བཏགས་པ་ཞིག་ཡིན་པར་ཐེ་མི་ཚོམ་མོ་སྙམ་དུ་སེམས་སོ། །
          དེང་གི་ཆར་རང་རེ་ཀྲུང་གོ་སྤྱི་དང་ནང་སྒོས་སུ་བོད་ཟལ་མོ་སྒང་གི་ཡུལ་ནི་འཛམ་བུའི་གླིང་གི་སངས་རྒྱས་ཆོས་ལུགས་ཀྱི་ཕ་ས་ཡིན་པར་མཁས་མང་གིས་བཞེད་ཅིང་བདེན་པར་ཡང་གྲུབ་མོད། སྐབས་བབས་བོད་འགྱུར་ཐ་སྙད་མུ་སྟེགས་ཞེས་པའི་གོ་དོན་བཀྲལ་ཚུལ་ཙམ་ལ་ཡང་ཐུབ་པའི་ལུགས་ཀྱི་དགོངས་པ་དང་མ་མཐུན་པའི་འགྲེལ་ཚུལ་བྱུང་དང་བྱུང་མུས་ཡིན་པའི་ཆ་ནས་དེ་དོན་“མོན་ཡུལ་རྡོར་གཅོད་”ཀྱི་དཔེ་ཟླར་གྱུར་ནས་གཅིག་ལ་གཅིག་བརྒྱུད་དུ་མ་དག་རྒྱུན་འབྱམས་སུ་མ་འགྱུར་ཆེད། ལུང་དང་རིགས་པའི་སྒྲུབ་བྱེད་གང་ནུས་གཅིག་གྲོགས་གཅིག་གི་ཚུལ་དུ་བཀོད་དེ། མུ་སྟེགས་ཞེས་པའི་གདོད་མའི་དོན་ལ་དམའ་འབེབས་ཀྱི་རང་བཞིན་མེད་ཚུལ་སྒྲུབ་ཁུལ་བྱས་ཡོད། དེ་ལྟར་ན་ཡང་། ང་གཉིས་ནི་བལྟ་རྒྱ་ཐོས་རྒྱས་ལྷག་པར་ཕོངས་ཤིང་རྣམ་དཔྱོད་ཤིན་ཏུ་རྩིང་བའི་ཆེས་ཐ་མལ་བ་ཞིག་ཡིན་པའི་ཆ་ནས་གོ་ལོག་གི་གཏམ་སྨྲས་པ་སྲིད་ངེས་ན། དཔྱོད་ལྡན་གཟུ་བོར་གནས་པ་དག་གིས་དགག་བཞག་གི་བཀའ་སློབ་གནད་སྨིན་ཡོད་པ་ཞུ། །
