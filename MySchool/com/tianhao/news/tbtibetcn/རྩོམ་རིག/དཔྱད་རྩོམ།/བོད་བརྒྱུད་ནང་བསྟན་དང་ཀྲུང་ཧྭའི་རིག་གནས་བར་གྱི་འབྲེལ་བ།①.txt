བོད་བརྒྱུད་ནང་བསྟན་དང་ཀྲུང་ཧྭའི་རིག་གནས་བར་གྱི་འབྲེལ་བ།①
[ནང་དོན་གནད་བསྡུས།] བོད་བརྒྱུད་ནང་བསྟན་ནི་ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་གྲུབ་ཆ་གལ་ཆེན་ཞིག་ཡིན་ལ། ཀྲུང་ཧྭའི་རིག་གནས་ཁྲོད་ཀྱི་ཐུན་མིན་ཁྱད་ཆོས་ལྡན་པའི་རིག་གནས་ཤིག་ཀྱང་ཡིན། བོད་བརྒྱུད་ནང་བསྟན་དང་ཀྲུང་ཧྭའི་རིག་གནས་བར་གྱི་འབྲེལ་བར་རྒྱུས་ལོན་དང་ཐག་གཅོད་ཡང་དག་པ་ཞིག་བྱེད་པ་ལ། བོད་བརྒྱུད་ནང་བསྟན་ནི་དད་མོས་ཤིག་ཡིན་ལ་རིག་གནས་ཤིག་ཀྱང་ཡིན་པ་དང་། ངོ་བོའི་སྟེང་ནས་རིག་གནས་ཀྱི་རང་བཞིན་ལྡན་ལ། སྒྲིག་གཞི་དང་ནང་དོན་གྱི་སྟེང་ནས་ཐུན་མོང་མ་ཡིན་པའི་རིག་གནས་མ་ལག་རིགས་ཤིག་ཏུ་གྱུར་འདུག་པ་དང་། བྱེད་ལས་དང་ནུས་པའི་སྟེང་ནས་རིག་གནས་ཀྱི་བྱེད་ལས་སྣ་མང་ལྡན་པ་ཞིག་ཡིན་པ་བཅས་ཤེས་དགོས། བོད་རིགས་ཀྱི་སྲོལ་རྒྱུན་རིག་གནས་གལ་ཆེན་ཞིག་ཡིན་པའི་ཆ་ནས། བོད་བརྒྱུད་ནང་བསྟན་གྱི་རྩ་འཛུགས་བྱེད་སྟངས་དང་། ཆོས་ཀྱི་ལྟ་དགོངས། སློབ་གཉེར་ལམ་ལུགས། རྒྱུད་འཛིན་བྱེད་ལུགས། རིག་གནས་གོམས་སྲོལ་སོགས་ཀྱི་ཕྱོགས་ཚང་མར་ཐུན་མིན་ཁྱད་ཆོས་ལྡན་པས། བོད་བརྒྱུད་ནང་བསྟན་ནི་ཀྲུང་ཧྭའི་རིག་གནས་ཁྲོད་ཀྱི་ཐུན་མིན་ཁྱད་ཆོས་ལྡན་པའི་རིག་གནས་ཤིག་ཏུ་གྱུར་ཡོད། རྒྱ་བོད་གཉིས་ཀྱི་ནང་བསྟན་གྱི་རིག་གནས་ཕན་ཚུན་བརྗེ་སྤེལ་དང་འབྲེལ་འདྲིས་བྱས་པ་དེས་ནང་བསྟན་ཀྲུང་གོ་ཅན་དུ་འགྱུར་བའི་འཕེལ་རིམ་ལ་སྐུལ་འདེད་བཏང་བ་རེད། ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་ཤུགས་རྐྱེན་ཁྲོད་དུ་བོད་བརྒྱུད་ནང་བསྟན་གྱིས་བོད་ཀྱི་སྲོལ་རྒྱུན་རིག་གནས་ལ་ཐེབས་པའི་ཤུགས་རྐྱེན་ནི་ཆེས་ཐད་ཀར་དང་གཏིང་ཟབ་ཤོས་ཡིན། བོད་བརྒྱུད་ནང་བསྟན་གྱིས་ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་གྲུབ་འཕེལ་ཐད་བྱས་རྗེས་གཞན་དྲིང་མི་འཇོག་པ་བཞག་ཡོད་ལ། ཀྲུང་ཧྭ་མི་རིགས་ཁག་གི་རིག་གནས་མཉམ་འདྲེས་ཀྱི་སྒྲོག་ཐག་ཏུའང་གྱུར་ཡོད། བོད་བརྒྱུད་ནང་བསྟན་ལ་ཀྲུང་ཧྭའི་རིག་གནས་དར་རྒྱས་ཕུན་ཚོགས་སུ་འགྲོ་བའི་ཁྲོད་དུ་རང་ཉིད་ཀྱི་འགན་འཁྲི་ཡོད་སྟབས། ངེས་པར་དུ་ཀྲུང་ཧྭའི་རིག་གནས་འཛུགས་སྐྲུན་ནང་ཞུགས་ཏེ་ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་ཁྱད་ཆོས་དང་གཤིས་སྤྱོད་ཕུན་སུམ་ཚོགས་པར་གཏོང་དགོས།
[བརྡ་ཆད་གཙོ་བོ།] བོད་བརྒྱུད་ནང་བསྟན། བོད་རིགས། རིག་གནས། ཀྲུང་ཧྭའི་རིག་གནས། ཀྲུང་ཧྭའི་རིག་གནས་འཛུགས་སྐྲུན།
[ཀྲུང་གོའི་དཔེ་དེབ་དབྱེ་རྟགས།] B946.6
[ཡིག་ཚང་མཚོན་བྱང་།] A
[རྩོམ་གྱི་གོ་རིམ་ཨང་སྒྲིག]  1002-9060-2020-02-0003-25
        བོད་བརྒྱུད་ནང་བསྟན་ནི་རང་རྒྱལ་གྱི་རང་སའི་ཆོས་ལུགས་རིག་གནས་ཡིན་ལ། ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་གྲུབ་ཆ་གལ་ཆེན་ཡིན་པ་དང་ཀྲུང་ཧྭའི་རིག་གནས་ཁྲོད་ཀྱི་ཐུན་མིན་ཁྱད་ཆོས་ལྡན་པའི་རིག་གནས་ཀྱང་ཡིན། དེང་གི་ཆར་རིག་གནས་རང་རྟོན་རྒྱུན་འཁྱོངས་བྱེད་པའི་ཆ་རྐྱེན་འོག་དང་སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་རིག་གནས་དར་རྒྱས་ཕུན་སུམ་ཚོགས་པར་གཏོང་བའི་རྒྱབ་ལྗོངས་འོག་ཏུ། བོད་བརྒྱུད་ནང་བསྟན་དང་ཀྲུང་ཧྭའི་རིག་གནས་བར་གྱི་འབྲེལ་བར་རྒྱུས་ལོན་དང་ཐག་གཅོད་ཡང་དག་པ་ཞིག་བྱེད་དགོས།ནང་བསྟན་ནི་གནའ་བོའི་རྒྱ་གར་གྱི་ཀ་པི་ལ་བསྟུའི་རྒྱལ་སྲས་ཤཱཀྱ་མུ་ནེ་ཡིས་སྤྱི་ལོ་སྔོན་གྱི་དུས་རབས་དྲུག་པ་ནས་ལྔ་པ་ལ་སྲོལ་བཏོད་ཅིང་། དེའི་འཕེལ་རྒྱས་ཀྱི་ལོ་རྒྱུས་ནི་དུས་རིམ་བཞི་རུ་བཀར་ཆོག་སྟེ། སོ་སོར་ནང་བསྟན་གྱི་ཐོག་མའི་དུས་རིམ་དང་། སྡེ་པ་སོ་སོར་གྱེས་པའི་དུས་རིམ། ཐེག་ཆེན་གྱི་དུས་རིམ། གསང་སྔགས་རྡོ་རྗེ་ཐེག་པའི་དུས་རིམ་བཅས་སོ། །
སྤྱི་ལོ་སྔོན་གྱི་དུས་རབས་དྲུག་པའི་དུས་དཀྱིལ་ཙམ་ནས་སྤྱི་ལོ་སྔོན་གྱི་དུས་རབས་བཞི་པའི་དུས་དཀྱིལ་བར་ནི། ཤཱཀྱ་ཐུབ་པས་ནང་བསྟན་གྱི་སྲོལ་བཏོད་པ་དང་རྗེས་འཇུག་བུ་སློབ་རྣམས་ཀྱིས་ཁོང་གི་ཆོས་བརྒྱུད་རྒྱུད་འཛིན་བྱས་པའི་དུས་རིམ་ཡིན་པས། དེ་ནི་ནང་བསྟན་གྱི་ཐོག་མའི་དུས་རིམ་ཡིན། སྤྱི་ལོ་དུས་རབས་བཞི་པའི་དུས་དཀྱིལ་ནས་འགོ་ཚུགས་ཏེ། ནང་བསྟན་ནང་ཁུལ་དུ་ཆོས་ཀྱི་ལྟ་དགོངས་དང་འདུལ་བའི་བཅས་མཚམས་སོགས་ལ་ངོས་འཛིན་ཚུལ་གྱི་འགལ་བ་བྱུང་ཞིང་རིམ་གྱིས་སྡེ་པ་མང་དུ་གྱེས་ལ། ཐོག་མར་གནས་བརྟན་སྡེ་པ་(བསླབ་པ་རྒན་པའི་གནས་བརྟན་ཁ་ཅིག་གཙོར་གྱུར་ཅིང་། རྙིང་ཞེན་ཆེ)དང་ཕལ་ཆེན་སྡེ་པ་(དགེ་འདུན་པ་ལོ་གཞོན་དག་གཙོར་གྱུར་ཅིང་། ཐལ་ཆ་ཅུང་ཆེ)གཉིས་སུ་གྱེས་ཤིང་། དེ་ལ་རྩ་བའི་སྡེ་པ་ཟེར། སྡེ་པ་དེ་གཉིས་ཀྱང་རིམ་བཞིན་གྱེས་པས་སྡེ་པ་མང་དུ་(བྱང་བརྒྱུད་ནང་བསྟན་གྱི་བརྗོད་ལུགས་ལྟར་ན་ཁྱོན་བསྡོམས་སྡེ་པ་20དང་། ལྷོ་བརྒྱུད་ནང་བསྟན་གྱི་བརྗོད་ལུགས་ལྟར་ན་ཁྱོན་བསྡོམས་སྡེ་པ་18བྱུང་ཟེར)བྱུང་ཞིང་དེ་ལ་ཡན་ལག་གི་སྡེ་པ་ཞེས་བརྗོད་ཆོག  དེ་ནི་སྡེ་པ་སོ་སོར་གྱེས་པའི་དུས་རིམ་ཡིན། ཐོག་མའི་དུས་རིམ་དང་སྡེ་པ་སོ་སོར་གྱེས་པའི་དུས་རིམ་ལ་སྤྱིར་བཏང་ཐེག་དམན་ཞེས་འབོད་ཅིང་། རང་དོན་(རང་ཉིད་ཐར་པ་འཐོབ་པ)བསམ་ཞིང་དགྲ་བཅོམ་པའི་གོ་འཕང་འཐོབ་རྒྱུ་ནི་མཐར་ཐུག་གི་དམིགས་ཡུལ་ཡིན། སྤྱི་ལོ་དུས་རབས་དང་པོ་ཡས་མས་ནས་ཐེག་ཆེན་གྱི་དུས་རིམ་འགོ་ཚུགས་ཤིང་། དབུ་མ་པ་དང་སེམས་ཙམ་པ་(རྣལ་འབྱོར་སྤྱོད་པ་བ་ཞེས་ཀྱང་ཟེར)གཉིས་སུ་གྱེས། ཐེག་ཆེན་ཞེས་པ་ནི་རང་གཞན་གྱི་དོན་གཉིས་སྒྲུབ་པའི་བྱང་ཆུབ་སེམས་དཔའི་སྤྱོད་པས་མཐར་ཐུག་སངས་རྒྱ་བ་དམིགས་ཡུལ་དུ་བྱ་བ་ཞིག་ཡིན། སྤྱི་ལོ་དུས་རབས་བདུན་པའི་རྗེས་སུ་ཐེག་ཆེན་གྱི་སྡེ་པ་ཁག་ཅིག་དང་བྲམ་ཟེ་ཆོས་ལུགས་གཉིས་མཉམ་འདྲེས་ཀྱིས་གསང་སྔགས་བྱུང་། དེ་ནི་གསང་སྔགས་ཀྱི་དུས་རིམ་མོ། །  
སྤྱི་ལོ་སྔོན་གྱི་དུས་རབས་གསུམ་པར་ཡུལ་དབུས་མ་ག་དྷའི་མོརྱ་རྒྱལ་རྒྱུད་ཀྱི་རྒྱལ་པོ་མྱ་ངན་མེད་ནས་འགོ་ཚུགས་ཤིང་། སྤྱི་ལོ་དུས་རབས་གཉིས་པར་ཀུ་ཤ་ནའི་རྒྱལ་པོ་ཀ་ནིཥྐ་བརྒྱུད། ནང་བསྟན་རྒྱ་གར་ནས་ཕྱི་ཕྱོགས་སུ་མཚམས་མེད་པར་མཆེད་དེ་འཛམ་གླིང་རང་བཞིན་གྱི་ཆོས་ལུགས་ཤིག་ཏུ་གྱུར་པར་མ་ཟད། རྒྱལ་ཁབ་མང་པོ་རུ་རང་རང་གི་མི་རིགས་ཀྱི་ཁྱད་ཆོས་ལྡན་པའི་གྲུབ་མཐའ་ཆགས། ཁྱབ་ཡུལ་དང་ཆོས་གཞུང་གི་ཐད་ནས་བལྟས་ན། རང་རྒྱལ་གྱི་ས་ཆ་ཕལ་མོ་ཆེ་དང་། ཁོ་རེ་ཡ། ཉི་ཧོང་། ཝེ་ཐེ་ནམ། སོག་ཡུལ།སི་བྷི་རི་ཡ། འབྲུག་སོགས་སུ་དར་བའི་ནང་བསྟན་ནི་ཐེག་ཆེན་གཙོ་བོ་ཡིན་ཞིང་། བྱང་བརྒྱུད་ནང་བསྟན་ཞེས་སུ་འབོད། དེའི་ནང་ནས་རང་རྒྱལ་གྱི་རྒྱ་རིགས་ས་ཁུལ་དང་ཉི་ཧོང་། ཝེ་ཐེ་ནམ་སོགས་སུ་དར་བ་ལ་བྱང་བརྒྱུད་ནང་བསྟན་ལས་རྒྱ་བརྒྱུད་ནང་བསྟན་ནམ་རྒྱ་འགྱུར་ནང་བསྟན་ཞེས་འབོད་ལ། བསྟན་བཅོས་ཕལ་ཆེར་རྒྱ་ཡིག་ཏུ་ཡོད་པས་རྒྱའི་སྐད་རྒྱུད་ཀྱི་ནང་བསྟན་ཞེས་ཀྱང་འབོད། ཡང་རང་རྒྱལ་གྱི་བོད་དང་། ནང་སོག མོང་གོལ། འབྲུག འབྲས་ལྗོངས།(དེང་སྐབས་རྒྱ་གར་གྱི་འབྲས་ལྗོངས་མངའ་སྡེ) བལ་པོ། སི་བྷི་རི་ཡ་སོགས་སུ་དར་བ་ནི་བྱང་བརྒྱུད་ནང་བསྟན་ལས་བོད་བརྒྱུད་ནང་བསྟན་ཡིན་ཞིང་། ཐེག་ཆེན་དང་གསང་སྔགས་རྡོ་རྗེ་ཐེག་པ་སྤེལ་མར་ཡོད། དེའི་བསྟན་བཅོས་ཕལ་མོ་ཆེ་བོད་ཡིག་ཏུ་གཏོགས་པས་བོད་སྐད་རྒྱུད་ཀྱི་ནང་བསྟན་ཞེས་ཀྱང་འབོད། དེང་སང་གི་སིང་གྷ་ལ་དང་། འབར་མ། ཐེ་ལན་ཌི། ཁམ་ཕུ་ཁྲེ། ལོ་སི་སོགས་དང་། རང་རྒྱལ་གྱི་ཏའེ་རིགས་ས་ཁུལ་སོགས་སུ་དར་བ་ནི་ཐེག་དམན་གཙོར་གྱུར་ཅིང་། ལྷོ་བརྒྱུད་ནང་བསྟན་ནམ་གནས་བརྟན་སྡེ་པའི་ལུགས་ཀྱི་ནང་བསྟན་ཟེར། དེའི་བསྟན་བཅོས་ཕལ་མོ་ཆེ་ལེགས་སྦྱར་པྲ་ཀྲི་ཏ་ཡི་གེའི་མ་ལག་ལ་གཏོགས་པས། ལེགས་སྦྱར་པྲ་ཀྲི་ཏའི་སྐད་རྒྱུད་ཀྱི་ནང་བསྟན་ཡང་ཟེར། དེ་ནི་སྐད་རྒྱུད་ཆེན་མོ་གསུམ་གྱི་ནང་བསྟན་ཡིན།
        སྤྱིའི་ཆ་ནས་བཤད་ན་ནང་བསྟན་གྱི་གྲུབ་མཐའ་ལ་དབྱེ་སྟངས་གསུམ་ཡོད། རིག་གཞུང་གི་ལྟ་བའི་ངོས་ནས་བལྟས་ན་སྤྱིའི་ཆ་ནས་ཐེག་ཆེན་དང་ཐེག་དམན་གཉིས་སུ་དབྱེ་བ་དང་། ད་དུང་ཐེག་ཆེན་དང་། ཐེག་དམན། གསང་སྔགས་རྡོ་རྗེ་ཐེག་པ་ཞེས་དབྱེ་བའང་ཡོད། ཁྱབ་སྤེལ་གྱི་ལམ་ཐིག་ལྟར་ན་ལྷོ་བརྒྱུད་ནང་བསྟན་དང་བྱང་བརྒྱུད་ནང་བསྟན་གཉིས་སུ་དབྱེ། བེད་སྤྱོད་པའི་སྐད་ཆ་ལྟར་ན་རྒྱའི་སྐད་རྒྱུད་ཀྱི་ནང་བསྟན་དང་། བོད་སྐད་རྒྱུད་ཀྱི་སངས་རྒྱས་ཆོས་ལུགས། ལེགས་སྦྱར་པྲ་ཀྲི་ཏའི་སྐད་རྒྱུད་ཀྱི་ནང་བསྟན་བཅས་རིགས་གསུམ་དུ་དབྱེ་ལ། དེར་སྐད་རྒྱུད་ཆེན་མོ་གསུམ་གྱི་ནང་བསྟན་ཞེས་ཟེར།

        2014ལོའི་ཟླ་3པའི་ཚེས་27ཉིན། སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིན་ཕིང་གིས་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་སློབ་གསོ་ཚན་རིག་རིག་གནས་རྩ་འཛུགས་ཀྱི་སྤྱི་ཁྱབ་པུའུ་རུ་གཏམ་བཤད་གནང་སྐབས་བསྟན་དོན། “མིའི་རིགས་ཀྱིས་གསར་དུ་བཏོད་པའི་དཔལ་ཡོན་ཐམས་ཅད་ངལ་རྩོལ་དང་ཤེས་རབ་ཀྱི་ཉིང་བཅུད་ཡིན་ལ། དཔལ་ཡོན་རིགས་རེ་རེ་བཞིན་ཐུན་མོང་མ་ཡིན་པ་ཞིག་ཡིན་”ཞེས་དང་། “ནང་བསྟན་ནི་གནའ་བོའི་རྒྱ་གར་དུ་བྱུང་ནའང་། དེ་ཉིད་ཀྲུང་གོ་རུ་དར་བའི་རྗེས་སུ་དུས་ཡུན་རིང་པོའི་འཕེལ་འགྱུར་བརྒྱུད། ནང་བསྟན་དང་ཀྲུང་གོའི་རུའུ་ལུགས་རིག་གནས་དང་ཏའོ་ལུགས་རིག་གནས་ལ་མཉམ་འདྲེས་འཕེལ་རྒྱས་བྱུང་ནས། མཐར་ཀྲུང་གོའི་ཁྱད་ཆོས་ལྡན་པའི་ནང་བསྟན་གྱི་རིག་གནས་ཆགས་པ་དང་། ཀྲུང་གོ་བའི་ཆོས་ལུགས་དད་མོས་དང་། མཚན་ཉིད་རིག་པའི་ལྟ་བ། རྩོམ་རིག་སྒྱུ་རྩལ། གུས་ལུགས་གོམས་སྲོལ་སོགས་ལ་ཤུགས་རྐྱེན་ཟབ་མོ་ཐེབས་ཡོད་”ཅེས་དང་། “ཀྲུང་གོ་བས་ཀྲུང་ཧྭའི་རིག་གནས་གཞིར་བཟུང་ནས་ནང་བསྟན་གྱི་བསམ་བློ་འཕེལ་རྒྱས་སུ་བཏང་བས། ནང་བསྟན་གྱི་གཞུང་ལུགས་ཁྱད་པར་བ་གྲུབ་པར་མ་ཟད། ནང་བསྟན་ཀྲུང་གོ་ནས་ཉི་ཧོང་དང་། ཁོ་རེ་ཡ། ཨེ་ཤ་ཡ་ཤར་ལྷོ་སོགས་སུ་དར་བར་བྱས་ཡོད།” ③ སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིན་ཕིང་གི་གཏམ་བཤད་ཀྱིས་ནང་བསྟན་ཀྲུང་གོ་ཅན་དུ་གྱུར་ཏེ་ཀྲུང་ཧྭའི་དཔལ་ཡོན་ལ་ཐེབས་པའི་ཤུགས་རྐྱེན་ཟབ་མོ་དང་འཛམ་གླིང་གི་རིག་གནས་ལ་བཞག་པའི་ཐུན་མིན་བྱས་རྗེས་ནན་བཤད་གནང་ཡོད་པ་དང་། ནང་བསྟན་ཀྲུང་ཧྭའི་རིག་གནས་འཛུགས་སྐྲུན་ལ་ཞུགས་པའི་འབྲེལ་ཡོད་བརྒྱུད་ལམ་དང་ནང་དོན་གསལ་སྟོན་གནང་ཡོད་ལ། ཟུར་ངོས་གཞན་ཞིག་ནས་ནང་བསྟན་གྱི་རིག་གནས་རང་བཞིན་དང་རིག་གནས་ནུས་པ་ནན་བཤད་ཀྱང་གནང་ཡོད།
        ངོ་བོའི་རང་བཞིན་ཐད་ནས་བལྟས་ན། བོད་བརྒྱུད་ནང་བསྟན་ལ་རིག་གནས་ཀྱི་རང་བཞིན་ལྡན་པ་རེད། བོད་བརྒྱུད་ནང་བསྟན་ནི་ཐོག་མར་ཆོས་ལུགས་དད་མོས་ཤིག་ཡིན་པའི་ཆ་ནས། གངས་ཅན་ལྗོངས་ཀྱི་ཆོས་དད་མང་ཚོགས་ཀྱི་ཆོས་ལུགས་དད་མོས་ཀྱི་དགོས་མཁོ་སྐོང་བཞིན་ཡོད་པ་སྨོས་མེད་རེད། འོན་ཀྱང་མིའི་རིགས་ཀྱི་རིག་གནས་ནི་སྣ་མང་ཡིན་ཞིང་། ཆོས་ལུགས་རང་ཉིད་ཀྱང་རིག་གནས་ཤིག་ཡིན་ལ། ཆོས་ལུགས་རིག་གནས་ནི་མིའི་རིགས་ཀྱི་རིག་གནས་སྣ་ཚོགས་པའི་གྲུབ་ཆ་གལ་ཆེན་ཞིག་རེད། མིའི་རིགས་ཀྱི་རིག་གནས་ལོ་རྒྱུས་སུ་ཆོས་ལུགས་རིག་གནས་དང་འཇིག་རྟེན་པའི་རིག་གནས་གཉིས་ཕན་ཚུན་འགལ་ཟླར་གནས་ལ་ཕན་ཚུན་འདྲེས་མཐུན་ཡང་འབྱུང་གི་ཡོད། ཆོས་དད་ཡོད་པའི་ས་ཆ་རུ་ཆོས་ལུགས་རིག་གནས་དེ་མིའི་རིགས་ཀྱི་འཚོ་བའི་ཕྱོགས་སོ་སོ་ལ་སིམ་ཡོད། དེའི་ཕྱིར་ངོ་བོའི་རང་བཞིན་ཐད་ནས་བལྟས་ན། བོད་བརྒྱུད་ནང་བསྟན་ནི་དད་མོས་ཤིག་ཡིན་ལ་རིག་གནས་ཤིག་ཀྱང་ཡིན། ཆོས་ཀྱི་ལྟ་བ་དང་བསམ་བློའི་ཐད་ནས་བལྟས་ན། བོད་ཡིག་གི་བཀའ་བསྟན་གྱིས་སྣ་འཛོམས་ཤིང་རྫོངས་མཐུག་པའི་རིག་གནས་ནང་དོན་ཞིག་རྒྱུད་འཛིན་བྱས་ཡོད་ལ། ཆོས་ལུགས་ཆོ་གའི་ངོས་ནས་བལྟས་ན། རྡོ་རྗེའི་གར་འཆམ་ལས་ཞི་འཇགས་ཡིད་འོང་གི་རིག་གནས་འདུན་པ་ཞིག་མངོན་བཞིན་ཡོད། འདུལ་བ་བསླབ་ཁྲིམས་ཐད་ནས་བལྟས་ན། དབྱར་གནས་ལས་མ་གྱུར་སེམས་ཅད་ཐམས་ཅད་ལ་བརྩེ་བའི་རིག་གནས་སེམས་ཁམས་ཤིག་མངོན་བཞིན་ཡོད། ཆོས་ལུགས་སྒྱུ་རྩལ་གྱི་ངོས་ནས་བལྟས་ན། འབུར་བཟོ་དང་ལྷ་བྲིས། ཐང་ག་དང་ཨར་ལས་སོགས་ལས་སྣ་མང་ཕུན་སུམ་ཚོགས་པའི་རིག་གནས་ནང་རྫོངས་ཤིག་མངོན་བཞིན་ཡོད། ཆོས་ལུགས་ཀྱི་དུས་སྟོན་ཐད་ནས་བལྟས་ན། ས་ག་ཟླ་བ་ལ་དགེ་སྒྲུབ་སྡིག་སྤོང་དང་ཉམས་དགའ་བའི་རིག་གནས་མོས་འདུན་ཞིག་གཅིག་སྡུད་བྱས་ཡོད།
        སྒྲིག་གཞིའི་ནང་དོན་ཐད་ནས་བལྟས་ན། བོད་བརྒྱུད་ནང་བསྟན་ནི་ཐུན་མིན་གྱི་རིག་གནས་མ་ལག་ཅིག་ཏུ་གྲུབ་ཡོད། ང་ཚོས་དེང་སྐབས་སྤྱིར་བཏང་རིག་གནས་ཞེས་པ་སྒྲིག་གཞིའི་ནང་དོན་ཆ་ནས་དངོས་པོའི་རིག་གནས་དང་། བསམ་པའི་རིག་གནས། ཆབ་སྲིད་རིག་གནས་བཅས་སུ་དབྱེ་ཞིང་། དེ་ནི་དངོས་པོའི་དཔལ་ཡོན་དང་། བསམ་པའི་དཔལ་ཡོན། ཆབ་སྲིད་དཔལ་ཡོན་བཅས་ཡིན། འོན་ཀྱང་སྒྲ་དོག་འཇུག་གི་རིག་གནས་ཀྱི་ངོས་ནས་བརྗོད་ན། རིག་གནས་ནི་ཡོ་ཆས་རིག་གནས་དང་། ལམ་ལུགས་རིག་གནས། བསམ་པའི་རིག་གནས་བཅས་སུ་དབྱེ་ལ། དེ་གསུམ་པོས་རིག་གནས་ཀྱི་མ་ལག་འཐུས་ཚང་བ་ཞིག་གྲུབ་ཡོད། བོད་བརྒྱུད་ནང་བསྟན་ནི་གཞུང་དོན་མ་ལག་ཅན་དང་ཆོ་ག་ཚད་ལྡན་ཅན་དུ་གྱུར་པའི་ནང་བསྟན་ཞིག་ཡིན་ལ། དུས་མཚུངས་སུ་ཐུན་མོང་མ་ཡིན་ཞིང་འཐུས་ཚང་བའི་རིག་གནས་ཀྱི་མ་ལག་ཅིག་ཏུའང་གྲུབ་ཡོད། 
བྱེད་ལས་དང་ནུས་པའི་ངོས་ནས་བལྟས་ན། བོད་བརྒྱུད་ནང་བསྟན་ལ་རིག་གནས་ཀྱི་བྱེད་ལས་སྣ་མང་ལྡན་པ་རེད། རིག་གནས་ཀྱི་སྣང་ཚུལ་ཞིག་ཡིན་པའི་ཆ་ནས་བོད་བརྒྱུད་ནང་བསྟན་ལ་བསམ་པའི་རིག་གནས་ཐད་ནས་ཕྱོགས་མང་པོའི་རིག་གནས་བྱེད་ལས་ལྡན་པ་སྟེ། གཙོ་བོ་འགྲེལ་བཤད་ཀྱི་བྱེད་ལས་དང་། རིག་གནས་བརྗེ་རེས་ཀྱི་བྱེད་ལས། སློབ་གསོའི་བྱེད་ལས། རིག་གནས་གོམས་སྲོལ་གྱི་བྱེད་ལས་སོགས་འདུ་ལ། ཁྱབ་ཁོངས་ངེས་ཅན་ནང་དུ་རང་ལ་ཡོད་པའི་སྤྱི་ཚོགས་བྱེད་ལས་ཐུན་མོང་མ་ཡིན་པ་དེ་འདོན་བཞིན་ཡོད།
        བོད་བརྒྱུད་ནང་བསྟན་གྱི་སངས་རྒྱས་ཆོས་དང་དགེ་འདུན་ཏེ་དཀོན་མཆོག་གསུམ་གྱི་ལྟ་བ་དང་མཐུན་པར། ཕྱག་འཚལ་མཆོད་འབུལ་དང་། དམ་ཆོས་འཆད་ཉན། དགེ་འདུན་པ་གསོ་སྐྱོང་གསུམ་ནི་བོད་བརྒྱུད་ནང་བསྟན་གྱི་དགོན་སྡེའི་རྩ་བའི་བྱེད་ལས་གསུམ་ཡིན་ལ། རྩ་བའི་བྱེད་ལས་དེ་གསུམ་ལས་བྱེ་བྲག་གི་བྱེད་ལས་གཞན་མང་པོ་བྱུང་ཡོད། 
        འགྲེལ་བཤད་ཀྱི་བྱེད་ལས། རིག་གནས་ཀྱི་བྱེད་ལས་ཐད་ནས་བཤད་ན་དམ་ཆོས་འཆད་ཉན་ནི་འགྲེལ་བཤད་ཀྱི་བྱེད་ལས་ཁོངས་སུ་གཏོགས་ཤིང་། དམ་ཆོས་འཆད་ཉན་སྐབས་སུ་ཐད་ཀར་སློབ་མར་ཁ་ཕྱོགས་ཏེ་ཆོས་འཆད་དགོས་པར་མ་ཟད། ད་དུང་བསྟན་བཅོས་ལོ་ཙཱ་དང་གསར་རྩོམ་སོགས་ཀྱང་བྱེད་དགོས། དེའི་ཕྱིར་བོད་བརྒྱུད་ནང་བསྟན་གྱི་དགོན་སྡེ་ལ་པར་འདེབས་དང་པར་སྐྲུན་གྱི་བྱེད་ལས་ལྡན་ལ། དགོན་པ་མང་པོ་ལ་རང་རང་གི་པར་ཁང་དང་པར་སྐྲུན་སྡེ་ཁག་ཡོད་ཅིང་། བཀའ་བསྟན་སོགས་བསྟན་བཅོས་དང་ཆོས་དཔེ་མང་པོ་དཔར་བཞིན་ཡོད་པས། དེར་འགྲེལ་བཤད་ཀྱི་བྱེད་ལས་ཧ་ཅང་ཆེན་པོ་ལྡན་པ་རེད། 
        སློབ་གསོའི་བྱེད་ལས། རིག་གནས་ཀྱི་བྱེད་ལས་ཐད་ནས་དགེ་འདུན་པ་གསོ་སྐྱོང་བྱེད་པ་ནི་སློབ་གསོའི་བྱེད་ལས་ཀྱི་ཁོངས་སུ་གཏོགས་ཤིང་། རྒྱུད་འཛིན་པ་མང་དུ་སྐྱོང་བ་ནི་བོད་བརྒྱུད་ནང་བསྟན་གྱི་དགོན་སྡེས་ནང་བསྟན་འཛིན་སྐྱོང་སྤེལ་གསུམ་བྱེད་པའི་རྩ་བའི་ལས་དོན་ཞིག་ཡིན་ལ།  དགོན་སྡེ་རང་ཉིད་ཀྱི་རྩ་བ་ཚུགས་སའང་ཡིན། བོད་བརྒྱུད་ནང་བསྟན་གྱི་དགོན་སྡེ་ཚང་མར་སློབ་གཉེར་གྱི་སྒྲིག་གཞི་ཅུང་འཐུས་ཚང་ཡོད་ཅིང་། གྲྭ་བཙུན་གྱི་སློབ་སྦྱོང་ནང་དོན་ཡང་ཆེད་ལས་མི་འདྲ་བར་ཕྱེ་ཡོད། དཔེར་ན་དགེ་ལུགས་པའི་དགོན་སྡེའི་གྲྭ་ཚང་བཞི་ནི་ཆེད་ལས་མི་འདྲ་བ་སློབ་སྦྱོང་བྱ་བའི་རྩ་འཛུགས་སྡེ་ཁག་ཡིན། དེའི་ཕྱིར་ལོ་རྒྱུས་སྟེང་དགོན་པའི་གྲྭ་བཙུན་ཡང་རིགས་མི་འདྲ་བར་ཕྱེ་ཡོད་ཅིང་། དེ་དང་འབྲེལ་ནས་རིམ་པའང་མི་འདྲ་བ་བྱུང་ཡོད། དུས་མཚུངས་སུ་གྲྭ་བཙུན་རྣམས་ཡོང་ཁུངས་མི་འདྲ་བས། སློབ་གཉེར་གྲུབ་ནས་རང་རང་གི་ཕ་ཡུལ་དུ་ཕྱིན་རྗེས་དགོན་སྡེ་གསར་འཛུགས་དང་གྲྭ་རྒྱུན་བསྡུས་ནས་འཆད་ཉན་སྤེལ་བ་དང་། རང་གིས་སློབ་གཉེར་བྱ་ཡུལ་གྱི་དགོན་སྡེ་དང་འབྲེལ་བ་འཛུགས་པས། མ་དགོན་གཅིག་ལ་བུ་དགོན་མང་པོ་ཡོད་པ་དང་། དགོན་པའི་མ་ལག་མི་འདྲ་བ་གྲུབ་ཡོད། དམངས་གཙོ་བཅོས་སྒྱུར་གྱི་རྗེས་སུ་ཆོས་སྲིད་ལོགས་འབྱེད་བྱས་ནས་ཆོས་ལུགས་ཀྱིས་སྲིད་འཛིན་དང་། ཁྲིམས་འཛིན། སློབ་གསོ། སྤྱི་ཚོགས་འཚོ་བ་སོགས་ལ་ཐེ་གཏོགས་བྱེད་མི་ཆོག་པར་གྱུར་པས། བོད་བརྒྱུད་ནང་བསྟན་གྱི་ཞི་ཞིང་རྣམ་པར་དག་པའི་གཞི་རྩའི་རྣམ་པ་སོར་ཆུད་བྱུང་ཡོད། དེའི་ལོ་རྒྱུས་སྟེང་གི་སློབ་གསོའི་ནུས་པ་ཉམས་པར་གྱུར་ནའང་། ཁྱབ་ཁོངས་ངེས་ཅན་ཏེ་དཔེར་ན་དགོན་སྡེ་སོགས་ཀྱི་སློབ་གསོའི་བྱེད་ལས་ནི་ཡལ་བར་འགྱུར་མི་སྲིད་དོ། །
        བློ་སྒྱུར་བྱེད་ལས། དགེ་འདུན་པ་གསོ་སྐྱོང་བྱེད་པའི་བྱེད་ལས་དེ་སྤྱི་ཚོགས་ཀྱི་སློབ་གསོས་བློ་ཁ་སྒྱུར་བའི་བྱེད་ལས་ཤིག་ཏུའང་འཕེལ་འགྱུར་འབྱུང་སྲིད་ལ། དེ་ནི་ནང་བསྟན་གྱི་དགོན་སྡེའི་རྩ་བའི་དམིགས་ཡུལ་ཡིན། ལོ་རྒྱུས་སྟེང་བོད་བརྒྱུད་ནང་བསྟན་གྱི་དགོན་སྡེས་སྤྱི་ཚོགས་ཀྱི་བློ་སྒྱུར་བྱེད་ལས་འདོན་སྤེལ་གང་ལེགས་བྱས་ཏེ། བཀས་བཀོད་རྒྱུད་འཛིན་ཞིང་བྲན་ལམ་ལུགས་ཀྱི་དུས་སྐབས་ཧྲིལ་པོའི་བོད་ལྗོངས་སྤྱི་ཚོགས་དེ་དམངས་ཡོངས་ཀྱིས་ཆོས་དད་བྱེད་པའི་སྤྱི་ཚོགས་སུ་བསྒྱུར་བ་རེད། དགོན་སྡེས་དྲང་ཕྱོགས་ནས་མི་རྣམས་ལ་རང་རྟོགས་ཀྱིས་ནང་བསྟན་གྱི་གཞུང་ལུགས་དེ་རང་ཉིད་ཀྱི་བསམ་བློ་དང་བྱ་སྤྱོད་ཀྱི་ཚད་གཞིར་འཛིན་པའི་བློ་སྒྱུར་སློབ་གསོ་གཏོང་བའི་མཚུངས་སུ། ད་དུང་ཧུར་བརྩོན་གྱིས་འཇིག་རྟེན་པའི་ཆབ་སྲིད་དང་ཟུང་འབྲེལ་བྱས་ཏེ། སྲིད་འཛིན་དང་བཅའ་ཁྲིམས་ཀྱི་བྱེད་ཐབས་སྣ་ཚོགས་ལ་བརྟེན་ནས་ནང་བསྟན་མ་ཡིན་པ་དང་ནང་བསྟན་ལ་རྒོལ་བའི་བསམ་བློ་དང་རིགས་པའི་གཞུང་ལུགས་ཅི་རིགས་ཁྱབ་སྤེལ་བྱེད་པ་བཀག་འགོག་བྱས་ནས། ཆོས་སྲིད་ཟུང་འབྲེལ་གྱི་བཀས་བཀོད་རྒྱུད་འཛིན་ཞིང་བྲན་ལམ་ལུགས་ཀྱི་སྤྱི་ཚོགས་ཁྲོད་དུ་རང་ཉིད་ཀྱི་ཆབ་སྲིད་དང་ཆོས་ལུགས་ཐད་ཀྱི་གོ་གནས་སྲ་བརྟན་དུ་བཏང་ཡོད། དེང་གི་དུས་སུའང་བོད་བརྒྱུད་ནང་བསྟན་གྱི་ལྟ་དགོངས་བསམ་བློ་དང་། ཆོ་ག་འདུལ་ཁྲིམས། རིག་གནས་གོམས་སྲོལ་སོགས་ཀྱིས་དད་ལྡན་མང་ཚོགས་ཀྱི་ཐོན་སྐྱེད་འཚོ་བ་དང་སྤྱི་ཚོགས་ཀྱི་བྱ་སྤྱོད་ལ་ཐད་ཀར་མཛུབ་ཁྲིད་ཀྱི་ནུས་པ་འདོན་པ་དང་ངང་ངམ་ཤུགས་ཀྱིས་བསྒྱུར་བའི་ཤུགས་རྐྱེན་གལ་ཆེན་ཐེབས་ཀྱི་ཡོད་པ་རེད། 
        རིག་གནས་གོམས་སྲོལ་གྱི་བྱེད་ལས། ཕྱག་འཚལ་མཆོད་འབུལ་ནི་རིག་གནས་གོམས་སྲོལ་གྱི་བྱེད་ལས་ཀྱི་ཁོངས་སུ་གཏོགས་ཤིང་། དགོན་པའི་ནང་དུ་སངས་རྒྱས་བྱང་སེམས་དང་ཆོས་སྐྱོང་སྣ་ཚོགས་ལ་ཕྱག་འཚལ་བ་དང་། དགོན་སྡེ་ཁག་གི་བླ་མ་སྐྱེས་ཆེན་སོགས་ཀྱིས་ད་དུང་དད་ལྡན་མང་ཚོགས་ཀྱི་མཇལ་ཕྱག་དང་ལེན་བྱེད་དགོས། དེ་དང་ཆབས་ཅིག་གྲུབ་མཐའ་ཁག་གི་དགོན་སྡེ་ཚང་མར་རང་ཉིད་ཀྱིས་གོང་བཀུར་བྱེད་པའི་སྲོལ་འབྱེད་པ་དང་ཆོས་སྐྱོང་སོགས་ཡོད་པས་གྲུབ་མཐའ་མི་འདྲ་བའི་དགོན་སྡེ་ལ་ཁྱད་ཆོས་མི་འདྲ་བ་ཆགས་པར་བྱས་ཡོད། དད་མཆོད་བྱ་ཡུལ་དེ་དག་དམངས་ཁྲོད་དུ་དར་ཁྱབ་བྱུང་བས་རྟག་པར་ས་ཁུལ་མི་འདྲ་བའི་ཐུན་མོང་མ་ཡིན་པའི་དད་གུས་གོམས་སྲོལ་དུ་ཆགས་ཡོད། དགོན་སྡེ་ཁ་ཤས་སུ་འཇིམ་རྐོས་དང་རི་མོ། ལྡེར་བཟོ་སྒྱུ་རྩལ་བཅས་དར་འཕེལ་བྱུང་ནས་སྒྱུ་རྩལ་གྱི་གྲུབ་མཐའ་མི་འདྲ་བ་ཆགས་ཡོད་པ་དང་། སྒྱུ་རྩལ་གྱི་གྲུབ་མཐའ་དེ་དག་གིས་མི་རིགས་དང་ས་ཁོངས་ཀྱི་རིག་གནས་འཕེལ་རྒྱས་ལ་ཤུགས་རྐྱེན་ཐེབས་ནས། རྟག་པར་མི་རིགས་སམ་ས་ཁུལ་གྱི་དམངས་ཁྲོད་སྒྱུ་རྩལ་དུ་གྱུར་ཡོད། བོད་བརྒྱུད་ནང་བསྟན་གྱི་རོལ་དབྱངས་ཞབས་བྲོ་དང་དུས་ཆེན་བྱེད་སྒོ་སོགས་ཀྱང་ཕུན་སུམ་ཚོགས་པ་ཡོད་པས། དམངས་ཁྲོད་རིག་གནས་དང་བསམ་པའི་འཚོ་བ་ཕུན་སུམ་ཚོགས་པོར་གཏོང་བའི་ཐབས་ལམ་གལ་ཆེན་ཞིག་ཏུ་གྱུར་ཡོད།ཆོས་ལུགས་ཀྱི་རིག་གནས་ནུས་པའི་དབང་གིས་ལོ་རྒྱུས་སྟེང་བོད་བརྒྱུད་ནང་བསྟན་ཁྱབ་འཕེལ་ས་ཁོངས་སུ་ལྷ་ལ་མཆོད་པ་དང་དགེ་འདུན་ལ་བཀུར་བ། རི་མོ་འབྲི་བ་དང་སྐུ་བརྙན་བཞེངས་པ། ནང་བསྟན་གྱི་རོལ་མོ་ལོངས་སུ་སྤྱོད་པ།  ཆོས་ལུགས་ཀྱི་དུས་ཆེན་ལ་རོལ་བ་སོགས་རིག་གནས་ཀྱི་སྲོལ་རྒྱུན་རྫོང་མཐུག་པོ་ཆགས་ཡོད། 
        བོད་བརྒྱུད་ནང་བསྟན་ལ་ངོ་བོའི་སྟེང་ནས་རིག་གནས་ཀྱི་རང་བཞིན་ལྡན་པ་དང་། སྒྲིག་གཞི་ནང་དོན་གྱི་ངོས་ནས་ཐུན་མོང་མ་ཡིན་པའི་རིག་གནས་མ་ལག་ཅིག་ཏུ་གྲུབ་པ་དང་། བྱེད་ལས་དང་ནུས་པའི་ཐད་ནས་རིག་གནས་ཀྱི་བྱེད་ལས་སྣ་མང་ལྡན་པ་བཅས་ཀྱི་རྐྱེན་ལས། བོད་བརྒྱུད་ནང་བསྟན་ནི་དད་མོས་ཤིག་ཡིན་ལ་རིག་གནས་ཤིག་དེ་བས་ཀྱང་ཡིན་ནོ། །
གཉིས་པ། བོད་བརྒྱུད་ནང་བསྟན་ནི་ཀྲུང་ཧྭའི་རིག་གནས་ཁྲོད་ཀྱི་ཐུན་མིན་ཁྱད་ཆོས་ལྡན་པའི་རིག་གནས་ཤིག་ཡིན།
        ཀྲུང་ཧྭའི་རིག་གནས་ནི་སྣ་མང་གཞི་གཅིག་གི་རིག་གནས་ཡིན་པ་དང་། བོད་ལྗོངས་ནི་ཀྲུང་ཧྭ་མི་རིགས་ཀྱི་ཐུན་མིན་ཁྱད་ཆོས་ལྡན་པའི་རིག་གནས་སྲུང་སྐྱོང་ས་ཁུལ་ཞིག་ཡིན། བོད་ཀྱི་སྲོལ་རྒྱུན་རིག་གནས་གལ་ཆེན་ཞིག་ཡིན་པའི་བོད་བརྒྱུད་ནང་བསྟན་ནི་ཀྲུང་ཧྭའི་རིག་གནས་ཁྲོད་ཀྱི་ཐུན་མིན་ཁྱད་ཆོས་ལྡན་པའི་རིག་གནས་ཤིག་ཏུ་གྱུར་ཡོད།
        བོད་བརྒྱུད་ནང་བསྟན་དེ་ནང་བསྟན་གཞན་དང་མི་འདྲ་བའི་རྒྱུ་མཚན་ནི། དེ་ཉིད་རྒྱ་གར་གྱི་ནང་བསྟན་དང་།ནང་ལོགས་ས་ཆའི་ནང་བསྟན། བོད་ཁུལ་གྱི་གདོད་མའི་བོན་ཆོས་ཕན་ཚུན་མཉམ་འདྲེས་བྱུང་བའི་འབྲས་བུ་ཡིན་པ་དང་། བོད་ཁུལ་གྱི་ས་གནས་ཅན་དུ་གྱུར་པའི་ནང་བསྟན་ཡིན་པས་རང་གི་ཐུན་མོང་མ་ཡིན་པའི་ཁྱད་ཆོས་ལྡན་པ་རེད། བོད་བརྒྱུད་ནང་བསྟན་ནི་ནང་བསྟན་གྱི་ལྟ་དགོངས་རྨང་གཞིར་བྱས་ཐོག བོན་གྱི་ལྷ་ཚོགས་དང་ཆོ་ག་སྡུད་ལེན་བྱས་ཡོད་པ་ཞིག་ཡིན་ལ། ལྟ་དགོངས་ཐད་ཐེག་ཆེན་དབུ་མ་པའི་ལྟ་བ་གཙོ་བོར་འཛིན་ཞིང་། མདོ་སྔགས་ཟུང་འབྲེལ་དུ་ཉམས་སུ་ལེན་པ་དང་། རྣལ་འབྱོར་བླ་མེད་རྒྱུད་ཀྱི་ཉམས་ལེན་དེ་རིམ་པ་མཐོ་ཤོས་སུ་བརྩི་བས། བོད་བརྒྱུད་ནང་བསྟན་གྱི་གསང་སྔགས་ཀྱི་གྲུབ་མཐའ་རིགས་ཤིག་ཆགས་ཡོད། བོད་བརྒྱུད་ནང་བསྟན་ལ་ཐུན་མོང་མ་ཡིན་པའི་ཁྱད་ཆོས་ལྡན་པའི་ཆོས་བརྒྱུད་མང་པོ་ཞིག་ཆགས་ཡོད་ཅིང་། དགོན་པའི་རྩ་འཛུགས་དང་། རྒྱུད་འཛིན་མ་ལག  སློབ་གཉེར་ལམ་ལུགས་བཅས་ཚགས་དམ་པོ་ཡོད་ལ། བསྟན་བཅོས་ལ་བོད་ཡིག་གི་《བཀའ་འགྱུར་》དང་《བསྟན་འགྱུར་》ཆ་ཚང་ཡང་ཡོད།
        སྐབས་དེའི་བོད་རིགས་སྤྱི་ཚོགས་ཀྱི་བཀས་བཀོད་རྒྱུད་འཛིན་གྱི་མངའ་བདག་གིས་མངའ་ཁོངས་བཅད་འཛིན་བྱས་པའི་དུས་བབ་དང་བསྟུན་པ་ལས། བོད་བརྒྱུད་ནང་བསྟན་ནི་ནང་བསྟན་གཞན་དག་དང་མི་མཚུངས་པར་གསང་སྔགས་ཀྱི་བརྒྱུད་པ་གཙོ་བོར་བྱས་ཏེ་གྲུབ་མཐའ་སྣ་ཚོགས་འཕེལ་རྒྱས་བྱུང་ཡོད། གསང་སྔགས་ཞེས་པ་དེ་ལ་གསང་སྔགས་གྲུབ་མཐའ་འམ་གསང་བ་སྔགས་ཀྱི་ཐེག་པ། ཡང་ན་གཟུངས་སྔགས་ཀྱི་གྲུབ་མཐའ་སོགས་སུ་འབོད་ལ། ཐེག་པ་ཆེན་པོའི་ཁྲོད་ཀྱི་གྲུབ་མཐའ་གལ་ཆེན་ཞིག་ཡིན་པ་དང་།  དེ་ནི་གྲུབ་མཐའ་གཙོ་བོ་གཞན་ཞིག་སྟེ་མདོ་ལུགས་སམ་མདོ་ཕྱོགས་ལ་བལྟོས་ནས་བཏགས་པ་ཡིན་ཞིང་། ངག་རྒྱུན་ལྟར་ན་ཀླུ་སྒྲུབ་ཀྱིས་རྒྱ་གར་ལྷོ་ཕྱོགས་ཀྱི་ལྕགས་ཀྱི་མཆོད་རྟེན་ལས་གསང་སྔགས་ཀྱི་གཞུང་བླངས་པ་ལས་མཆེད་པར་གྲགས། དེ་ནི་རྩ་འཛུགས་ཚད་མཐོ་པོ་ཅན་གྱི་གཟུངས་སྔགས་དང་། ཆོ་ག འཇིག་རྟེན་དད་མོས་བཅས་ཁྱད་རྟགས་གཙོ་བོར་གྱུར་པ་ཞིག་ཡིན་ལ། ངག་གིས་གཟུངས་སྔགས་འདོན་པ་(“གསུང་གི་གསང་བ”)དང་། ལག་པས་ཕྱག་རྒྱ་བྱེད་པ། (“སྐུ་ཡི་གསང་བ་”སྟེ། ལག་པའི་ཕྱག་རྒྱ་དང་ལུས་ཀྱི་རྣམ་འགྱུར་བཀོལ་བ) སེམས་ཀྱིས་ལྷ་སྒོམ་པ་(“ཐུགས་ཀྱི་གསང་བ”)བཅས་ཀྱི་གསང་བ་གསུམ་དང་སངས་རྒྱས་ཐམས་ཅད་ཀྱི་སྐུ་གསུང་ཐུགས་རྣམས་ངོ་འཕྲོད་དབྱེར་མེད་དུ་གྱུར་ཚེ་སངས་རྒྱས་པར་འདོད། རྒྱ་གར་གྱི་ནང་བསྟན་དུ་གཟུངས་སྔགས་འདོན་པའི་སྲོལ་བྱུང་བ་ཤིན་ཏུ་སྔ་སྟེ། ཕལ་ཆེར་ཤཱཀྱ་མུ་ནེའི་དུས་སྐབས་ནས་ཡོད་པ་རེད། ཧ་ལམ་སྤྱི་ལོ་དུས་རབས་བདུན་པའི་དུས་དཀྱིལ་དུ་རྩ་འཛུགས་ཡོད་པའི་གསང་སྔགས་ཀྱི་ཆོས་བརྒྱུད་ཆགས་ཤིང་གྲུབ་མཐའ་མི་འདྲ་བ་མང་པོ་བྱུང་། སྤྱི་ལོའི་དུས་རབས་བརྒྱད་པའི་སྐབས་སུ་དགེ་བ་འཇིགས་མེད་དང་། རྡོ་རྗེ་ཡེ་ཤེས། དོན་ཡོད་གྲུབ་སོགས་ཀྱིས་རང་རྒྱལ་དུ་སྤེལ་བ་དང་། རིམ་བཞིན་རྒྱ་བརྒྱུད་ནང་བསྟན་ཁྲོད་ཀྱི་གསང་སྔགས་ཆོས་བརྒྱུད་དུ་གྲུབ་མོད། རྗེས་སུ་རུའུ་ལུགས་དང་མ་མཐུན་པས་རིམ་བཞིན་ཉམས་རྒུད་དུ་གྱུར། འོན་ཀྱང་གསང་སྔགས་ཀྱི་གྲུབ་མཐའ་སྣ་ཚོགས་རང་རྒྱལ་གྱི་བོད་ཁུལ་དུ་དར་ཁྱབ་བྱུང་ནས་མཇུག་མཐར་བོད་བརྒྱུད་ནང་བསྟན་གྲུབ་པ་རེད།
        རྒྱ་བོད་གཉིས་ཀྱི་ནང་བསྟན་གང་ཡིན་ན་ཡང་ལྟ་དགོངས་སུ་དབུ་མའི་དགོངས་པ་དང་སེམས་ཙམ་གྱི་དགོངས་པའི་ནང་དོན་ཡོད་མོད། བོད་བརྒྱུད་ནང་བསྟན་གྱི་གྲུབ་མཐའ་ཚང་མས་ཀླུ་སྒྲུབ་ཀྱི་དབུ་མའི་ལྟ་དགོངས་གཙོ་བོར་འཛིན་པ་རེད། གྲུབ་མཐའ་ཁག་གིས་དབུ་མའི་དོན་དམ་པའི་བདེན་པ་དང་ཀུན་རྫོབ་ཀྱི་བདེན་པ་གཉིས་ལ་གོ་བ་ལེན་ཚུལ་འདྲ་མིན་སྣ་ཚོགས་ཡོད་ན་ཡང་། ཚང་མས་དབུ་མའི་དགོངས་པ་རང་ལུགས་ཀྱི་རིགས་པའི་གཞུང་ལུགས་ཀྱི་རྨང་གཞིར་འཛིན། སེམས་ཙམ་པའི་དགོངས་པ་གནས་ཡོད་ཀྱང་རང་རྐྱ་འཕེར་བའི་གྲུབ་མཐའ་ཆགས་མེད། སེམས་ཙམ་པར་ཞིབ་འཇུག་བྱེད་པ་ཡང་ལྡོག་ཕྱོགས་ནས་དབུ་མའི་དགོངས་པ་རྟོགས་ཆེད་ཁོ་ན་ཡིན། རྒྱ་བརྒྱུད་ནང་བསྟན་གྱི་སྔ་རབས་དུས་སུ་དམ་ཆོས་པད་དཀར་གྱི་གྲུབ་མཐའ་དང་། ཟབ་མོ་ལྟ་བརྒྱུད་དང་། སྡོང་པོ་བཀོད་པའི་གྲུབ་མཐའ་དང་། སྙིང་པོ་དོན་བརྒྱུད་པའི་གྲུབ་མཐའ་སོགས་གྲུབ་མཐའ་ཚང་མས་དབུ་མའི་ལྟ་བ་བཟུང་མོད། འོན་ཀྱང་ཐང་ཞོན་ཙང་གིས་སེམས་ཙམ་པའི་སློབ་དཔོན་བསྟེན་པ་གཞི་བྱས་ནས། ནང་ལོགས་ས་ཁུལ་དུ་སེམས་ཙམ་པའི་རིག་པ་དར་སྤེལ་བཏང་ལ། དེས་ཐང་རྒྱལ་རབས་རྗེས་ཀྱི་རྒྱ་བརྒྱུད་ནང་བསྟན་ལ་ཤུགས་རྐྱེན་ཧ་ཅང་ཆེན་པོ་ཐེབས་ཤིང་། ད་ལྟའི་རྒྱ་བརྒྱུད་ནང་བསྟན་ཕལ་མོ་ཆེར་སེམས་ཙམ་པའི་ཁྱད་ཆོས་ལྡན་པ་རེད། ནང་བསྟན་གྱི་དབུ་མའི་བསམ་བློས་ཐེག་ཆེན་གྱི་ཐོག་མའི་དུས་ཀྱི་སྟོང་ཉིད་ཀྱི་བསམ་བློ་རྩལ་འདོན་བྱས་ཡོད་ཅིང་། འཇིག་རྟེན་གྱི་བྱ་དངོས་ཡོད་ཚད་དང་དེ་བཞིན་དུ་མི་རྣམས་ཀྱི་ཤེས་རྟོགས། ཐ་ན་ནང་བསྟན་དང་བཅས་པ་ཐམས་ཅད་ལྟོས་བཅས་དང་ཕན་ཚུན་རྟེན་ཅིང་འབྲེལ་བར་འདོད་པ་དང་། དེ་དག་རང་ངོས་ནས་ངོ་བོས་གྲུབ་པ་མེད་ཅིང་རང་བཞིན་དང་བྲལ་བ་དང་། བློས་དབང་གིས་བཞག་པ་ཙམ་དུ་འདོད། དེར་བརྟེན་ཆོས་ཀུན་མི་རྟག་ཅིང་རང་བཞིན་གྱིས་སྟོང་པར་བརྗོད། དབུ་མའི་གྲུབ་མཐའ་དང་འདྲ་བར་སེམས་ཙམ་པའི་གྲུབ་མཐའ་ཡིས་ཀྱང་འཇིག་རྟེན་ཁམས་ཀྱི་ཆོས་ཐམས་ཅད་སྟོང་པའི་རང་བཞིན་དུ་ཁས་ལེན་མོད། འོན་ཀྱང་འཇིག་རྟེན་གྱི་ངོ་བོ་སྟོང་པ་ཁས་ལེན་པའི་དུས་མཚུངས་སུ། རང་སྣང་སེམས་ཀྱི་རྣམ་པར་ཤེས་པའི་ནུས་པ་ནན་བཤད་བྱས་ནས། བྱ་དངོས་ཐམས་ཅད་ཀྱི་ཉོག་ཅིང་འཛིང་བའི་ཁྱད་པར་ཚང་མ་རང་སྣང་སེམས་ཀྱི་འཕྲུལ་སྣང་དང་རྟོག་པ་ཡིན་པ་སྟེ། ཕྱི་རོལ་གྱི་འཇིག་རྟེན་ནི་དོན་དངོས་སུ་ཡིད་ཀྱི་རྣམ་པར་ཤེས་པའི་རྟོག་པ་ལས་བྱུང་བར་འདོད།
        རྒྱ་བརྒྱུད་ནང་བསྟན་གྱི་སློབ་གཉེར་ལམ་ལུགས་ད་དུང་མ་ལག་ཅན་དུ་གྲུབ་མེད་པ་དང་། གྲུབ་མཐའ་སོ་སོའི་སློབ་སྦྱོང་བྱེད་སྟངས་ཀྱང་མི་འདྲ་བ་རེད། སྤྱིའི་ཆ་ནས་དགོན་སྡེ་དང་ནང་བསྟན་སློབ་གླིང་རྟེན་གཞིར་བཟུང་སྟེ། བླ་མའང་དགེ་རྒན་གྱི་སྣེ་ཁྲིད་འོག་ནང་བསྟན་གྱི་གཞི་རྩའི་ཤེས་བྱ་དང་། སྡོམ་ཁྲིམས། གྲུབ་མཐའ་ཁག་སོ་སོས་གཙོ་བོར་འཛིན་པའི་གསུང་རབ་བཅས་ལ་སློབ་སྦྱོང་བྱེད་ལ།སྲོལ་རྒྱུན་ལྟར་ན་སློབ་གཉེར་བརྒྱུད་རིམ་ལ་དུས་ཚོད་ཀྱི་ཚོད་འཛིན་ནན་པོ་མེད། རྒྱ་བརྒྱུད་ནང་བསྟན་ཁྲོད་ཤུགས་རྐྱེན་ཆེ་ཤོས་དང་གཞི་ཁྱོན་ཆེ་ཤོས་ཡིན་པའི་བསམ་གཏན་པར་མཚོན་ན། ངེས་པར་བསྲུང་དགོས་པའི་སྡོམ་ཁྲིམས་དང་། ཚུལ་ཁྲིམས། དབྱར་གནས་ཀྱི་གཏན་འབེབས་བཅས་ཕུད། གཙོ་བོ་《ལང་ཀར་གཤེགས་པའི་མདོ་》དང་། 《རྡོ་རྗེ་གཅོད་པ》 《མེས་རབས་དྲུག་པའི་གསུང་མདོ་》སོགས་སློབ་གཉེར་བྱེད་པ་ཡིན་ལ་མདོ་དེ་དག་གི་ནང་དོན་ལྟར་སྒོམ་སྒྲུབ་བྱེད། གཞན་ཡང་ཐུན་མོང་མ་ཡིན་པའི་ཉམས་ལེན་བྱེད་སྟངས་ཤིག་ཡོད་ཅིང་དེས་སྒོམ་རྒྱག་རྒྱུ་གཙོ་བོར་འཛིན། སློབ་གཉེར་ཉམས་ལེན་གྱི་རིམ་པའི་ངོས་ནས་བཤད་ན།གྲུབ་མཐའ་ཁག་ཚང་མར་སློབ་གཉེར་ཉམས་ལེན་གྱི་རིམ་པའི་སྐོར་གྱི་གཞུང་ལུགས་ངེས་ཅན་ཞིག་ཡོད་དེ། དཔེར་ན་དམ་ཆོས་པད་དཀར་གྱི་གྲུབ་མཐའི་དཔྱད་འཇོག་ཟུང་འཇུག་གི་རིམ་པ་དང་། སེམས་ཙམ་པའི་གྲུབ་མཐའ་ཡི་སེམས་ཙམ་པའི་ལྟ་བ་ལྔའི་བཤད་པ་ལྟ་བུའོ། ། འོན་ཀྱང་ད་ཆ་གྲུབ་མཐའ་རང་ཉིད་དཀའ་ཁག་དང་ཕྲད་ཡོད་པས། གྲུབ་མཐའ་ཁ་ཤས་ཀྱི་སློབ་གཉེར་ཉམས་ལེན་གྱི་རིམ་པ་དེ་གཞི་རྩའི་ཆ་ནས་གཞུང་ལུགས་ཁོ་ན་ལས་ལག་ལེན་ཡོད་མཁན་ཧ་ཅང་དཀོན། ད་ལྟ་མ་ཉམས་པར་ཡོད་པ་ནི་སྙིང་པོ་དོན་བརྒྱུད་པའི་ཏིང་འཛིན་སྒོམ་པ་དང་། བདེ་ཆེན་ཞིང་སྒྲུབ་པའི་འདོན་སྒོམ་སོགས་ཀྱི་ཉམས་ལེན་ཙམ་མོ། །
        བོད་བརྒྱུད་ནང་བསྟན་ལ་སློབ་གཉེར་ལམ་ལུགས་ཚགས་དམ་པོ་ཡོད་ལ། གྲུབ་མཐའ་སོ་སོར་རང་རང་གི་ཁྱད་ཆོས་ལྡན། དེ་ལས་དགེ་ལུགས་པའི་སློབ་གཉེར་ལམ་ལུགས་ལ་དཔེ་མཚོན་རང་བཞིན་ལྡན་ཏེ། གྲུབ་མཐའ་དེའི་སློབ་གཉེར་སྒྲིག་ལམ་ནན་མོ་ཡིན་ལ། ལམ་གྱི་རིམ་པར་གཙིགས་ཆེན་བླ་ལྷག་ཏུ་བྱས་ནས་སྔོན་ལ་མདོ་ཕྱོགས་དང་དེའི་རྗེས་སུ་སྔགས་ཕྱོགས་སློབ་གཉེར་དང་ཉམས་ལེན་བྱེད་པ་ནན་སྲུང་བྱེད་ཅིང་། མདོ་ནི་ཀུན་གྱིས་བསླབ་ཏུ་རུང་བ་དང་། སྔགས་ནི་“སྣོད་རུང་”ཁོ་ནར་བརྒྱུད་པ་ཡིན་ནོ། །མདོ་ཕྱོགས་སུ། ངེས་པར་དུ་སྦྱང་དགོས་པ་གཞུང་བཀའ་པོད་ལྔ་སྟེ། མངོན་རྟོགས་རྒྱན་དང་། དབུ་མ་འཇུག་པ། ཚད་མ་རྣམ་འགྲེལ། མངོན་པ་མཛོད། འདུལ་བ་མདོ་རྩ་བཅས་སོ། ། དགེ་ལུགས་པའི་དགོན་པར་ཞུགས་པའི་དགེ་འདུན་པ་ཚོས་མདོ་ཕྱོགས་ཀྱི་བསླབ་ཚན་སྦྱངས་རྗེས་དགེ་བཤེས་ཀྱི་མཚན་གནས་བླངས་ཆོག་ལ། “དགེ་བཤེས་”ཞེས་པ་ནི་“དགེ་བའི་བཤེས་གཉེན་”ཏེ། མ་གཞི་ཡང་དག་པའི་ལམ་སྟོན་པའི་དགེ་རྒན་དང་གྲོགས་བཟང་པོར་གོ་ཞིང་།  ཕྱིས་སུ་དགོན་པར་ཆོས་རྩོད་དམ་བཅའ་བཞག་ནས་བླངས་པའི་བསླབ་གནས་མཚན་བཏགས་ལ་ཆེད་དུ་སྤྱོད་པ་ལྟར་ཆགས། ལྷ་སའི་དགོན་སྡེ་ཆེན་པོ་གསུམ་དུ་དགེ་བཤེས་ཀྱི་མཚན་གནས་ལ་རིམ་པ་བཞི་རུ་ཕྱེ་ཡོད་ལ། དེ་ལས་རིམ་པ་དང་པོ་ནི་དགེ་བཤེས་“ལྷ་རམས་པ་” སྟེ། ལྷ་སའི་བོད་ཟླ་དང་པོའི་ཆོ་འཕྲུལ་སྨོན་ལམ་ཆེན་མོའི་སྟེང་རྩོད་རྒྱུགས་བརྒྱུད་ནས་ལེན་པ་དང་། རིམ་པ་གཉིས་པ་ནི་དགེ་བཤེས་“ཚོགས་རམས་པ་”སྟེ། དགོན་པ་ཧྲིལ་པོའི་ཚོགས་རྒྱུགས་བརྒྱུད་ནས་ལེན། རིམ་པ་གསུམ་པ་ནི་“གླིང་བསྲེ་”སྟེ། དགོན་པའི་གྲྭ་ཚང་ཁག་ནས་ལེན། རིམ་པ་བཞི་པ་ནི་དགེ་བཤེས་“རྡོ་རམས་པ་”སྟེ། ལྷ་ཁང་གི་སྒོ་མདུན་རྡོ་གཅལ་ཐང་དུ་ཆོས་རྩོད་བརྒྱུད་ནས་ལེན་པའོ། ། གསང་སྔགས་ཀྱི་ཕྱོགས་སུ། བརྒྱུད་རིམ་ལྡན་པའི་སྒོ་ནས་བྱ་རྒྱུད་དང་། སྤྱོད་རྒྱུད། རྣལ་འབྱོར་རྒྱུད། རྣལ་འབྱོར་བླ་མེད་རྒྱུད་བཅས་རྒྱུད་སྡེ་བཞི་ལ་སྦྱང་བ་བྱེད་དགོས། ལྷ་སའི་རྒྱུད་སྟོད་སྨད་གླིང་དང་འབྲས་སྤུངས་སེ་ར་སོགས་དགོན་སྡེ་ཆེ་ཁག་གི་རྒྱུད་སྟོད་སྨད་ནི་དགེ་ལུགས་པའི་གསང་སྔགས་དར་སྤེལ་གཏོང་སའི་རྩ་བའི་ཆོས་ར་ཡིན། དུས་མཚུངས་སུ། ཚད་མ་རིག་པ་ལ་སློབ་སྦྱོང་དང་ཞིབ་འཇུག་དང་བཀོལ་སྤྱོད་བཅས་བྱེད་པ་ནི་བོད་བརྒྱུད་ནང་བསྟན་གྱི་ཁྱད་ཆོས་མངོན་གསལ་དོད་པོ་ཞིག་རེད། གཞུང་ཆེན་བཀའ་པོད་ལྔའི་ནང་གི་《ཚད་མ་རྣམ་འགྲེལ་》ནི་ཚད་མ་རིག་པའི་གཞུང་ཡིན་པ་དང་། ཚད་མ་རིག་པ་ནི་ནང་བསྟན་གྱིས་མཚན་ཉིད་རིག་པའི་བསམ་བློ་གསེད་འགྲེལ་བྱེད་པའི་རྣམ་པ་དང་བྱེད་ཐབས་ཡིན་ལ། རྩོད་གླེང་གི་ཤེས་རབ་རིག་པའང་ཡིན། དེའི་བྱེད་ཐབས་གཙོ་བོ་ནི་རང་ལུགས་འཐད་ལྡན་འཇོག་པ་དང་དེར་སྒྲུབ་བྱེད་འགོད་པ་ཡིན་ལ། དོན་གྱི་འཁྲུལ་བ་ཚར་གཅོད་པའི་རིག་པ་ཞིག་ཡིན་སྟབས། ནང་བསྟན་གྱི་ཁྲོད་དུ་གོ་གནས་གལ་ཆེན་བཟུང་ཡོད། ལྷག་པར་བོད་བརྒྱུད་ནང་བསྟན་ཁྲོད་དུ་ཚད་མ་རིག་པ་ནི་ནང་དོན་རིག་པའི་བསླབ་གཞི་དང་རྩོད་གླེང་བྱེད་ཐབས་གལ་ཆེན་ཞིག་ཡིན་ཞིང་། དགོན་སྡེའི་ནང་ཡོངས་ཁྱབ་ཏུ་ཚད་མ་རིག་པའི་རྩོད་གླེང་བྱེད་ཐབས་སྤྱད་དེ་ཆོས་གཞུང་ལ་སློབ་གཉེར་བྱེད་པ་རེད།
        རྒྱ་བོད་གཉིས་ཀྱི་ནང་བསྟན་པའི་གོ་གནས་གཙོ་བོ་རྒྱུད་འཛིན་བྱེད་སྟངས་དེ་དང་ཐོག་བླ་སློབ་རྒྱུད་འཛིན་གཙོར་བྱེད་པ་ཡིན་མོད། ལོ་རྒྱུས་ཀྱི་འཕེལ་ཕྱོགས་དང་བསྟུན་ནས་བོད་བརྒྱུད་ནང་བསྟན་ལ་རང་ཉིད་ཀྱི་ཐུན་མོང་མ་ཡིན་པའི་སྤྲུལ་སྐུའི་ཡང་སྲིད་ངོས་འཛིན་ལམ་ལུགས་ཆགས་ཏེ། བོད་བརྒྱུད་ནང་བསྟན་གྱི་ཆོས་ཀྱི་གོ་ཐོབ་ལམ་ལུགས་ལ་དམིགས་བསལ་གྱི་ནང་དོན་ཞིག་བསྣན་པ་རེད། དེ་ནི་བོད་བརྒྱུད་ནང་བསྟན་དེ་རྒྱ་བརྒྱུད་ནང་བསྟན་དང་ལྷོ་བརྒྱུད་ནང་བསྟན་དང་མི་འདྲ་བའི་ཁྱད་རྟགས་གལ་ཆེན་ཞིག་ཏུའང་གྱུར་ཡོད། སྤྲུལ་སྐུའི་ཡང་སྲིད་ངོས་འཛིན་ལམ་ལུགས་ནི་བོད་བརྒྱུད་ནང་བསྟན་གྱིས་གསར་གཏོད་བྱས་པ་ཞིག་ཡིན་ཞིང་། ནང་བསྟན་གྱི་རྟེན་འབྲེལ་རིག་པ་དང་“སྤྲུལ་སྐུ་”སོགས་ཀྱི་གཞུང་ལུགས་དང་། བོད་བརྒྱུད་ནང་བསྟན་དགོན་པའི་ཚོགས་པའི་ཁེ་ཕན་དང་ཟུང་འབྲེལ་བྱས་ནས་གྲུབ་པའི་ཆོས་ལུགས་ཀྱི་ལམ་ལུགས་ཤིག་རེད། བྱེད་ཐབས་དེ་ཐོག་མར་ཀརྨ་བཀའ་བརྒྱུད་ཀྱི་ཀརྨ་པ་ནས་བྱུང་། ཡོན་རྒྱལ་རབས་ཀྱི་གོང་མ་ཞན་ཙུང་སྐུ་གཤེགས་རྗེས་ཀྱི་རྒྱལ་ཁྲིའི་འཐབ་རྩོད་ཁྲོད། ཀརྨ་བཀའ་བརྒྱུད་པའི་སྐྱེས་ཆེན་ཀརྨ་པཀྴི་(1204-1283)ལ་བརྩི་མཐོང་ཤོར་ཏེ། 1264ལོར་ཀརྨ་བཀའ་བརྒྱུད་ཀྱི་མཚུར་ཕུ་དགོན་དུ་ལོག་པ་དང་1283ལོར་སྐུ་གཤེགས། 1288ལོར་ཨོ་རྒྱན་པ་རིན་ཆེན་དཔལ་གྱིས་ཀརྨ་པཀྵིའི་ཤུལ་བཞག་དངོས་ཆས་རྩད་གཅོད་ངོས་འཛིན་སོགས་ཀྱི་བྱེད་ཐབས་བརྒྱུད་དེ། དེང་དུས་ཀྱི་བོད་ལྗོངས་སྐྱིད་གྲོང་རྫོང་དུ་འཁྲུངས་པའི་རང་བྱུང་རྡོ་རྗེ་(1284-1339)ཀརྨ་པཀྴིའི་ཡང་སྲིད་དུ་གཏན་འཁེལ་བྱས་པས།④ རང་བྱུང་རྡོ་རྗེ་ནི་བོད་བརྒྱུད་ནང་བསྟན་ལོ་རྒྱུས་ཐོག་གི་སྤྲུལ་སྐུ་དང་པོར་གྱུར་ལ། དེ་ནས་སྤྲུལ་སྐུའི་ཡང་སྲིད་ངོས་འཛིན་ལམ་ལུགས་ཆགས་པ་དང་བོད་བརྒྱུད་ནང་བསྟན་ཁོ་ནར་ཡོད་པའི་ཐུན་མོང་མ་ཡིན་པའི་རྒྱུད་འཛིན་བྱེད་སྟངས་སུ་གྱུར། 
        རྩ་འཛུགས་བྱེད་སྟངས་དང་། ཆོས་ཀྱི་ལྟ་གྲུབ། སློབ་གཉེར་ལམ་ལུགས། རྒྱུད་འཛིན་བྱེད་སྟངས། རིག་གནས་གོམས་སྲོལ་སོགས་ཀྱི་ཕྱོགས་སུ་ཐུན་མོང་མ་ཡིན་པའི་ཁྱད་ཆོས་ལྡན་པའི་དབང་གིས་བོད་བརྒྱུད་ནང་བསྟན་དེ་ཀྲུང་ཧྭའི་རིག་གནས་ཁྲོད་ཀྱི་ཐུན་མིན་ཁྱད་ཆོས་ལྡན་པའི་རིག་གནས་སུ་གྱུར་པ་རེད།
གསུམ་པ། རྒྱ་བོད་ནང་བསྟན་གྱི་རིག་གནས་ཕན་ཚུན་སྤེལ་རེས་བྱས་ཏེ་ནང་བསྟན་ཀྲུང་གོ་ཅན་དུ་འགྱུར་བའི་འཕེལ་རིམ་ལ་སྐུལ་འདེད་བཏང་བ།
        སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིང་ཕིང་གིས་ཀྲུང་དབྱང་འཐབ་ཕྱོགས་གཅིག་གྱུར་ལས་དོན་ཚོགས་འདུའི་ཐོག་ནན་བཤད་གནང་དོན། ཆོས་ལུགས་སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་སྤྱི་ཚོགས་དང་མཐུན་འཇུག་ཡོང་བར་ཁྲིད་སྟོན་ཧུར་ཐག་བྱེད་པ་ལ།ཀྲུང་གོ་ཅན་དུ་འགྱུར་བའི་ཁ་ཕྱོགས་ངེས་པར་རྒྱུན་འཁྱོངས་བྱེད་དགོས་གསུངས། སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་སྤྱི་ཚོགས་དང་མཐུན་འཇུག་ཡོང་བའི་ཀྲུང་གོ་ཅན་གྱི་ཁ་ཕྱོགས་དང་། ལོ་རྒྱུས་ཐོག་གི་ནང་བསྟན་ཀྲུང་གོ་ཅན་དང་། ཀྲུང་གོ་ཅན་གྱི་ནང་བསྟན་དང་། ནང་བསྟན་རུའུ་ལུགས་ཅན་སོགས་ཀྱི་དབར་འབྲེལ་བ་ཡོད་ལ་ཁྱད་པར་ཡང་ཡོད། བོད་བརྒྱུད་ནང་བསྟན་དེ་སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་སྤྱི་ཚོགས་དང་མཐུན་པའི་ཀྲུང་གོ་ཅན་གྱི་ཁ་ཕྱོགས་དེ་གཙོ་བོ་ཀྲུང་གོ་ཅན་གྱི་ནང་བསྟན་འཕེལ་རྒྱས་ཀྱི་རྨང་གཞིའི་ཐོག བོད་བརྒྱུད་ནང་བསྟན་གྱི་རྩ་བའི་གཞུང་ལུགས་དང་། ཀྲུང་གོར་དར་སྤེལ་བྱས་པའི་ལག་ལེན། ཀྲུང་གོའི་ལོ་རྒྱུས་སྲོལ་རྒྱུན། མི་རིགས་རིག་གནས། སྲོག་ཤིང་གི་རིན་ཐང་ལྟ་བ་སོགས་ཟུང་འབྲེལ་གྱིས་དུས་རབས་ཅན་དང་མི་རིགས་ཅན་གྱི་ཁྱད་རྟགས་ལྡན་ལ་ཀྲུང་གོའི་དོན་དངོས་དང་མཐུན་པ་དང་ཀྲུང་གོའི་ཁྱད་ཆོས་ལྡན་པའི་བོད་བརྒྱུད་ནང་བསྟན་འཕེལ་རྒྱས་ཀྱི་བགྲོད་ལམ་ཆགས་པར་བྱེད་དགོས། ཟུར་ངོས་གང་ནས་བལྟས་ཀྱང་། ནང་བསྟན་གྱི་ཀྲུང་གོ་ཅན་ནམ་ཀྲུང་གོ་ཅན་གྱི་ནང་བསྟན་ནི་ཀྲུང་ཧྭའི་རིག་གནས་ཀྱིས་འཛམ་གླིང་གི་དཔལ་ཡོན་ལ་བཞག་པའི་མཛད་རྗེས་གལ་ཆེན་རེད། ཀྲུང་གོའི་རྒྱ་བོད་ནང་བསྟན་སོགས་སྐད་རྒྱུད་ཆེན་པོ་གསུམ་གྱི་ནང་བསྟན་གྱིས་ཕན་ཚུན་བརྗེ་སྤེལ་འབྲེལ་འདྲིས་གོང་སྤེལ་བཅས་ཀྱི་སྒོ་ནས་ནང་བསྟན་ཀྲུང་གོ་ཅན་དུ་འགྱུར་བའི་འཕེལ་རིམ་ལེགས་འགྲུབ་བྱས་པ་རེད། 
        རྒྱ་བརྒྱུད་ནང་བསྟན་ནི་བོད་བརྒྱུད་ནང་བསྟན་གྱི་འབྱུང་ཁུངས་གཙོ་བོའི་གྲས་ཤིག་རེད། འུན་ཤིང་ཀོང་ཇོ་བོད་དུ་ཕེབས་པ་ནས་རྒྱ་བརྒྱུད་ནང་བསྟན་བོད་དུ་དར་བ་ཡིན་ཏེ། སྤྱི་ལོ་དུས་རབས་བདུན་པའི་དུས་དཀྱིལ་དུ་འུན་ཤིང་ཀོང་ཇོ་བོད་དུ་ཕེབས་དུས་སངས་རྒྱས་ཤཱཀྱ་ཐུབ་པ་དགུང་གྲངས་བཅུ་གཉིས་ལ་ཕེབས་པའི་སྐུ་ཚད་ཅན་གྱི་སྐུ་བརྙན་དང་ཆོས་དཔེ་སུམ་བརྒྱ་དྲུག་ཅུ་བོད་དུ་ཁྱེར་ཡོད་ལ།  དེ་ལས་ཇོ་བོ་ཤཱཀྱ་མུ་ནེའི་སྐུ་བརྙན་ད་ལྟའི་བར་དུ་ལྷ་ལྡན་གཙུག་ལག་ཁང་དུ་བཞུགས་སུ་གསོལ་ཅིང་། བོད་བརྒྱུད་ནང་བསྟན་གྱི་སྐུ་བརྙན་རྩ་ཆེ་ཤོས་དང་བོད་རིགས་དད་ལྡན་མང་ཚོགས་ཀྱིས་དད་གུས་ཆེ་ཤོས་བྱེད་པའི་སྐུ་བརྙན་དུ་གྱུར་ཡོད། འུན་ཤིང་ཀོང་ཇོས་སྐུ་བརྙན་དང་ཆོས་དཔེ་ཁྱེར་ཡོང་བ་ཕུད། ད་དུང་ལྷ་ཁང་བཞེངས་ལུགས་དང་དགོན་སྡེའི་ཁྲིམས་སྲོལ་ཁྱེར་ཡོང་བ་དང་། རྒྱ་ཡིག་གི་ཆོས་གཞུང་བོད་ཡིག་ཏུ་ལོ་ཙཱ་བྱེད་པའི་བྱ་གཞག་གི་དབུ་བརྙེས།⑤ བོད་བཙན་པོའི་རྒྱལ་རབས་སུ་ནང་བསྟན་ཐོག་མར་དར་སྐབས།རྒྱ་བརྒྱུད་ནང་བསྟན་གྱི་སྐུ་བརྙན་དང་། ཆོས་དཔེ། ཆོས་གཞུང་ལོ་ཙཱ། དགོན་པ་བཞེངས་སྐྲུན། རྒྱའི་བཙུན་པ། རྒྱའི་དགོན་སྡེའི་སྒྲིག་སྲོལ་བཅས་སྔ་རྗེས་སུ་བོད་བཙན་པོའི་རྒྱལ་རབས་སུ་སླེབས་ནས། བཙན་པོའི་རྒྱལ་རབས་ཀྱི་ནང་བསྟན་གྱི་འཕེལ་རྒྱས་ལ་ཤུགས་རྐྱེན་ཆེན་པོ་ཐེབས། དེ་ལས་བོད་བཙན་པོའི་རྒྱལ་རབས་ཀྱི་ནང་བསྟན་དེ་དང་པོ་ཉིད་ནས་རྒྱ་བརྒྱུད་ནང་བསྟན་དང་འབྲེལ་བ་དམ་ཟབ་ཡོད་པའང་མངོན་པར་མཚོན་ཡོད།
        སྤྱི་ལོ་710པར་ཁྲི་ལྡེ་གཙུག་བཙན་གྱིས་གྱིམ་ཤིང་ཀོང་ཇོ་ཁབ་ཏུ་བཞེས་དུས། ཐང་ཀྲུང་ཙུང་སྐུ་ངོ་མས་ཧྲི་ཕིང་(ད་ལྟའི་ཧྲའན་ཞི་ཞིང་ཆེན་ཞིང་ཕིང་རྫོང)དུ་བསྐྱལ་ཅིང་། དར་གོས་དང་སྒྱུ་རྩལ་པ་དང་བཟོ་རྩལ་པ་བརྒྱ་དང་། ཆིའུ་ཚིའི་རོལ་ཆ་བཅས་རྫོངས་སྐལ་དུ་སྤྲད་པ་དང་། གོང་མའི་ལུང་གིས་ཝེ་གཡོན་པའི་དམག་དཔོན་ཆེན་མོ་དབྱང་ཅུ་ཞེས་པ་གསེར་ཡིག་པར་བསྐོས་ཏེ་ཀོང་ཇོ་བོད་ལ་སྐྱེལ་དུ་བཅུགབཙན་པོས་ཀོང་ཇོར་སྡོད་མཁར་གསར་དུ་བརྩིགས། གྱིམ་ཤིང་ཀོང་ཇོས་བཟོ་རྩལ་པ་བརྒྱ་ཕྲག་ཁྲིད་ཅིང་། ཐུགས་ཞིབ་མོའི་སྒོ་ནས་བཟོ་རྩལ་མི་སྣར་ལག་རྩལ་སྣ་ཚོགས་སྤེལ་དུ་བཅུག་པས། བོད་ཀྱི་ཞིང་ལས་ཐོན་སྐྱེད་དང་། ལྕགས་རིགས་བཞུ་སྦྱོང་། ཨར་ལས། འཁེལ་འཐག ཆང་སྙལ་བ།ཤོག་བཟོ། རོལ་དབྱངས་སོགས་ལས་རིགས་ཁག་གི་ལག་རྩལ་ཚང་མར་འཕེལ་རྒྱས་བྱུང་། གྱིམ་ཤིང་ཀོང་ཇོས་བཙན་པོས་ནང་བསྟན་སྤེལ་བར་གཞོགས་འདེགས་ཧུར་ཐག་བྱས་པ་སྟེ། ངག་རྒྱུན་ལྟར་ན་ཀོང་ཇོས་མི་མངགས་ནས་སངས་རྒྱས་ཤཱཀྱ་ཐུབ་པ་དགུང་གྲངས་བཅུ་གཉིས་ལ་ཕེབས་པའི་སྐུ་ཚད་ཅན་གྱི་སྐུ་བརྙན་སྔོན་ས་འོག་ཏུ་སྦས་པ་བཏོན་ཏེ་ལྷ་ལྡན་གཙུག་ལག་ཁང་དུ་བཞུགས་སུ་གསོལ་བ་དང་། མོ་དང་མཉམ་པོར་བོད་དུ་ཡོང་བའི་ནང་ལོགས་ས་ཁུལ་གྱི་དགེ་འདུན་པར་དགོན་པར་དོ་དམ་བྱེད་དུ་བཅུག་པར་གྲགས། གཞན་ཡང་བཙན་པོར་བོད་ནས་ནང་ལོགས་ས་ཁུལ་དུ་ཆོས་སྦྱོང་དུ་མི་མངག་རྒྱུ་དང་། ནང་ལོགས་ས་ཁུལ་གྱི་དགེ་འདུན་པ་གདན་འདྲེན་རྒྱུ། མི་སྣ་རྩ་འཛུགས་བྱས་ནས་ཆོས་དཔེ་དང་། གནམ་རིག སྐར་དཔྱད། སྐར་རྩིས། གསོ་རིག་སོགས་ཀྱི་དཔེ་དེབ་ལོ་ཙཱ་བྱ་རྒྱུ་སོགས་ཀྱི་གྲོས་འགོ་བཏོན། དུས་མཚུངས་སུ་རང་གིས་བསྣམས་ཡོང་བའི་《མའོ་ཧྲི》 《ཙུའོ་ཀྲོན》 《ཁྲུན་ཆིའུ་》སོགས་རྒྱའི་གསུང་རབ་བོད་དུ་སྤེལ།⑥ ད་དུང་ཆེད་དུ་ནང་ལོགས་ས་ཁུལ་དུ་མི་མངགས་ཏེ་སྔོན་ཐོན་རིག་གནས་སློབ་སྦྱོང་བྱེད་དུ་བཅུག་པས། རྒྱ་བོད་ནང་བསྟན་རིག་གནས་ཕན་ཚུན་བརྗེ་སྤེལ་འབྲེལ་འདྲིས་བྱེད་པ་དང་། ནང་བསྟན་ཀྲུང་གོ་ཅན་དུ་འགྱུར་བའི་འཕེལ་རིམ་ལ་སྔར་བས་སྐུལ་འདེད་བཏང་བར་མ་ཟད། རྒྱ་བོད་རིག་གནས་མཉམ་འདྲེས་ལ་ནུས་པ་བླ་ལྷག་བཏོན་པ་རེད།
        ཧྭ་ཤང་མ་ཧཱ་ཡཱ་ན་(750-830？)ཡིས་བོད་དུ་ནང་ལོགས་ས་ཆའི་སྙིང་པོ་དོན་བརྒྱུད་པའི་གྲུབ་མཐའ་ཁྱབ་སྤེལ་བྱས་ནས། སྙིང་པོ་དོན་བརྒྱུད་པ་བོད་དུ་འཕེལ་རྒྱས་ཅུང་ཆེན་པོ་འབྱུང་དུ་བཅུག་པས་བོད་དུ་བསམ་གཏན་པའི་ཆོས་བརྒྱུད་ཆགས་པར་བྱས་ཤིང་། དེས་རྙིང་མ་པའི་རྫོགས་ཆེན་དང་བཀའ་བརྒྱུད་པའི་ཕྱག་རྒྱ་ཆེན་པོར་ཤུགས་རྐྱེན་ལྷག་ཏུ་ཆེ་བ་ཐེབས་ཤིང་། ས་སྐྱ་དང་ཇོ་ནང་ལའང་དེའི་ཤུགས་རྐྱེན་ཐེབས་ཡོད། བླ་ཆེན་དགོངས་པ་རབ་གསལ་(892-975)གྱིས་སྡོམ་པ་བཞེས་པ་དང་བོད་བརྒྱུད་ནང་བསྟན་གྲུབ་པའི་དབར་འབྲེལ་བ་དམ་པོ་ཡོད་དེ། བསྟན་པ་ཕྱི་དར་སྐབས་ཀྱི་མཁས་པ་མི་གསུམ་སྟེ་གཙང་རབ་གསལ་དང་གཡོ་དགེ་འབྱུང་། དམར་ཤཱཀྱ་མུ་ནེ་བཅས་ཀྱིས་བླ་ཆེན་དགོངས་པ་རབ་གསལ་ལ་སྡོམ་པ་གནང་སྐབས། དགེ་སློང་ལྔ་སྡེའི་གྲངས་མ་ཚང་བས་མཚོ་སྔོན་བ་ཡན་ཤར་ཕྱོགས་ཀྱི་མཐུ་བ་དགོན་ནས་རྒྱ་རིགས་ཀྱི་དགེ་སློང་ཀེ་ཝང་དང་གྱི་བན་གཉིས་གདན་དྲངས་ཏེ་བླ་ཆེན་ལ་དགེ་སློང་གི་སྡོམ་པ་བསྩལ་བས་སྡོམ་རྒྱུན་མ་ཆད་པར་མཐུད་ཐུབ་པ་བྱུང་།⑦ བོད་བརྒྱུད་ནང་བསྟན་གྱི་ལྷ་ཚོགས་ནི་བོད་བརྒྱུད་ནང་བསྟན་གྱི་དད་ལྡན་མང་ཚོགས་ཀྱིས་མཆོད་འབུལ་བྱ་ཡུལ་ཡིན་ཞིང་། དེ་རུ་སྲོལ་རྒྱུན་གྱི་བླ་མ་ཡི་དམ་སངས་རྒྱས་བྱང་སེམས་ཆོས་སྐྱོང་སོགས་ལས་གཞན་ད་དུང་ནང་ལོགས་ས་ཁུལ་གྱི་ལོ་རྒྱུས་མི་སྣ་ཁག་ཅིག་སྟེ། དཔེར་ན་ཁུང་ཙི་དང་། ཀོན་ཡུས། ཞོན་ཙང་།ཧྭ་ཤང་དགེ་སྲུང་། ཡུའེ་ཧྥེ། ལིའུ་ཆི་(མིང་རྒྱལ་རབས་ཀྱི་དཔོན་རིགས་ཁོང་བོད་ས་གནས་སྲིད་གཞུང་གིས་ནོར་སྲིད་དང་ཉོ་ཚོང་དཔོན་དུ་བསྐོས)སོགས་ཀྱང་དེའི་ནང་ཚུད་ཡོད་ཅིང་། དེ་དག་ལ་བོད་བརྒྱུད་ནང་བསྟན་གྱི་དད་ལྡན་མང་ཚོགས་ཀྱིས་བརྩི་བཀུར་དང་དད་གུས་ཆེན་པོ་བྱེད་པ་རེད། 
        རྒྱ་བོད་ནང་བསྟན་རིག་གནས་ཕན་ཚུན་བརྗེ་སྤེལ་བྱས་པ་དེས་རྒྱ་བོད་ནང་བསྟན་ཕན་ཚུན་སྡུད་ལེན་དང་མཉམ་འདྲེས་ཡོང་བར་བྱས་པར་མ་ཟད། དེ་གཉིས་ཀྲུང་ཧྭའི་སྲོལ་རྒྱུན་རིག་གནས་དང་དམ་པོར་སྦྲེལ་དུ་བཅུག་ནས་ཀྲུང་གོའི་ཐུན་མིན་ཁྱད་ཆོས་ལྡན་པའི་བསམ་བློའི་མ་ལག་དང་། དད་མོས་མ་ལག རྩ་འཛུགས་མ་ལག སློབ་གཉེར་མ་ལག རྒྱུད་འཛིན་མ་ལག་སོགས་གྲུབ་པར་བྱས་ཤིང་། ཐུན་མོང་ཐོག་ནང་བསྟན་ཀྲུང་གོ་ཅན་དུ་འགྱུར་བའི་འཕེལ་རིམ་ལ་སྐུལ་འདེད་དང་ལེགས་འགྲུབ་ཡོང་བར་བྱས་སོ། །⑧

        བོད་བརྒྱུད་ནང་བསྟན་གྱི་ཆོས་ལུགས་དུས་ཆེན་དེ་དམངས་སྲོལ་དུས་ཆེན་རིགས་ཤིག་ཏུ་གྱུར་ནས། བོད་ཀྱི་དམངས་སྲོལ་རིག་གནས་ལ་ཐུན་མོང་མ་ཡིན་པའི་ནང་དོན་ལྡན་པར་བྱས་པ་སྟེ། དབྱར་གནས་མཇུག་བསྒྲིལ་ཏེ་གྲོང་ལ་ཡོང་བའི་དགེ་འདུན་པ་བསུ་བ་ཞོ་སྟོན་དུ་གྲུབ་པ་དང་། སངས་རྒྱས་ཤཱཀྱ་ཐུབ་པ་སྐུ་བལྟམས་པ་དང་། སངས་རྒྱས་པ། སྐུ་མྱ་ངན་ལས་འདས་པ་བཅས་ལ་དྲན་གསོ་ཞུ་བའི་ས་ག་ཟླ་བ་དང་། སངས་རྒྱས་ཤཱཀྱ་ཐུབ་པས་ཆོ་འཕྲུལ་སྣ་ཚོགས་བསྟན་ནས་མུ་སྟེགས་པ་རྣམས་བཏུལ་བར་དྲན་གསོ་ཞུ་བའི་ཆོ་འཕྲུལ་དུས་ཆེན་སོགས་དམངས་སྲོལ་རང་བཞིན་གྱི་དུས་ཆེན་རིག་གནས་སུ་གྱུར་ཡོད་ལ། རྗེ་ཙོང་ཁ་པ་སྐུ་ཞི་བར་གཤེགས་པར་དྲན་གསོ་ཞུ་བའི་ལྔ་མཆོད་དུས་ཆེན་སོགས་དམངས་སྲོལ་གྱི་རང་བཞིན་ལྡན་པའི་བྱེད་སྒོ་ཞིག་ཏུ་གྱུར་ཡོད། 
        རྒྱུན་ལྡན་གྱི་འཚོ་བའི་ཕྱོགས་སུ། མི་ཚེའི་གུས་ལུགས་གོམས་སྲོལ་ནང་དུ་བོད་བརྒྱུད་ནང་བསྟན་གྱི་ནང་དོན་ཐིམ་ཡོད་པ་སྟེ། སྐྱེ་བོ་མང་པོས་ནངས་སྔ་མོ་ནས་དགོང་ཕྱི་དྲོའི་བར་དང་། སྐྱེ་བ་ནས་འཆི་བའི་བར་དུ་བྱ་བ་ཅི་བྱེད་ཀྱང་ནང་བསྟན་དང་མ་འབྲེལ་བ་མེད་པ་ལྟ་བུ་ཆགས་ཡོད་དེ། དཔེར་ན། མི་ལ་ལས་བླ་གྲྭ་ནི་རང་ཉིད་སངས་རྒྱས་ཀྱི་ལམ་ལ་འཁྲིད་མཁན་ཡིན་པ་དང་། ཤི་གསོན་གྱི་གནས་སྐབས་གང་དུའང་རང་ལ་དེའི་སྣེ་ཤན་མཁོ་བར་སྙམ་གྱི་ཡོད། མི་ལ་ལའི་སེམས་ནང་དུ་བླ་གྲྭ་ནི་སངས་རྒྱས་ཆོས་དགེ་འདུན་བཅས་དཀོན་མཆོག་གསུམ་གྱི་ཡ་གྱལ་ཡིན་པས་ཚང་མས་བརྩི་བཀུར་དང་དད་གུས་བྱེད་ཀྱི་ཡོད། བོད་བརྒྱུད་ནང་བསྟན་གྱི་ཤུགས་རྐྱེན་ཐེབས་པའི་དབང་གིས་མི་ལ་ལར་སྐྱེས་པ་ནས་འཆི་བའི་བར་བླ་གྲྭའི་རོགས་རམ་མཁོ་བ་སྟེ། བུ་ཕྲུག་སྐྱེས་མ་ཐག་བླ་གྲྭས་མིང་འདོགས་དགོས་པ་དང་། བུ་ཕྲུག་ནར་སོན་ཏེ་ཆང་ས་རྒྱག་དུས་བླ་མས་ཕྱག་མོ་གཟིགས་དགོས་ལ། ནམ་རྒྱུན་ན་ཚ་ཕོག་དུས་ཀྱང་བླ་གྲྭར་སྨན་བཅོས་བྱེད་དུ་འཇུག་ཅིང་། ཕྱི་རུ་དོན་དག་བསྒྲུབ་ཏུ་འགྲོ་དུས་ཀྱང་བླ་གྲྭར་ཕྱག་མོ་གཟིགས་སུ་འཇུག་པ། ཚེ་ལས་འདས་རྗེས་ལྷག་ཏུ་བླ་གྲྭ་གདན་དྲངས་ནས་རྣམ་ཤེས་འདྲེན་དགོས་པ་རེད། མང་ཚོགས་ཀྱི་རྒྱུན་ལྡན་གྱི་ཐོན་སྐྱེད་འཚོ་བར་ཡང་བོད་བརྒྱུད་ནང་བསྟན་གྱི་ཤུགས་རྐྱེན་ཐེབས་ཡོད་དེ། མི་ལ་ལས་ཟས་མ་ཟོས་སྔོན་ལ་མཆོད་པ་འདོན་པ་དང་། ཐག་རིང་ལ་འགྲོ་སྐབས་གསོལ་བ་འདེབས་པ། རྨོ་འདེབས་བྱེད་སྐབས་རིམ་གྲོ་སྒྲུབ་པ། འབྲུ་རིགས་བསྡུ་སྐབས་གསོལ་མཆོད་བྱེད་པ་རེད།   ས་ཆ་ལ་ལར་ནམ་རྒྱུན་གྱི་ཐོན་སྐྱེད་ཁྲོད་ཆར་འབེབ་པ་དང་སེར་བ་འགོག་པ། ལོ་ལེགས་ཡོང་བར་སྨོན་ལམ་འདེབས་པ་སོགས་ཀྱི་བྱེད་སྒོ་ལའང་བླ་གྲྭ་གདན་འདྲེན་བཞིན་ཡོད། མི་ལ་ལ་འཚོ་བའི་ཐད་དཀའ་ངལ་སྣ་ཚོགས་ལ་འཕྲད་སྐབས་བཅའ་བ་རིམ་གྲོ་སྒྲུབ་པའམ་ཡང་ན་གྲྭ་བཙུན་ལ་སྐྱབས་བཅོལ་ཞུ་བ་དང་། ཐ་ན་གྱོད་གཞི་བྱུང་སྟེ་མི་སྐྱ་ཚོས་སྡུམ་ཐབས་བྲལ་བའི་ཚེ་སྐབས་འགར་བླ་མས་སྡུམ་གྱི་ཡོད། དེ་བས་བོད་བརྒྱུད་ནང་བསྟན་ནི་བོད་རིགས་ཀྱི་དམངས་སྲོལ་རིག་གནས་ཀྱི་ཕྱོགས་སོ་སོར་སིམ་ཡོད་པ་རེད།
        གཉིས། མཆོག་གྱུར་རང་བཞིན་དང་ཆབ་སྲིད་རང་བཞིན་ཟུང་འབྲེལ་བྱས་ཏེ་ལམ་ལུགས་རིག་གནས་ལ་ཆོས་ལུགས་ཀྱི་རྟགས་མཚན་བཀོད་པ།
        ལོ་རྒྱུས་སྟེང་ཆོས་ལུགས་དང་། ཆབ་སྲིད་དང་དཔལ་འབྱོར་བར་གྱི་འབྲེལ་བའི་“ཆོས་གཉིས་གཞི་གཅིག་”རང་བཞིན་དང་། བོད་བརྒྱུད་ནང་བསྟན་དང་བོད་ཀྱི་སྲོལ་རྒྱུན་གྱི་སྤྱི་ཚོགས་སྒྲིག་གཞི་ཟུང་འབྲེལ་དམ་ཟབ་བྱས་པས། དཔེ་མཚོན་རང་བཞིན་ལྡན་པའི་“ཆོས་སྲིད་ཟུང་འབྲེལ་”གྱི་ལམ་ལུགས་ཆགས་པ་རེད། ལམ་ལུགས་དེར་ཁྱད་ཆོས་གལ་ཆེན་ཞིག་ཡོད་པ་ནི་བོད་ས་གནས་ཀྱིས་ཆབ་སྲིད་དང་། དཔལ་འབྱོར། སྤྱི་ཚོགས། དམག་དོན། བཅའ་ཁྲིམས། དམངས་ཁྲོད་ལས་དོན་སོགས་ཐག་གཅོད་བྱེད་དུས་ནང་བསྟན་གྱི་བསྟན་དོན་གཞི་རྩའི་ཚད་གཞིར་འཛིན་པར་མ་ཟད། ཆོས་ལུགས་ཀྱི་གཙོ་འཛིན་གྱི་བཀའ་བརྩི་བ་དང་ནང་བསྟན་གྱི་བསྟན་དོན་བརྩི་སྲུང་ཞུ་རྒྱུ་དེ་རྩ་དོན་མཐོ་ཤོས་སུ་འཛིན་གྱི་ཡོད། དུས་མཚུངས་སུ་བོད་བརྒྱུད་ནང་བསྟན་གྱི་སྡོམ་ཁྲིམས་གཞིར་བཟུང་ནས་གཏན་ལ་ཕབ་པའི་བོད་རྙིང་པའི་བཅའ་ཁྲིམས་དང་ཁྲིམས་སྲོལ་ལའང་ཆོས་ལུགས་ཀྱི་ཁྱད་ཆོས་ལྡན་པ་རེད། བོད་ས་གནས་སྲིད་གཞུང་རྙིང་པའི་བཅའ་ཁྲིམས་ནི་བཀས་བཀོད་རྒྱུད་འཛིན་མངའ་བདག་ཚོགས་པའི་གྲལ་རིམ་ཁེ་ཕན་དང་གྲལ་རིམ་དབང་སྒྱུར་སྲུང་སྐྱོང་བྱེད་པའི་ལག་ཆ་གལ་ཆེན་ཞིག་རེད། “ཆོས་ཁྲིམས་”དང་འཇིག་རྟེན་པའི་ཁྲིམས་ལུགས་ཟུང་འབྲེལ་བྱས་ནས་གྲུབ་པའི་“ཆོས་སྲིད་ཟུང་འབྲེལ་”གྱི་བཅའ་ཁྲིམས་ཆ་ཚང་སྟེ།《ཞལ་ལྕེ་བཅོ་ལྔ་》དང་། 《ཞལ་ལྕེ་བཅུ་གསུམ》 《ཞལ་ལྕེ་བཅུ་དྲུག་》སོགས་ཀྱིས་མངའ་བདག་ཚོགས་པའི་དམིགས་བསལ་གྱི་ཁེ་དབང་མཚོན་ཡོད།   དཔེར་ན། གཏམ་སྤྱོད་གང་ཡང་ཆོས་དང་མཐུན་དགོས་ཤིང་། སྤྲུལ་སྐུ་བསད་ན་མིག་འདོན་པ་དང་རྐང་ལག་གཅོད་པ་སོགས་ཀྱི་ཁྲིམས་ཆད་ཐུ་མོ་སྣ་ཚོགས་དང་། མི་བསད་ན་སྟོང་འཇལ་དགོས་པ་དང་། སྤྲུལ་སྐུ་ཆེན་པོ་སོགས་རབ་ཀྱི་རབ་ཏུ་གཏོགས་པའི་མིའི་སྲོག་སྟོང་ལ་གསེར་ལུས་དང་ལྗིད་ཚད་མཉམ་པ་དང་། རིམ་པ་ཐ་མའི་ཐ་མར་གཏོགས་པའི་སྲོག་ཚབ་ལ་གྲེས་ཐག་གཅིག་འཇལ་བ་སོགས་སོ། །ཆོས་ལུགས་ལམ་ལུགས་ལ་དམངས་གཙོའི་བཅོས་སྒྱུར་བྱས་པ་བརྒྱུད་དེ། དགོན་པའི་མངའ་བདག་ཚོགས་པའི་བཀས་བཀོད་རྒྱུད་འཛིན་གྱི་དམིགས་བསལ་དབང་ཆ་དང་བཀས་བཀོད་རྒྱུད་འཛིན་གྱི་གཉའ་གནོན་བཤུ་གཞོག་ལམ་ལུགས་མེད་པར་བཟོས་པ་དང་། བོད་བརྒྱུད་ནང་བསྟན་གྱི་ཞི་ཞིང་རྣམ་པར་དག་པའི་གཞི་རྩའི་རྣམ་པ་སོར་ཆུད་བྱུང་ཡོད། འོན་ཀྱང་སྤྱི་ཚོགས་ཀྱི་འཕོ་འགྱུར་ནི་མིས་འགྱུར་ལྡོག་བཏང་བ་ཙམ་གྱིས་ཕན་འབྲས་འཐོན་ཐུབ་པ་ཞིག་མ་ཡིན་ཞིང་། ལོ་རྒྱུས་གོམས་ཤུགས་ཀྱི་དབང་གིས་“ཆོས་སྲིད་ཟུང་འབྲེལ་”གྱི་དུག་ལྷག་དུས་ཚོད་ཐུང་ངུའི་ནང་རྦད་དེ་གཙང་སེལ་བྱེད་ཐུབ་མི་སྲིད་དོ། །
        གསུམ། མཆོག་གྱུར་རང་བཞིན་དང་ཚད་ལྡན་རང་བཞིན་ཟུང་འབྲེལ་བྱས་ཏེ། སྤྱོད་ལམ་རིག་གནས་ལ་ཆོས་ལུགས་ཀྱི་བྱེད་ལས་སྡུད་ལེན་བྱེད་དུ་བཅུག་པ།
        བོད་བརྒྱུད་ནང་བསྟན་གྱི་སྡོམ་ཁྲིམས་དེ་བོད་མི་རིགས་ཀྱི་རྒྱུན་ལྡན་འཚོ་བའི་གཏིང་རིམ་དུ་འདྲེས་ཤིང་། “སྡོམ་པ་ལྔ་”དང་“དགེ་བ་བཅུ་”ཡི་བསམ་བློ་མི་རྣམས་ཀྱི་རུས་རྐང་དུ་ཟུག་ནས་ཁོ་ཚོའི་ནམ་རྒྱུན་གྱི་སྤྱོད་ལམ་ལ་ཚོད་འཛིན་ཐེབས་ཀྱི་ཡོད། སྲིད་ལས་ངེས་པར་འབྱུང་འདོད་ཀྱི་བསམ་བློ་དང་སྐྱེ་བ་སྔ་ཕྱིའི་འདུ་ཤེས་ཀྱིས་ཀྱང་ཚད་ངེས་ཅན་ཞིག་གི་ཐོག་ནས་མི་རྣམས་ཀྱི་བརྩོན་ལེན་གྱི་བྱ་སྤྱོད་ལ་ཚོད་འཛིན་བྱེད་ཀྱི་ཡོད། བོད་བརྒྱུད་ནང་བསྟན་གྱི་མི་ཚེའི་ལྟ་ཚུལ་གྱི་སྙིང་པོ་ནི་དེའི་“སྡུག་བསྔལ་བདེན་པ་”ཡི་ལྟ་བ་རུ་གཅིག་བསྡུས་ཀྱིས་མཚོན་ཡོད། ནང་བསྟན་གྱི་དགོངས་པ་རྒྱུད་འཛིན་བྱས་ནས་བོད་བརྒྱུད་ནང་བསྟན་གྱིས་ཀྱང་མི་ཚེ་ནི་སྡུག་བསྔལ་གྱི་རང་བཞིན་ཅན་ཡིན་པ་དང་། མི་ཚེར་“སྐྱེ་རྒ་ན་འཆི་”ཡི་སྡུག་བསྔལ་གྱིས་ཁེངས་ཤིང་། སྐྱེ་འཆིས་བསྡུས་པའི་འཁོར་བའི་ངོ་བོ་ཡང་“སྡུག་བསྔལ་”ཡིན་པར་འདོད་ལ། “སྡུག་བསྔལ་”གྱི་འབྱུང་རྩ་ནི་འདོད་ཆགས་དང་མ་རིག་པར་ཐུག་ཡོད། དེར་བརྟེན་མི་ནི་རྫོགས་མཐའ་མེད་པའི་“རིགས་དྲུག་འཁོར་བ་”ཡི་ཁྲོད་དུ་འཁོར་བ་རེད། སྡུག་བསྔལ་དང་ཉོན་མོངས་ལས་གྲོལ་བ་དང་། སྐྱེ་འཆི་འཁོར་བའི་གནས་ལས་ཐར་དགོས་ན། ངེས་པར་དུ་ལུས་ངག་ཡིད་གསུམ་གྱི་ལས་མི་དགེ་བ་སྡོམ་དགོས་པ་དང་། ལུས་སེམས་གཉིས་གཅུན་དང་། ཕལ་པའི་ཚུལ་ལས་འདས་པ། དཀའ་སྡུག་བཟོད་པ། སྡིག་སྤོང་དགེ་སྒྲུབ། བློ་བདེ་གུ་ཡངས། དགྲ་གཉེན་མེད་པ། འདོད་ཆུང་ཆོག་ཤེས། སྡུག་བརྙས་བཟོད་སྒོམ། ཚུལ་ཁྲིམས་ཏིང་ངེ་འཛིན་ཤེས་རབ་བཅས་ལམ་བསླབ་པ་གསུམ་བརྒྱུད་དེ། སྐྱེ་བ་ཕྱི་མར་ཐར་པ་དང་ཐམས་ཅད་མཁྱེན་པའི་གོ་འཕང་བརྙེས་རྒྱུ་དེའོ། །བསླབ་བྱ་འདི་དག་གིས་དོན་དངོས་ཐོག་མི་རྣམས་ཀྱི་བསམ་བློ་དང་བྱ་སྤྱོད་འཆིང་རྒྱར་ཚུད་དུ་བཅུག་ཅིང་། ཚེ་ཕྱི་མར་རེ་བ་བཅོལ་ནས་ཚེ་འདིར་རང་མོས་རང་འཐད་ཀྱི་སྒོ་ནས་མངོན་སུམ་གྱི་གནས་ཚུལ་ལ་བཟོད་བསྲན་བྱེད་དུ་འཇུག་པའམ། ཡང་ན་སྒྱིད་ལུག་དང་སྲིད་པའི་ཕུན་ཚོགས་ལ་ཞེན་པ་ལོག་སྟེ་མངོན་སུམ་གྱི་དཀའ་སྡུག་ལ་འཐབ་རྩོད་བྱེད་པའི་འདུན་པ་མེད་པར་གཏོང་བ་རེད། དེར་བརྟེན་མི་ཁ་ཤས་ཀྱི་ཆོས་བྱེད་སྟངས་དང་དགེ་བ་སྒྲུབ་སྟངས་ལས་ཚད་ལས་བརྒལ་བའི་བྱ་སྤྱོད་མངོན་གྱི་ཡོད་ལ། ཁོ་ཚོ་ཕྱག་འཚལ་བ་དང་། ཁ་ཏོན་འདོན་པ། སྦྱིན་པ་གཏོང་བ། མཆོད་པ་འབུལ་བ། ཚེ་ཐར་གཏོང་བ་སོགས་ཀྱི་དགེ་བའི་ལས་ལ་ཧ་ཅང་མོས་པ་རེད།
        བཞི། མཆོག་གྱུར་རང་བཞིན་དང་ཞབས་ཞུའི་རང་བཞིན་ཟུང་འབྲེལ་བྱས་ཏེ་སྤྱི་ཕན་རིག་གནས་ལ་ཆོས་ལུགས་ཀྱི་ཁྱད་ཆོས་ལྡན་པར་བྱས་པ།
        བོད་བརྒྱུད་ནང་བསྟན་གྱི་“ཡར་རྫོགས་པའི་སངས་རྒྱས་ཀྱི་གོ་འཕང་དོན་དུ་གཉེར་བ་དང་། མར་སེམས་ཅན་གྱི་དོན་སྒྲུབ་པའི་”བྱང་ཆུབ་ཀྱི་སེམས་ནི། ནང་བསྟན་གྱི་ཟུར་ངོས་ནས་བཤད་ན་“སྤྱི་ཚོགས་ལ་ཞབས་འདེགས་ཞུ་བ་དང་། མང་ཚོགས་ལ་ཁེ་ཕན་གཉེར་བ་”ཡི་ཡང་དག་པའི་རྩ་དོན་ལག་ལེན་བསྟར་ཆོག་པ་དང་། འཇིག་རྟེན་གྱི་ཟུར་ངོས་ནས་བཤད་ན། གཞན་ཕན་རོགས་རམ་དང་དབུལ་སྐྱོར་རྒུད་སེལ་གྱི་དར་སྲོལ་བཟང་པོ་བསྐྲུན་ཆོག་པ་རེད། བོད་བརྒྱུད་ནང་བསྟན་གྱི་སྡེ་ཡིས་སྤེལ་བའི་སེམས་ཁམས་ཞབས་ཞུ་དང་། དགེ་སྒྲུབ་ཞབས་ཞུ། མི་ཆོས་ཞབས་ཞུ།  སྨན་བཅོས་ཞབས་ཞུ། དམངས་ཕན་ཞབས་ཞུ་སོགས་ཀྱིས་ཚད་ངེས་ཅན་གྱི་ཐོག་ནས་བོད་རིགས་ས་ཁུལ་གྱི་སྤྱི་ཚོགས་སྤྱི་ཕན་བྱ་གཞག་ལ་ཤུགས་རྐྱེན་ཐེབས་པ་དང་སྐུལ་ཁྲིད་བྱས་ཡོད་ལ། སྤྱི་ཚོགས་ཀྱི་སྤྱི་ཕན་རིག་གནས་དེ་ཆོས་ལུགས་པའི་སྡེ་ཡི་དགེ་སྒྲུབ་བྱེད་སྒོ་དང་མཉམ་དུ་སྤེལ་གྱི་ཡོད་པ་རེད།
ལྔ་པ། བོད་བརྒྱུད་ནང་བསྟན་གྱིས་ཀྲུང་ཧྭའི་རིག་གནས་བྱུང་འཕེལ་ཐད་ཐུན་མིན་བྱས་རྗེས་བཞག་པ།

        ད་ལྟ་བོད་བརྒྱུད་ནང་བསྟན་ལ་དགོན་པ་3500ལྷག་ཡོད་ཅིང་། དེའི་ཁྲོད་བོད་རང་སྐྱོང་ལྗོངས་སུ་1780ལྷག་ཡོད་པའི་ནང་ཚབ་མཚོན་རང་བཞིན་ལྡན་པའི་དགོན་པ་མང་པོ་ཚུད་ཡོད་པ་དང་། དེ་དག་གིས་བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་ནང་བསྟན་འཕེལ་རྒྱས་ཀྱི་ལོ་རྒྱུས་སྟེང་གོ་གནས་གལ་ཆེན་བཟུང་ཡོད། དེ་ཡང་ལྷ་ལྡན་གཙུག་ལག་ཁང་དང་། བསམ་ཡས་དགོན་པ། ས་སྐྱ་དགོན་པ། མཚུར་ཕུ་དགོན་པ། དགེ་ལུགས་པའི་དགོན་ཆེན་དྲུག་སོགས་སོ་སོར་རང་རང་གི་ཐུན་མིན་ཁྱད་ཆོས་རེ་ལྡན་ལ། འཐུས་ཚབ་རང་བཞིན་གྱི་དོན་སྙིང་ངེས་ཅན་ལྡན་པ་རེད། པོ་ཏ་ལ་དང་། གཙུག་ལག་ཁང་། དགའ་ལྡན། བཀྲ་ཤིས་ལྷུན་པོ། འབྲས་སྤུངས། བསམ་ཡས་སོགས་རྒྱལ་ཁབ་ཀྱི་རིག་དངོས་སྲུང་སྐྱོང་སྡེ་ཚན་གནད་ཆེའི་གྲས་སུ་བཞག་ཡོད་ལ། པོ་ཏ་ལ་དང་གཙུག་ལག་ཁང་ནི་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་སློབ་གསོ་ཚན་རྩལ་རིག་གནས་རྩ་འཛུགས་ཀྱིས་འཛམ་གླིང་གི་རིག་གནས་ཤུལ་བཞག་གྲས་སུ་བཞག་ཡོད།
        ལྡེར་བཟོའི་སྒྱུ་རྩལ་གྱི་ཕྱོགས་སུ། ལི་མ་དང་། ཤིང་བརྐོས། འཇིམ་སྐུ། རྡོ་བརྐོས། མེ་ཏོག་མཆོད་པའི་ལྡེར་སྐུ་སོགས་རིགས་དུ་མ་ཡོད་ལ། གཙོ་བོར་ལི་མ་དང་། འཇིམ་སྐུ། ཤིང་བརྐོས་སོགས་ཡོད། དཔེར་ན། སི་ཁྲོན་སྡེ་དགེ་དཔལ་སྤུངས་དགོན་པའི་བྱམས་པའི་ཟངས་སྐུ་ལ་“ཁམས་ཕྱོགས་ཀྱི་བྱམས་སྐུ་ཆེ་ཤོས་”ཞེས་པའི་མཚན་སྙན་ཡོད་ལ། སི་ཁྲོན་དཀར་མཛེས་ལི་ཐང་དགོན་པའི་ཇོ་བོ་ཤཱཀྱ་མུ་ནེའི་སྐུ་བརྙན་ལ་“འཛམ་གླིང་ལྷ་ཁང་ནང་གི་ཤཱཀྱ་ཐུབ་པའི་ཟངས་སྐུ་ཆེ་ཤོས་”ཞེས་རྗོད་ཀྱི་ཡོད་པ་ལྟ་བུའོ། །
        བོད་བརྒྱུད་ནང་བསྟན་གྱི་མཆོད་རྟེན་གྱི་བཟོ་བཀོད་ཕུན་སུམ་ཚོགས་པོ་ཡོད་དེ། རང་རྒྱལ་གྱི་བོད་རང་སྐྱོང་ལྗོངས་དང་ཞིང་ཆེན་བཞིའི་བོད་ཁུལ་ཙམ་དུ་མ་ཟད་པར་བོད་བརྒྱུད་ནང་བསྟན་གྱི་ཤུགས་རྐྱེན་འཕོས་སའི་ས་ཕྱོགས་ཁག་ཏུ་བཟོ་དབྱིབས་འདྲ་མིན་སྣ་ཚོགས་ཀྱི་མཆོད་རྟེན་མཐོང་ཐུབ་ལ། དེས་བོད་བརྒྱུད་ནང་བསྟན་གྱི་བཟོ་སྐྲུན་སྒྱུ་རྩལ་ལ་ཚོན་མདངས་རྣམ་པར་བཀྲ་བ་ཞིག་བསྣན་ཡོད། བོད་ལྗོངས་ལྷོ་ཁའི་བསམ་ཡས་དགོན་པའི་མཆོད་རྟེན་དང་ཟུར་མཁར་གྱི་རྡོའི་མཆོད་རྟེན་ནི་བོད་བཙན་པོའི་དུས་ཀྱི་མཆོད་རྟེན་སྒྱུ་རྩལ་གྱི་དཔེ་མཚོན་དུ་འཇོག་པ་དང་། རྒྱལ་རྩེ་དཔལ་འཁོར་མཆོད་རྟེན་དང་། མཚོ་སྔོན་ཡུལ་ཤུལ་གྱི་འཛོམ་ཉག་མཆོད་རྟེན། སྐུ་འབུམ་དགོན་པའི་གསེར་གདུང་ཆེན་མོ་དང་བདེ་བཤེགས་མཆོད་རྟེན་བརྒྱད། ནང་སོག་མཁར་སྔོན་པོའི་བློ་གསལ་གླིང་གི་རྡོར་གདན་རིང་བསྲེལ་མཆོད་རྟེན། པེ་ཅིན་མཆོད་རྟེན་དཀར་པོ་དགོན་པའི་ཤཱཀྱ་ཐུབ་པའི་རིང་བསྲེལ་མཆོད་རྟེན་སོགས་ནི་ལོ་རྒྱུས་སྟེང་གི་བོད་བརྒྱུད་ནང་བསྟན་མཆོད་རྟེན་སྒྱུ་རྩལ་གྱི་དཔེ་མཚོན་ལེགས་ཤོས་སུ་བརྩི་བ་དང་། པཎ་ཆེན་སྐུ་ཕྲེང་ལྔ་པ་ནས་དགུ་པའི་བར་གྱི་སྐུ་གདུང་ལྷན་བཞུགས་ཀྱི་མཆོད་སྡོང་དང་མཆོད་ཁང་ཤར་བཀྲ་ཤིས་རྣམ་རྒྱལ། པཎ་ཆེན་སྐུ་ཕྲེང་བཅུ་པའི་སྐུ་གདུང་མཆོད་རྟེན་དང་མཆོད་ཁང་སྲིད་གསུམ་རྣམ་རྒྱལ་ནི་དེང་དུས་ཀྱི་བོད་བརྒྱུད་ནང་བསྟན་མཆོད་རྟེན་སྒྱུ་རྩལ་གྱི་དཔེ་མཚོན་ཡིན་ནོ། །
        གཉིས། རྩ་འཛུགས་ཆོ་གས་ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་ལམ་ལུགས་དཔལ་ཡོན་ཕུན་སུམ་ཚོགས་པར་བཏང་བ།
        བོད་བརྒྱུད་ནང་བསྟན་གྱི་ལམ་ལུགས་རིག་གནས་ལ་གཙོ་བོ་བོད་བརྒྱུད་ནང་བསྟན་གྱི་རྩ་འཛུགས་སྒྲིག་གཞི་དང་། སློབ་གཉེར་ལམ་ལུགས། གུས་སྲོལ་ཚད་གཞི། འདུལ་ཁྲིམས་སྒྲིག་ལམ། ཕྱག་ལེན་ཆོ་ག དུས་མཆོད་མཛད་སྒོ་སོགས་ཡོད་ལ། དེ་ནི་བོད་བརྒྱུད་ནང་བསྟན་འདུ་ཤེས་ཀྱི་སྒྲིག་གཞི་རང་བཞིན་གྱི་མཚོན་སྟངས་ཡིན་པ་དང་། བོད་བརྒྱུད་ནང་བསྟན་རྐང་ཚུགས་ཐུབ་པའི་རྩ་འཛུགས་སྟོབས་ཤུགས་ཡིན་པས། དེས་ཀྱང་ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་ལམ་ལུགས་དཔལ་ཡོན་ཕུན་སུམ་ཚོགས་པར་བཏང་ཡོད།
        དགོན་སྡེའི་དོ་དམ་ལམ་ལུགས་ཕྱོགས་སུ། གྲུབ་མཐའ་སོ་སོས་དོ་དམ་མ་ལག་དང་དོ་དམ་མི་སྣའི་ལས་འགན་མི་འདྲ་བ་གཏན་ལ་ཕབ་ཡོད། དཔེར་ན་སློབ་གཉེར་རྩ་འཛུགས་དང་དགོན་པའི་དཔལ་འབྱོར་རྩ་འཛུགས་སོགས་ལྟ་བུ་ཡིན་ལ། བསྐོས་མིས་གཞུང་དོན་ལ་གྲོས་མོལ་བྱེད་པའི་རྣམ་པ་སྤྱད་ནས་ཉིན་རྒྱུན་གྱི་ལས་དོན་ཐག་གཅོད་བྱེད་པ་རེད། གྲྭ་བཙུན་གྱི་གོ་གནས་རིགས་སྣ་ཚོགས་ལ་ད་དུང་འགན་ཡུན་ངེས་ཅན་ཡོད། དམངས་གཙོ་བཅོས་སྒྱུར་བྱས་རྗེས་སྲོལ་རྒྱུན་གྱི་བླ་སྤྲུལ་མཁན་པོ་སོགས་གཙོ་བྱས་ཏེ་དགོན་པར་དོ་དམ་བྱེད་པའི་ལམ་ལུགས་དེ་དམངས་གཙོའི་དོ་དམ་བྱེད་ཐབས་སུ་བསྒྱུར། སྒྲིག་གཞི་རང་བཞིན་གྱི་དོ་དམ་ཕྱོགས་སུ། དགོན་པ་ཆེན་པོ་དག་ཏུ་སྤྱིར་བཏང་དུ་གླིང་དང་གྲྭ་ཚང་། ཁམས་ཚན་བཅས་རིམ་པ་གསུམ་དུ་བཀར་ལ། གྲྭ་ཚང་ནི་རྨང་གཞིའི་སྡེ་ཚན་ཡིན་པ་དང་སློབ་གླིང་དང་འདྲ་པོ་ཡོད། དཔེར་ན། མཚན་ཉིད་གྲྭ་ཚང་། རྒྱུད་པ་གྲྭ་ཚང་། སྨན་པ་གྲྭ་ཚང་། དུས་འཁོར་གྲྭ་ཚང་སོགས་ལྟ་བུའོ། ། སློབ་གཉེར་ནི་གཙོ་བོ་“རིག་པའི་གནས་བཅུ་”(རིག་གནས་ཆེ་བ་ལྔ་ནི་ནང་དོན་རིག་པ། ཚད་མ་རིག་པ། སྒྲ་རིག་པ། བཟོ་བ་རིག་པ། གསོ་བ་རིག་པ་དང་། རིག་གནས་ཆུང་བ་ལྔ་ནི་སྐར་རྩིས། སྙན་ངག མངོན་བརྗོད། སྡེབ་སྦྱོར། ཟློས་གར)སོགས་ལ་བྱེད་དོ། །
        བོད་བརྒྱུད་ནང་བསྟན་གྱི་སློབ་གཉེར་ལམ་ལུགས་ལ་སོ་སོར་ཁྱད་ཆོས་མི་འདྲ་བ་ལྡན། དགེ་ལུགས་པ་ཕུད་གྲུབ་མཐའ་གཞན་པས་གསང་སྔགས་ཉམས་ལེན་ལ་དེ་བས་ཀྱང་མཐོང་ཆེན་བླ་ལྷག་ཏུ་བྱེད་ལ་སོ་སོར་ཁྱད་ཆོས་ཀྱང་ལྡན་པ་རེད། དཔེར་ན་རྙིང་མ་པས་གཙོ་བོ་《སྒྱུ་འཕྲུལ་དྭ་བ་》དང་《ལང་ཀར་གཤེགས་པའི་མདོ》 《དགོངས་པ་འདུས་པ》《རྫོགས་པ་ཆེན་པོ་》སོགས་ལ་སློབ་གཉེར་བྱེད། དེ་ལས་《རྫོགས་པ་ཆེན་པོ་》ཡི་ཁྲིད་ལ་སེམས་སྡེ། ཀློང་སྡེ། མན་ངག་སྡེ་གསུམ་དུ་དབྱེ། དུས་རབས་བཅུ་བཞི་པར་ཀློང་ཆེན་རབ་འབྱམས་པ་དྲི་མེད་འོད་ཟེར་(1308-1364)གྱིས་རྫོགས་པ་ཆེན་པོ་གསང་བ་སྙིང་ཐིག་གི་བཅུད་ཕྱུངས་པའི་《ཀློང་ཆེན་མཛོད་བདུན་》(《ཡིད་བཞིན་མཛོད》 《མན་ངག་མཛོད》《ཆོས་དབྱིངས་མཛོད》 《གྲུབ་མཐའ་མཛོད》 《ཐེག་མཆོག་མཛོད》 《ཚིག་དོན་མཛོད》 《གནས་ལུགས་མཛོད》)བརྩམས་ཤིང་། དེ་ནི་རྙིང་མ་པའི་རྩ་བའི་ཆོས་གཞུང་ཡིན།⑨རྙིང་མ་པའི་ཆོས་གཞུང་གལ་ཆེན་གཞན་ཞིག་ནི་“གཏེར་མ་”ཡིན། ས་སྐྱ་པའི་གསང་སྔགས་སྐོར་ལ་ལུགས་གསུམ་ཡོད་ཅིང་། དེ་ནི་ངོར་ལུགས་དང་། ཀུན་དགའི་ལུགས། ཚར་ལུགས་སོ། །ས་སྐྱ་པའི་བླ་མ་སྐྱེས་ཆེན་དག་གིས་ལམ་འབྲས་སྐོར་གྱི་གཞུང་འགྲེལ་སོགས་གསུང་ཆོས་འབོར་ཆེན་བརྩམས་ཡོད་ལ།དེ་ལས་ས་སྐྱའི་མེས་པོ་ས་ཆེན་ཀུན་དགའ་སྙིང་པོ་(1092-1158)ཡིས་ལམ་འབྲས་གཞུང་ལུགས་སྐོར་ལ་བསྟན་བཅོས་ཆེན་པོ་དགུ་ལྷག་བརྩམས་ཡོད།⑩ དཔེར་ན་《ལམ་འབྲས་རྡོ་རྗེའི་ཚིག་རྐང་》སོགས་ཀྱིས་དབུ་མའི་ལྟ་དགོངས་ཀྱིས་བོད་བརྒྱུད་ནང་བསྟན་ཁྲོད་དུ་གཙོ་ཁྲིད་ནུས་པ་འདོན་པར་ཤུགས་རྐྱེན་གལ་ཆེན་བཞག་ཡོད།
        བོད་བརྒྱུད་ནང་བསྟན་གྱི་བསླབ་གནས་མི་ཆོས་ལའང་རང་ཉིད་ཀྱི་ཁྱད་ཆོས་ཡོད། བོད་བརྒྱུད་ནང་བསྟན་ཁོ་ནར་ཡོད་པའི་སྤྲུལ་སྐུ་ངོས་འཛིན་ལམ་ལུགས་ཀྱིས་བོད་བརྒྱུད་ནང་བསྟན་གྱི་བསླབ་གནས་ལམ་ལུགས་ལ་ཐུན་མོང་མ་ཡིན་པའི་ནང་དོན་བསྩལ་ཡོད་ཅིང་། སྤྲུལ་སྐུ། མཁན་པོ། ཡོངས་འཛིན་སོགས་གོང་རིམ་མི་སྣ་བར་གྱི་འབྲེལ་བ་སྣེ་མང་རྙོག་འཛིང་ཡིན་པས། ལམ་ལུགས་རིག་གནས་ལ་ཁྱད་ཆོས་སྣ་མང་ལྡན་ཞིང་ཚོན་མདངས་རྣམ་པར་བཀྲ་བ་བྱས་ཡོད། 
        བོད་བརྒྱུད་ནང་བསྟན་གྱི་ཆོས་ལུགས་ཆོ་ག་ལ་བསྟན་པ་སྔ་དར་དུས་ཀྱི་བརྒྱུད་པ་ཡོད་པར་མ་ཟད་རྒྱ་གར་གྱི་ཐེག་ཆེན་ཆོས་ལུགས་དར་རྒྱས་སུ་འགྲོ་བའི་དུས་མཇུག་གི་རྣལ་འབྱོར་བླ་མེད་རྒྱུད་ཀྱི་གསང་སྔགས་ཀྱང་དང་ལེན་བྱས་ཡོད་པས། བརྒྱུད་ཚུལ་འདྲ་མིན་སྣ་ཚོགས་སུ་སྣང་ལ་ཆོ་ག་ཤིན་ཏུ་སྣེ་མང་འདྲེས་འཛིང་ཡིན། ལྷག་པར་དུ་གསང་སྔགས་དབང་བསྐུར་དང་སྒྲིབ་སྦྱོང་ཕྱག་ལེན་སོགས་ལ་རིགས་ཤིན་ཏུ་མང་ལ། རྒྱ་བརྒྱུད་ནང་བསྟན་ལ་དེ་འདྲ་མང་པོ་ཁྱོན་ནས་བྱུང་མེད། ཆོས་ལུགས་ཀྱི་དུས་ཆེན་བྱེད་སྒོ་ཡང་ཧ་ཅང་ཕུན་སུམ་ཚོགས་པར་སྣང་། ད་ལྟ་བོད་བརྒྱུད་ནང་བསྟན་གྱི་སྲོལ་རྒྱུན་དུས་ཆེན་ལ་སྨོན་ལམ་ཆེན་མོ་དང་། ས་ག་ཟླ་བ། ལྷ་བབས་དུས་ཆེན། དགའ་ལྡན་ལྔ་མཆོད་སོགས་མང་པོ་ཡོད་པ་དང་། ཆོས་ལུགས་དུས་ཆེན་ནས་རིམ་འགྱུར་བྱུང་བའི་ཞོ་སྟོན་དུས་ཆེན་སོགས་ཡོད། གྲུབ་མཐའ་སོ་སོའི་ཤིང་རྟའི་སྲོལ་འབྱེད་ཀྱི་འཁྲུངས་སྐར་དང་འདས་སྐར་སོགས་ཀྱང་ཆོས་མཐུན་ཚུལ་མཐུན་གྱི་སྒོ་ནས་སྤེལ་བཞིན་ཡོད། དགོན་སྡེ་སོ་སོར་རང་གི་ཁྱད་ཆོས་དང་ལྡན་པའི་དུས་ཆེན་ལ་དཔེར་ན་འབྲས་སྤུངས་ཀྱི་གོས་སྐུ་ཆེན་མོ་དང་། སེ་རའི་ཕུར་མཇལ། གཙང་བཀྲ་ཤིས་ལྷུན་པོའི་གོས་སྐུ་ཆེན་མོ་སོགས་དུས་ལྟར་སྤེལ་ནས་དད་ལྡན་མང་ཚོགས་དང་མང་ཚོགས་གཞན་དག་རིག་གནས་ཀྱི་ལོངས་སྤྱོད་ལ་རོལ་དུ་འཇུག་གི་ཡོད་ལ། ཀྲུང་ཧྭའི་རིག་གནས་ལ་ཐུན་མོང་མ་ཡིན་པའི་ནང་དོན་ཁ་སྣོན་བྱས་པ་རེད །
        གསུམ། ཆོས་ལུགས་ལྟ་དགོངས་ཀྱིས་ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་བསམ་པའི་དཔལ་ཡོན་ཕུན་སུམ་ཚོགས་པར་བཏང་བ།
        བོད་བརྒྱུད་ནང་བསྟན་གྱི་བསམ་པའི་རིག་གནས་ལ་གཙོ་བོར་ཆོས་གཞུང་གསུང་རབ་དང་། ནང་བསྟན་ལྟ་གྲུབ། ལྷ་སྒྲུང་གཏམ་རྒྱུད། རྩོམ་རིག་སྒྱུ་རྩལ། རོལ་དབྱངས་ཞབས་བྲོ་སོགས་འདུས་ལ། དེ་དག་ནི་བོད་བརྒྱུད་ནང་བསྟན་གྱི་ཆོས་ལུགས་འདུ་ཤེས་ཀྱི་མཚོན་སྟངས་སྣ་མང་ཡིན་ལ། བོད་བརྒྱུད་ནང་བསྟན་རྐང་ཚུགས་ཐུབ་པའི་སྲོག་ཤིང་ཡང་ཡིན། ཡོ་ཆས་རིག་གནས་དང་ལམ་ལུགས་རིག་གནས་ནི་བོད་བརྒྱུད་ནང་བསྟན་རིག་གནས་ཀྱི་ཕྱི་ཤུན་ལྟ་བུ་ཡིན་ལ། བསམ་པའི་རིག་གནས་ནི་བོད་བརྒྱུད་ནང་བསྟན་རིག་གནས་ཀྱི་ནང་སྙིང་ཡིན་པས། ཕྱོགས་ག་གེ་མོའམ་ཚད་ངེས་ཅན་གྱི་སྟེང་ནས་ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་བསམ་པའི་དཔལ་ཡོན་ཕུན་སུམ་ཚོགས་པར་བཏང་ཡོད།
        ནང་བསྟན་གྱི་རིག་གཞུང་མ་ལག་ནི་ལྟ་བ་སྒོམ་པ་སྤྱོད་པ་འབྲས་བུ་སྟེ་ཕྱོགས་བཞིའི་སྒོ་ནས་སྒྲིག་འཛུགས་བྱས་ཡོད་པ་དང་། དད་པ་རིག་པ་སྤྱོད་པ་རྟོགས་པ་བཅས་བཞི་རུ་བཀར་བ་ཡང་ཡོད། ཕྱོགས་དེ་བཞིའི་སྒྲོམ་གཞི་ནི་བོད་བརྒྱུད་ནང་བསྟན་གྱིས་ནན་སྟོན་བྱེད་པའི་གཞི་ལམ་འབྲས་གསུམ་དུ་གོ་བ་བླངས་ཀྱང་ཆོག་ལ། དེ་ལས་སྒོམ་པ་དང་སྤྱོད་པ་གཉིས་ནི་ལམ་མོ། །
        “ལྟ་བ་”རུ་ནང་བསྟན་གྱི་རྨང་གཞིའི་ལྟ་དགོངས་དང་སྲོག་ཤིང་གི་བསམ་བློ་བསྡུས་ཡོད་ལ། གཙོ་བོ་ནང་བསྟན་གྱི་འཇིག་རྟེན་ལྟ་ཚུལ་དང་མི་ཚེའི་ལྟ་ཚུལ། གཞན་ཡང་རྟེན་འབྱུང་སྟོང་ཉིད་དང་། འཕགས་པའི་བདེན་པ་བཞི། ལྟ་བ་བཀར་བཏགས་ཀྱི་ཕྱག་རྒྱ་བཞི། རྟེན་འབྲེལ་ཡན་ལག་བཅུ་གཉིས་སོགས་སྲོག་ཤིང་གི་འདུ་ཤེས་ཚུད་ཡོད། 
        ནང་བསྟན་གྱི་འཇིག་རྟེན་ལྟ་ཚུལ་ལྟར་ན། འཇིག་རྟེན་ལ་མུ་མཐའ་མེད་ཅིང་། དེ་ལ་བལྟོས་མེད་ཀྱི་ལྟེ་གནས་ཤིག་ཀྱང་ངོས་འཛིན་དུ་མེད། འཇིག་རྟེན་གྱི་རྣམ་པ་ལ་“ཁམས་གསུམ་རིགས་དྲུག་”དང་། “སྟོང་ཆེན་པོའི་འཇིག་རྟེན” “ཕྱོགས་བཅུའི་འཇིག་རྟེན་”བཅས་ཀྱི་གཞུང་ལུགས་ཡོད། དེ་ཡང་“ཁམས་གསུམ་”ཟེར་བ་ནི་འདོད་ཁམས་དང་གཟུགས་ཁམས་དང་གཟུགས་མེད་ཁམས་བཅས་གསུམ་ཡིན་ལ། འདོད་ཁམས་ལ་ཡང་“རིགས་དྲུག་”སྟེ་ལྷ་དང་མི། ལྷ་མ་ཡིན། དམྱལ་བ། ཡི་དྭགས། དུད་འགྲོ་བཅས་སུ་དབྱེ་བ་རེད། ནང་བསྟན་གྱིས་ཕྱི་སྣོད་ཀྱི་འཇིག་རྟེན་དང་ནང་བཅུད་ཀྱི་སེམས་ཅན་གྱི་ངོས་འཛིན་ནི་གཙོ་བོར་“འདུས་བྱས་ཐམས་ཅད་མི་རྟག་པ། ཟག་བཅས་ཐམས་ཅད་སྡུག་བསྔལ་བ། ཆོས་ཐམས་ཅད་སྟོང་ཞིང་བདག་མེད་པ། མྱ་ངན་ལས་འདས་པ་ཞི་བ་”བཅས་ལྟ་བ་བཀའ་བཏགས་ཀྱི་ཕྱག་རྒྱ་བཞི་ཡིས་མཚོན་པར་བྱེད། དེ་ཡང་ནང་བསྟན་གྱིས་“རྟེན་འབྱུང་སྟོང་ཉིད་”ཀྱི་འདོད་ཚུལ་འཛིན་པས། འཇིག་རྟེན་ནི་བདེན་པར་མ་གྲུབ་པར་འདོད། 
        ནང་བསྟན་གྱི་མི་ཚེའི་ལྟ་བའི་སྲོག་ཤིང་གི་བསམ་བློ་ནི་“སྡུག་བསྔལ་བདེན་པ་དང་། ཀུན་འབྱུང་བདེན་པ། འགོག་པའི་བདེན་པ། ལམ་གྱི་བདེན་པ་”བཅས་འཕགས་པའི་བདེན་པ་བཞི་ལས་མཚོན་པར་བྱེད་ལ། མི་ཚེ་ནི་“སྐྱེ་རྒ་ན་འཆི་”སོགས་སྡུག་བསྔལ་གྱིས་ཁེངས་པ་དང་། སྐྱེ་འཆི་རེས་མོས་སུ་འཁོར་བའི་ངོ་བོ་ནི་“སྡུག་བསྔལ་”ཡིན་པ། “སྡུག་བསྔལ་”གྱི་རྒྱུ་ནི་ཀུན་འབྱུང་དང་། དེ་ནི་“ལས་”ལས་བྱུང་བ་ཡིན་ལ། ཆོས་ཀྱི་བདེན་པ་དང་འཇིག་རྟེན་དང་མི་ཚེའི་གནས་ལུགས་མ་རྟོགས་པའི་དབང་གིས་འགྲོ་བ་རྣམས་འདོད་ཆགས་དང་། ཞེ་སྡང་དང་། གཏི་མུག་དང་། ང་རྒྱལ་དང་། ཕྲག་དོག་དང་། ལོག་ལྟ་སྟེ་“རྩ་ཉོན་དྲུག་”ལ་གནས་པ་དང་། དེ་ལ་བརྟེན་ནས་“ལས་”བསགས། རྒྱུ་འབྲས་བསླུ་མེད་ཀྱི་དབང་གིས་འབྲས་བུ་སྣ་ཚོགས་མྱོང་བར་བྱེད་ཅིང་། “རྟེན་འབྲེལ་ཡན་ལག་བཅུ་གཉིས་”བརྒྱུད་ནས་སྐྱེ་འཆི་རེས་མོས་སུ་“རིགས་དྲུག་འཁོར་བ་”རུ་འཁྱམས་པ་རེད། དེ་ཡང་སྡུག་བསྔལ་སྣ་ཚོགས་ལས་ཐར་བར་འདོད་ན་ཀུན་འབྱུང་ཐམས་ཅད་ངེས་པར་བཀག་ནས། ཁམས་གསུམ་འཁོར་བའི་ཉོན་མོངས་པའི་སྐྱེད་བྱེད་རྒྱུ་དང་སྐྱེ་འཆི་རེས་མོས་སུ་འཁོར་བར་བྱེད་པའི་བསྐྱེད་བྱ་འབྲས་བུ་ཡོངས་སུ་བསལ་ནས། མྱ་ངན་ལས་འདས་པའི་ཞི་བའི་གོ་འཕང་འཐོབ་པར་བྱ་བ་དེ་ལ་འགོག་པའི་བདེན་པ་ཞེས་བྱའོ། །མྱ་ངན་ལས་འདས་པའི་ཞི་བའི་གོ་འཕང་འཐོབ་པའི་ལམ་ནི་ལམ་གྱི་བདེན་པའོ། །དེའི་ཕྱིར། ནང་བསྟན་གྱི་ལྟ་བ་ལྟར་ན། མི་ཚེ་ཡི་སྙིང་པོ་ནི་“སེམས་འདུལ་བ་”དེ་ཡིན་ལ།སངས་རྒྱས་ཀྱི་ཆོས་ཚུལ་བཞིན་བསྒྲུབས་ཚེ་གཞི་ནས་སྡུག་བསྔལ་ཤེས་པ་དང་། ཀུན་འབྱུང་འགོག་པ་དང་། ལམ་སྒྲུབ་པ་དང་། འགོག་བདེན་མངོན་འགྱུར་བཅས་ཡོང་ཞིང་། ཁམས་གསུམ་ཉོན་མོངས་ཀྱི་“སྐྱེད་བྱེད་རྒྱུ་”ཚར་གཅོད་པ་དང་། སྐྱེ་འཆི་རེས་མོས་སུ་འཁོར་བ་ཡི་“བསྐྱེད་བྱ་འབྲས་བུ་”ཚར་གཅོད་ལ། དུག་གསུམ་སྟེ་འདོད་ཆགས་ཞེ་སྡང་གཏི་མུག་གསུམ་ཡོངས་སུ་དག འཁོར་བའི་སྡུག་བསྔལ་གྱི་རྒྱ་མཚོ་ལས་ཐར། “བདག་མེད་”རྟོགས་ནས་མྱང་འདས་ཞི་བའི་གོ་འཕང་འཐོབ་པར་བྱེད་དོ། །
        “སྒོམ་པ་”ནི་གྲུབ་པ་འཐོབ་པའི་ཐབས་ལམ་དང་བྱེད་སྟངས་སྟེང་གཅིག་བསྡུས་ཀྱིས་མཚོན་ལ། དེ་ཉིད་ནང་བསྟན་གྱི་བརྟག་ཐབས་སྨྲ་བར་བརྩིས་ཆོག དེ་ལ་གཙོ་བོར་བྱང་ཆུབ་མཆོག་ཏུ་སེམས་བསྐྱེད་པ་དང་། ལུང་སྡེ་སྣོད་གསུམ་དང་ལམ་བསླབ་པ་གསུམ་ལ་འཇུག་པ་དང་། “ཕ་རོལ་ཏུ་ཕྱིན་པ་དྲུག་”(སྦྱིན་པ། ཚུལ་ཁྲིམས། བཟོད་པ། བརྩོན་འགྲུས། བསམ་གཏན། ཤེས་རབ)དང་“འཕགས་ལམ་ཡན་ལག་བརྒྱད་”(ཡང་དག་པའི་ལྟ་བ། ཡང་དག་པའི་རྟོག་པ། ཡང་དག་པའི་ངག ཡང་དག་པའི་ལས་ཀྱི་མཐའ། ཡང་དག་པའི་འཚོ་བ། ཡང་དག་པའི་རྩོལ་བ། ཡང་དག་པའི་དྲན་པ། ཡང་དག་པའི་ཏིང་ངེ་འཛིན)སོགས་“བྱང་ཕྱོགས་སོ་བདུན་” (དྲན་པ་ཉེར་བཞག་བཞི། ཡང་དག་སྤོང་བ་བཞི། རྫུ་འཕྲུལ་གྱི་རྐང་པ་བཞི། རྣམ་བྱང་དབང་པོ་ལྔ། རྣམ་བྱང་གི་སྟོབས་ལྔ། བྱང་ཆུབ་ཡན་ལག་བདུན། འཕགས་ལམ་ཡན་ལག་བརྒྱད)ལ་སྤྱོད་པ། ཉོན་མོངས་མ་ལུས་པའི་གཉེན་པོ་ཆོས་ཀྱི་ཕུང་པོ་བརྒྱད་ཁྲི་བཞི་སྟོང་འདུའོ། །
        “སྤྱོད་པ་”ནི་ནང་བསྟན་པའི་བྱ་སྤྱོད་ལ་བཏོན་པའི་བླང་བྱ་ཡིན་ཞིང་། ནང་བསྟན་གྱིས་ནང་བསྟན་པ་རྣམས་ལ་ཡང་དག་པའི་དད་པ་དང་། ཡང་དག་པའི་ལྟ་བ། ཡང་དག་པའི་སྤྱོད་པ་ལྡན་པའི་བླང་བྱ་འདོན་ལ། དེ་ནི་གཙོ་བོར་དཀོན་མཆོག་གསུམ་ལ་སྐྱབས་སུ་འགྲོ་བ་དང་། བསླབ་པ་ལྔ་དང་དགེ་བ་བཅུ་སྲུང་བརྩི། འདུལ་ཁྲིམས་ནན་དུ་སྲུང་བ་དང་སྡིག་པ་ཅི་ཡང་མི་བྱ་བ། དགེ་བ་ཕུན་སུམ་ཚོགས་པར་སྤྱོད་པ། ཐོས་བསམ་སྒོམ་གསུམ་དང་འཆད་རྩོད་རྩོམ་གསུམ་བྱེད་པ། སེམས་ཅན་ཐམས་ཅད་མ་རུ་བསམ་པ། ཡར་སངས་རྒྱས་ཀྱི་གོ་འཕང་གཉེར་ཞིང་། མར་འགྲོ་བ་སེམས་ཅན་གྱི་དོན་མཛད་པ། བྱམས་པ་ཚད་མེད་དང་། སྙིང་རྗེ་ཚད་མེད། དགའ་བ་ཚད་མེད། བཏང་སྙོམས་ཚད་མེད་བཅས་“ཚད་མེད་བཞིའི་སེམས་བསྐྱེད་”མ་ཉམས་པར།“སེམས་ཅན་རྒྱ་མཚོ་མ་ལུས་སྒྲོལ་བར་བགྱི། །ཉོན་མོངས་མཐའ་ཡས་མ་ལུས་སྤང་བར་བགྱི། །ཆོས་ཕུང་རྒྱ་མཚོ་མ་ལུས་རྟོགས་པར་བགྱི། །བླ་མེད་བྱང་ཆུབ་མངོན་འཚང་རྒྱ་བར་བགྱི། །”ཞེས་པའི་བྱང་ཆུབ་སེམས་ཀྱི་དམ་བཅའ་ཆེན་མོ་བཞི་ལ་གནས་པར་བྱ་བའོ། །
        “འབྲས་བུ་”ཡིས་ནི་དོན་ངོ་མར་མི་ཚེའི་འབྲས་བུ་སྟོན་པ་ཡིན་ལ། ནང་བསྟན་ལ་“ལམ་གསུམ་”གྱི་བཤད་ཚུལ་ཡོད་པ་སྟེ། དང་པོ་ནི་ཐར་པའི་གོ་འཕང་འཐོབ་པ། གཉིས་པ་ནི་བདེ་འགྲོ་གསུམ་(ལྷ་དང་མི་དང་ལྷ་མ་ཡིན)གྱི་གནས་སུ་སྐྱེ་བ། གསུམ་པ་ནི་ངན་འགྲོ་གསུམ་(དམྱལ་བ་ཡི་དྭགས་དུད་འགྲོ་གསུམ)དུ་ལྟུང་བ་བཅས་སོ། །ཡིན་ནའང་ནང་བསྟན་གྱིས་སྙེག་པ་ནི་བདག་རྣམ་པར་གྲོལ་བ་དང་གཞན་རྣམ་པར་གྲོལ་བར་བྱེད་པ། རྣམ་གྲོལ་གྱི་ལམ་ཆ་ཡོངས་སུ་རྫོགས་པ་སྟེ། མཇུག་མཐའི་དམིགས་ཡུལ་ནི་རྣམ་གྲོལ་ཐར་པའི་གོ་འཕང་འཐོབ་རྒྱུ་དང་། འཁོར་བའི་གནས་ནས་ཐར་ཏེ་རྟག་པ། བདེ་བ། བདག དག་པ་བཅས་ཀྱི་མྱང་འདས་ཀྱི་གོ་འཕང་འཐོབ་པ་དེའོ། །
        བོད་བརྒྱུད་ནང་བསྟན་ལ་གསུང་རབ་བསྟན་བཅོས་རི་བོའི་གཏོས་དང་མཉམ་པ་བཞུགས་ཤིང་། དེ་ལས་གྲགས་ཆེ་བ་ནི་བོད་ཡིག་བཀའ་བསྟན་རྣམ་གཉིས་ཡིན་ལ། དེ་ལ་བཀའ་འགྱུར་དུ་སངས་རྒྱས་ཀྱིས་གསུངས་པའི་བཀའ་སྡེ་སྣོད་གསུམ་དང་རྒྱུད་སྡེ་བཞི་བོད་དུ་འགྱུར་རོ་ཅོག་ཕྱོགས་གཅིག་ཏུ་བསྒྲིགས་ཡོད་པ་དང་། བསྟན་འགྱུར་དུ་བཀའ་མདོ་རྒྱུད་ཀྱི་དགོངས་པ་འགྲེལ་བར་བྱེད་པའི་བསྟན་བཅོས་བོད་ཡིག་ཏུ་བསྒྱུར་བ་འདུས། པར་གཞིའི་ཕྱོགས་སུ། བཀའ་བསྟན་གཉིས་ཀར་པར་གཞི་ཧ་ཅང་མང་པོ་ཡོད་ཅིང་གླེགས་བམ་གྱི་གྲངས་ཀ་ཡང་མཚུངས་པ་མིན། དཔེར་ན་སྣར་ཐང་པར་དང་། པེ་ཅིན་པར། སྡེ་དགེའི་པར། ཅོ་ནེའི་པར། ལི་ཐང་པར་སོགས་ཡོད། པར་གཞི་མི་འདྲ་བའི་བོད་ཡིག་བཀའ་བསྟན་གཉིས་དཔེ་བསྡུར་ཞུས་རྗེས་ཀྱི་སྤྱིའི་དཀར་ཆག་ཏུ་《བཀའ་འགྱུར་》དུ་རྒྱུད་སྡེའི་སྐོར་ལ་བམ་པོ་757དང་། མདོ་སྡེའི་སྐོར་ལ་བམ་པོ་382བཅས་བསྡོམས་པས་བམ་པོ་1139བཞུགས། 《བསྟན་འགྱུར་》དུ་བསྟོད་ཚོགས་སྐོར་ལ་བམ་པོ་73དང་། སྔགས་རྒྱུད་སྐོར་ལ་བམ་པོ་2938དང་། མདོ་གཞུང་དང་རིག་གནས་སྐོར་ལ་བམ་པོ་698བཅས་བསྡོམས་པས་བམ་པོ་3700བཞུགས།⑾ དེ་ལས་བོད་ཡིག་བཀའ་བསྟན་གཉིས་ནི་བོད་བརྒྱུད་ནང་བསྟན་ལ་དད་མོས་བྱེད་པའི་ཡུལ་ལུང་དུ་ཁྱབ་རྒྱ་ཆེ་བ་དང་ཤུགས་རྐྱེན་ཟབ་པ་གསལ་པོར་རྟོགས་ཐུབ། ཀྲུང་གོའི་བོད་རིག་པ་ཞིབ་འཇུག་ལྟེ་གནས་ཀྱིས་བོད་ཡིག་བཀའ་བསྟན་ལེགས་སྒྲིག་དང་ཡིག་སྒྱུར་ཐད་བྱ་བ་མང་པོ་བསྒྲུབས་ཏེ། ལོ་རྒྱུས་སྟེང་སྔར་ཡོད་མ་མྱོང་བའི་གཞི་རྒྱ་ཆེ་ཤོས་དང་། དཔེ་བསྡུར་པར་གཞི་ཆ་ཚང་ཤོས། དཔེ་བསྡུར་མ་ལག་ཚང་ཤོས། དབང་གྲགས་ཆེ་ཤོས་བཅས་ཡིན་པའི་ཀྲུང་གོའི་བོད་རིག་པ་ཞིབ་འཇུག་ལྟེ་གནས་ཀྱི་པར་གཞིའི་《བཀའ་བསྟན་དཔེ་བསྡུར་མ་》པར་སྐྲུན་གྱི་ལས་དོན་ལེགས་འགྲུབ་བྱས། པར་གཞི་གསར་པའི་བོད་ཡིག་བཀའ་བསྟན་ལ་ཁྱོན་བསྡོམས་ཆོས་ཚན་4842ཡོད་པ་དང་ ། དེ་ལས་《བཀའ་འགྱུར་》དཔེ་བསྡུར་མར་དེབ་108དང་། 《བསྟན་འགྱུར་》དཔེ་བསྡུར་མར་དེབ་124ཡོད།⑫ ད་ཆ་ཀྲུང་གོའི་བོད་རིག་པ་ཞིབ་འཇུག་ལྟེ་གནས་ཀྱིས་《ཀྲུང་ཧྭའི་བོད་ཡིག་རིག་མཛོད་ཆེན་མོ་》ལེགས་སྒྲིག་གི་བྱ་བ་འགོ་བརྩམས་ཟིན་པ་དང་། ལོ་བཅོ་ལྔའི་ནང་བཀའ་བསྟན་མ་ཡིན་པའི་བོད་བརྒྱུད་ནང་བསྟན་གྱི་མཁས་དབང་དག་གི་གསུང་རབ་བསྟན་བཅོས་ཁག་བསྡུ་སྒྲིག་པར་སྐྲུན་གྱི་འཆར་གཞི་བཟོས་ཡོད།
        བོད་བརྒྱུད་ནང་བསྟན་གྱི་རི་མོ་དང་ཐང་གའི་སྒྱུ་རྩལ་ལ་སྒྱུ་རྩལ་གྲུབ་མཐའ་གྲགས་ཅན་མང་པོ་གྲུབ་ཡོད་པ་དང་། དེའི་ནང་སྨན་ཐང་ལུགས་དང་། མཁྱེན་བརྩེ་ལུགས། ཀརྨ་སྒར་བྲིས་ལུགས་བཅས་ལ་བོད་བརྒྱུད་ནང་བསྟན་གྱི་རི་མོའམ་ཐང་གའི་སྒྱུ་རྩལ་གྱི་གྲུབ་མཐའ་ཆེན་པོ་གསུམ་དུ་འབོད། དེ་མིན་ད་རུང་བལ་པོའི་ལུགས་དང་། ཇོ་ནང་ལུགས། རེབ་ཀོང་ལུགས་སོགས་ཡོད་པ་དང་། དེ་དག་ཚང་མར་རང་གི་ཐུན་མིན་གྱི་ཉམས་འགྱུར་དང་ཁྱད་ཆོས་མངའ་བ་རེད། དེང་རབས་བོད་བརྒྱུད་ནང་བསྟན་གྱིས་མིང་དུ་གྲགས་པའི་རི་མོ་འབྲི་ལུགས་དེ་དག་འཛིན་སྐྱོང་སྤེལ་གསུམ་བྱས་པས། དེང་རབས་བོད་བརྒྱུད་ནང་བསྟན་གྱི་སྒྱུ་རྩལ་ལ་ཚོན་མདངས་རྣམ་པར་བཀྲ་བ་བསྩལ་ཡོད། ད་ཆ་སྨན་ཐང་དང་མཁྱེན་བརྩེ། ཀརྨ་སྒར་བྲིས་བཅས་གསུམ་རྒྱལ་སྲིད་སྤྱི་ཁྱབ་ཁང་གི་ཆོག་མཆན་ཐོབ་པ་བརྒྱུད་རྒྱལ་ཁབ་རིམ་པའི་མངོན་མིན་རིག་གནས་ཤུལ་བཞག་གི་མིང་ཐོ་ཁག་དང་པོའི་ནང་བཅུག་ཡོད། དེ་དང་ཆབས་ཅིག་འཛམ་གླིང་ཐོག་སྐད་གྲགས་འབར་བའི་ཕོ་བྲང་པོ་ཏ་ལ་དང་། ནོར་བུ་གླིང་ག ལྷ་ལྡན་གཙུག་ལག་ཁང་། བསམ་ཡས་དགོན། ས་སྐྱ་དགོན། རྒྱལ་རྩེ་དཔལ་འཁོར་ཆོས་སྡེ་བཅས་ཀྱི་ལྡེབས་རིས་སོགས་ཀྱི་ཉམས་གསོ་དང་སྲུང་སྐྱོང་ལ་འཛམ་གླིང་སྐྱེ་བོ་ཀུན་གྱིས་དོ་སྣང་བྱེད་པའི་གྲུབ་འབྲས་བླངས་ཡོད་པ་དང་། དེང་རབས་བོད་བརྒྱུད་ནང་བསྟན་གྱི་ལྡེབས་རིས་ཀྱི་རྨད་བྱུང་ཆུ་ཚད་ཀྱང་དངོས་སུ་མཚོན་ཡོད།
བོད་བརྒྱུད་ནང་བསྟན་གྱི་རོལ་མོའི་མ་ལག་ཁྲོད་དུ་རོལ་མོའི་གཞུང་ལུགས་མ་ལག་ལ་ལྷན་ཅིག་སྐྱེས་པའི་རོལ་མོ་(མགྲིན་དབྱངས)དང་རྐྱེན་གཞན་ལས་བྱུང་བའི་རོལ་མོ་(རོལ་དབྱངས)གཉིས་ཀྱིས་གཙོས་པའི་མ་ལག་ཆ་ཚང་བ་ཞིག་གྲུབ་ཡོད། དེང་རབས་བོད་བརྒྱུད་ནང་བསྟན་གྱི་གྲུབ་མཐའ་དང་དགོན་སྡེ་ཁག་གི་འདོན་དབྱངས་རྟ་ཅན་གྱི་བྱེད་སྒོ་དང་། རོལ་ཆས་དཀྲོལ་བའི་ཆོས་ལུགས་བྱེད་སྒོ་སོགས་ནི་དེང་རབས་ཀྱི་བོད་བརྒྱུད་ནང་བསྟན་གྱི་རོལ་མོ་རྒྱུད་འཛིན་དང་དར་སྤེལ་བྱེད་པའི་ཐབས་ལམ་གཙོ་བོ་ཡིན། གཞན་ཡང་ལོ་རྒྱུས་འཕེལ་རྒྱས་ཀྱི་བརྒྱུད་རིམ་རིང་པོའི་ནང་ཆགས་ཤིང་ཐུན་མིན་ཁྱད་ཆོས་ལྡན་པའི་བོད་བརྒྱུད་ནང་བསྟན་རོལ་མོའི་སྒྱུ་རྩལ་གྱི་རྣམ་པར་ཡང་རྒྱུད་འཛིན་དང་གོང་འཕེལ་ཐོབ་ཡོད་ལ། དེང་རབས་ཀྲུང་གོ་ཙམ་དུ་མ་ཟད་འཛམ་གླིང་གི་རོལ་མོའི་སྡེ་རུ་ཡང་ཤུགས་རྐྱེན་ངེས་ཅན་བཞག་ཡོད། དཔེར་ན་ཀན་སུའུ་བླ་བྲང་དགོན་པ་ཁོ་ནར་ཡོད་པའི་བོད་བརྒྱུད་ནང་བསྟན་གྱི་རོལ་མོ་(མདོ་དར)དང་། མཚོ་སྔོན་བོད་རིགས་ཀྱི་ཆོས་ཀྱི་འདོན་རྟ་གཉིས་ནི་རྒྱལ་ཁབ་རིམ་པའི་མངོན་མིན་རིག་གནས་ཤུལ་བཞག་གི་མིང་ཐོ་ཁག་གཉིས་པའི་ནང་ཚུད་ཡོད། ཇོ་ནང་པའི་རོལ་མོ་ནི་ཀྲུང་གོའི་བོད་བརྒྱུད་ནང་བསྟན་གྱི་ད་ཡོད་རོལ་མོ་སྔ་ཤོས་ཀྱི་གྲས་ཡིན་པས། དེ་ལ་“ཀྲུང་གོའི་རོལ་མོའི་འགྱུར་རྡོ་གསོན་པོ་”རུ་འབོད་པ་དང་། རྒྱལ་ཁབ་རིམ་པའི་མངོན་མིན་རིག་གནས་ཤུལ་བཞག་གི་མིང་ཐོ་ཁག་གསུམ་པའི་ནང་ཚུད་ཡོད། འབྲི་གུང་བཀའ་བརྒྱུད་པའི་མ་དགོན་འབྲི་གུང་མཐིལ་དགོན་པའི་ནང་ཆོས་རོལ་མོ་དེ་2008ལོའི་ཟླ་6པའི་ནང་རྒྱལ་ཁབ་རིམ་པའི་མངོན་མིན་རིག་གནས་ཤུལ་བཞག་གི་མིང་ཐོ་རུ་ཚུད་ཡོད།
        བོད་བརྒྱུད་ནང་བསྟན་གྱི་གར་འཆམ་གྱིས་ཀྱང་བོད་བརྒྱུད་ནང་བསྟན་གྱི་རིག་གནས་སྒྱུ་རྩལ་ལ་ཚོན་མདངས་རྣམ་པར་བཀྲ་བ་བསྩལ་ཡོད། བོད་ཀྱི་གདོད་མའི་ཆོས་ལུགས་བོན་ཆོས་དང་ཕྱིས་བྱུང་བོད་བརྒྱུད་ནང་བསྟན་ཁྲོད་ཀྱི་རྙིང་མ་པ་དང་། བཀའ་བརྒྱུད་པ། ས་སྐྱ་པ། དགེ་ལུགས་པ་སོགས་ཚང་མར་རང་ཉིད་ཀྱི་གར་འཆམ་ཡོད་པ་དང་། དེ་དག་སོ་སོའི་རྣམ་པ་དང་ནང་དོན་སོགས་འདྲ་མི་འདྲ་ཅི་རིགས་ཡོད་ནའང་། སྤྱིར་བཏང་དུ་ཚང་མར་“རྡོ་རྗེ་བདུད་འདུལ་གར་འཆམ་”ཞེས་པའམ། “རྡོ་རྗེའི་གར་”ཞེས་པའམ་ཡང་ན་“རྡོ་རྗེའི་ཆོས་གར་”ཞེས་འབོད་ཀྱི་ཡོད་པ་དང་། བོད་སྐད་ནང་སྤྱིར་བཏང་དུ་“འཆམ་”ཞེས་རྗོད་ཀྱི་ཡོད།དཔེར་ན་ཕོ་བྲང་པོ་ཏ་ལའི་བདུད་འདུལ་གར་འཆམ་དང་། བསམ་ཡས་དགོན་པའི་རྡོ་རྗེའི་གར་འཆམ། སི་ཁྲོན་མཛོད་དགེ་རྫོང་གི་ས་སྐྱ་པའི་དགོན་པ་ཆོས་སྐྱིད་དགོན་པའི་རྡོ་རྗེ་ཕུར་པའི་གར་འཆམ། མཚོ་སྔོན་སྐུ་འབུམ་བྱམས་པ་གླིང་གི་རྡོ་རྗེ་འཇིགས་བྱེད་ཀྱི་གར་འཆམ་དང་རྟ་མགྲིན་གྱི་གར་འཆམ། ཀན་སུའུ་བླ་བྲང་དགོན་པའི་དུས་འཁོར་རྡོ་རྗེ་རྩ་རྒྱུད་ཀྱི་གར་འཆམ། བོད་ལྗོངས་འབྲི་གུང་མཐིལ་དགོན་གྱི་འབྲི་གུང་ཆོས་སྐྱོང་གི་གར་འཆམ་སོགས་ལ་ཐུན་མོང་མ་ཡིན་པའི་ཁྱད་ཆོས་ལྡན་པས། བོད་བརྒྱུད་ནང་བསྟན་གར་འཆམ་ཁྲོད་ཀྱི་མཚོན་རྟགས་ལྟ་བུར་གྱུར་ཡོད་དོ། །
        བཞི། བོད་བརྒྱུད་ནང་བསྟན་དེ་ཀྲུང་ཧྭའི་མི་རིགས་ཁག་གི་རིག་གནས་མཉམ་འདྲེས་ཀྱི་སྒྲོག་ཐག་ལྟ་བུར་གྱུར་པ།
        སུང་དང་ཡོན་གྱི་དུས་སྐབས་ནས་བཟུང་། བོད་བརྒྱུད་ནང་བསྟན་ནང་ལོགས་ས་ཁུལ་དུ་མཚམས་མེད་པར་དར་ཁྱབ་བྱུང་བ་དང་། ནང་ལོགས་ས་ཁུལ་གྱི་པེ་ཅིན་དང་རི་བོ་རྩེ་ལྔ། ཁྲེང་ཏེ་ལ་སོགས་པར་བོད་བརྒྱུད་ནང་བསྟན་གྱི་གནས་མཆོག་མང་པོ་ཆགས་ཏེ། དགོན་སྡེ་དང་མཆོད་རྟེན་ལ་སོགས་ཀྱིས་མཚོན་པའི་བཟོ་སྐྲུན་རིག་གནས་ཀྱི་ཤུལ་བཞག་མང་པོ་བྱུང་བ་དང་། བོད་བརྒྱུད་ནང་བསྟན་གྱི་ལྟ་དགོངས་དང་ཉམས་ལེན་བྱེད་སྟངས། རིག་གནས་སྒྱུ་རྩལ་ལ་སོགས་ཀྱང་ནང་ལོགས་ས་ཁུལ་དུ་དར་བ་ཡིས། ཀྲུང་ཧྭའི་མི་རིགས་ཁག་བར་རིག་གནས་འབྲེལ་འདྲིས་དང་བརྗེ་སྤེལ་མཉམ་འདྲེས་ཡོང་བར་སྐུལ་སྤེལ་བཏང་།
        ནང་བསྟན་རིག་གནས་ཕན་ཚུན་བརྗེ་རེས་བྱེད་པ་ནི་ཡོན་རྒྱལ་རབས་སྐབས་སུ་རྒྱ་བོད་སོགས་མི་རིགས་ཕན་ཚུན་བར་རིག་གནས་སྤེལ་རེས་བྱས་པའི་ནང་དོན་གལ་ཆེན་ཞིག་ཡིན། 1260ལོར་སེ་ཆེན་རྒྱལ་པོས་རྒྱལ་ས་གོང་མ་རུ་ནང་བསྟན་ལ་སྤྱི་འདོམས་བྱེད་པའི་སྒྲིག་གཞི་ཙུང་ཐུང་སུའོ་ཞེས་པ་ཞིག་བཙུགས་པ་དང་། དེ་ནི་ཡོན་རྒྱལ་རབས་ཀྱིས་ཀྲུང་དབྱང་རིམ་པའི་གྲྭ་བཙུན་ལས་དོན་སྡེ་ཁག་བཙུགས་པ་སྔ་ཤོས་ཡིན་ལ། འགྲོ་མགོན་འཕགས་པ་ཏི་ཤཱིར་བཀུར་ཏེ་གཡང་ཊི་ལས་བྱས་པའི་ཐམ་ག་དང་ཀྲུང་ཡོན་གྱི་དགེ་འདུན་པ་ཡོངས་ལ་ཁ་ལོ་སྒྱུར་བའི་ལུང་ལས་གནང་ངོ་། །1264ལོར་ཡང་བསྐྱར་ཙུང་ཀྲི་ཡོན་ཞེས་པ་བཙུགས་ཏེ་རྒྱལ་ཡོངས་ཀྱི་ནང་བསྟན་ལས་དོན་དང་བོད་ཁུལ་གྱི་ཆོས་སྲིད་ལས་དོན་ལ་དོ་དམ་མཛད་པས། རྒྱ་བོད་གཉིས་ཀྱི་ནང་བསྟན་སྔར་བས་མཉམ་འདྲེས་ཡོང་བར་ཆ་རྐྱེན་བསྐྲུན་པ་དང་། ཁྱབ་ཁོངས་གཞན་གྱི་རིག་གནས་མཉམ་འདྲེས་ལའང་སྔར་མེད་སྐུལ་འདེད་བཏང་། དེ་ལས་ཀྱང་གལ་ཆེ་བ་ཞིག་ལ། 1285ལོ་ནས་1288ལོའི་བར་དུ་སེ་ཆེན་རྒྱལ་པོས་རྒྱ་བོད་ཡུ་གུར་སོགས་མི་རིགས་ཁག་གི་བླ་ཆེན་དང་མཁས་པ་རྒྱལ་སའི་ཆོས་རྒྱས་གླིང་དགོན་པར་གདན་དྲངས་ནས་རྒྱ་བོད་གཉིས་ཀྱི་བཀའ་བསྟན་ཕན་ཚུན་བསྡུར་དུ་བཅུག་སྟེ་རྒྱལ་བའི་བཀའ་བསྟན་གཉིས་ལ་དཀར་ཆག་གསར་པ་《དཱའི་ཡུའན་ཧྥ་པོའུ་ཥན་ཐུང་ལུས་མཛད་པ་》ཞེས་པ་བམ་པོ་10ཅན་ཞིག་བཞེངས། དེ་ནི་རྒྱ་བོད་སྐད་ཡིག་གཉིས་ལ་འཁོད་པའི་རྒྱལ་བའི་བཀའ་བསྟན་འགྱུར་རིན་པོ་ཆེ་ཐོག་དང་པོར་ཕན་ཚུན་བསྡུར་དཔྱད་མཛད་པ་ཡིན་ལ། རྒྱ་བོད་གཉིས་ཀྱི་སྐད་གང་དུ་བྱས་པའི་བཀའ་བསྟན་ལ་“ཆོས་ཚན་གང་ཡོད་མེད་དང་བམ་གྲངས་ག་ཚོད་དུ་བགྱིས་པ་”སོགས་ཀྱི་གནས་ཚུལ་ཆེས་ཐོག་མར་གསལ་ཁ་བྱུང་བའང་ཡིན་ནོ། ། རྒྱ་བོད་སྐད་གཉིས་ཀྱི་ཚིག་སྦྱོར་དང་ནང་དོན་ལའང་“གང་ཞིག་འགྱུར་ཆད་པ་”དང་། “གང་ཞིག་འགྱུར་མང་བ” “གང་ཞིག་འགྱུར་ཅུང་ཟད་ཉུང་བ” “གང་ཞིག་འགྱུར་གཅིག་མཐུན་ཡིན་པ་”ལ་སོགས་པའི་མཆན་བཏབ་སྟེ་སོ་སོར་བམ་པོ་1644ཡི་རྗེས་སུ་བཀོད་ཡོད།⑬ དེ་ནི་རིག་གནས་ལས་གཞི་གཏོས་ཆེན་ཞིག་ཡིན་ལ། ཕྱིས་སུ་རྒྱ་བོད་གཉིས་ཀྱི་བཀའ་བསྟན་ཕན་ཚུན་དཔེ་སྡུར་བྱེད་པར་རྨང་གཞི་བཏིང་ཡོད་དོ། ། རྒྱ་བོད་གཉིས་ཀྱི་ནང་བསྟན་རིག་གནས་ཕན་ཚུན་བརྗེ་རེས་བྱས་པ་དང་འབྲེལ། རྒྱ་བོད་གཉིས་ཀྱི་ནང་བསྟན་གྱིས་ཀྲུང་དབྱང་སྲིད་གཞུང་དང་བོད་ས་གནས་ཀྱི་ཆབ་སྲིད་ཀྱི་འབྲེལ་བའི་ཐད་ལ་སྒྲོག་ཐག་ལྟ་བུའི་ནུས་པ་སྔར་བས་འདོན་སྤེལ་བྱས་པ་དང་། རྒྱ་བོད་གཉིས་ཀྱི་ནང་བསྟན་གྱིས་རྒྱ་བོད་དང་། བོད་དང་མི་རིགས་གཞན་དག་བར་དུ་རིག་གནས་བརྗེ་སྤེལ་བྱེད་ཤུགས་ཆེ་རུ་འགྲོ་བར་ཟམ་པའི་ནུས་པ་སྔར་བས་འདོན་སྤེལ་གང་ལེགས་བྱས་ཤིང་།  མེས་རྒྱལ་གཅིག་གྱུར་སྲུང་སྐྱོང་དང་མི་རིགས་མཐུན་སྒྲིལ་ལ་ཤུགས་སྣོན་པར་བྱས་རྗེས་གལ་ཆེན་བཞག་ཡོད།
མིང་རྒྱལ་རབས་ཀྱི་“ཆོ་ལོ་མང་དུ་སྩོལ་བ་”ཡི་སྲིད་ཇུས་ཀྱིས་བོད་ཀྱི་ཆབ་སྲིད་ལམ་ལུགས་ལ་ཤུགས་རྐྱེན་ཟབ་མོ་ཐེབས། མིང་རྒྱལ་རབས་ཀྱི་སྲིད་གཞུང་གིས་བོད་བརྒྱུད་ནང་བསྟན་གྱི་བཀའ་བརྒྱུད་པའི་དབུ་ཁྲིད་མི་སྣ་ལ་ “རིན་ཆེན་ཆོས་རྒྱལ་”དང་། ས་སྐྱ་པའི་དབུ་ཁྲིད་མི་སྣ་ལ་ “ཐེག་ཆེན་ཆོས་རྒྱལ” དགེ་ལུགས་པའི་དབུ་ཁྲིད་མི་སྣ་ལ་ “བྱམས་ཆེན་ཆོས་རྒྱལ་”ཞེས་པའི་མཚན་བཏགས་གནང་བ་དེ་ནི་མིང་རྒྱལ་རབས་ཀྱིས་གནང་བའི་“ཆོས་རྒྱལ་ཆེན་པོ་གསུམ་”ཞེས་བྱ་བ་ཡིན། ཆོས་རྒྱལ་ཆེན་པོ་གསུམ་ལས་གཞན་མིང་རྒྱལ་རབས་ཀྱིས་ད་དུང་ས་གནས་ཀྱི་ཝང་ལྔ་ཞེས་པའི་ལུང་ལས་འཇའ་ས་བཙུགས། དེ་དག་ནི་ཁྲན་ཧྭ་ཝང་དང་། ཙན་ཧྲན་ཝང་། ཧུའུ་ཅོ་ཝང་། ཁྲན་ཅོ་ཝང་། ཧྥུ་ཅོ་ཝང་བཅས་སོ། །ཆོས་རྒྱལ་ནི་གནས་གཅིག་ཏུ་གཏན་བཞུགས་གནང་བ་མི་མཛད་ཅིང་། ས་གནས་ཀྱི་ཝང་རྣམས་ལ་སོ་སོར་མངའ་རིས་བསྩལ་ཡོད་པས་ཕན་ཚུན་གྱི་ཁོངས་སུ་མི་གཏོགས་པར་ས་གནས་སོ་སོའི་ཆོས་སྲིད་གཉིས་ཀྱི་བདག་པོ་བྱེད། ས་གནས་ཀྱི་ཝང་ལྔ་པོ་ནི་གྲུབ་མཐའ་མི་འདྲ་བ་དང་གྲུབ་མཐའི་ཡན་ལག་མི་འདྲ་བར་གཏོགས་པས། དེ་དག་གི་དབང་བསྒྱུར་མངའ་རིས་ཀྱིས་ཕལ་ཆེར་བོད་ཡུལ་ཧྲིལ་པོར་ཁེབས་ཡོད་པ་རེད། ཝང་དེ་དག་ཚང་མར་གོང་མས་ཐམ་ག་འཇའ་ས་དང་རིན་པོ་ཆེ་ཅི་རིགས་བསྩལ་ཡོད་པ་དང་། གོ་ཐོབ་རྒྱུད་འཛིན་བྱས་ཆོག་ལ་དུས་ངེས་གཏན་ནང་གོང་མར་མཇལ་ཁ་ཞུ་བ་དང་སྐྱེས་འབུལ་དགོས། དེར་མ་ཟད་ཝང་ཚང་མའི་ས་ཐོག་ཏུ་རྟ་ཟམ་བཙུགས་ནས་ཀྲུང་དབྱང་སྲིད་གཞུང་དང་བོད་ཁུལ་བར་གྱི་འབྲེལ་བར་འགན་ལེན་བྱེད། དེ་ལྟར་ཡོན་རྒྱལ་རབས་ཀྱིས་ཆོས་སྲིད་སྟོབས་ཤུགས་གཅིག་ཁོ་ནར་མཐོང་ཆེན་བྱེད་པའི་རྣམ་པ་བསྒྱུར་ཏེ། བོད་ས་གནས་ཀྱི་ཆབ་སྲིད་ལམ་ལུགས་ལ་འགྱུར་བ་བཏང་ངོ་། །
        མིང་རྒྱལ་རབས་ཀྱིས་བཙུགས་པའི་བོད་ཀྱི་སྐྱེས་འབུལ་ལམ་ལུགས་དེར་ཆབ་སྲིད་ཀྱི་དོན་སྙིང་གལ་ཆེན་ལྡན་པར་མ་ཟད། དཔལ་འབྱོར་དང་རིག་གནས་ཐད་ལའང་དོན་སྙིང་གལ་ཆེན་ལྡན་ཏེ། ཆབ་སྲིད་ཀྱི་འབྲེལ་བར་ཤུགས་སྣོན་བྱས་པ་དང་དུས་མཚུངས་སུ་དཔལ་འབྱོར་དང་རིག་གནས་བརྗེ་སྤེལ་ལའང་སྐུལ་འདེད་བཏང་ཡོད། མིང་རྒྱལ་རབས་དུས་སུ་ཇ་རྟ་བརྗེ་རེས་ཀྱི་ཚོང་ར་མུ་མཐུད་འཕེལ་རྒྱས་སུ་སོང་།1371ལོར་ནོར་སྡེས་ཧྲའན་ཞི་དང་སི་ཁྲོན་གྱི་ཇ་རིགས་བོད་ཀྱི་རྟ་ལ་བརྗེ་ཆོག་པའི་གཏན་འཁེལ་བྱས་པས། ཇ་རིགས་ཐོན་སྐྱེད་བྱ་ཡུལ་ཁག་ཏུ་ཇ་ཁྲལ་སྡེ་ཁག་བཙུགས་པ་དང་ཁྲལ་ཚད་གཏན་ཁེལ་བྱས་ཡོད། གཞན་ཡང་ཆིན་ཇུ་(དེང་གི་ཀན་སུའུ་ཐེན་ཧྲུའེ)དང་། ཐའོ་ཇུ་(དེང་གི་ཀན་སུའུ་བ་ཙེ)དང་། ཧོ་ཇུ་(དེང་གི་ཀན་སུའུ་ཀ་ཇུ)དང་། ཡ་ཇུ་(དེང་གི་སི་ཁྲོན་གཡག་རྔ)སོགས་སུ་ཆེད་དུ་ཇ་རྟ་དོ་དམ་སྡེ་ཁག་བཙུགས་ཏེ་ཇ་རྟ་བརྗེ་ཚོང་གི་ལས་དོན་ལ་དོ་དམ་བྱེད་དུ་བཅུག་ནས།⑭ བོད་ལྗོངས་དང་བོད་ཁུལ་གཞན་དག་ལ་ཇ་རྟ་བརྗེ་ཚོང་བྱེད་པའི་བདག་དབང་ཀྲུང་དབྱང་སྲིད་གཞུང་གིས་དམ་དུ་བཟུང་། ཇ་རྟ་དོ་དམ་སྡེ་ཁག་ཡོད་ས་དེ་བོད་དང་ནང་ལོགས་ས་ཁུལ་བར་གྱི་ཚོང་དོན་ལྟེ་བར་གྱུར་ལ། རིག་གནས་བརྗེ་སྤེལ་གྱི་ལྟེ་བར་ཡང་གྱུར་པས། བོད་ཀྱི་རིག་གནས་དང་ནང་ལོགས་ས་ཁུལ་སོགས་ཀྱི་རིག་གནས་མཉམ་འདྲེས་ཡོང་བ་ལ་སྐུལ་འདེད་ཧ་ཅང་ཆེན་པོ་བཏང་། སྐྱེས་འབུལ་མཁན་ཇེ་མང་དང་ཕན་ཚུན་གྱི་ཉོ་ཚོང་འབྲེལ་བ་རྒྱ་བསྐྱེད་པ་དང་བསྟུན་ནས། རིག་གནས་བརྗེ་སྤེལ་བྱེད་པར་ཟམ་པ་དང་སྒྲོག་ཐག་གསར་པ་བསྣན། སྐྱེས་འབུལ་གྲྭ་བཙུན་དང་ཚོང་པ་འབོར་ཆེན་ནང་ལོགས་ས་ཁུལ་དང་བོད་ཀྱི་བར་འགྲོ་འོང་བྱས་པ་ནི་བོད་ཀྱི་རིག་གནས་སྒྲོག་སྤེལ་མཁན་དུ་གྱུར་ལ། ལྷག་པར་དུ་ནང་ལོགས་ས་ཁུལ་གྱི་དཔལ་ཡོན་སྒྲོག་སྤེལ་མཁན་དུ་གྱུར་ཏེ། ཁོ་ཚོས་ཀྲུང་ཡོན་ས་ཁུལ་སོགས་ཀྱི་དཔལ་ཡོན་རྒྱུན་ཆད་མེད་པར་བོད་ཡུལ་དུ་ནང་འདྲེན་བྱས་སོ། །
        ཆིང་རྒྱལ་རབས་སྐབས་སུ་བོད་དང་ནང་ལོགས་ས་ཁུལ་བར་གྱི་རིག་གནས་སྒྱུ་རྩལ་བརྗེ་སྤེལ་ལ་སྔ་ན་མེད་པའི་དར་རྒྱས་གོང་འཕེལ་བྱུང་ལ། རྒྱ་བོད་གཉིས་ཀྱི་བཟོ་རྩལ་དང་མཛེས་རྩལ་བརྗེ་སྤེལ་ལ་ཕྱོགས་ཡོངས་ནས་འཕེལ་རྒྱས་བྱུང་ཞིང་། རིགས་དང་གྲངས་ཀ་གང་གི་ཐད་ནས་ཀྱང་སྔ་མའི་རྒྱལ་རབས་རིམ་བྱུང་ལས་བརྒལ་འདུག་པ་དང་། བོད་བརྒྱུད་ནང་བསྟན་གྱིས་སྔར་ལྟར་ཟམ་པ་དང་སྒྲོག་ཐག་གི་ནུས་པ་ཐོན་ཡོད་དེ། དཔེར་ན། རྒྱལ་རྩེ་ས་ཆར་ས་གདན་ཐོན་སྐྱེད་བྱེད་པ་ལ་ནང་ལོགས་ས་ཁུལ་གྱི་དར་གོས་རྒྱན་རིས་ཀྱི་ཤུགས་རྐྱེན་ཐེབས་ཏེ། ས་གདན་སྟེང་དུ་གཡུ་འབྲུག་ཁ་སྤྲོད་དང་། པདྨ་རྒྱ་བྱ། ཁྱུང་འབྲུག་ཁ་སྤྲོད་སོགས་ཀྱི་རི་མོ་ཐོན་ཡོད་ལ།⑮རི་མོ་དེ་དག་ཀྱང་ཀྲུང་ཧྭ་མི་རིགས་ཐུན་མོང་ལ་ཡོད་པའི་བཀྲ་ཤིས་པའི་མཚོན་རྟགས་ཤིག་ཏུ་གྱུར་ཡོད། ནང་ལོགས་ས་ཁུལ་གྱི་དར་གོས་བཟོ་རྩལ་དང་བོད་ཀྱི་སྲོལ་རྒྱུན་ཐང་ག་གཉིས་ཟུང་འབྲེལ་བྱས་ནས་ཐང་ག་གསར་པ་སྟེ་འཚེམ་དྲུབ་ཐང་ག་བྱུང་། བོད་ཀྱི་བཟོ་རིག་མཛེས་རྩལ་ཡང་ནང་ལོགས་ས་ཁུལ་དུ་དར་ཁྱབ་བྱུང་བ་དང་། ཁྱད་པར་དུ་བོད་བརྒྱུད་ནང་བསྟན་གྱི་སྒྱུ་རྩལ་ནང་ལོགས་ས་ཁུལ་དུ་དར་ཁྱབ་བྱུང་བ་དེས་ནང་ལོགས་ས་ཁུལ་གྱི་བཟོ་རིག་མཛེས་རྩལ་ལ་ཤུགས་རྐྱེན་ངེས་ཅན་ཞིག་ཐེབས་ཡོད། དཔེར་ན། ནང་ལོགས་ས་ཁུལ་གྱི་བཟོ་རྩལ་ཐོན་རྫས་ཁྲོད་དུ་བོད་བརྒྱུད་ནང་བསྟན་གྱི་བཀྲ་ཤིས་པའི་དོན་ཅན་གྱི་རི་མོ་དང་བརྙན་རིས་སོགས་འབོར་ཆེན་སྡུད་ལེན་བྱས་ཡོད་པ་ལྟ་བུའོ། །
        ཆིང་རྒྱལ་རབས་ཀྱི་དུས་སུ་བོད་ཡིག་གི་བཀའ་བསྟན་སོགས་རྒྱ་དང་སོག་པོ། མན་ཇུ་སོགས་མི་རིགས་བར་དུ་བསྒྱུར་བ་དང་བསྟུན་ནས། བོད་ཀྱི་རྩོམ་རིག་དེ་ནང་ལོགས་ས་ཁུལ་གྱི་མི་རིགས་ཁག་བར་དུ་བརྗེ་སྤེལ་བྱེད་པ་ཡང་ཇེ་མང་དུ་གྱུར། ཚུལ་དེ་རྒྱ་བོད་སོག་མན་སོགས་ཡིག་རིགས་སྣ་ཚོགས་ཀྱི་ཚད་ལྡན་ཡིག་ཆའི་དཔར་འགྲེམ་དང་ཕན་ཚུན་བར་ལོ་ཙཱ་བྱས་པ་སོགས་ལས་མངོན་ཡོད་པར་མ་ཟད། རྒྱ་བོད་སོགས་མི་རིགས་རྩོམ་རིག་གི་མཉམ་འདྲེས་སྟེང་དུ་ཡང་མངོན་ཡོད། རྒྱ་ཡིག་གི་རྩོམ་རིག་བརྩམས་ཆོས་མང་པོ་སྟེ་དཔེར་ན་《ཆུ་ངོགས་གཏམ་རྒྱུད་》དང་། 《རྒྱལ་ཁབ་གསུམ་གྱི་གཏམ་རྒྱུད》 《ཐང་སིན་བླ་མའི་རྣམ་ཐར》 《ངོ་མཚར་འདྲེ་སྒྲུང》 《པའོ་ཀུང་གྱོད་བཤད་》སོགས་བོད་ཡུལ་དུ་ཁྱབ་པ་དང་བོད་ཀྱི་དམངས་ཁྲོད་སྒྱུ་རྩལ་པ་མང་པོས་“རྒྱ་སྒྲུང་”ངམ་“རྒྱ་ཡི་གཏམ་རྒྱུད་”བཤད་ནས་སྐད་གྲགས་འབར་ཡོད།⑯
མི་སེར་རྒྱལ་ཁབ་ཀྱི་སྐབས་སུ། ནང་བསྟན་རིག་གནས་བརྗེ་སྤེལ་བྱེད་པ་ནི་བོད་ལྗོངས་རིག་གནས་ཀྱིས་ནང་ལོགས་ས་ཁུལ་དང་མི་རིགས་གཞན་པའི་རིག་གནས་བར་སྤེལ་རེས་བྱེད་པའི་ཕྱོགས་གལ་ཆེན་ཞིག་ཏུ་གྱུར་ཡོད། སོག་བོད་ནང་བསྟན་གྱི་སྐྱེས་ཆེན་དམ་པ་རྣམས་ཀྱིས་ཀྲུང་དབྱང་སྲིད་གཞུང་གི་སྲིད་ཇུས་དང་མ་དངུལ་གྱི་སྐུལ་ལྕག་དང་རྒྱབ་སྐྱོར་འོག་ཟམ་མ་ཆད་པར་ནང་ལོགས་ས་ཁུལ་དུ་ཕེབས་ནས་ཆོས་འཁོར་བསྐོར་བ་དང་། ནང་བསྟན་སློབ་གླིང་དུ་བཀའ་ཆོས་གསུངས་ཤིང་རྒྱལ་སྲུང་གནོད་སེལ་དང་ལུགས་གཉིས་བསླབ་བྱ་སོགས་ཀྱི་བྱེད་སྒོ་སྤེལ་བས། རྒྱ་བོད་ནང་བསྟན་སྤེལ་རེས་བྱེད་པ་དང་ཕྱོགས་གཉིས་མཉམ་འདྲེས་ཡོང་བ་གཏིང་ཟབ་ཏུ་སྤེལ་བ་བཅས་ཀྱི་ཐད་ནུས་པ་མི་དམན་པ་བཏོན། དེའི་ནང་དུ་སྐད་གྲགས་ཆེ་བ་ནི་པཎ་ཆེན་སྐུ་ཕྲེང་དགུ་པ་དང་། དགའ་ལྡན་བྱིན་ཆགས་གླིང་གི་བླ་མ་པེ་ཕུ་རིན། ལྷ་སའི་འབྲས་སྤུངས་དགོན་པའི་དགེ་བཤེས་རྡོ་རྗེ་གཅོད་པ། བོད་ལྗོངས་རི་བོ་ཆེ་ནོར་ལྷ་དགོན་པའི་ནོར་ལྷ་ཧོ་ཐོག་ཐུ་སྐུ་ཕྲེང་བདུན་པ། མི་ཉག་ཁམས་གསུམ་གྲགས་དགོན་པའི་གངས་དཀར་ཧོ་ཐོག་ཐུ་སྐུ་ཕྲེང་ལྔ་པ་སོགས་ཡོད་ལ། སྐྱེས་ཆེན་དམ་པ་དེ་དག་གིས་ནང་ལོགས་ས་ཁུལ་དུ་བོད་བརྒྱུད་ནང་བསྟན་དར་སྤེལ་བཏང་བར་མ་ཟད། རྒྱ་བརྒྱུད་ནང་བསྟན་རིག་གནས་སོགས་བོད་ལྗོངས་སུ་ཡང་དར་སྤེལ་བཏང་བ་རེད།
        རྒྱལ་དམངས་སྲིད་གཞུང་གིས་1935ལོ་དང་1942ལོར་སོ་སོར་《རྒྱ་བོད་གྲྭ་བཙུན་སློབ་གཉེར་ལ་འགྲོ་བར་ཁ་གསབ་བྱེད་པའི་སྒྲིག་སྲོལ་》དང་། 《སོག་བོད་ཨུ་ཡོན་ལྷན་ཁང་གིས་ནང་ལོགས་ས་ཁུལ་གྱི་གྲྭ་བཙུན་བོད་ལྗོངས་སུ་སློབ་གཉེར་དུ་མངག་གཏོང་དང་ཁ་གསབ་སྟེར་བའི་སྒྲིག་སྲོལ་》འགྲེམ་བསྟར་བྱས་ཏེ།⑰ རྒྱ་བརྒྱུད་ནང་བསྟན་གྱི་གྲྭ་བཙུན་བོད་ལྗོངས་སུ་སློབ་གཉེར་དུ་འགྲོ་བར་སྐུལ་བ་ལ་ལམ་ལུགས་དང་སྲིད་ཇུས་ཀྱི་སྟེང་ནས་འགན་ལེན་བྱས་པས། རྒྱ་བརྒྱུད་ནང་བསྟན་གྱི་གྲྭ་བཙུན་རྣམས་བོད་ལྗོངས་སུ་སོང་ནས་ཆོས་ལུགས་སློབ་གཉེར་བྱེད་པ་དེ་རིམ་བཞིན་མི་སེར་རྒྱལ་ཁབ་དུས་སྐབས་ཀྱི་རྒྱ་བོད་ནང་བསྟན་རིག་གནས་བརྗེ་སྤེལ་བྱེད་པའི་ནང་དོན་གལ་ཆེན་ཞིག་ཏུ་གྱུར། དེའི་ཁྲོད་དུ་མིང་གྲགས་ཆེ་བ་ནི་དགེ་བའི་བཤེས་གཉེན་ཏ་ཡུང་དང་།དགེ་བའི་བཤེས་གཉེན་ཧྥ་ཙུན། དགེ་བའི་བཤེས་གཉེན་ནེང་ཧའེ། དགེ་བའི་བཤེས་གཉེན་པེ་སུང་སོགས་ཡོད།
རྒྱལ་དམངས་སྲིད་གཞུང་གི་རྒྱབ་སྐྱོར་དང་རྒྱ་བོད་ནང་བསྟན་སྡེའི་མི་སྣའི་ཧུར་བརྩོན་འོག་ཏུ། པེ་ཅིན་ནང་བསྟན་བོད་ཡིག་སློབ་གླིང་དང་། ཧྲང་ཧའེ་བོད་ཡིག་སློབ་གླིང་། ཧྲང་ཧའེ་བྱང་ཆུབ་སློབ་ཚོགས་བཅས་དང་། གཞན་ད་རུང་རྒྱ་བོད་ནང་བསྟན་གླིང་སོགས་ནང་ལོགས་ས་ཁུལ་དུ་རྒྱ་བོད་ནང་བསྟན་ཞིབ་འཇུག་ལས་ཁུངས་མང་པོ་ཞིག་གཅིག་རྗེས་གཉིས་མཐུད་དུ་ཚུགས་ཤིང་། ནང་བསྟན་ཞིབ་འཇུག་ལས་ཁུངས་དེ་དག་གིས་བོད་ཡིག་དང་བོད་ཀྱི་གསང་སྔགས་ལ་སློབ་སྦྱོང་བྱེད་པ་གཙོ་བོར་འཛིན་པ་དང་། འཇོན་མི་སྐྱེད་སྲིང་དང་རྒྱ་བོད་རིག་གནས་བརྗེ་སྤེལ། རྒྱ་བོད་རིག་གནས་བརྗེ་རེས་ལ་སྐུལ་འདེད་བྱེད་པ་བཅས་ཀྱི་ཐད་དུ་ནུས་པ་གལ་ཆེན་བཏོན་ཡོད།

        ཀྲུང་ཧྭ་མི་རིགས་ཀྱི་ལོ་རྒྱུས་འཕེལ་རིམ་ལ་བལྟས་ན། རྒྱ་བོད་སོགས་མི་རིགས་ཁག་གི་རིག་གནས་བརྗེ་རེས་མཉམ་འདྲེས་དང་། སྒོས་སུ་རྒྱ་བོད་ནང་བསྟན་གྱི་བརྗེ་རེས་མཉམ་འདྲེས་ལས་ཆབ་སྲིད་རྒྱུན་སྐྱོང་གི་བྱེད་ལས་དང་། མི་རིགས་མཐུན་སྒྲིལ་གྱི་བྱེད་ལས། རིག་གནས་མཉམ་འདྲེས་ཀྱི་བྱེད་ལས་བཅས་འདོན་སྤེལ་བྱས་ཤིང་། ཀྲུང་དབྱང་སྲིད་གཞུང་དང་བོད་ལྗོངས་ས་གནས་གཉིས་མཉམ་དུ་སྦྲེལ་བའི་ཆབ་སྲིད་སྒྲོག་ཐག་ཏུ་གྱུར་ཏེ། ཀྲུང་དབྱང་སྲིད་གཞུང་དང་བོད་ལྗོངས་ས་གནས་གཉིས་ཆབ་སྲིད་ཀྱི་སྟེང་ནས་ཕན་ཚུན་དམ་པོར་སྦྲེལ་ཅིང་། ཀྲུང་དབྱང་སྲིད་གཞུང་གིས་བོད་ལྗོངས་ས་གནས་ལ་སྤྱོད་པའི་བདག་དབང་སྲུང་སྐྱོང་ནུས་ལྡན་བྱས་པ་དང་མེས་རྒྱལ་གཅིག་གྱུར་སྲུང་སྐྱོང་ནུས་ལྡན་བྱས་པ་རེད།རྒྱ་བོད་ནང་བསྟན་སོགས་མི་རིགས་རིག་གནས་ཀྱི་བརྗེ་རེས་མཉམ་འདྲེས་དེ་མི་རིགས་མཐུན་སྒྲིལ་ཡོང་བའི་སྐུལ་རྫས་དང་འདྲེས་རྫས་ལྟ་བུར་གྱུར་ལ། རིག་གནས་སྟེང་གི་མཉམ་སྡུད་མཉམ་གསོག་གིས་དཔལ་འབྱོར་སྟེང་གི་ཕན་ཚུན་གཅིག་ལ་གཅིག་བརྟེན་དང་། བརྩེ་དུང་སྟེང་གི་ཕན་ཚུན་ཉེ་བར་བཅར་བ་གཉིས་སྐུལ་སྤེལ་དང་གཏིང་ཟབ་ཏུ་བཏང་བས། ཁྱོད་ཀྱི་ཁྲོད་དུ་ང་ཡོད་ལ་ང་ཡི་ཁྲོད་དུ་ཁྱོད་ཡོད་ཅིང་། སུ་ཡང་སུ་ལ་འབྲལ་ཐབས་མེད་པའི་སྣ་མང་གཞི་གཅིག་ཅན་གྱི་རྣམ་པ་ཆགས་པར་བྱས། རྒྱ་བོད་ནང་བསྟན་སོགས་མི་རིགས་རིག་གནས་ཀྱི་བརྗེ་རེས་མཉམ་འདྲེས་དེ་རིག་གནས་བརྗེ་སྤེལ་བྱེད་པའི་ཟམ་པ་ལྟ་བུར་གྱུར་ལ། ཁྱེད་དགའ་ང་སྐྱིད་ཀྱིས་གཅིག་འདྲེས་སུ་སོང་ནས། བོད་ལྗོངས་རིག་གནས་ཀྱིས་རྒྱ་བོད་སོགས་ཀྱི་རིག་གནས་བརྗེ་སྤེལ་ཁྲོད་རིམ་བཞིན་རྒྱའི་རིག་གནས་དང་རིག་གནས་གཞན་པ་སྡུད་ལེན་དང་སྒྲིག་བཅོས་བྱས་ཤིང་། དེ་ནས་རིག་གནས་རང་རྟོགས་བརྒྱུད་དེ་རང་ཉིད་གསར་འཛུགས་དང་རང་ཉིད་ཐོད་བརྒལ་མངོན་འགྱུར་བྱས་ནས་ཀྲུང་ཧྭའི་རིག་གནས་ལ་ཁྱད་ཆོས་དང་ནང་དོན་ཕུན་སུམ་ཚོགས་པ་བསྩལ་ཏོ། །
དྲུག་པ། ཀྲུང་ཧྭའི་རིག་གནས་དར་འཕེལ་ཁྲོད་ཀྱི་བོད་བརྒྱུད་ནང་བསྟན་གྱི་འགན་འཁྲི།
        ཏང་གི་ཚོགས་ཆེན་ཐེངས་བཅུ་དགུ་པའི་སྙན་ཞུའི་ནང་བསྟན་དོན། “རིག་གནས་ནི་རྒྱལ་ཁབ་དང་མི་རིགས་ཤིག་གི་བླ་སྲོག་ཡིན། རིག་གནས་དར་རྒྱས་སུ་སོང་ན་རྒྱལ་ཁབ་དར་རྒྱས་སུ་འགྲོ་ལ། རིག་གནས་སྟོབས་ལྡན་དུ་སོང་ན་མི་རིགས་སྟོབས་ལྡན་དུ་འགྲོ། ཚད་མཐོ་བའི་རིག་གནས་རང་རྟོན་མེད་པ་དང་། རིག་གནས་ཀྱི་དར་རྒྱས་གོང་འཕེལ་མེད་ཚེ། ཀྲུང་ཧྭ་མི་རིགས་ཀྱི་རླབས་ཆེན་བསྐྱར་དར་ཡང་ཡོད་མི་སྲིད། ཀྲུང་གོའི་ཁྱད་ཆོས་ལྡན་པའི་སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་རིག་གནས་འཕེལ་རྒྱས་འགྲོ་ལམ་རྒྱུན་འཁྱོངས་དང་། མི་རིགས་ཡོངས་ཀྱི་རིག་གནས་གསར་སྐྲུན་དང་གསར་གཏོད་ཀྱི་གསོན་ཤུགས་ཡོངས་སུ་བསླངས་ནས། སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་རིག་གནས་སྟོབས་ལྡན་རྒྱལ་ཁབ་འཛུགས་སྐྲུན་བྱེད་དགོས།” ⑲ བོད་བརྒྱུད་ནང་བསྟན་ནི་ལོ་རྒྱུས་རིང་ཞིང་ཐུན་མོང་མ་ཡིན་པའི་ཆོས་ལུགས་རིག་གནས་ཤིག་ཡིན་པའི་ཆ་ནས། ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་སྣེ་ཁྲིད་དང་ཤུགས་རྐྱེན་འོག་རང་ཉིད་ཀྱི་ཐུན་མོང་མ་ཡིན་པའི་རིག་གནས་ནང་རྫོངས་དང་ཁྱད་གཤིས་གྲུབ་ཡོད་ལ། ཕྱོགས་ག་གེ་མོ་ནས་ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་སྲོག་ཤིང་དང་བླ་སྲོག་ཀྱང་གསལ་པོར་མངོན་ཡོད་པ་དང་། ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་ཁྱད་ཆོས་དང་རང་གཤིས་ཕུན་སུམ་ཚོགས་པར་བཏང་ཡོད། བོད་བརྒྱུད་ནང་བསྟན་ནི་ཀྲུང་ཧྭ་མི་རིགས་ཀྱི་ཁྱད་ཆོས་ལྡན་པའི་རིག་གནས་སྲུང་སྐྱོང་ཁུལ་ནང་གི་ཐུན་མིན་རིག་གནས་ཡིན་ལ། ཀྲུང་ཧྭའི་རིག་གནས་ཁྲོད་ཀྱི་སྟོབས་ཤུགས་གལ་ཆེན་ཞིག་ཀྱང་ཡིན། དེར་བརྟེན་བོད་བརྒྱུད་ནང་བསྟན་གྱིས་ཀྲུང་ཧྭའི་རིག་གནས་དར་རྒྱས་གོང་འཕེལ་ཁྲོད་ངེས་པར་དུ་ལོ་རྒྱུས་ཀྱི་འགན་འཁྲི་ཕྲག་ཏུ་ཁུར་ནས་རང་ཉིད་ཀྱི་ཐུན་མིན་ནུས་པ་འདོན་དགོས།

        རིག་གནས་རྒྱུ་མང་རང་བཞིན་དང་དཔལ་ཡོན་སྣ་མང་རང་བཞིན་ནི་འགྲོ་བ་མིའི་རིགས་ཀྱི་སྤྱི་ཚོགས་ལོ་ངོ་སྟོང་ཕྲག་དུ་མར་འཕེལ་འགྱུར་བྱུང་བའི་རིང་གི་དངོས་ཡོད་སྣང་ཚུལ་ཞིག་ཡིན་པས། དེང་དུས་ཀྱི་འཛམ་གླིང་གི་རྩ་བའི་ཁྱད་ཆོས་སུ་གྲུབ་ཅིང་། འགྲོ་བ་མིའི་རིགས་མུ་མཐུད་མདུན་སྐྱོད་བྱེད་པར་སྐུལ་འདེད་བྱེད་པའི་སྒུལ་ཤུགས་གཙོ་བོ་ཡང་ཡིན། མིའི་རིགས་ཀྱི་ལོ་རྒྱུས་འཕེལ་རྒྱས་སུ་ཕྱིན་པའི་གོ་རིམ་ནི་དཔལ་ཡོན་སྣ་ཚོགས་བསྟུད་མར་གདོང་ཐུག་དང་། བརྗེ་སྤེལ། མཉམ་འདྲེས།གསར་སྐྲུན་བཅས་བྱས་པའི་གོ་རིམ་ཡིན་པ་རེད།
        དེ་ལྟ་བུའི་རྒྱབ་ལྗོངས་འོག་ཏུ་བོད་བརྒྱུད་ནང་བསྟན་རིག་གནས་འཛུགས་སྐྲུན་གྱིས་ངེས་པར་དུ་ཀྲུང་ཧྭའི་རིག་གནས་འཕེལ་རྒྱས་ལ་བརྩི་བཀུར་བྱེད་དགོས་ཤིང་། ངེས་པར་དུ་ཀྲུང་གོ་ཅན་གྱི་ཁ་ཕྱོགས་རྒྱུན་འཁྱོངས་བྱས་ཏེ། རང་ཉིད་བཅོས་སྒྱུར་དང་འཐུས་ཚང་དུ་གཏོང་དགོས་པ་དང་། ཆབ་སྲིད་སྟེང་ནས་རང་འགུལ་གྱིས་ཁས་ལེན་དང་རིག་གནས་སྟེང་ནས་རང་འགུལ་གྱིས་མཉམ་འདྲེས། སྤྱི་ཚོགས་སྟེང་ནས་རང་འགུལ་གྱིས་མཐུན་འཇུག་བཅས་བྱས་ནས། སྤྱི་ཚོགས་འཕེལ་རྒྱས་དང་འགྲོས་མཉམ་པ་དང་དུས་རབས་ཡར་ཐོན་དང་འགྲོས་མཉམ་པ་བྱས་ཏེ། ཀྲུང་ཧྭའི་རིག་གནས་འཛུགས་སྐྲུན་བྱེད་པའི་འཆམ་མཐུན་གྱི་བྱེད་པོ་དང་ཧུར་བརྩོན་གྱི་སྟོབས་ཤུགས་སུ་ཆགས་པར་བྱེད་དགོས། རང་ཉིད་དུས་རབས་དང་མཐུན་པར་འཛུགས་སྐྲུན་བྱ་རྒྱུར་སྐུལ་སྤེལ་ཧུར་ཐག་བྱེད་པ་ལ། སྤྱི་ཚོགས་རང་ལུགས་ཀྱི་སྲོག་ཤིང་ལྟ་བ་སྤྱད་ནས་ཆོས་ལུགས་སྡེའི་མི་སྣ་དང་དད་ལྡན་མང་ཚོགས་ལ་སྣེ་ཁྲིད་དང་སློབ་གསོ་གཏོང་དགོས་པར་མ་ཟད། ད་རུང་ཆོས་ཁྲིམས་དང་ལྟ་དགོངས་ལ་དེང་རབས་ཀྲུང་གོའི་འཕེལ་རྒྱས་ཡར་ཐོན་གྱི་བླང་བྱ་དང་མཐུན་ལ། ཀྲུང་ཧྭའི་ཕུལ་བྱུང་སྲོལ་རྒྱུན་རིག་གནས་དང་ཡང་མཐུན་པའི་འགྲེལ་བཤད་རྒྱག་དགོས། འོན་ཀྱང་དེའི་དམིགས་ཡུལ་མཐར་ཐུག་ནི་རང་ཉིད་དང་སྤྱི་ཚོགས་རིང་ལུགས་སྤྱི་ཚོགས་གཉིས་མཐུན་ཐབས་བྱེད་པར་སྐུལ་འདེད་ཧུར་ཐག་བྱེད་པ་དེ་ཡིན།   བོད་བརྒྱུད་ནང་བསྟན་རང་ཉིད་བདེ་ཐང་དང་འཕེལ་རྒྱས་བྱུང་ཚེ་དེ་ནི་ཀྲུང་ཧྭའི་རིག་གནས་དར་རྒྱས་ཕུན་ཚོགས་སུ་འགྲོ་བར་ཡང་བྱས་རྗེས་བཞག་པའོ། །
        དུས་མཚུངས་སུ་བོད་བརྒྱུད་ནང་བསྟན་རིག་གནས་ཀྱིས་རིག་གནས་བརྗེ་སྤེལ་གྱི་བྱེད་ལས་སྤྱད་དེ་ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་གསར་སྐྲུན་ལ་སྐུལ་འདེད་བྱེད་ཆོག་ལ། ཀྲུང་ཧྭའི་རིག་གནས་ཀྱིས་དུས་རབས་ཀྱི་སྙིང་སྟོབས་སྡུད་ལེན་དང་མཐུན་འཇུག་བྱེད་པའི་ནང་དོན་ཕུན་སུམ་ཚོགས་པོ་དང་རིག་གནས་སྣ་མང་གི་དཔལ་ཡོན་གྲུབ་འབྲས་ལ་རྒྱབ་སྐྱོར་དང་རོགས་འདེགས་བྱས་ཏེ། དུས་རབས་ཀྱི་ནང་དོན་སྔར་བས་ཕུན་སུམ་ཚོགས་པོ་ལྡན་པ་དང་སྔར་བས་སྣ་མང་བའི་དཔལ་ཡོན་གྲུབ་འབྲས་དང་རིག་གནས་བསམ་བློ་བཅས་ཐོན་ཁོངས་རྩ་ཆེན་དུ་བརྩིས་ནས་སྡུད་ལེན་དང་མིག་དཔེར་བཟུང་ནས། ཀྲུང་ཧྭའི་རིག་གནས་མུ་མཐུད་དུ་ཕུན་སུམ་ཚོགས་པོ་དང་འཐུས་ཚང་དུ་གཏོང་དགོས། དུས་རབས་ཀྱི་གནད་དོན་ལ་དོ་ཁུར་བྱས་ཏེ། རིག་གནས་ཀྱི་འཕེལ་རྒྱས་དང་དེང་རབས་དངོས་པོའི་དཔལ་ཡོན། བསམ་པའི་དཔལ་ཡོན། ཆབ་སྲིད་དཔལ་ཡོན། སྐྱེ་ཁམས་དཔལ་ཡོན་བཅས་བར་གྱི་འབྲེལ་བར་བསམ་གཞིག་བྱེད་ཕོད་པའི་སྒོ་ནས་དུས་རབས་གོང་འཕེལ་དང་སྤྱི་ཚོགས་ཡར་ཐོན་གསར་སྐྲུན་ལ་རིག་ལམ་དང་གསོན་ཤུགས་སྦྱིན་དགོས།
        ད་ལྟ་ཀྲུང་གོའི་ཁྱད་ཆོས་ལྡན་པའི་སྤྱི་ཚོགས་རིང་ལུགས་དེ་དུས་རབས་གསར་པར་བསྐྱོད་ལ། དེ་ནི་རང་རྒྱལ་འཕེལ་རྒྱས་ཀྱི་ལོ་རྒྱུས་ཁ་ཕྱོགས་གསར་པ་རེད།   བོད་བརྒྱུད་ནང་བསྟན་སྡེས་ངེས་པར་དུ་ཡང་དག་པའི་ཆོས་ལུགས་ཀྱི་ལྟ་བ་འཛུགས་པ་དང་འཕེལ་རྒྱས་ཀྱི་རྣམ་པ་གསར་སྐྲུན་ཧུར་ཐག་བྱས་ནས། དེང་རབས་ཀྲུང་གོའི་འཕེལ་རྒྱས་ཡར་ཐོན་གྱི་བླང་བྱ་དང་མཐུན་པ་དང་ཀྲུང་ཧྭའི་ཕུལ་བྱུང་སྲོལ་རྒྱུན་རིག་གནས་ཀྱི་འཕེལ་རྒྱས་དང་ཡང་མཐུན་པའི་ལམ་དུ་བགྲོད་པ་དང་། ཆོས་ལུགས་དུས་རབས་ཅན་དང་སྒྲིག་ལམ་རྒྱུན་ལྡན་ཅན་གྱི་ལམ་དུ་བགྲོད་དེ། ཆོས་ལུགས་བར་གྱི་འབྲེལ་བ་འཆམ་མཐུན་ངང་མངོན་འགྱུར་ཧུར་ཐག་བྱས་ཏེ།   ཁྱད་ཆོས་མངོན་པར་གསལ་ཞིང་ཕྱི་མཛེས་ནང་གྲུང་། འཆམ་མཐུན་དཀྱེལ་ཡངས། དཔལ་ཡོན་ཡར་ཐོན་ལྡན་པའི་བོད་བརྒྱུད་ནང་བསྟན་གྱི་སྣང་བརྙན་གསར་པ་བསྐྲུན་ཏེ། ཀྲུང་ཧྭའི་རིག་གནས་ཀྱི་དར་རྒྱས་དང་གོང་འཕེལ་ལ་རང་ཉིད་ཀྱིས་འཇོག་འོས་པའི་བྱས་རྗེས་འཇོག་དགོས།
