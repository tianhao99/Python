རི་ཆེན་ཇོ་མོ་གླང་མ་ངོ་སྤྲོད་མདོར་བསྡུས།
ཇོ་མོ་གླང་མ།
         གངས་རི་དེའི་མིང་ཡོངས་ལ་ཇོ་མོ་གླང་མ་རི་བོ་ཞེས་གྲགས་ཤིང་དེ་ནི་རང་རྒྱལ་གྱི་གཞུང་རྒྱུག་རི་རྒྱུད་ཆེ་བ་གསུམ(ཁུ་ནུ་ལའི་རི་རྒྱུད་དང་། གངས་ཏི་སེའི་རི་རྒྱུད། ཧི་མ་ལ་ཡའི་རི་རྒྱུད)དུ་གྲགས་པའི་ནང་ཚན་བོད་ཀྱི་ལྷོ་མཐའི་ཧི་མ་ལ་ཡའི་རི་རྒྱུད་ཀྱི་རྒྱུད་དུ་ཆགས་ཡོད།          ཇོ་མོ་གླང་མ་ནི་དེ་སྔ་སྐམ་ས་ཞིག་མིན་པར་གནའ་བོའི་རྒྱ་མཚོ་ཞིག་ཡིན་པ་དང་། རྒྱ་མཚོ་དེའི་ལྷོ་དང་བྱང་ཕྱོགས་གཉིས་ནས་གཙང་ཆུ་ལྷུང་ལྷུང་དུ་འབབ་ཀྱི་ཡོད་པས་མཚོ་མོའི་ནང་དུ་འདམ་རྫབ་བྱེ་མ་ལ་སོགས་འབོར་ཆེན་ཞིག་ཟམ་མི་ཆད་པར་མཚོ་ནང་ཁྱེར་ཡོང་བའི་དབང་གིས་དེ་དག་རྗེས་སུ་རིམ་བཞིན་བྲག་རྡོར་གྱུར་ཏེ་ཇོ་མོ་གླང་མ་རི་བོ་འདི་ལྟ་བུ་ཞིག་ཆགས་རྒྱུར་དངོས་པོའི་རྨང་གཞི་ལྷུན་གྱིས་གྲུབ་པ་སྐྲུན་ཡོད་པ་རེད།       བཤད་ཚུལ་དེ་ནི་སྔར་གྱི་ལོ་རྒྱུས་ཀྱི་ཡིག་རིགས་ལས་བོད་འདི་ཐོག་མར་ཆུ་ཡི་ཀློང་ཆེན་པོ་ཞིག་ཏུ་གྲུབ་པ་ལ་དེ་རྗེས་རིམ་བཞིན་ཆུ་རྣམས་གོང་རོལ་དུ་སོང་ཞིང་ས་ཡི་ཁམས་ཆེན་པོ་ཆགས་པ་ཡིན་ཞེས་པ་དང་ཡོངས་སུ་མཐུན་ཞིང་། ཧ་ལམ་ལོ་ངོ་དུང་ཕྱུར་བཅུ་ཕྲག་ལྷག་འདས་ཟིན་པ་དང་། ད་ལྟའི་བར་དུ་ལོ་ཁྲི་ཚོ་སུམ་སྟོང་ཡས་མས་སོང་མཚམས་ཀྱི་རྒྱུན་ལྡན་མིན་པའི་ལོ་ཟླར་ནམ་མཁའ་དབྲལ་དང་མཚུངས་པའི་འབྲུག་སྒྲ་ལྡིར་བ་དང་། སའི་གོ་ལ་ཧྲིལ་བོ་གཡོ་ཞིང་འགུལ་བའི་དུས་སུ་ཇོ་མོ་གླང་མ་འདི་བཞིན་ས་ཡི་གོ་ལའི་ངོས་ལ་དངོས་སུ་བལྟམས་ནས་ཧི་མ་ལ་ཡའི་རི་རྒྱུད་དུ་ཀེ་རེར་ལངས་འགོ་ཚུགས་སོ།།      ཇོ་མོ་གླང་མ་ནི་འཛམ་གླིང་གི་རི་བོ་མཐོ་ཤོས་ཡིན་ཞིང་མཐོ་ཚད་ལ་སྤྱི་ཁྲེ༨༨༤༤.༤༣ཡོད། ཧི་མ་ལ་ཡའི་རི་རྒྱུད་ཀྱི་དཀྱིལ་རྒྱུད་ཀྲུང་བལ་མཚམས་ཀྱི་རྒྱུད་དུ་ཆགས་ཡོད། དེང་སྐབས་སྲིད་འཛིན་ས་ཁུལ་ནས་བཤད་ན་གཞིས་རྩེ་ཀྲོན་ཆུས་དིང་རི་རྫོང་ཁུངས་སུ་ཡོད་ཅིང་། གཞིས་རྩེ་གྲོང་ཁྱེར་ནས་ནུབ་ངོས་ཇོ་མོ་གླང་མའི་རི་བོ་བར་སྤྱི་ལེ་ཉིས་བརྒྱ་ལྷག་ཙམ་ཡོད། རི་བོ་འདི་ནི་ལྷུན་པོའི་ཚོགས་ལས་མཐོན་པོར་འཕགས་ཤིང་དུས་བཞིར་དབུ་རྩེར་ཁ་བའི་མདངས་འཛིན་ལ་ན་བུན་སྤྲིན་གྱི་ཅོད་པན་གྱིས་མཛེས་པ་སོགས་ངོ་མཚར་ཆེ་བའི་བཀོད་པ་དུ་མས་ཕྱུག་ཡོད།    ཇོ་མོ་གླང་མ་ནི་རྒྱལ་ཁབ་ཕྱི་ནང་གི་ཡུལ་སྐོར་བ་རྣམས་ཡུལ་སྐོར་བྱ་ཡུལ་གྱི་གནས་ཁྱད་དུ་འཕགས་པ་ཞིག་ཡིན་པ་ལགས་སོ།།
