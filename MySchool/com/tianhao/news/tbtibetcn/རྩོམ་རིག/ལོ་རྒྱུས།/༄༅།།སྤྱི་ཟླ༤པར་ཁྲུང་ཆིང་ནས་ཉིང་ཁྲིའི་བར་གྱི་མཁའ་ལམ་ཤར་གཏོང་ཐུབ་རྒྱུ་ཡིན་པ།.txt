༄༅།།སྤྱི་ཟླ༤པར་ཁྲུང་ཆིང་ནས་ཉིང་ཁྲིའི་བར་གྱི་མཁའ་ལམ་ཤར་གཏོང་ཐུབ་རྒྱུ་ཡིན་པ།
བོད་རང་སྐྱོང་ལྗོངས་ལྗོངས་རྒྱུ་ཁང་གི་འགན་འཛིན་གཞོན་པ་ཝང་སུང་ཕིང་གིས་ངོ་སྤྲོད་གནང་བ་ལྟར་ན། བོད་ལྗོངས་ཀྱི་དུས་བཞིའི་ལྗོངས་རྒྱུའི་སྐོར་ལམ་དང་པོ་དེ་ན་ནིང་ཉིང་ཁྲིར་སྐྲུན་ཚར་བ་རེད། ལམ་བུ་དེའི་སྟེང་ནས་ལྗོངས་རྒྱུ་བ་རྣམས་ཀྱིས་འཛམ་གླིང་ཐོག་གི་གྲོགས་རོང་ཆེ་ཤོས་སུ་གྲགས་པའི་ཡར་རླུང་གཙང་པོའི་གྲོགས་རོང་དང་། ཀྲུང་གོའི་རི་བོ་མཛེས་ཤོས་སུ་གྲགས་པའི་གནམ་ལྕགས་འབར་བ་རི་བོ། ཀྲུང་གོའི་མི་འབོར་ཉུང་ཤོས་ཡིན་པའི་ལྷོ་པ་མི་རིགས་འདུས་སྡོད་བྱེད་ཁུལ་གྱི་གནས་ཡུལ་ལུང་པ་སོགས་འཛམ་གླིང་རིམ་པའི་ལྗོངས་རྒྱུའི་ཡུལ་ལྗོངས་ལ་གཟིགས་ཐུབ་ཡོད་ཅིང་། ལྗོངས་རྒྱུའི་སྐོར་ལམ་དེ་ནི་ན་ནིང་དགུན་དུས་སུ་ལྗོངས་རྒྱུའི་གཙོ་ཆེའི་ཐོན་རྫས་འགྲེམ་སྟོན་བྱེད་གནས་ཤིག་ཀྱང་ཡིན་པ་རེད།
        བོད་ལྗོངས་ཉིང་ཁྲི་ས་ཁུལ་ནི་རྒྱ་མཚོའི་ངོས་ལས་ཅུང་དམའ་བའི་སར་ཆགས་པ་དང་། གནམ་གཤིས་དྲོ་ཞིང་གཤེར་ཚད་འཚམ་པ། བརྒྱ་ཆའི༥༠ཙམ་ནགས་ཚལ་གྱིས་ཁེབས་ཡོད་པ། ལྗོངས་རྒྱུའི་ཐོན་ཁུངས་ཕུན་སུམ་ཚོགས་པ་སོགས་ཀྱི་དགེ་མཚན་གྱིས་ཕྱུག་པས་མི་རྣམས་ཀྱིས་རྒྱུན་དུ་དེར“བོད་ལྷོའི་མུ་ཏིག”ཅེས་འབོད་ཀྱི་ཡོད།
        གཞན་ཡང་ཤེས་རྟོགས་བྱུང་བ་ལྟར་ན། སྤྱི་ཟླ༤པའི་ཟླ་མཇུག་ལ་ཁྲུང་ཆིང་ནས་ཉིང་ཁྲིའི་བར་གནམ་གྲུ་ཐད་གཏོང་ཐུབ་རྒྱུ་ཡིན་པ་དང་། མིག་སྔར་ཁྲེང་ཏུའུ་དང་ཁུན་མིང་ནས་ཉིང་ཁྲིའི་བར་གནམ་གྲུ་ཐད་གཏོང་ཐུབ་ཀྱི་ཡོད་པ་རེད།
