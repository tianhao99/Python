ཀྲུང་དབྱང་དང་རང་སྐྱོང་ལྗོངས་ཀྱིས་བོད་ཀྱི་ཕུལ་བྱུང་སྲོལ་རྒྱུན་རིག་གནས་དང་སྒྱུ་རྩལ་རྒྱུན་འཛིན་དང་འཕེལ་རྒྱས་གཏོང་བར་ཐོག་མཐའ་བར་གསུམ་དུ་སྣང་ཆེན་གནང་གི་ཡོད།
དམངས་གཙོའི་བཅོས་སྒྱུར་བྱས་ནས་ལོ་ལྔ་བཅུའི་རིང་བོད་ཀྱི་སྲོལ་རྒྱུན་མི་རིགས་རོལ་དབྱངས་དང་། ཞབས་བྲོ། དམངས་ཁྲོད་གཏམ་རྒྱུད་དང་དེ་བཞིན་གནའ་གཞུང་དང་ཡིག་ཚགས་བཅས་སྔོག་འདོན་དང་ལེགས་སྒྲིག་སོགས་བྱེད་པར་མེ་ཏོག་ཀུན་བཞད་ཀྱི་རྣམ་པ་ཐོན་ཞིང་དུས་མཚུངས་བོད་རང་སྐྱོང་ལྗོངས་དང་དེ་བཞིན་ས་ཁུལ（གྲོང་ཁྱེར）ཁག་གིས་མི་རིགས་ཀྱི་ཕུལ་བྱུང་སྲོལ་རྒྱུན་རིག་གནས་ཤུལ་བཞག་མྱུར་སྐྱོབ་དང་། ལེགས་སྒྲིག ཞིབ་འཇུག་བཅས་ཀྱི་སྒྲིག་གཞི་བཙུགས་ནས་ལོ་རྒྱུས་ཐོག་བྱུང་མྱོང་མེད་པའི་གཞི་ཁྱོན་གྱི་ཐོག་ནས་མི་རིགས་ཀྱི་དམངས་ཁྲོད་རྩོམ་རིག་དང་སྒྱུ་རྩལ་ཤུལ་བཞག་མྱུར་སྐྱོབ་དང་། འཚོལ་བསྡུ། ལེགས་སྒྲིག ཞིབ་འཇུག རྩོམ་འབྲི པར་སྐྲུན་བཅས་ཀྱི་ལས་ཀ་བྱས་པས་བོད་ཀྱི་ཕུལ་བྱུང་སྲོལ་རྒྱུན་རིག་གནས་སྔར་བྱུང་མྱོང་མེད་པའི་གོང་འཕེལ་དར་རྒྱས་བྱུང་ཡོད། རགས་རྩིས་ལ་གཞིགས་ན།ལོ་ལྔ་བཅུའི་རིང་བོད་ཀྱི་ཕུལ་བྱུང་སྲོལ་རྒྱུན་གྱི་མི་རིགས་རིག་གནས་སྲུང་སྐྱོང་བྱེད་པར་རྒྱལ་ཁབ་ཀྱིས་མ་དངུལ་སྒོར་དུང་ཕྱུར་༡༣བཏང་ཡོད།
     མ་དངུལ་འབོར་ཆེན་བཏང་ནས་རིག་དངོས་གནའ་ཤུལ་སྲུང་སྐྱོང་བྱེད་པ། 
     བོད་དུ་དམངས་གཙོའི་བཅོས་སྒྱུར་བྱས་རྗེས་རིག་དངོས་སྲུང་སྐྱོང་ལས་ཀར་ཀྲུང་དབྱང་མི་དམངས་སྲིད་གཞུང་གིས་ཚད་མཐོའི་མཐོང་ཆེན་གནང་བ་དང་ལྷག་པར་དུ་དུས་རབས་གོང་མའི་ལོ་རབས་བརྒྱད་ཅུ་པ་ནས་བཟུང་རྒྱལ་ཁབ་ཀྱིས་མ་དངུལ་འབོར་ཆེན་བཏང་ནས་བོད་ཀྱི་གལ་ཆེའི་རིག་དངོས་གནའ་ཤུལ་ལ་སྲུང་སྐྱོང་དང་ཉམས་གསོ་བྱས་ཡོད། དུས་རབས་གོང་མའི་ལོ་རབས་བརྒྱད་ཅུ་པ་དང་དགུ་བཅུ་པར་ཀྲུང་དབྱང་མི་དམངས་སྲིད་གཞུང་གིས་མ་དངུལ་མི་དམངས་ཤོག་སྒོར་དུང་ཕྱུར་༣ལྷག་ཙམ་བཏང་ནས་བོད་ལ་དགོན་སྡེ་༡༤༠༠ལྷག་ཙམ་ཉམས་གསོ་དང་སྒོ་དབྱེ་བྱེད་རོགས་བྱས་པ་དང་བཀྲིས་ལྷུན་དགོན་པ་དང་།ས་སྐྱ་དགོན་པ། བསམ་ཡས་དགོན་པ། གཙུག་ལག་ཁང་། རྒྱལ་རྩེའི་རྫོང་རི་དབྱིན་འགོག་གནའ་ཤུལ་སོགས་གནའ་བོའི་བཟོ་སྐྲུན་དང་གནའ་ཤུལ་ལ་གཙོ་གནད་དུ་ཉམས་གསོ་དང་སྲུང་སྐྱོང་བྱས་ཡོད། ལྷག་པར་དུ་༡༩༨༩ལོ་ནས་༡༩༩༤ལོའི་བར་རྒྱལ་ཁབ་ཀྱིས་མ་དངུལ་མི་དམངས་ཤོག་སྒོར་ཁྲི་༥༥༠༠དང་གསེར་དངུལ་སོགས་རྩ་ཆེའི་དངོས་པོ་འབོར་ཆེན་བཏང་སྟེ་ཕོ་བྲང་པོ་ཏ་ལར་གཞི་ཁྱོན་ཆེ་བའི་ཉམས་གསོ་ཐེངས་གཅིག་བྱས་པ་དང་༢༠༠༡ལོ་ནས་བཟུང་ཕོ་བྲང་པོ་ཏ་ལ་དང་། ནོར་བུ་གླིང་ཁ། ས་སྐྱ་དགོན་པ་བཅས་རིག་དངོས་གནའ་ཤུལ་ཆེ་ཁག་གསུམ་ཉམས་གསོ་བྱེད་པར་སླར་ཡང་ཆེད་དངུལ་མི་དམངས་ཤོག་སྒོར་དུང་ཕྱུར་༣.༨བཏང་ཡོད།
       ༢༠༠༦ལོ་ནས་༢༠༡༠ལོའི་བར་བོད་ཀྱི་གཙོ་གནད་རིག་དངོས་སྲུང་སྐྱོང་ཚན་པ་ཁག་༢༢ལ་ཉམས་གསོ་དང་སྲུང་སྐྱོང་བྱ་ཕྱིར་ཀྲུང་དབྱང་མི་དམངས་སྲིད་གཞུང་གིས་སླར་ཡང་མ་དངུལ་སྒོར་དུང་ཕྱུར་༥.༧བཀོད་སྒྲིག་བྱས་ཤིང་།   མ་དངུལ་འབོར་ཆེན་དེ་ལྟ་བུ་དང་ཉམས་གསོ་གཞི་ཁྱོན་དེ་ལྟ་བུ་བྱས་པ་ནི་ཀྲུང་གོའི་རིག་དངོས་སྲུང་སྐྱོང་ལོ་རྒྱུས་ཐོག་བྱུང་མྱོང་མེད་པ་ཞིག་རེད། 
     མི་རིགས་དང་དམངས་ཁྲོད་སྲོལ་རྒྱུན་རིག་གནས་ལ་ཆེས་གཅེས་སྐྱོང་བྱེད་པ།
 ༡༩༥༠ལོར་དམག་དཔུང་དང་མཉམ་དུ་བོད་དུ་ཡོང་མཁན་མི་རིགས་ཁག་གི་རིག་གནས་ལས་དོན་པས་བོད་རིགས་རིག་རྩལ་ལས་དོན་པ་དང་མཉམ་དུ་དམངས་ཁྲོད་དུ་འཚོལ་བསྡུ་བྱེད་པར་བསྐྱོད་དེ་རོལ་དབྱངས་དང་། ཞབས་བྲོ། དམངས་ཁྲོད་གཏམ་རྒྱུད། གཏམ་དཔེ། དམངས་གཞས་སོགས་དཔྱད་གཞིའི་ཡིག་རིགས་འབོར་ཆེན་འཚོལ་བསྡུ་བྱས་ནས《བོད་ཀྱི་དམངས་གཞས》དང《བོད་ཀྱི་དམངས་ཁྲོད་རོལ་དབྱངས་ནང་མ》སོགས་རིག་རྩལ་གྱི་དཔེ་དེབ་རིམ་མཐུད་ནང་ལེགས་སྒྲིག་དང་པར་སྐྲུན་བྱས་པས་དུས་རབས་ཕྱེད་ཀ་ལྷག་ཙམ་རིང་རྒྱལ་ཁབ་ཀྱིས་བོད་ཀྱི་ཕུལ་བྱུང་སྲོལ་རྒྱུན་རིག་གནས་སྲུང་སྐྱོང་བྱེད་པས་དགའ་འོས་པའི་ལེའུ་ཞིག་བརྩམས་ཡོད། 
      དུས་རབས་གོང་མའི་ལོ་རབས་བདུན་ཅུ་པ་ནས་བཟུང་རྒྱལ་ཁབ་ཀྱིས་བོད་ཀྱི་མི་རིགས་དམངས་ཁྲོད་རིག་གནས་ཤུལ་བཞག་ལ་བརྒྱུད་རིམ་དང་གཞི་ཁྱོན་ལྡན་པའི་ཡོངས་ཁྱབ་ཞིབ་བཤེར་དང་།འཚོལ་བསྡུ། འཚོལ་ཞིབ་ཐོ་འགོད། ལེགས་སྒྲིག་བཅས་ཀྱི་ལས་ཀ་བྱས། དུས་རབས་གོང་མའི་ལོ་རབས་བརྒྱད་ཅུ་པ་ནས་བཟུང་རང་སྐྱོང་ལྗོངས་དང་དེ་བཞིན་ས་ཁུལ་དང་གྲོང་ཁྱེར་ཁག་གིས་སྔར་བྱུང་མྱོང་མེད་པའི་གཞི་ཁྱོན་ལྡན་པའི་མྱུར་སྐྱོབ་དང་། འཚོལ་བསྡུ། ལེགས་སྒྲིག   ཞིབ་འཇུག རྩོམ་འབྲི། དཔེ་སྐྲུན་བཅས་ཀྱི་ལས་ཀ་བྱས་ནས་སྔ་རྗེས་སུ་
ལས་དོན་ཚོགས་ཆུང་བཅུ་ཕྲག་ཁ་ཤས་ལྗོངས་ཡོངས་ཀྱི་གྲོང་ཁྱེར་དང་གྲོང་རྡལ་དང་། ཤང་གྲོང་། དགོན་སྡེ་བཅས་སུ་ཕྱོགས་ཡོངས་ནས་བརྟག་དཔྱད་དང་འཚོལ་ཞིབ་ཐོ་འགོད་བྱེད་པར་བཏང་ནས་རྒྱ་བོད་ཡིག་རིགས་གཉིས་ཀྱི་དཔྱད་གཞིའི་ཡིག་རིགས་ཡིག་འབྲུ་ཁྲི་༣༠༠༠ལྷག་ཙམ་འཚོལ་བསྡུ་ལེགས་སྒྲིག་དང་། སྐད་པར་ཕེན་ཙི་དང་སྤྱིན་ཤོག་གི་དཔྱད་གཞིའི་ཡིག་རིགས་འབོར་ཆེན་བསྡུ་བཟོ། པར་ཁྲི་ཙམ་བརྒྱབ་པ་བཅས་བྱས་པ་བརྒྱུད་སྔ་རྗེས་སུ་བོད་མི་རིགས་ཀྱི་སྲོལ་རྒྱུན་རིག་གནས་དང་འབྲེལ་བ་ཡོད་པའི་རིག་གཞུང་ཆེད་རྩོམ་༡༠༠༠ལྷག་ཙམ་སྤེལ་བ་དང་། 《ཀྲུང་གོའི་ཟློས་གར་ལོ་རྒྱུས་བོད་ཀྱི་བམ་པོ》སོགས་རིག་རྩལ་གྱི་གཅིག་སྡུད་ལོ་རྒྱུས་དེབ་ཐེར་ཆེ་ཁག་བཅུ་དཔེ་སྐྲུན་བྱས་པ། མི་རིགས་རིག་རྩལ་གྱི་ཞིབ་འཇུག་ཆེད་རྩོམ་༣༠ལྷག་ཙམ་སྤེལ་བ་བཅས་རེད།
      བོད་ལྗོངས་སྤྱི་ཚོགས་ཚན་རིག་ཁང་མི་རིགས་ཞིབ་འཇུག་སུའོ་ཡི་སུའོ་ཀྲང་པ་སངས་དབང་འདུས་ཀྱིས“ད་ཆ་བོད་ཀྱིས་མི་རིགས་ཀྱི་ཕུལ་བྱུང་སྲོལ་རྒྱུན་རིག་གནས་སྲུང་སྐྱོང་དང་འཕེལ་རྒྱས་གཏོང་རྒྱུ་དེ་ཆུ་ཚད་མཐོ་བ་དང་། འཁོར་རྒྱུག་བཟང་བ། སྐྱོན་བྲལ་ངང་འཕེལ་རྒྱས་འགྲོ་བ་བཅས་ཀྱི་ལོ་རྒྱུས་ཀྱི་དུས་སྐབས་ཡག་ཤོས་སུ་སྐྱོད་འགོ་ཚུགས་ཡོད”ཅེས་གསུངས། 
      མི་རིགས་རིག་གནས་ཤུལ་བཞག་ལ་འོད་སྟོང་རབ་ཏུ་འཕྲོ་བ། 
      འཛམ་གླིང་ཐོག་སྐད་གྲགས་ལྡན་པའི《གླིང་གེ་སར་རྒྱལ་པོའི་སྒྲུང》ནི་ཀྲུང་གོ་དང་ཐ་ན་འཛམ་གླིང་གི་རྩོམ་རིག་ནོར་མཛོད་ནང་གི་རྩ་ཆེའི་རྩོམ་རིག་ཅིག་རེད། 
      རིག་གནས་ཀྱི་མཆོག་ནོར་དེ་སྲུང་སྐྱོང་བྱ་ཕྱིར་རང་སྐྱོང་ལྗོངས་ཀྱིས་༡༩༧༩ལོར《གླིང་གེ་སར་རྒྱལ་པོའི་སྒྲུང》མྱུར་སྐྱོབ་དང་ལེགས་སྒྲིག་བྱེད་མཁན་ཆེད་དོན་སྒྲིག་གཞི་བཙུགས་ཏེ་འཚོལ་བསྡུ་དང་། འཚོལ་ཞིབ་ཐོ་འགོད། ལེགས་སྒྲིག ཞིབ་འཇུག པར་སྐྲུན་བཅས་ཀྱི་ལས་ཀ་ཕྱོགས་ཡོངས་ནས་བྱེད་འགོ་ཚུགས། རྒྱལ་ཁབ་ཀྱིས《གེ་སར་སྒྲུང》དེ་རྩ་ཆེའི་ཚན་རིག་ཞིབ་འཇུག་གི་རྣམ་གྲངས་གྲས་སུ་བཅུག་པས་ཡུན་རིང་ནང་ཁ་ཐོར་དུ་བརྒྱུད་གཏོང་བྱེད་པའི་ངག་ཐོག་རྩོམ་རིག་དེ་བརྒྱུད་རིམ་ལྡན་ཞིང་ཆ་ཚང་བའི་བརྩམས་དེབ་ཆེན་པོ་ཞིག་ཏུ་འགྱུར་བ་དང་དུས་མཚུངས་རྒྱ་ཡིག་ཏུ་བསྒྱུར་བའི་དེབ་ཁག་ཉི་ཤུ་ལྷག་ཙམ་པར་སྐྲུན་བྱས་པ་མ་ཟད།ཁག་གཅིག་ནི་དབྱིན་ཡིག་དང་། རི་པིན་གྱི་ཡི་གེ ཕ་རན་སིའི་ཡི་གེ་བཅས་སུ་བསྒྱུར་ནས་པར་སྐྲུན་ཡང་བྱས་ཡོད། དེ་ནི་བོད་རིགས་ཀྱི་དམངས་ཁྲོད་རིག་གནས་ཤུལ་བཞག་སྲུང་སྐྱོང་དང་དེ་བཞིན་པར་སྐྲུན་བྱས་པའི་ལོ་རྒྱུས་ཐོག་བྱུང་མྱོང་མེད་པ་ཞིག་རེད། 
     ད་ཆ་བོད་ཀྱི་སྲོལ་རྒྱུན་མི་རིགས་རིག་གནས་སྲུང་སྐྱོང་བྱ་རྒྱུའི་ལས་ཀ་དེ་ཁྱབ་ཁོངས་དེ་བས་ཆེ་བ་དང་རིམ་པ་དེ་བས་ཟབ་པའི་ཐོག་བྱེད་བཞིན་ཡོད་པ་དང་རྒྱལ་ཁབ་ཀྱིས་མངོན་མིན་རིག་གནས་ཤུལ་བཞག་སྲུང་སྐྱོང་འདུ་འགོད་ལག་ལེན་བསྟར་ནས་སྔ་རྗེས་སུ་བོད་དུ་གཅུང་པ་ལྷ་མོ་སོགས་རྒྱལ་ཁབ་རིམ་པའི་དམངས་ཁྲོད་སྒྱུ་རྩལ་ཕ་ཡུལ་ལྔ་དང་ཁྱད་ལྡན་སྒྱུ་རྩལ་ཕ་ཡུལ་གཅིག་གི་མིང་བཏགས་ཡོད།   ད་བར་ལྗོངས་དེ་ཡོངས་སུ་རྒྱལ་ཁབ་རིམ་པའི་མངོན་མིན་རིག་གནས་ཤུལ་བཞག་གི་མཚོན་བྱེད་༦༠དང་རང་སྐྱོང་ལྗོངས་རིམ་པའི་མངོན་མིན་རིག་གནས་ཤུལ་བཞག་གི་མཚོན་བྱེད་༡༢༡ཡོད་པ་དང་གླིང་གེ་སར་རྒྱལ་པོའི་སྒྲུང་བརྒྱུད་གཏོང་བྱེད་མཁན་སྒྱུ་རྩལ་པ་ཚེ་རིང་དབང་འདུས་སོགས་མི་༣༡སྔ་རྗེས་སུ་རྒྱལ་ཁབ་རིམ་པའི་མངོན་མིན་རིག་གནས་ཤུལ་བཞག་རྣམ་གྲངས་ཀྱི་མཚོན་བྱེད་རང་བཞིན་གྱི་རྒྱུན་འཛིན་པར་གྱུར་ཡོད། 
     དམངས་ཁྲོད་རིག་གནས་ཀྱི་གཞུང་ལུགས་འབོར་ཆེན་སྔོག་འདོན་དང་ལེགས་སྒྲིག་བྱུང་བ། 
     བོད་དུ་དམངས་གཙོའི་བཅོས་སྒྱུར་བྱས་ནས་ལོ་ལྔ་བཅུའི་རིང་དམངས་ཁྲོད་རིག་གནས་ཀྱི་གཞུང་ལུགས་འབོར་ཆེན་ཞིག་སྔོག་འདོན་དང་ལེགས་སྒྲིག་བྱུང་ཞིང་། ༡༩༥༩ལོའི་ཟླ་༦པར་རྒྱལ་སྲིད་སྤྱི་ཁྱབ་ཁང་གི་མཛུབ་སྟོན་གཞིར་བཟུང་བོད་རང་སྐྱོང་ལྗོངས་གྲ་སྒྲིག་ཨུ་ཡོན་ཨུ་ལྷན་ཁང་གིས་དེ་སྔའི་བོད་ས་གནས་སྲིད་གཞུང་དང་དེ་བཞིན་དེའི་ཁོངས་གཏོགས་ཀྱི་སྡེ་ཚན་ཁག་གི་ཡིག་ཆ་དང་ཡིག་ཚགས་ཀྱི་དཔྱད་གཞིའི་ཡིག་རིགས་དང་དེ་བཞིན་དགོན་སྡེ་ཁག་དང་སྐུ་དྲག་གིས་གཅེས་ཉར་པའི་ཡིག་ཆ་དང་ཡིག་ཚགས་ལེགས་སྒྲིག་དང་། མྱུར་སྐྱོབ། འཚོལ་བསྡུ། བདག་གཉེར་བཅས་བྱེད་འགོ་ཚུགས་ཏེ་གཅེས་ཉར་ཡིག་ཚགས་ཅུང་ཆ་ཚང་བ་ཞིག་བཟོས་ཡོད། ༡༩༨༤ལོར་ཀྲུང་དབྱང་མི་དམངས་སྲིད་གཞུང་གིས་མ་དངུལ་འབོར་ཆེན་བཏང་སྟེ་ནུས་པ་ཅུང་ཆ་ཚང་བ་དང་སྒྲིག་ཆས་ཅུང་དེང་རབས་ཅན་ཡིན་པའི་བོད་རང་སྐྱོང་ལྗོངས་ཡིག་ཚགས་ཁང་གསར་སྐྲུན་བྱས་པས་ཡིག་ཚགས་བདག་གཉེར་བྱེད་པའི་ཆ་རྐྱེན་ལེགས་སུ་ཤུགས་ཆེན་སོང་ཡོད། བོད་ཀྱི་རིམ་ཁག་དང་སྡེ་ཚན་ཁག་གིས་བོད་ཤོག་དང་། དར་འཇམ། ཤིང་། ལྕགས་རིགས། རྡོ། ལོ་མ་བཅས་ལས་གྲུབ་པའི་ཡིག་ཚགས་བམ་པོ་ཁྲི་༤༠༠ལྷག་ཙམ་གཅེར་ཉར་བྱས་པ་དེའི་ནང་༩༠%ལྷག་ཙམ་བོད་ཡིག་ཡིན་པ་ཕུད་ད་དུང་རྒྱ་ཡིག་དང་། མན་ཇུའི་ཡི་གེ སོག་པོའི་ཡི་གེ དབྱིན་ཏིའི་ཡི་གེ ལེགས་སྦྱར་ཡི་གེ བལ་པོའི་ཡི་གེ དབྱིན་ཇིའི་ཡི་གེ ཨུ་རུ་སུའི་ཡི་གེ་སོགས་ཡིག་རིགས་བཅུ་ལྷག་ཙམ་ཡོད་པ་དང་ཡིག་ཚགས་ནང་ཡོན་རྒྱལ་རབས་ནས་དེང་རབས་བར་གྱི་ནང་དོན་ལྡན་ཡོད་པས་དུས་རབས་ཀྱི་མ་ལག་ཆ་ཚང་བའི་ལོ་རྒྱུས་ཡིག་ཚགས་ཀྱི་ནོར་མཛོད་ཅིག་རེད།
    དུས་མཚུངས་བོད་ཡིག་གི་གནའ་གཞུང་པར་སྐྲུན་བྱ་རྒྱུར་གྲུབ་འབྲས་མངོན་གསལ་དོད་པོའང་ཐོབ་པ་དང་ད་ཆ་བོད་ཡིག་གི་གནའ་གཞུང་ཁག་༢༦༡མྱུར་སྐྱོབ་དང་། ལེགས་སྒྲིག དཔེ་སྐྲུན་བཅས་བྱས་པ་དང་ཡིག་འབྲུ་དུང་ཕྱུར་༡.༣ལྷག་ཙམ་ཟིན་ཡོད།
