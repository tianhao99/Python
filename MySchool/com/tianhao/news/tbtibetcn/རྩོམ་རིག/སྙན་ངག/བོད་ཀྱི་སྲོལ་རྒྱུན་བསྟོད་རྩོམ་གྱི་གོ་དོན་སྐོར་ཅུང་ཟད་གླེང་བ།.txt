བོད་ཀྱི་སྲོལ་རྒྱུན་བསྟོད་རྩོམ་གྱི་གོ་དོན་སྐོར་ཅུང་ཟད་གླེང་བ།
བསྟོད་པའི་རྩོམ་རིག་ཅེས་པའི་གོ་དོན་སྐོར་ལ་ད་ཡོད་ང་ཚོའི་མིག་ལམ་དུ་མཇལ་རྒྱུ་ཡོད་པའི་དཔྱད་གཞིའི་ཡིག་ཆ་ཁག་ལས་ཤེས་རྟོགས་བྱུང་བར་གཞིགས་ན། བོད་སྐད་དུ་བསྟོད་པ་ཞེས་པར་སྒྲ་ལྕི་ཡང་གི་ཁྱད་པར་སོགས་སྐབས་མི་འདྲ་བར་གཞིགས་ཏེ་མིང་ཚིག་དང་། བ་ཚིག་གཉིས་སུ་གོ་བ་ལེན་རྒྱུ་ཡོད་ཅིང་། དེ་ཡང་བ་ཚིག་གི་དབང་དུ་བཏང་སྟེ་བཤད་ན། བསྔགས་བརྗོད་བས་པ་དང་། ཆེ་བ་བརྗོད་པ། གཟེངས་སུ་བསྟོད་པ།དེ་བཞིན་གསོལ་བ་འདེབས་པ་སོགས་ཀྱི་དོན་ལ་གོ་དགོས་ཏེ།  བསྟོད་པ་ཞེས་པ་བསྔགས་བརྗོད་བས་པ་དང་། མཆོད་པ་བརྗོད་པ། གོང་དུ་བཀུར་བ།བསྟོད་བསྔགས་བཏང་བ། ཆེ་བ་བརྗོད་པ། བསྟོད་ཕག་བས་པ། ཤིས་པ་བརྗོད་པ། གུས་འདུད་པ་སོགས་ཀྱི་དོན་ལ་འཇུག་པ་དང་། བསྟོད་ཚིག་ཞེས་པ་བསྟོད་པ་བརྗོད་པའི་ཚིག་ལ་གོ་བ་དང་། བསྟོད་ཚིག་ནི་བསྟན་བཅོས་སྤྱིའི་སྔོན་འགྲོ་ཐུན་མོང་བ་ཁང་པའི་ནང་ལ་འཇུག་པའི་སྒོ་ལྟ་བུ་སྟེ། རྩོམ་ལ་ཐོག་མར་འཇུག་དུས་ཀྱི་སྒོ་ལྟ་བུ་མཆོད་པར་བརྗོད་པ་དང་། བརྩམ་བའི་གཞུང་ངམ་བསྒྱུར་བའི་གཞུང་དེ་ཉིད་སྡེ་སྣོད་གང་དུ་གཏོགས་མིན་མཚོན་བྱེད་ཀྱི་ལོ་པཎ་འགྱུར་ཕག དེ་བཞིན་སྙན་ངག་མེ་ལོང་དང་དེའི་རྩ་འགྲེལ་ཁག་ཏུ་གསལ་བའི་བསྟོད་པའི་དཔེར་བརྗོད་ཚིག་རྐང་རེ་གཉིས་དང་ཤོ་ལོ་ཀ་རེ་གཉིས། རྣམ་ཐར་དང་རྟོགས་བརྗོད་སོགས་ཀྱི་ནང་འཁོད་པའི་བསྟོད་པ་དེ་དག་ཀྱང་འདིར་བརྗོད་པའི་བསྟོད་ཚིག་ཙམ་ལས་བསྟོད་རྩོམ་དུ་གོ་བ་མིན། བསྟོད་ཚོགས་ཞེས་པའི་ཐ་སྙད་ལྟ་བུར་མཚོན་ན་བསྟན་པ་སྔ་དར་གྱི་ལོ་ཙཱ་བ་རྨ་རིན་ཆེན་མཆོག་དང་ཀ་ཅོག་ཞང་གསུམ་སོགས་དང་། ཕྱི་དར་གྱི་ལོ་ཙཱ་བ་རིན་ཆེན་བཟང་པོ་དང་ནག་ཚོ་ལོ་ཙཱ་བ་ཚུལ་ཁྲིམས་རྒྱལ་བ་སོགས་དུས་རབས་རིམ་བྱུང་གི་ལོ་ཆེན་རྣམ་པས་བོད་གངས་ཅན་པའི་བརྡ་སྤྲོད་ཀྱི་ལུགས་དང་མཐུན་པར་བསྒྱུར་ཞིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ་རྣམས་ཕྱོགས་གཅིག་ཏུ་བསྒྲིགས་ཏེ་བསྟན་བསྒྱུར་ཁྲོད་དུ་བཞུགས་པའི་བསྟོད་རྩོམ་བོད་འགྱུར་མ་དེ་དག་དང་། བོད་ཀྱི་མཁས་པས་ཕག་རྩོམ་མཛད་པའི་བསྟོད་པའི་རྩོམ་རིག་ཁག་ཕྱོགས་བསྒྲིགས་བས་སྟེ་བསྟོད་པའི་ཚོགས་སུ་བསྒྲིགས་པའི་བསྟོད་པ་རྣམས་ལ་བསྟོད་ཚོགས་ཞེས་བརྗོད་པ་ཡིན། བསྟོད་རྩོམ་ཞེས་པ་ནི་བསྟོད་པའི་རྩོམ་རིག་སྟེ། དེ་ཡང་རྩོམ་རིག་གི་ལུས་སྲོག་རྒྱན་གསུམ་ལས་སྲོག་གམ་བརྗོད་བ་ནི་བསྟོད་པའི་ཡུལ་དུ་གྱུར་པའི་རང་གི་དད་པའམ་ཕག་གི་ཡུལ་སྟོན་པའམ་བསྟན་པའི་མངའ་བདག་བླ་མ་དགེ་བའི་བཤེས་གཉེན་ནམ། ལྷ་ཀླུ་གཉན་བཙན། ཆོས་སྐྱོང་སྲུང་མ་སོགས་གང་རུང་ཞིག་གི་སྐུ་གསུང་ཐུགས་འཕྲིན་ལས་ཀྱི་ཡོན་ཏན་ནམ་ཡང་ན་གནས་ཆེན་གང་རུང་ཞིག་གིས་ཡོན་ཏན་ལ་དམིགས་སྟེ་ཚིགས་སུ་བཅད་པའི་སྒོ་ནས་བསྔགས་པ་རྒྱ་ཆེར་བརྗོད་པ་ཞིག་སྟེ། རྩོམ་ཡིག་ཧྲིལ་པོའི་ནང་དོན་གཙོ་བོ་བསྟོད་པ་བརྗོད་པ་ཞིག་དང་། རྒྱན་ནམ་རྗོད་བྱེད་ཚིག་སྙན་ཞིང་འཇེབས་པ་སྙན་ངག་གི་རྒྱན་གང་རུང་གིས་རྣམ་པར་བརྒྱན་པའི་བསྟོད་པའི་རྣམ་པ་ཅན་གྱི་རྩོམ་རིག་བྱེ་བྲག་པ་ཞིག་ལ་གོ་དགོས་པ་ཡིན། སྤྱིར་བསྟོད་རྩོམ་ཞེས་པར་སྒྲ་ཡངས་དོག་གི་གོ་དོན་གཉིས་སུ་བཤད་ཆོག་སྟེ། སྒྲ་ཡངས་འཇུག་གི་དབང་དུ་བས་ན་བོད་ཀྱི་རྩོམ་རིག་གར་སྟེགས་ཐོག་བསྟོད་པའི་རྣམ་པའམ་ཆ་ཤས་ཡོད་པའི་རྩོམ་རིག་ནི་ཤིན་ཏུ་མང་སྟེ། རྣམ་ཐར་རྟོགས་བརྗོད་དང་། སྐྱེས་རབས། གདན་རབས། གདུང་རབས་དེ་བཞིན་བསྔོ་སྨོན་གསོལ་འདེབས་སོགས་གང་ལ་བལྟས་ཀྱང་དེ་དག་ཚང་མ་བསྟོད་པའི་ཁོངས་སུ་བསྡུ་ཆོག་སྟེ། དཔེར་ན། རྣམ་ཐར་ལ་མཚོན་ནའང་སངས་རྒྱས་བང་སེམས་དང་། བླ་མ་སློབ་དཔོན་སོགས་ཡོན་ཏན་གྱི་བདག་ཉིད་གོང་དུ་བཀུར་འོས་པའི་སྐྱེས་ཆེན་དམ་པ་གང་དེའི་རྣམ་པར་ཐར་ཚུལ་ཏེ། སྐུ་ཚེ་ཧྲིལ་པོའི་ལོ་རྒྱུས་རིགས་ཕུན་སུམ་ཚོགས་པར་སྐྱེ་བ་བཞེས་ཚུལ་དང་། རྟེན་དེ་ལ་ཡོན་ཏན་བཪྙེས་ཚུལ། བརྙེས་ནས་འགྲོ་དོན་མཛད་ཚུལ་བཅས་སྐྱེས་ཆེན་གསེར་རི་འདྲ་བ་གང་དེའི་ཆེ་བའི་ཡོན་ཏན་བརྗོད་པ་ལས་མ་འདས་ཤིང་། གང་དེའི་ཆེ་བའི་ཡོན་ཏན་བརྗོད་པར་བསྟོད་པ་ནི་མེད་དུ་མི་རུང་སྟེ། ཆེ་བར་བསྟོད་པ་བརྗོད་པ་དང་། བསྟོད་པས་ཆེ་བ་ཤེས་ཐུབ་པས། རྣམ་ཐར་རྟོགས་བརྗོད་ཀྱི་བར་སྐབས་གང་དུའང་བསྟོད་པ་ནི་མེད་དུ་མི་རུང་བ་ཞིག་ཡིན་པས་བརྗོད་བ་རྗོད་བྱེད་གང་གི་ཆ་ནས་བལྟས་ཀྱང་རྣམ་ཐར་རྟོགས་བརྗོད་ལ་བསྟོད་རྩོམ་དང་མཐུན་པའི་ཆ་ཤིན་ཏུ་མང་བས་བསྟོད་རྩོམ་ལྟ་བུར་སྣང་མོད། འོན་ཀྱང་རྣམ་ཐར་རྟོགས་བརྗོད་དུ་བསྟོད་པའི་ཆ་ཇི་ཙམ་འཛོམས་ཡོད་རུང་དེ་དག་ལ་བསྟོད་རྩོམ་གྱི་མིང་གདགས་མི་རུང་སྟེ། གཅིག་ནས་རྣམ་ཐར་རྟོགས་བརྗོད་དང་བསྟོད་རྩོམ་གྱི་བསྟོད་བའི་གཞི་ལ་རྒྱ་ཆེ་ཆུང་གི་ཁྱད་པར་ཡོད་པ་ལས་བསྟོད་རྩོམ་གྱི་བརྗོད་གཞི་ནི་གནས་དང་། གང་ཟག དངོས་པོའི་གནས་ལུགས་བཅས་ལས་གང་རུང་ཞིག་ཡིན་དགོས་པ་དང་། རྣམ་ཐར་རྟོགས་བརྗོད་ཀྱི་བརྗོད་གཞི་ནི་གང་ཟག་ཁོ་ནར་ངེས་པས་རྩོམ་གྱི་བརྗོད་གཞིར་ཁྱབ་ཆེ་ཆུང་གི་ཁྱད་པར་ཡོད་པ། གཉིས་ནས་བསྟོད་ཚུལ་གྱི་རྣམ་པའང་མཐའ་གཅིག་ཏུ་གཅིག་མཚུངས་སུ་མ་ངེས་སྟེ། རྗེ་བཙུན་ས་པཎ་ཀུན་དགའ་རྒྱལ་མཚན་དཔལ་བཟང་པོའི《བོད་ཡུལ་ལ་བསྔགས་པ》ཞེས་པ་ལྟ་བུའི་དམིགས་གསལ་རེ་གཉིས་མ་གཏོགས་གཞན་ཕལ་ཆེ་བ་མགོ་མཇུག་བར་གསུམ་དུ་ཚིགས་བཅད་ཁོ་ནའི་རྣམ་པར་ངེས་ཤིང་། རྣམ་ཐར་རྟོགས་བརྗོད་ནི་དེ་ལས་ལྡོག་སྟེ་བཅད་ལྷུག་སྤེལ་མ་གང་རུང་བྱུང་བས་རྩོམ་གྱི་ལུས་ཀྱང་གཅིག་ཁོ་ནར་མ་ངེས་པའོ། འོན་ཀྱང་བསྟོད་རྩོམ་ཡིན་ན་རྣམ་ཐར་རྟོགས་བརྗོད་མ་ཡིན་པས་མ་ཁྱབ་སྟེ།《ཇོ་བོ་རྗེའི་བསྟོད་པ་སུམ་ཅུ་པ》ལྟ་བུ་རྣམ་ཐར་ཡིན་ལ་བསྟོད་རྩོམ་ཡང་ཡིན་པའི་ཕྱིར།བསྟོད་རྩོམ་ནི་རྩོམ་ཡིག་ཧྲིལ་པོའི་བརྗོད་བའི་ནང་དོན་གཙོ་བོ་བསྟོད་པ་བརྗོད་པ་ཞིག་དགོས་པ་དང་། རྗོད་བྱེད་ཀྱི་ཚིག་སྦྱོར་ཡང་བརྗོད་བ་དེ་ཉིད་མཚོན་པར་བྱེད་པའི་བསྟོད་པའི་རྣམ་པ་ཅན་ཞིག་ཡིན་དགོས་པས་བསྟན་བཅོས་ཀྱི་ཀླད་དོན་དུ་གྱུར་པའི་མཆོད་བརྗོད་དེ་ཉིད་བསྟན་བཅོས་སམ་གཞུང་ཧྲིལ་པོ་དེའི་གྲུབ་ཆ་བྱེ་བྲག་པ་ཞིག་ཡིན་པ་ཙམ་ལས་བརྗོད་བའི་ནང་དོན་གཙོ་བོ་མིན་ལ་དེ་བས་དམིགས་བསལ་དུ་རྩོམ་གྱི་ལུས་བྱེ་བྲག་པ་སྤྱད་དེ་བསྟོད་པ་ཆེད་དུ་བརྗོད་པ་ཞིག་མིན་པས་རྗོད་བྱེད་ཀྱི་ཚིག་ཀྱང་ཤོ་ལོ་ཀ་འགའ་ཤས་ལས་མི་འབྱུང་བས་བསྟོད་ཚིག་ཙམ་དུ་འགྲོ་སྲིད་པ་ལས་བསྟོད་རྩོམ་དུ་མི་འགྲོ་བས་མཆོད་བརྗོད་ནི་མཚན་ཉིད་ཡོངས་སུ་ཚང་བའི་རང་སྐྱ་ཐུབ་པའི་བསྟོད་རྩོམ་ཞིག་མིན་ནོ་ཞེས་བརྗོད་པར་འདོད་དོ། མདོར་ན། བསྟོད་རྩོམ་ནི་རྗོད་བྱེད་སྙན་ཞིང་འཇེབས་པའི་ཚིག་གི་སྦྱོར་བ་ལ་བརྟེན་ནས་བརྗོད་བ་བསྟོད་ཡུལ་གྱི་ཡོན་ཏན་རྣམས་ཀློག་པ་པོའི་རྒྱུད་ལ་གསོན་ཉམས་དོད་པོས་ངོ་སྤྲོད་པའི་རྩོམ་རིག་སྒྱུ་རྩལ་གྱི་རྣམ་པ་བྱེ་བྲག་པ་ཞིག་ལ་ཟེར་བ་ཡིན་པས་སྐབས་འདིར་སྒྲ་དོག་འཇུག་གི་དབང་དུ་བས་པའི་བསྟོད་རྩོམ་ལ་གོ་དགོས་པས་སྙན་ངག་གི་ཆ་དང་ལྡན་པའི་བསྟོད་རྩོམ་ཁོ་ན་ལ་ཟེར་བ་ཡིན་ནོ།
