རྒྱལ་སྤྱིའི་གེ་སར་དང་འཛམ་གླིང་ལོ་རྒྱུས་སྙན་རྩོམ་སྐོར་གྱི་རིག་གཞུང་བགྲོ་གླེང་ཚོགས་འདུ་ཟི་ལིང་དུ་འཚོགས་པ།
རྒྱལ་སྤྱིའི་གེ་སར་དང་འཛམ་གླིང་ལོ་རྒྱུས་སྙན་རྩོམ་སྐོར་གྱི་རིག་གཞུང་བགྲོ་གླེང་ཚོགས་འདུ་ཟི་ལིང་དུ་འཚོགས་པ། 
ཟླ༧པའི་ཚེས༡༧ཉིན། མཚོ་སྔོན་ཞིང་ཨུའི་དྲིལ་སྒྲོག་པུའུ་དང་། ཀྲུང་གོའི་སྤྱི་ཚོགས་ཚན་རིག་གླིང་མི་རིགས་རྩོམ་རིག་ཞིབ་འཇུག་ཁང་གཉིས་ཀྱིས་གཙོ་སྒྲུབ་དང་། མཚོ་སྔོན་ཞིང་ཆེན་སྤྱི་ཚོགས་ཚན་རིག་གླིངགིས་འགན་སྒྲུབ། ཀྲུང་གོའི་དམངས་སྲོལ་སློབ་ཚོགས་དང་ཀྲུང་གོའི་གྲངས་ཉུང་མི་རིགས་རྩོམ་པ་པོའི་སློབ་ཚོགས། མཚོ་སྔོན་ཞིང་ཆེན་རྩོམ་རིག་མཉམ་འབྲེལ་ལྷན་ཚོགས། མཚོ་སྔོན་ཞིང་ཆེན་དམངས་སྲོལ་སློབ་ཚོགས་བཅས་ཀྱིས་རོགས་སྒྲུབ་གནང་བའི་རྒྱལ་སྤྱིའི་གེ་སར་དང་འཛམ་གླིང་ལོ་རྒྱུས་སྙན་རྩོམ་སྐོར་གྱི་རིག་གཞུང་བགྲོ་གླེང་ཚོགས་འདུ་ཟི་ལིང་དུ་འཚོགས་ཤིང་། མཚོ་སྔོན་ཞིང་ཨུའི་རྒྱུན་ཨུའམ་དྲིལ་སྒྲོག་པུའི་འགོ་གཙོ་ཅི་ཏེ་མཱ་ཅཱ་དང་། ཀྲུང་གོའི་སྤྱི་ཚོགས་ཚན་རིག་གླིང་གི་ཏང་རྩ་འཛུགས་ཀྱི་ཚོགས་མིའམ་གླིང་གཙོ་གཞོན་པ་ཝུའུ་ཡན། ཀྲུང་གོའི་སྤྱི་ཚོགས་ཚན་རིག་གླིང་མི་རིགས་རྩོམ་རིག་ཞིབ་འཇུག་གླིང་གི་འགོ་གཙོ་ཁྲོའོ་ཀི་ཅིན་སོགས་འགོ་ཁྲིད་རིམ་པ་དང་། དེ་མིན་ཨ་རི་དང་། དབྱིན་ཇི། འཇར་མན། ཧྥ་རན་སི། ཨུ་རུ་སི། པོ་ཧེ། ཡ་མེ་ཉི་ཡ། མཱ་ལེ་སོགས་རྒྱལ་ཕྱི་དང་རྒྱལ་ནང་གི་འབྲེལ་ཡོད་ཆེད་མཁས་པ༡༥༠ལྷག་ཐེངས་འདིའི་བགྲོ་གླེང་དུ་ཞུགས། 
     མཚོ་སྔོན་ཞིང་ཨུའི་རྒྱུན་ཨུའམ་དྲིལ་སྒྲོག་པུའི་འགོ་གཙོ་ཅི་ཏེ་མཱ་ཅཱས་བསྟན་དོན། རིག་གནས་མང་པོའི་ཁྲོད་མི་རིགས་རིག་གནས་ཇི་ལྟར་རྟོག་ཞིབ་དང་། བསྣོལ་སྦྲེལ་གྱི་རིག་ཚན་ཞིབ་འཇུག་ཁྲོད་མི་རིགས་རིག་གནས་སྲོལ་རྒྱུན་ཇི་ལྟར་གཟེངས་སུ་གཏོད་པ་ནི་ཞིབ་ཏུ་བསམ་དགོས་པའི་གནད་དོན་ཞིག་ཡིན། 
  ཀྲུང་གོའི་སྤྱི་ཚོགས་ཚན་རིག་གླིང་གི་ཏང་རྩ་འཛུགས་ཀྱི་ཚོགས་མིའམ་གླིང་གཙོ་གཞོན་པ་ཝུའུ་ཡན་གྱིས“ངག་ཐོག་གི་ལོ་རྒྱུས་སྙན་རྩོམ་འདི་དག་ནི་མི་རིགས་ལ་རྒྱུས་ལོན་བྱེད་པའི་རིག་གནས་ཀུན་བཏུས་ཆེན་མོ་ཞིག་ཡིན་པ་མ་ཟད། མི་རིགས་ཀྱི་བསམ་པའི་བློ་སྟོབས་མཚོན་པའི་བཤམས་སྟོན་ཁང་ཞིག་ཀྱང་རེད། ཉེ་བའི་ལོ་བཅུའི་ནང་། འདིས་དུས་རབས༢༠བའི་དུས་དཀྱིལ་ནས་བཟུང་ཡོ་རོབ་དང་ཨ་རིའི་ལོ་རྒྱུས་སྙན་རྩོམ་རིག་པའི་ཞིབ་འཇུག་ལ་གླེང་མོལ་དང་འབྲེལ་འདྲིས་ཀྱི་ཐད་ནས་ཤུགས་སྣོན་བྱས་ཤིང་། རྒྱལ་སྤྱི་དང་གླེང་མོལ་བྱེད་པ་བརྒྱུད་རྒྱལ་སྤྱིའི་རིགས་མཚུངས་ཉམས་མྱོང་མི་དམན་པ་ཉམས་ལེན་བྱས་ཡོད”ཅེས་བཤད། 
     མཚོ་སྔོན་ཞིང་ཆེན་སྤྱི་ཚོགས་ཚན་རིག་གླིང་གི་འགོ་གཙོ་ཀྲོའོ་ཙུང་ཧྥུ་ཡིས་ནན་གྱིས་བསྟན་དོན། 《གེ་སར》ནི་འཛམ་གླིང་དུ་ལོ་རྒྱུས་ཀྱི་དུས་ཡུན་རིང་ཞིང་ནང་དོན་ཕུན་སུམ་ཚོགས་པའི་ལོ་རྒྱུས་སྙན་རྩོམ་ཞིག་ཡིན་པའི་ཆ་ནས། མཚོ་བོད་མཐོ་སྒང་གི་མི་རིགས་རིག་གནས་ཀྱི་མཆོག་གྱུར་ནོར་བུ་ཞིག་ཡིན་ལ། འཛམ་གླིང་རྩོམ་རིག་སྒྱུ་རྩལ་ཁྲོད་ཀྱི་གཅེས་ནོར་དང་འཛམ་གླིང་དཔའ་བོའི་ལོ་རྒྱུས་བརྗོད་པའི་སྙན་རྩོམ་ཁྲོད་ཀྱི་ཚབ་བྱེད་བརྩམས་ཆོས་ཤིག་ཀྱང་ཡིན་པ་རེད། དེ་ཉིད་ཏག་ཏག་འཛམ་གླིང་ལོ་རྒྱུས་ཀྱི་སྐད་ཆའི་ཁོར་ཡུག་ཏུ་བཞག་སྟེ་གླེང་མོལ་མཉམ་འབྲེལ་བྱས་ན་ད་གཟོད་འཛམ་གླིང་རིག་གནས་ཁྲོད་གཟི་བརྗིད་ཀྱི་གོ་གནས་ཤིག་ཟིན་ངེས་པ་དང་། དོན་དངོས་ཐོག་དེའི་རིག་གནས་ཀྱི་ནང་དོན་དང་སྒྱུ་རྩལ་གྱི་མཐིལ་རྟོགས་ནུས་ལ། དེའི་ཞིབ་འཇུག་གི་རིམ་གྲས་ཀྱང་མཐོར་འདེགས་སུ་གཏོང་ཐུབ་པ་རེད། 
   ཤར་ཕྱོགས་ཡེ་ལེ་ཡ་ཐིར་གྲགས་པའི་བོད་ཀྱི་དཔའ་བོའི་ལོ་རྒྱུས་སྙན་རྩོམ《གེ་སར》ཞེས་པའི་ནང་གེ་སར་རྒྱལ་པོའི་མི་ཚེ་ཧྲིལ་བོའི་མཛད་རྣམ་བརྗོད་ཡོད་ཅིང་། གཙོ་ཆེར་བོད་རིགས་དང་། སོག་རིགས། ཧོར་རིགས། འཇང་རིགས་སོགས་ཀྱི་ས་ཁུལ་དུ་ཁྱབ་འདུག་ལ། འདི་ནི་ཕུལ་དུ་བྱུང་བའི་རྩོམ་རིག་བརྩམས་ཆོས་ཤིག་ཡིན་པ་མ་ཟད། གནའ་བོའི་བོད་ཀྱི་སྤྱི་ཚོགས་འཚོ་བ་དང་། མི་རིགས་ལོ་རྒྱུས། དཔལ་འབྱོར་རིག་གནས། གྲལ་རིམ་འབྲེལ་བ། མི་རིགས་འབྲེལ་འདྲིས། བསམ་བློའི་འདུ་ཤེས། མི་ཚེའི་ལྟ་བ། ཡུལ་སྲོལ་གོམས་གཤིས། ཆོས་ལུགས་དད་མོས་སོགས་ལ་ཞིབ་འཇུག་བྱེད་པའི་རིག་གནས་ཀུན་བཏུས་ཆེན་མོ་ཞིག་ཡིན་པ་རེད།རྒྱལ་སྤྱིའི་གེ་སར་དང་འཛམ་གླིང་ལོ་རྒྱུས་སྙན་རྩོམ་སྐོར་གྱི་རིག་གཞུང་བགྲོ་གླེང་ཚོགས་འདུ་ཟི་ལིང་དུ་འཚོགས་པ།
རྒྱལ་སྤྱིའི་གེ་སར་དང་འཛམ་གླིང་ལོ་རྒྱུས་སྙན་རྩོམ་སྐོར་གྱི་རིག་གཞུང་བགྲོ་གླེང་ཚོགས་འདུ་ཟི་ལིང་དུ་འཚོགས་པ། 
ཟླ༧པའི་ཚེས༡༧ཉིན། མཚོ་སྔོན་ཞིང་ཨུའི་དྲིལ་སྒྲོག་པུའུ་དང་། ཀྲུང་གོའི་སྤྱི་ཚོགས་ཚན་རིག་གླིང་མི་རིགས་རྩོམ་རིག་ཞིབ་འཇུག་ཁང་གཉིས་ཀྱིས་གཙོ་སྒྲུབ་དང་། མཚོ་སྔོན་ཞིང་ཆེན་སྤྱི་ཚོགས་ཚན་རིག་གླིངགིས་འགན་སྒྲུབ། ཀྲུང་གོའི་དམངས་སྲོལ་སློབ་ཚོགས་དང་ཀྲུང་གོའི་གྲངས་ཉུང་མི་རིགས་རྩོམ་པ་པོའི་སློབ་ཚོགས། མཚོ་སྔོན་ཞིང་ཆེན་རྩོམ་རིག་མཉམ་འབྲེལ་ལྷན་ཚོགས། མཚོ་སྔོན་ཞིང་ཆེན་དམངས་སྲོལ་སློབ་ཚོགས་བཅས་ཀྱིས་རོགས་སྒྲུབ་གནང་བའི་རྒྱལ་སྤྱིའི་གེ་སར་དང་འཛམ་གླིང་ལོ་རྒྱུས་སྙན་རྩོམ་སྐོར་གྱི་རིག་གཞུང་བགྲོ་གླེང་ཚོགས་འདུ་ཟི་ལིང་དུ་འཚོགས་ཤིང་། མཚོ་སྔོན་ཞིང་ཨུའི་རྒྱུན་ཨུའམ་དྲིལ་སྒྲོག་པུའི་འགོ་གཙོ་ཅི་ཏེ་མཱ་ཅཱ་དང་། ཀྲུང་གོའི་སྤྱི་ཚོགས་ཚན་རིག་གླིང་གི་ཏང་རྩ་འཛུགས་ཀྱི་ཚོགས་མིའམ་གླིང་གཙོ་གཞོན་པ་ཝུའུ་ཡན། ཀྲུང་གོའི་སྤྱི་ཚོགས་ཚན་རིག་གླིང་མི་རིགས་རྩོམ་རིག་ཞིབ་འཇུག་གླིང་གི་འགོ་གཙོ་ཁྲོའོ་ཀི་ཅིན་སོགས་འགོ་ཁྲིད་རིམ་པ་དང་། དེ་མིན་ཨ་རི་དང་། དབྱིན་ཇི། འཇར་མན། ཧྥ་རན་སི། ཨུ་རུ་སི། པོ་ཧེ། ཡ་མེ་ཉི་ཡ། མཱ་ལེ་སོགས་རྒྱལ་ཕྱི་དང་རྒྱལ་ནང་གི་འབྲེལ་ཡོད་ཆེད་མཁས་པ༡༥༠ལྷག་ཐེངས་འདིའི་བགྲོ་གླེང་དུ་ཞུགས། 
     མཚོ་སྔོན་ཞིང་ཨུའི་རྒྱུན་ཨུའམ་དྲིལ་སྒྲོག་པུའི་འགོ་གཙོ་ཅི་ཏེ་མཱ་ཅཱས་བསྟན་དོན། རིག་གནས་མང་པོའི་ཁྲོད་མི་རིགས་རིག་གནས་ཇི་ལྟར་རྟོག་ཞིབ་དང་། བསྣོལ་སྦྲེལ་གྱི་རིག་ཚན་ཞིབ་འཇུག་ཁྲོད་མི་རིགས་རིག་གནས་སྲོལ་རྒྱུན་ཇི་ལྟར་གཟེངས་སུ་གཏོད་པ་ནི་ཞིབ་ཏུ་བསམ་དགོས་པའི་གནད་དོན་ཞིག་ཡིན། 
  ཀྲུང་གོའི་སྤྱི་ཚོགས་ཚན་རིག་གླིང་གི་ཏང་རྩ་འཛུགས་ཀྱི་ཚོགས་མིའམ་གླིང་གཙོ་གཞོན་པ་ཝུའུ་ཡན་གྱིས“ངག་ཐོག་གི་ལོ་རྒྱུས་སྙན་རྩོམ་འདི་དག་ནི་མི་རིགས་ལ་རྒྱུས་ལོན་བྱེད་པའི་རིག་གནས་ཀུན་བཏུས་ཆེན་མོ་ཞིག་ཡིན་པ་མ་ཟད། མི་རིགས་ཀྱི་བསམ་པའི་བློ་སྟོབས་མཚོན་པའི་བཤམས་སྟོན་ཁང་ཞིག་ཀྱང་རེད། ཉེ་བའི་ལོ་བཅུའི་ནང་། འདིས་དུས་རབས༢༠བའི་དུས་དཀྱིལ་ནས་བཟུང་ཡོ་རོབ་དང་ཨ་རིའི་ལོ་རྒྱུས་སྙན་རྩོམ་རིག་པའི་ཞིབ་འཇུག་ལ་གླེང་མོལ་དང་འབྲེལ་འདྲིས་ཀྱི་ཐད་ནས་ཤུགས་སྣོན་བྱས་ཤིང་། རྒྱལ་སྤྱི་དང་གླེང་མོལ་བྱེད་པ་བརྒྱུད་རྒྱལ་སྤྱིའི་རིགས་མཚུངས་ཉམས་མྱོང་མི་དམན་པ་ཉམས་ལེན་བྱས་ཡོད”ཅེས་བཤད། 
     མཚོ་སྔོན་ཞིང་ཆེན་སྤྱི་ཚོགས་ཚན་རིག་གླིང་གི་འགོ་གཙོ་ཀྲོའོ་ཙུང་ཧྥུ་ཡིས་ནན་གྱིས་བསྟན་དོན། 《གེ་སར》ནི་འཛམ་གླིང་དུ་ལོ་རྒྱུས་ཀྱི་དུས་ཡུན་རིང་ཞིང་ནང་དོན་ཕུན་སུམ་ཚོགས་པའི་ལོ་རྒྱུས་སྙན་རྩོམ་ཞིག་ཡིན་པའི་ཆ་ནས། མཚོ་བོད་མཐོ་སྒང་གི་མི་རིགས་རིག་གནས་ཀྱི་མཆོག་གྱུར་ནོར་བུ་ཞིག་ཡིན་ལ། འཛམ་གླིང་རྩོམ་རིག་སྒྱུ་རྩལ་ཁྲོད་ཀྱི་གཅེས་ནོར་དང་འཛམ་གླིང་དཔའ་བོའི་ལོ་རྒྱུས་བརྗོད་པའི་སྙན་རྩོམ་ཁྲོད་ཀྱི་ཚབ་བྱེད་བརྩམས་ཆོས་ཤིག་ཀྱང་ཡིན་པ་རེད། དེ་ཉིད་ཏག་ཏག་འཛམ་གླིང་ལོ་རྒྱུས་ཀྱི་སྐད་ཆའི་ཁོར་ཡུག་ཏུ་བཞག་སྟེ་གླེང་མོལ་མཉམ་འབྲེལ་བྱས་ན་ད་གཟོད་འཛམ་གླིང་རིག་གནས་ཁྲོད་གཟི་བརྗིད་ཀྱི་གོ་གནས་ཤིག་ཟིན་ངེས་པ་དང་། དོན་དངོས་ཐོག་དེའི་རིག་གནས་ཀྱི་ནང་དོན་དང་སྒྱུ་རྩལ་གྱི་མཐིལ་རྟོགས་ནུས་ལ། དེའི་ཞིབ་འཇུག་གི་རིམ་གྲས་ཀྱང་མཐོར་འདེགས་སུ་གཏོང་ཐུབ་པ་རེད། 
   ཤར་ཕྱོགས་ཡེ་ལེ་ཡ་ཐིར་གྲགས་པའི་བོད་ཀྱི་དཔའ་བོའི་ལོ་རྒྱུས་སྙན་རྩོམ《གེ་སར》ཞེས་པའི་ནང་གེ་སར་རྒྱལ་པོའི་མི་ཚེ་ཧྲིལ་བོའི་མཛད་རྣམ་བརྗོད་ཡོད་ཅིང་། གཙོ་ཆེར་བོད་རིགས་དང་། སོག་རིགས། ཧོར་རིགས། འཇང་རིགས་སོགས་ཀྱི་ས་ཁུལ་དུ་ཁྱབ་འདུག་ལ། འདི་ནི་ཕུལ་དུ་བྱུང་བའི་རྩོམ་རིག་བརྩམས་ཆོས་ཤིག་ཡིན་པ་མ་ཟད། གནའ་བོའི་བོད་ཀྱི་སྤྱི་ཚོགས་འཚོ་བ་དང་། མི་རིགས་ལོ་རྒྱུས། དཔལ་འབྱོར་རིག་གནས། གྲལ་རིམ་འབྲེལ་བ། མི་རིགས་འབྲེལ་འདྲིས། བསམ་བློའི་འདུ་ཤེས། མི་ཚེའི་ལྟ་བ། ཡུལ་སྲོལ་གོམས་གཤིས། ཆོས་ལུགས་དད་མོས་སོགས་ལ་ཞིབ་འཇུག་བྱེད་པའི་རིག་གནས་ཀུན་བཏུས་ཆེན་མོ་ཞིག་ཡིན་པ་རེད།
