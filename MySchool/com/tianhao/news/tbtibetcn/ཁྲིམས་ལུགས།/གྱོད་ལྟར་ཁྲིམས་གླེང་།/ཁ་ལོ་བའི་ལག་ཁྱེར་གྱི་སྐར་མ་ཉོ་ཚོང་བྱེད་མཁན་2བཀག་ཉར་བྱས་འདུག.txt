ཟླ་4ཚེས་1ཉིན། སྣེ་གདོང་ལམ་མངགས་གཏོང་ཁང་ལ་ལྷོ་ཁ་གྲོང་ཁྱེར་སྤྱི་བདེ་ཅུས་འགྲིམ་འགྲུལ་ཉེན་རྟོག་ཉིང་ལག་རུ་ཁག་ཁྲིམས་འགལ་ཐག་གཅོད་ལྟེ་གནས་ཀྱིས་ཁ་ལོ་བའི་ལག་ཁྱེར་གྱི་སྐར་མ་ཉོ་ཚོང་བྱས་པའི་གྱོད་གཞི་ཞིག་རྩིས་སྤྲད་པ་མ་ཟད། ཁྲིམས་འགལ་བ་2ཀྱང་རྩིས་སྤྲད་པ་རེད། བརྟག་དཔྱད་བྱས་པ་བརྒྱུད། ཁྲིམས་འགལ་བ་ཟླ་ཆེ་གེ་མོས（ཕོ། བོད་རིགས）སྐད་འཕྲིན་ལས་སྟེགས་ཐོག་ནས“ཁ་ལོ་བའི་ལག་ཁྱེར་གྱི་སྐར་མ་ཉོས་ཏེ་སྒྲིག་འགལ་སྐར་མ་ཚབ་སྤྲོད་ཆོག་པའི”བརྡ་ཁྱབ་ཆུང་ཆུང་ཞིག་མཐོང་བ་བརྒྱུད་སུན་ཆེ་གེ་མོ（ཕོ། རྒྱ་རིགས）ངོ་ཤེས་པ་རེད། མི་དེ་གཉིས་ཀྱིས་ཁ་པར་ནས་སྐད་ཆ་བྱས་པ་བརྒྱུད། ཟླ་ཆེ་གེ་མོས་དངུལ་སྒོར་1500སྤྲད་དེ་སུན་ཆེ་གེ་མོའི་ས་ནས་ཁ་ལོ་བའི་ལག་ཁྱེར་གྱི་སྐར་མ་9ཉོས་ཏེ་རང་གི་ཟོག་འདྲེན་རླངས་འཁོར་གྱི་འགྲིམ་འགྲུལ་སྒྲིག་འགལ་གྱི་ཁྲིམས་འགལ་བྱེད་སྤྱོད་ཐག་གཅོད་ཀྱི་ཆད་པར་སྤྲོད་རྒྱུ་བྱས། འདི་ལོའི་ཟླ་3ཚེས་30ཉིན་མི་དེ་གཉིས་སྔོན་ལ་སྐད་ཆ་བྱས་ནས་ལྷོ་ཁ་གྲོང་ཁྱེར་སྤྱི་བདེ་ཅུས་འགྲིམ་འགྲུལ་ཉེན་རྟོག་ཉིང་ལག་རུ་ཁག་ཁྲིམས་འགལ་ཐག་གཅོད་ལྟེ་གནས་སུ་འབྲེལ་ཡོད་ཀྱི་འགྲོ་ལུགས་གཉེར་སྐབས་མི་དམངས་ཉེན་རྟོག་པས་གནས་ཚུལ་དེ་ཤེས་པ་རེད།
          འདྲི་རྩད་བྱས་པ་བརྒྱུད། མི་དེ་གཉིས་ཀྱིས་སྲིད་འཛིན་ཁྲིམས་འཛིན་ལས་ཁུངས་ལ་དཔང་པོ་རྫུས་མ་མཁོ་འདོན་བྱས་ཏེ་ཁ་ལོ་བའི་ལག་ཁྱེར་གྱི་སྐར་མ་ཉོ་ཚོང་བྱས་ནས་འགྲིམ་འགྲུལ་སྒྲིག་འགལ་གྱི་ཆད་པ་གསབ་ཐབས་བྱས་པའི་ཁྲིམས་འགལ་དོན་དངོས་སྦ་གསང་མེད་པར་ངོས་ལེན་ཞུས་པ་རེད།

          ཉེན་རྟོག་པའི་དྲན་སྐུལ།
          ཉེན་རྟོག་ཕྱོགས་ཀྱིས་རྒྱ་ཆེའི་རླངས་འཁོར་གཏོང་མཁན་ཚོར་བཅའ་ཁྲིམས་ཁྲིམས་སྲོལ་ལ་རང་རྟོགས་ངང་བརྩི་སྲུང་ཞུས་ཏེ། འགྲིམ་འགྲུལ་ཁྲིམས་འགལ་གྱི་བྱེད་སྤྱོད་ཉུང་དུ་གཏོང་གང་ཐུབ་དང་། ལྷག་པར་དུ་སྐར་མ་ཉོ་བ་དང་སྐར་མ་འཚོང་བའི་ཁྲིམས་འགལ་གྱི་བྱེད་སྤྱོད་ནང་མི་ཞུགས་པ། “ཚབ་གཉེར་བས”འགྲིམ་འགྲུལ་ཁྲིམས་འགལ་གྱི་ཆད་པ་དང་སྐར་གསོག་གི་ལས་སྒོ་ཚབ་གཉེར་ལ་ཡིད་ཆེས་དང་དེའི་ནང་ཞུགས་མི་ཉན་པ་དང་། གལ་སྲིད་སྐར་མ་ཉོ་ཚོང་དང“ཚབ་གཉེར་བས”ཚབ་གཉེར་བྱེད་པ་སོགས་ཀྱི་ཁྲིམས་འགལ་བྱེད་སྤྱོད་ཤེས་རྟོགས་ཐུབ་ཚེ། སྤྱི་བདེ་ལས་ཁུངས་སུ་ཐེར་འདོན་ཧུར་ཐག་བྱེད་དགོས་པའི་དྲན་སྐུལ་བྱས་འདུག
