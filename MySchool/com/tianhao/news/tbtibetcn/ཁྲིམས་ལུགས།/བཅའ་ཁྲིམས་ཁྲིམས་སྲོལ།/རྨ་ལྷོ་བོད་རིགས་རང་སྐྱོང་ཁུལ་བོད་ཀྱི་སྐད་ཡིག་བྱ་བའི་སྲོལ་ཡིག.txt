རྨ་ལྷོ་བོད་རིགས་རང་སྐྱོང་ཁུལ་བོད་ཀྱི་སྐད་ཡིག་བྱ་བའི་སྲོལ་ཡིག
༡༩༩༣ལོའི་ཟླ་༥ཚེས་༣༠ཉིན་རྨ་ལྷོ་བོད་རིགས་རང་སྐྱོང་ཁུལ་སྐབས་བཅུ་བའི་མི་དམངས་འཐུས་མི་ཚོགས་ཆེན་ཚང་འཛོམས་གྲོས་ཚོགས་ཐེངས་གསུམ་པའི་སྟེང་གྲོས་འཆམ་བྱུང་ཞིང་༡༩༩༣ལོའི་ཟླ་༩པའི་ཚེས་༡༨ཉིན་མཚོ་སྔོན་ཞིང་ཆེན་སྐབས་བརྒྱད་པའི་མི་དམངས་འཐུས་མི་ཚོགས་ཆེན་རྒྱུན་ལས་ཨུ་ཡོན་ལྷན་ཁང་གི་ཐེངས་ལྔ་བའི་ཚང་འཛོམས་གྲོས་ཚོགས་སྟེང་གཏན་ལ་ཕབ།༡༩༩༤ལོའི་ཟླ་༡ཚེས་༡ཉིན་ནས་བཟུང་ལག་བསྟར་བྱ་རྒྱུ།　　དོན་ཚན་དང་པོ།བོད་ཀྱི་སྐད་ཡིག་སློབ་སྦྱོང་དང༌།བཀོལ་སྤྱོད་གོང་སྤེལ་སོགས་ལ་ཤུགས་སྣོན་དང་ཁག་ཐེག་བྱེད་པའི་ཆེད་དུ་《 ཀྲུང་ཧྭ་མི་དམངས་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་རྩ་ཁྲིམས》དང་《རྨ་ལྷོ་བོད་རིགས་རང་སྐྱོང་ཁུལ་ཀྱི་རང་སྐྱོང་སྲོལ་ཡིག》《 ཀྲུང་ཧྭ་མི་མི་དམངས་སྤྱི་མཐུན་རྒྱལ་ཁབ་མི་རིགས་རང་སྐྱོང་ས་ཁུལ་གྱི་བཅའ་ཁྲིམས》བཅས་གཞིར་བཟུང་ཞིང༌།རང་ཁུལ་དངོས་ཡོད་་་་་དང་བསྟུན་ནས་སྲོལ་ཡིག་འདི་གཏན་ལ་ཕབ་པ་ཡིན།　　དོན་ཚན་གཉིས་པ།རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱིས་ཏང་དངརྒྱལ་ཁབ་ཀྱི་མི་རིགས་སྐད་ཡིག་སྲིད་ཇུས་དང་འབྲེལ་ཡོད་ཁྲིམས་ལུགས།ཁྲིམས་སྲོལ་གྱི་གཏན་ཁེལ་སོགས་ལག་ལེན་མཐར་ཕྱིན་དང་མི་རིགས་སོ་སོའི་སྐད་ཡིག་འདྲ་མཉམ་གྱི་གནས་ལུགས་རྒྱུན་འཁྱོངས།རིགས་སོ་སོས་་་རང་ཉིད་ཀྱི་སྐད་ཡིག་བཀོལ་སྤྱོད་དང་འཕེལ་རྒྱས་གཏོང་བའི་རང་དབང་ཁག་ཐེག་བྱ་རྒྱུ།མི་རིགས་སྐད་ཡིག་གིས་འདྲ་མཉམ་དང་མཐུན་སྒྲིལ་རོགས་སྐྱོར་ཅན་གྱི་སྤྱི་ཚོགས་རིང་ལུགས་མི་རིགས་འབྲེལ་བ་ཇེ་བརྟན་དང་འཕེལ་རྒྱས་སུ་གཏོང་རྒྱུ་དང༌།སྤྱི་ཚོགས་རིང་ལུགས་དངོས་པོའི་དཔལ་ཡོན་དང་བསམ་པའི་དཔལ་ཡོན་ལ་ཤུགས་སྣོན་པར་ཞབས་འདེགས་ཞུ་རྒྱུ།　　དོན་ཚན་གསུམ་པ།བོད་ཀྱི་སྐད་ཡིག་ནི་རང་སྐྱོང་ཁུལ་གྱིས་མི་རིགས་རང་སྐྱོང་ས་ཁུལ་གྱི་སྲིད་དབང་ལག་བསྟར་བྱེད་པའི་སྐད་ཡིག་གཙོ་བོ་ཞིག་ཡིན་ཞིང༌།རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱི་ལས་གནས་ལག་བསྟར་བྱེད་སྐབས་རྒྱ་ཡིག་དང་བོད་ཡིག་གཉིས་མཉམ་དུ་བཀོལ་རྒྱུ།　　དོན་ཚན་བཞི་བ།རང་སྐྱོང་ཁུལ་གྱིས་བོད་ཀྱི་སྐད་ཡིག་གི་བྱ་བ་ཁྱབ་བརྡལ་དང་མཐོར་འདེགས་རྒྱུན་འཁྱོངས་དང༌།བོད་ཀྱི་ཕུལ་བྱུང་རིག་གནས་ལོ་རྒྱུས་རྒྱུན་འཛིན་གསར་སྤེལ་དང་སྲུང་སྐྱོབ་བྱེད་རྒྱུ།བོད་ཀྱི་སྐད་ཡིག་འཕེལ་རྒྱས་ལ་ཤུགས་སྣོན་དང་བོད་ཀྱི་སྐད་ཡིག་གིས་རང་སྐྱོང་ཁུལ་གྱི་བཅོས་བསྒྱུར་སྒོ་འབྱེད་དང་དཔའ་འབྱོར་འཛུགས་སྐྲུན་ཐད་ནུས་པ་གལ་ཆེན་ཐོན་དུ་འཇུག་རྒྱུ།　　དོན་ཚན་ལྔ་པ།རང་སྐྱོང་ཁུལ་རང་སྐྱོང་ལས་ཁུངས་ཀྱིས་རིགས་སོ་སོའི་ལས་བཟོ་བར་ཕན་ཚུན་གྱི་སྐད་ཡིག་སློབ་སྦྱོང་དང་སྐུལ་འདེད་བྱེད་པར་སློབ་གསོ་གཏོང་རྒྱུ།བོད་རིགས་ལས་བཟོ་པས་མི་་་རིགས་རང་ཉིད་ཀྱི་སྐད་ཡིག་སློབ་སྦྱོང་དང་བེད་སྤྱོད་བྱེད་པའི་དུས་མཚུངས་སུ་རྒྱལ་ཡོངས་སྤྱི་བཀོལ་གྱི་རྒྱ་སྐད་དང་རྒྱ་ཡིག་ལ་སློབ་སྦྱོང་བྱེད་པ་དང༌།རྒྱ་རིགས་དང་གྲངས་ཉུང་མི་རིགས་གཞན་པའི་ལས་བཟོ་པ་རྣམས་ལ་བོད་ཡིག་སློབ་པར་སྐུལ་འདེད་བྱེད་རྒྱུ།
