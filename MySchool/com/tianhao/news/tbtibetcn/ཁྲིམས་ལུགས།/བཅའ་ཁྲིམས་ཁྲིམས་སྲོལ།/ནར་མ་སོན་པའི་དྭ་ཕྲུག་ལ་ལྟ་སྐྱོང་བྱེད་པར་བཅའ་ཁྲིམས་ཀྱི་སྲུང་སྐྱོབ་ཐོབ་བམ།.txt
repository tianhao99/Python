ནར་མ་སོན་པའི་དྭ་ཕྲུག་ལ་ལྟ་སྐྱོང་བྱེད་པར་བཅའ་ཁྲིམས་ཀྱི་སྲུང་སྐྱོབ་ཐོབ་བམ།
གྱོད་དཔེ།
          ང་ནི་ཡུལ་ཤུལ་ཁུལ་སྐྱེ་རྒུ་གྲོང་རྡལ་གྱི་ལས་བཟོ་བ་ཞིག་ཡིན། ད་ལོའི་ཟླ་བཞི་བར། ངའི་ཁྱིམ་མཚེས་དུང་མཚོ་སྒྲོལ་མ་གློ་བུར་དུ་ཚེ་ལས་འདས་ཤིང་ཁོ་མོའི་ཁྱོ་ག་ཡུལ་ཤུལ་ས་ཡོམ་ཆེན་མོའི་ཁྲོད་ཚེ་ལས་འདས།ད་ལྟ་ལོ་དྲུག་ལ་སོན་མ་ཐག་པའི་བུ་དཔལ་མཁར་དོན་གྲུབ་ཁོ་ན་མ་གཏོགས་ལུས་མེད། སྤྱིར་ཁོ་ལ་ཤ་ཉེ་ཡི་འབྲེལ་བ་ཡོད་པའི་མི་འགའ་རེ་ཡོད་ནའང་དེ་ཚོ་གཅིག་གིས་ཀྱང་དཔལ་མཁར་དོན་གྲུབ་ལ་ཁ་ཡ་མ་བྱས་པས། ད་རག་བར་དུ་ངས་སྙོར་སྐྱོང་བྱས་པ་ཡིན། འདིར་འདྲི་རྒྱུ་ཞིག་ལ།དཔལ་མཁར་དོན་གྲུབ་ཀྱི་ཤ་ཉེ་ཚོར་ཁོ་ལ་བདག་སྐྱོང་བྱེད་པའི་འོས་འགན་ཡོད་དམ།
ཁྲིམས་བཤད།
          ནར་མ་སོན་པའི་མིའི་ལྟ་སྐྱོང་པ་ནི་ཕ་མ་ཡིན།ནར་མ་སོན་པའི་མི་དེའི་ཕ་མ་ཤི་བའམ་ཡང་ན་ལྟ་སྐྱོང་གི་ནུས་པ་མེད་ཚེ། གཤམ་གྱི་མི་སྣའི་ནང་ནས་ལྟ་སྐྱོང་གི་ནུས་པ་ཡོད་པའི་མིས་ལྟ་སྐྱོང་གི་འགན་ཁུར་དགོས། ཨ་མྱེ་དང་ཨ་ཕྱི། སྤོ་བོ་དང་རྨོ་བོ།  གཉེན་པོ་དང་གཉེན་མོ། འབྲེལ་བ་དམ་པོ་ཡོད་པའི་གཉེན་ཉེ་གཞན་པ་དང་། གྲོགས་པོའི་ཁྲོད་ན་ལྟ་སྐྱོང་གི་འགན་ལེན་བྱེད་འདོད་པ་ཅན། ནར་མ་སོན་པའི་མིའི་ཕ་མ་ཡོད་སའི་ལས་ཁུངས་སམ་ཡང་ན་དེར་སྡོད་དམངས་ཡུ་ཡོན་ལྷན་ཚོགས་དང་། སྡེ་དམངས་ཡུ་ཡོན་ལྷན་ཚོགས། ཡང་ན་ས་གནས་དམངས་སྲིད་ལས་ཁུངས་ཀྱིས་ལྟ་སྐྱོང་གི་འགན་ལེན་དགོས།
          དམངས་དོན་གྱི་བྱེད་ནུས་མེད་པའང་ཡང་ན་དམངས་དོན་གྱི་བྱེད་ནུས་ལ་ཚོད་འཛིན་ཐེབས་པའི་སྨྱོ་ནད་ཅན་དང་། ཤེས་ཚོར་ཉམས་པ་ཅན་གྱི་ལྟ་སྐྱོང་པ་ནི་གཤམ་གསལ་གྱི་མི་སྣས་འགན་ལེན་བྱས་ཆོག གཅིག བཟའ་ཟླ། གཉིས། ཕ་མ། གསུམ ། ནར་སོན་པའི་བུ་དང་བུ་བུ་མོ། བཞི། ཤ་ཉེ་དང་གཞན་དག་དང་འབྲེལ་བ་དམ་པོ་ཡོད་པའི་ཉེ་འབྲེལ་དང་། གྲོགས་པོ་གཞན་པའི་ཁྲོད་ན་ལྟ་སྐྱོང་གི་འགན་ལེན་འདོད་པ་ཅན། ནད་པ་དེ་ཡོད་སའི་ལས་ཁུངས་སམ་ཡང་ན་སྡོད་ཁུངས་སྡོད་དམངས་ཨུ་ཡོན་ལྷན་ཚོགས་དང་སྡེ་དམངས་ཨུ་ཡོན་ལྷན་ཚོགས་བཅས་འཐད་པ་བྱུང་རྗེས་ལྟ་སྐྱོང་བྱས་ཆོག གལ་ཏེ་གོང་གསལ་གྱི་གཉེན་ཉེ་དང་གྲོགས་པོ་མེད་ཚེ། སྨྱོ་ནད་ཅན་ནམ་ཤེས་པ་ཉམས་པ་ཅན་ཡོད་སའི་ལས་ཁུངས་དང་ཡང་ན་སྡོད་ཁུལ་གྱི་སྡོད་དམངས་ཨུ་ཡོན་ལྷན་ཚོགས སྡེ་དམངས་ཨུ་ཡོན་ལྷན་ཚོགས ། ཡང་ན་ས་གནས་དམངས་སྲིད་ལས་ཁུངས་ཀྱིས་འགན་ལེན་དགོས།
          ལྟ་སྐྱོང་པའི་ལས་འགན་གཙོ་བོ་ནི། འཚོ་བ་དང་རྒྱུ་ནོར། ཁྲིམས་མཐུན་གྱི་ཁེ་དབང་གཞན་དག་ལ་སྲུང་སྐྱོབ་བྱེད་པ་དང་། གཟིགས་རྟོག་བྱེད་པ། དོ་དམ་བྱེད་པ་བཅས་བྱས་ཏེ། ལྟ་སྲུང་བྱེད་ཡུལ་གྱི་ཚབ་བྱས་ཏེ་དམངས་དོན་བྱ་འགུལ་དང་གཏུག་བཤེར་བྱ་འགུལ་སྤེལ་བ། ལྟ་སྐྱོང་བྱེད་ཡུལ་ལ་དོ་དམ་དང་སློབ་གསོ་གཏོང་བ་བཅས་བྱེད་དགོས་སོ།།
          གོང་གི་གྱོད་དཔེ་འདིའི་གནས་ཚུལ་ལ་གཞིགས་ན། དཔལ་མཁར་དོན་གྲུབ་ཀྱི་ཤ་ཉེ་ཚོས་ཁོ་རང་དྭ་ཕྲུག་ཏུ་ལུས་ཡོད་པའི་གནས་བབས་ཤེས་བཞིན་དུ་ཁ་ཡ་མ་བྱེད་པ་ནི་ཁྲིམས་ལུགས་དང་གནས་ལུགས་གང་ལའང་མི་མཐུན་ཞིང་། དཔལ་མཁར་དོན་གྲུབ་ཀྱི་ཁྲིམས་མཐུན་ཁེ་དབང་ལ་ལྟོས་བཅས་ཀྱིས་ཕོག་ཐུག་བཟོས་པ་རེད།
          དེར་བརྟེན་ཐོག་མར་ཁྱོད་ཀྱིས་དཔལ་མཁར་དོན་གྲུབ་ཀྱི་གཉེན་ཉེ་ཚོར་ཁོ་རང་སྙོར་སྐྱོང་བྱེད་དགོས་པའི་གྲོས་གཞི་བཏོན་ཆོག་པ་དང་། གལ་སྲིད་ཁོ་ཚོས་སྔར་བཞིན་ཁ་ཡ་མི་བྱེད་ཚེ་ཁྱོད་ཀྱིས་མི་དམངས་ཁྲིམས་ཁང་ལ་ཞུ་གཏུག་བྱས་ནས་ལྟ་སྐྱོང་གི་འགན་འཁྲི་རྩད་གཅོད་བྱེད་དུ་བཅུག་ཆོག་པར་མ་ཟད། བཅའ་ཁྲིམས་ལས་དྭ་ཕྲུག་དཔལ་མཁར་དོན་གྲུབ་ཀྱི་ཁྲིམས་མཐུན་ཁེ་དབང་ལ་སྲུང་སྐྱོང་བྱེད་དུ་བཅུག་ཆོག
