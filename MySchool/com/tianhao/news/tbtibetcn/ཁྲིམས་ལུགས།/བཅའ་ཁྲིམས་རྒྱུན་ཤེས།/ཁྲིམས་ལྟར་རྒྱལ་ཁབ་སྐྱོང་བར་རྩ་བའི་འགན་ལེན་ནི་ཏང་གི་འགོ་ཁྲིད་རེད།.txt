ཁྲིམས་ལྟར་རྒྱལ་ཁབ་སྐྱོང་བར་རྩ་བའི་འགན་ལེན་ནི་ཏང་གི་འགོ་ཁྲིད་རེད།
ཏང་གི་འགོ་ཁྲིད་ནི་ཕྱོགས་ཡོངས་ནས་ཁྲིམས་ལྟར་རྒྱལ་ཁབ་སྐྱོང་བར་སྐུལ་འདེད་གཏོང་བའི་གཞི་རྩའི་འགན་ལེན་ཞིག་རེད། རྒྱལ་སྤྱི་དང་རྒྱལ་ནང་གི་ཁོར་ཡུག་རྙོག་འཛིང་ཇི་ཙམ་ཆེ་ན། སྒྱུར་བཅོས་སྒོ་དབྱེ་དང་སྤྱི་ཚོགས་རིང་ལུགས་དེང་རབས་ཅན་འཛུགས་སྐྲུན་གྱི་ལས་འགན་དེ་ཙམ་གྱིས་ལྕི་རུ་འགྲོ་གི་ཡོད་པས། དེ་བས་ཁྲིམས་སྐྱོང་གི་བསམ་གཞི་དང་ཁྲིམས་སྐྱོང་གི་བྱེད་ཐབས་སྤྱད་དེ་སྲིད་འཛིན་གྱི་གནས་བབ་བརྟན་དུ་དང་། སྲིད་འཛིན་གྱི་བྱེད་སྟངས་ལེགས་སུ། སྲིད་འཛིན་གྱི་འཇོན་ཐང་ཆེ་རུ་བཅས་བཏང་ནས་ཏང་དང་རྒྱལ་ཁབ་ཡུན་རིང་བདེ་འཁོད་ཡོང་བའི་འགན་ལེན་བྱེད་དགོས། ཕྱོགས་ཡོངས་ནས་ཁྲིམས་ལྟར་རྒྱལ་ཁབ་སྐྱོང་བ་ནི་ཏང་གི་འགོ་ཁྲིད་ལ་ཤུགས་སྣོན་རྒྱག་པ་དང་ལེགས་སུ་གཏོང་བ་དང་། ཏང་གིས་ཕྱོགས་ཡོངས་ནས་ཁྲིམས་ལྟར་རྒྱལ་ཁབ་སྐྱོང་བར་འགོ་ཁྲིད་བྱེད་པའི་ལམ་ལུགས་དང་ལས་ཀའི་ལམ་སྲོལ་འཐུས་སྒོ་ཚང་དུ་བཏང་སྟེ། ཏང་གི་འགོ་ཁྲིད་ལམ་ལུགས་ཅན་དང་ཁྲིམས་སྐྱོང་ཅན་ལ་སྐུལ་འདེད་གཏོང་དགོས་པ་དང་། ཁྲིམས་སྐྱོང་བརྒྱུད་ཏང་གི་ལམ་ཕྱོགས་དང་། མཛད་ཕྱོགས། སྲིད་ཇུས་བཅས་ལག་ལེན་ནུས་ལྡན་ཐུབ་པའི་འགན་ལེན་བྱེད་དགོས་པ་དེ་རེད།
 ——2020ལོའི་ཟླ་11ཚེས་16དང་17ཉིན། ཞི་ཅིན་ཕིང་གིས་ཀྲུང་དབྱང་གི་ཕྱོགས་ཡོངས་ནས་ཁྲིམས་ལྟར་རྒྱལ་སྐྱོང་ལས་དོན་ཚོགས་འདུའི་ཐོག་གནང་བའི་གསུང་བཤད་ལས་དྲངས།
