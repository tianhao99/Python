གཉེན་བསྐྱར་སྒྲིག་བྱེད་དུས་ཐོ་འགོད་མ་བྱས་ན་ཤུལ་འཛིན་དབང་ཆ་མེད།
གྱོད་དཔེ།
                 ཉིས་སྟོང་བདུན་ལོའི་དཔྱིད་ཀར། ང་དང་རིག་འཛིན་གཉིས་མི་གཞན་གྱི་མཚམས་སྦྱོར་བརྒྱུད་ནས་ངོ་ཤེས་པ་མ་ཟད། དུས་ཡུན་མང་པོ་མ་འགོར་བར་གཉེན་སྒྲིག་ཀྱང་བྱས། གཉེན་སྒྲིག་བྱས་རྗེས་རིག་འཛིན་དུས་རྒྱུན་དུ་ཕྱི་ལ་སོང་ནས་ཤོ་བརྒྱབ་ནས་ནམ་གུང་མ་གཏོགས་ཕྱིར་མི་ཡོང་བའི་ངན་གཤིས་ཤིག་ཡོད་པ་ངས་ད་གཟོད་ཤེས། དོན་དེའི་རྐྱེན་གྱིས་ངེད་གཉིས་ཐེངས་གང་མང་ཞིག་ལ་ཁ་རྩོད་དང་འཛིང་རེས་བྱས། གཉེན་ཉེ་དང་གྲོགས་པོ་མང་པོས་ཁོ་ལ་ཁ་ཏ་སློབ་གསོ་བྱས་ཀྱང་ཕན་འབྲས་ཅི་ཡང་མ་ཐོབ་པའི་གནས་ཚུལ་འོག ངེད་གཉིད་རང་འཐད་སྒོས་དམངས་སྲིད་ལས་ཁུངས་ལ་སོང་སྟེ་བཟའ་འཐོར་གྱི་འགྲོ་ལུགས་བསྒྲུབས་པ་ཡིན། བཟའ་འཐོར་བྱས་རྗེས། རིག་འཛིན་གྱིས་གློ་བར་དུ་བསམ་ཤེས་བཞིན་ཁྱིམ་ཚང་གི་འཕེལ་རྒྱས་ཐད་ལ་དཀའ་སྤྱད་འབད་བརྩོན་བྱས་ཤིང་། ལོ་གཉིས་ཀྱི་རིང་ལ་ཁོས་ཁྱིམ་ལ་སྡོད་ཁང་གསར་བ་ལས་པར་མ་ཟད། དངོས་འདྲུད་རླངས་འཁོར་ཞིག་ཀྱང་ཉོས་ཏེ་ག་ས་གང་དུ་དངོས་པོ་དྲུད་དེ་ཁྱིམ་ཚང་ལ་ཡོང་འབབ་བཙལ། གྲོགས་པོ་ཚང་མས་ཡང་ཡང་ཞུ་བ་བྱས་མཐར་ང་ཡང་བསྐྱར་རིག་འཛིན་ཚང་སོང་སྟེ་ཁོ་དང་མཉམ་དུ་འཚོ་བ་རོལ། སྐབས་དེར་གཉེན་བསྐྱར་སྒྲིག་གི་ཐོ་འགོད་འགྲོ་ལུགས་ཅི་ཡང་མ་བསྒྲུབ་མོད་མི་གཞན་ལ་ངེད་གཉིས་ངེད་གཉིས་ཡང་བསྐྱར་གཉེན་སྒྲིག་བྱས་ཟིན་ཞེས་བཤད་པ་ཡིན། ན་ནིང་རིག་འཛིན་ས་ཆ་གཞན་དུ་དངོས་པོ་སྐྱེལ་འདྲུད་དུ་སོང་བའི་ལམ་བར་དུ་སྟབས་མ་ལེགས་པར་འགྲིམ་འགྲུལ་དོན་རྐྱེན་བྱུང་སྟེ་ཚེ་ལས་འདས་སོང་། འགྲོམ་འགྲུལ་དོན་རྐྱེན་དེ་བྱུང་རྗེས། འགྲིམ་འགྲུལ་དོན་རྐྱེན་གྱི་འགན་འཁྲི་གཙོ་བོ་འཁུར་མཁན་དང་ཉེན་འགོག་ཀུང་སི་གཉིས་ཀྱིས་ངེད་ཚང་ལ་ས་གནས་འགྲིམ་འགྲུལ་ཉེན་རྟོག་པས་ཐད་བཅད་བྱས་པ་ལྟར་སྐྱིན་ཚབ་སྒོར་ཁྲི་བརྒྱད་བྱིན་ནའང་།སྒོར་ཁྲི་བརྒྱད་པོ་རིག་འཛིན་གྱི་ཨ་ཕས་བདག་བཟུང་བྱས་པ་མ་ཟད། ད་ཁོ་ཚང་ནས་ཀྱང་འདུག་དུ་མ་བཅུག་པར་སྒོར་བདའ། དེ་བས། ངས་རིག་འཛིན་གྱི་ཕ་མར། ཤུལ་བཞག་རྒྱུ་ནོར་ཤུལ་འཛིན་བྱེད་པའི་རེ་བ་བཏོན་ཡང་ཁོས་ཁ་ཡ་མ་བྱས། བདག་གིས་ས་གནས་མི་དམངས་ཁྲིམས་ཁང་ལ་གཏུག་བཤེར་བྱས་ན་རིག་འཛིན་ལ་སྐྱིན་ཚབ་ཏུ་བྱིན་པའི་སྒོར་ཁྲི་བརྒྱད་ཀྱི་ཤུལ་འཛིན་དབང་ཆ་བདག་ལ་ཡོད་དམ།
ཁྲིམས་བཤད།
                 ང་ཚོའི་བསམ་པར། ཁྱོད་དང་རིག་འཛིན་གཉིས་སྔོན་ཆད་ཁྱོ་ཤུག་གི་འབྲེལ་བ་ཡིན་མོད། འོན་ཀྱང་ཉིས་སྟོང་བདུན་ལོར་བཟའ་འཐོར་བྱས་ཤིང་བཟའ་འཐོར་བྱེད་དུས་རྒྱུ་ནོར་ཕྱོགས་ནས་ཐག་གཅོད་བྱས་ཡོད་པ་རེད། ཕྱིས་སུ། ཁྱོད་གཉིས་ཡང་བསྐྱར་འགྲིག་མཐུན་བྱུང་སྟེ་ཁྱིམ་གཅིག་ཏུ་འཚོ་བ་རོལ་མོད། ཡིན་ནའང་། གཉེན་སྒྲིག་བཅའ་ཁྲིམས་ལྟར་ཡང་བསྐྱར་གཉེན་སྒྲིག་ཐོ་འགོད་སྡེ་ཁག་ལ་སོང་སྟེ་གཉེན་སྒྲིག་ཐོ་འགོད་ཀྱི་འགྲོ་ལུགས་མ་བསྒྲུབ་པས། ཁྱོད་གཉིས་ཀྱི་གཉེན་བསྐྱར་སྒྲིག་བྱས་པར་བཅའ་ཁྲིམས་ཀྱི་སྲུང་སྐྱོབ་མི་ཐོབ། རང་རྒྱལ་གྱི་གཉེན་སྒྲིག་བཅའ་ཁྲིམས་ཀྱི་གཏན་ཁེལ་ལྟར་ན། གཉེན་སྒྲིག་ཐོ་འགོད་མ་བསྒྲུབ་པར་ཁྱོ་ཤུག་གི་མིང་ཐོག་ནས་མཉམ་གཅིག་ཏུ་འཚོ་བ་རོལ་བའི་ཕོ་མོ་ཕྱོགས་གཅིག་ཚེ་ལས་འདས་ན། ཕྱོགས་གཞན་དེས་བཟའ་ཟླའི་ཐོབ་ཐང་གི་ཤུལ་འཛིན་དབང་ཆ་ལོངས་སྤྱོད་བྱས་མི་ཆོག དུས་མཚུངས་སུ། རང་རྒྱལ་གྱི་ཤུལ་འཛིན་བཅའ་ཁྲིམས་ཀྱི་དོན་ཚན་བཅུ་བའི་གཏན་ཁེལ་ལྟར་ན། ཤུལ་བཞག་རྒྱུ་ནོར་ལ་གཤམ་གྱི་གོ་རིམ་ལྟར་ཤུལ་འཛིན་བྱེད་དགོས།
                 གོ་རིམ་དང་པོ་ནི་བཟའ་ཟླ་དང་བྱིས་པ། ཕ་མ་བཅས་ཡིན། རིག་འཛིན་གྱི་ཕ་མ་ནི་གོ་རིམ་དང་པོའི་ཤུལ་འཛིན་མཁན་ལ་གཏོགས་པས། རིག་འཛིན་གྱི་ཤུལ་བཞག་རྒྱུ་ནོར་ཁོའི་ཕ་མ་གཉིས་ཀྱི་ཤུལ་འཛིན་བྱེད་འོས། ཡིན་ནའང་། ཁྱོད་དང་རིག་འཛིན་གཉིས་ཁྱིམ་གཅིག་ཏུ་སྡོད་དུས་མཉམ་དུ་ངལ་རྩོལ་བྱས་ཏེ་བསགས་པའི་རྒྱུ་ནོར་དང་སྒོར་མོ་ཡོད་པ་བདེན་དཔང་བྱེད་ཐུབ་ཚེ། ཁྱོད་ལ་རྒྱུ་ནོར་དེའི་ཕྱེད་ཀ་ཐོབ་པའི་དབང་ཆ་ཡོད་ཅིང་། གཞན་པའི་ཕྱེད་ཀ་དེ་རིག་འཛིན་གྱི་ཤུལ་བཞག་རྒྱུ་ནོར་དུ་གཏོགས་པས་ཁྱོད་ལ་ཤུལ་འཛིན་བྱེད་པའི་དབང་ཆ་མེད་ལ། རིག་འཛིན་གྱི་ཕ་མ་གཉིས་ཀྱི་ཤུལ་འཛིན་བྱེད་དགོས། རིག་འཛིན་ཤི་རྨས་བྱུང་བར་སྐྱིན་ཚབ་འཇལ་བའི་སྒོར་ཁྲི་བརྒྱད་ནི། རིག་འཛིན་ལ་དབང་བའི་ཤུལ་ནོར་ཡིན་ལ་དེ་རྒྱུན་འཛིན་དང་ཤུལ་འཛིན་བྱེད་མཁན་གྱི་བདག་དབང་ཁོའི་ཕ་མ་གཉིས་ལ་ཡོད་དོ།  །
