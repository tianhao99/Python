རྒྱལ་ཁབ་ཁྲལ་དོན་སྤྱི་ཁྱབ་ཅུས་ཀྱིས“ཁྲལ་གྲོན་ཚད་ལས་བརྒལ་ནས་བསྡུ་བ”སོགས་ཀྱི་སྒྲིག་འགལ་བྱེད་སྤྱོད་ལ་ཞིབ་བཤེར་ཐག་གཅོད་མཐའ་གཅིག་ཏུ་བྱས་པ།
ཚེས་30ཉིན་གསར་འགོད་པར་རྒྱལ་ཁབ་ཁྲལ་དོན་སྤྱི་ཁྱབ་ཅུས་ནས་ཤེས་རྟོགས་བྱུང་བར། རྒྱལ་སྲིད་སྤྱི་ཁྱབ་ཁང་གཞུང་དོན་ཐིང་ལྟ་བཤེར་ཁང་གིས་བརྡ་ཁྱབ་བཏང་བའི་ཧོ་པེ་ཞིང་ཆེན་ཅིང་རྫོང་གིས་སྒྲིག་འགལ་ངང་ཁྲལ་བསྡུས་པ་སོགས་ཀྱི་གནད་དོན་ལ་དམིགས་ནས། ཉེ་ཆར་ཁྲལ་དོན་སྤྱི་ཁྱབ་ཅུས་ཀྱིས་རྒྱལ་ཡོངས་ཀྱི་ཁྲལ་དོན་ཁོངས་གཏོགས་ཀྱི་བརྙན་ལམ་ཚོགས་འདུ་འཚོགས་ཤིང་། ཚོགས་འདུའི་ཐོག་སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིན་ཕིང་གི་གསུང་བཤད་གལ་ཆེན་གྱི་དགོངས་དོན་དང་མཆན་བཀོད་མཛུབ་སྟོན་གལ་ཆེན་གྱི་དགོངས་དོན་སློབ་སྦྱོང་དང་ལག་བསྟར་གཏིང་ཟབ་བྱེད་པ་དང་། ཏང་ཀྲུང་དབྱང་དང་རྒྱལ་སྲིད་སྤྱི་ཁྱབ་ཁང་གི་འཐབ་ཇུས་བཀོད་སྒྲིག་དོན་འཁྱོལ་ནན་ཏན་བྱེད་པ། “ཁྲལ་གྲོན་ཚད་ལས་བརྒལ་ནས་བསྡུ་བ”སོགས་ཀྱི་སྒྲིག་འགལ་བྱེད་སྤྱོད་ལ་ཞིབ་བཤེར་ཐག་གཅོད་མཐའ་གཅིག་ཏུ་བྱེད་པ། ཁྲལ་རིན་གཉིས་གཅོག་གི་སྲིད་ཇུས་དང་ཁྲིམས་བཞིན་ཡོང་འབབ་བསྡུ་བའི་རྩ་དོན་དོན་འཁྱོལ་མཐའ་གཅིག་ཏུ་བྱེད་པ་བཅས་བྱས་ཏེ“བརྟན་པོ་དྲུག”དང“འགན་ལེན་དྲུག”གི་སྤྱི་ཡོངས་ལ་ཞབས་འདེབས་དེ་བས་ལེགས་པ་ཞུ་དགོས་པའི་བླང་བྱ་བཏོན་ཡོད།
          འདི་ལོ་ཚུན་དུ། ཏོག་གསར་གློ་ཚད་རིམས་ནད་ཀྱི་ཤུགས་རྐྱེན་ཐེབས་ནས་རིམ་ཁག་ནོར་སྲིད་དང་ལྷག་པར་དུ་གཞི་རིམ་ནོར་སྲིད་ཀྱི་གཏོང་ཡོང་གི་འགལ་ཟླ་མངོན་གསལ་དོད་དུ་སོང་བ་དང་ཁྲལ་དོན་སྡེ་ཚན་གྱིས་ཡོང་འབབ་བསྡུ་བར་གནོན་ཤུགས་ཆེ་ཙམ་འཕྲད་ཡོད། ཁྲལ་དོན་སྤྱི་ཁྱབ་ཅུས་ཀྱིས་སྔ་རྗེས་སུ་ཡིག་ཆ་ཐེངས་མང་བཀྲམ་པ་དང་ཚོགས་འདུ་ཐེངས་མང་འཚོགས་ཏེ། རིམ་ཁག་ཁྲལ་དོན་ལས་ཁུངས་ལ་ཆབ་སྲིད་ཀྱི་ཚུགས་གནས་སྔར་བས་མཐོ་རུ་གཏོང་བ་དང་། ཏང་ཀྲུང་དབྱང་དང་རྒྱལ་སྲིད་སྤྱི་ཁྱབ་ཁང་གི་འཐབ་ཇུས་བཀོད་སྒྲིག་གཞིར་བཟུང་ཁྲལ་རིན་གཉིས་གཅོག་གི་སྲིད་ཇུས་དོན་འཁྱོལ་ནན་ཏན་བྱེད་པ། ཁྲིམས་བཞིན་ཡོང་འབབ་བསྡུ་བའི་རྩ་དོན་ནན་པོ་སྲུང་བ། “ཁྲལ་གྲོན་ཚད་ལས་བརྒལ་ནས་མི་བསྡུ་བའི”མཐའ་ཚད་ནན་པོ་སྲུང་བ་བཅས་ལས། གནོན་ཤུགས་ཆེ་བའི་དབང་གིས་རྩ་དོན་བློས་གཏོང་བྱེད་པའི་བྱ་བ་གཏན་ནས་བྱེད་མི་རུང་ལ། གནོན་ཤུགས་ཀྱིས་རྐྱེན་པས་མཐའ་ཚད་ལས་བརྒལ་བ་ཡང་གཏན་ནས་བྱེད་མི་རུང་བའི་བླང་བྱ་བཏོན་ཡོད།
