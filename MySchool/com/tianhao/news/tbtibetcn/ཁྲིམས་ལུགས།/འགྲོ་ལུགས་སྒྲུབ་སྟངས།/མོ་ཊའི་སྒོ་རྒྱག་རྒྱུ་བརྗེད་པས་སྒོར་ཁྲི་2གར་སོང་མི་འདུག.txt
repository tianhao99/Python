མོ་ཊའི་སྒོ་རྒྱག་རྒྱུ་བརྗེད་པས་སྒོར་ཁྲི་2གར་སོང་མི་འདུག
ཟླ་7ཚེས་8ཉིན་གྱི་དགོང་མོའི་ཆུ་ཚོད་9པ་ཙམ་ལ། དོ་བདག་བཀྲ་ཆེ་གེ་མོས་ལྷ་ས་གྲོང་ཁྱེར་སྤྱི་བདེ་ཅུས་ཉེས་དོན་ཉེན་རྟོག་ཉིང་ལག་རུ་ཁག་ལ། ལྷ་ས་གྲོང་ཁྱེར་ཁྲེང་ཀོན་ཆུས་འོ་བརྒྱལ་ཐང་གྲོང་ལྷན་ཉེ་འགྲམ་དུ་བཞག་པའི་མོ་ཊའི་ནང་གི་དངུལ་སྨར་སྒོར་ཁྲི་2བསྡད་མི་འདུག་ཅེས་ཁ་པར་བརྒྱབ།
          ཁ་པར་འབྱོར་རྗེས། མི་དམངས་ཉེན་རྟོག་པ་མགྱོགས་མྱུར་གནས་ཚུལ་ཐོན་ཡུལ་དུ་བསྐྱོད། རྩད་ཞིབ་བྱས་པ་བརྒྱུད། མོ་ཊ་དེའི་སྒེའུ་ཁུང་གི་ཤེལ་སྒོ་དང་སྒོ་སོགས་ལ་གཏོར་སྐྱོན་བཏང་ཤུལ་མེད་པས། མི་དམངས་ཉེན་རྟོག་པའི་ཚོད་དཔག་ལྟར་ན། དོ་བདག་གིས་མོ་ཊའི་སྒོ་རྒྱག་རྒྱུ་བརྗེད་པས། དོགས་ཡོད་ཉེས་ཅན་གྱིས་མོ་ཊའི་སྒོ་ཕྱེ་ནས་བརྐུས་པ་ཞིག་རེད། དོ་བདག་གིས་མཁོ་འདོན་བྱས་པའི་གནས་ཚུལ་དང་། དེ་བཞིན་གྱོད་གཞི་ཐོན་སའི་ཉེ་འཁོར་གྱི་གློག་མིག་ལ་བལྟས་རྗེས། གྱོད་སྒྲུབ་མི་དམངས་ཉེན་རྟོག་པས་ཉེས་གསོག་དོགས་ཅན་གྱི་བྱ་སྤྱོད་སྤེལ་ཤུལ་ཤེས་པ་མ་ཟད། གྱོད་སྒྲུབ་མི་དམངས་ཉེན་རྟོག་པས་འབྲེལ་ཡོད་སྡེ་ཚན་དང་མཉམ་དུ་ལྷ་ས་གྲོང་ཁྱེར་ཁྲེང་ཀོན་ཆུས་ཀརྨ་དགོན་གསར་གྱི་ཚོགས་ཆུང་བརྒྱད་པའི་བོགས་གཏོང་ཁང་པ་ག་གེ་མོ་ཞིག་ནས་ཉེས་གསོག་དོགས་ཅན་བློ་ཆེ་གེ་མོ་བཟུང་བ་རེད། 
          དོགས་ཡོད་ཉེས་ཅན་པས“ཉིན་དེའི་ཕྱི་དྲོའི་ཆུ་ཚོད་16ཙམ་ལ། ང་ཁྲེང་ཀོན་ཆུས་འོ་བརྒྱལ་ཐང་གྲོང་ལྷན་གྱི་ཉེ་འགྲམ་ན་ཡོད། སྐབས་དེར་མོ་ཊ་འགའ་ཤས་ཀྱི་སྒོ་བརྒྱབ་ཡོད་མེད་ལ་ལྟ་སྐབས། སྒོ་འབྱེད་ཐུབ་མ་བྱུང་། ཡིན་ནའང་། ང་རང་དེ་ནས་ཁ་བྲལ་རྩིས་བྱེད་སྐབས། སྒོ་བརྒྱབ་མེད་པའི་མོ་ཊ་ཞིག་ཕྲད་བྱུང་། ངས་མྱུར་དུ་མོ་ཊའི་ནང་འཛུལ་ནས། ཁ་ལོ་བའི་གཡས་ཀྱི་སྡོད་སའི་ཟོག་སྒམ་ཁ་འབྱེད་སྐབས། དེ་ནས་དངུལ་སྨར་སྒོར་ཁྲི་2རག་བྱུང”ཞེས་མི་དམངས་ཉེན་རྟོག་པར་གནས་ཚུལ་གྱི་བྱུང་རྐྱེན་ཚང་མ་བཤད། 
          མིག་སྔར། གྱོད་སྒྲུབ་མི་དམངས་ཉེན་རྟོག་པས་དངུལ་སྒོར་19800ཚུར་བླངས་ནས་དོ་བདག་ལ་རྩིས་སྤྲད་པ་མ་ཟད། ཉེས་གསོག་དོགས་ཅན་བློ་ཆེ་གེ་མོ་ཁྲིམས་ལྟར་ཉེས་དོན་བཀག་ཉར་བྱས་པ་རེད།
