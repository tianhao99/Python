ཞན་ཆ་གསབ་རྒྱུ་ནི་དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་དམག་འཐབ་ལ་རྒྱལ་ཁ་ལེན་པའི་འགག་རྩ་ཡིན།

ཟླ་6ཚེས་1ཉིན་དཔེ་སྐྲུན་བྱས་པའི་དུས་ཐེངས་11པའི《བདེན་འཚོལ》དུས་དེབ་ཐོག་སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིན་ཕིང་གི་གསུང་རྩོམ་གལ་ཆེན《འབྱོར་འབྲིང་སྤྱི་ཚོགས་སྒོ་ཀུན་ནས་འཛུགས་སྐྲུན་ལེགས་གྲུབ་བྱེད་པའི་ཞན་ཆ་གསབ་རྒྱུའི་གནད་དོན་སྐོར། 》ཞེས་པ་བཀོད་ཡོད། གསུང་རྩོམ་ནང་འབྱོར་འབྲིང་སྤྱི་ཚོགས་སྒོ་ཀུན་ནས་བསྐྲུན་རྒྱུ་ནི་ཕྱོགས་གང་ས་ཅི་ཐད་དང་འབྲེལ་བ་ཡོད་མོད། འོན་ཀྱང་ཞན་ཆ་གསབ་རྒྱུ་ནི་ལྡོག་མེད་ཀྱི་ལས་འགན་ཞིག་རེད་ཅེས་ནན་བཤད་གནང་ཡོད།
ཏང་གི་ཚོགས་ཆེན་བཅོ་བརྒྱད་པ་ཚོགས་ཚུན། ལྗོངས་ཡོངས་ཀྱི་མི་རིགས་ཁག་གི་མི་དམངས་ཀྱིས་དཀའ་ངལ་ལ་མི་འཛེམ་པར་མདུན་སྐྱོད་བྱེད་པ་དང་དཀའ་སྤྱད་འབད་འཐབ་ཀྱི་འབད་བརྩོན་འོག བོད་ཀྱི་དབུལ་སྒྲོལ་འགག་སྒྲོལ་ལ་ཐག་གཅོད་རང་བཞིན་གྱི་འཕེལ་རྒྱས་བྱུང་ཡོད་ཅིང་། ལྗོངས་ཡོངས་ཀྱི་ཡིག་ཚགས་དང་གཱ་བཟོས་པའི་དབུལ་པོའི་མི་འབོར་ཁྲི་62.8ཚང་མ་དབུལ་སྒྲོལ་ཐུབ་པ་དང་། རྫོང་(ཆུས)74ཚང་མ་དབུལ་ཞྭ་ཕུད་ཅིང་། ཀྲུང་དབྱང་གིས་དཔྱད་ཞིབ་བྱས་ནས་ལོ་4བསྟུད་མར“ཕྱོགས་བསྡུས་གདེང་འཇོག་ཡག་པོར”གཏན་འཁེལ་བྱས་པ་རེད། འོན་ཀྱང་གྲུབ་འབྲས་མཐོང་བ་དང་དུས་མཚུངས་སུ་ང་ཚོས་ད་དུང་རིག་པ་དྭངས་གསལ་གྱི་སྒོ་ནས་བོད་ཀྱི་དཔལ་འབྱོར་འཕེལ་རྒྱས་འགྲོ་འགོར་བ་དང་། ཉམས་མྱོང་ཉུང་བ། འཕེལ་རྒྱས་དོ་མི་མཉམ་ཞིང་གང་ལེགས་མ་བྱུང་བའི་མངོན་གསལ་དོད་པའི་གནད་དོན་ལ་ངོས་འཛིན་གསལ་པོ་བྱས་ཏེ། རྒྱལ་ཡོངས་མི་རིགས་ཁག་གི་མི་དམངས་དང་ལྷན་དུ་འབྱོར་འབྲིང་སྤྱི་ཚོགས་སྒོ་ཀུན་ནས་སྐྲུན་རྒྱུར་འགན་ལེན་བྱེད་པར་ད་དུང་ཡང་གཤམ་གསལ་གྱི་ཞན་ཆ་འགའ་ཞིག་གསབ་དགོས།
དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་རྒྱལ་ཁ་ཕུ་ཐག་གཅོད་པར“ཐོན་ལས་ཀྱི་ཞན་ཆ”གསབ་དགོས། ཐོན་ལས་ཀྱི་ཞན་ཆ་གསབ་རྒྱུ་ནི་གྲོང་ཚོ་དབུལ་པོའི་རང་ངོས་འཕེལ་རྒྱས་ཀྱི་ནུས་པ་ཆེ་རུ་གཏོང་བ་དང་། ཤིན་ཏུ་དབུལ་བའི་ས་ཁུལ་གྱི་མང་ཚོགས་ལ་འབབ་འཕར་ཕྱུག་འགྱུར་བརྟན་པོ་ཡོང་རྒྱུར་སྐུལ་འདེད་གཏོང་བ། དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་རྒྱལ་ཁ་ཕུ་ཐག་གཅོད་པའི་ལྡོག་མེད་ལམ་བུ་ཞིག་བཅས་རེད། རྨང་གཞིའི་སྒྲིག་བཀོད་འཛུགས་སྐྲུན་ནི་ཐོན་ལས་འཕེལ་རྒྱས་ཀྱི་ཞན་ཆ་ཆེ་ཤོས་ཡིན། ཐོན་ལས་འཛུགས་སྐྲུན་བྱེད་ཤུགས་ཆེ་རུ་གཏོང་བ་དང་། ཞིང་ནང་ཆུ་བེད་དང་། འཐུང་ཆུའི་བདེ་འཇགས། གྲོང་གསེབ་ཀྱི་གློག་ཤུགས། དྲ་རྒྱའི་འཕྲིན་གཏོང་སོགས་རྨང་གཞིའི་སྒྲིག་བཀོད་འཛུགས་སྐྲུན་བྱེད་ནུས་ཆེ་རུ་བཏང་ནས་ཞིང་འབྲོག་ཁུལ་གྱི་ཐོན་ལས་ཐོན་རྫས་འཁོར་རྒྱུག་སྐྱེལ་འདྲེན་དང་ཐོན་སྐྱེད་ཀྱི་རྒྱུ་རྐྱེན་གཙོ་བོའི་འཁོར་རྒྱུག་ནུས་པ་དངོས་སུ་ཆེ་རུ་གཏོང་དགོས། ཐོན་ལས་ཀྱི་བཀོད་པ་རྒྱུན་མཐུད་ངང་ལེགས་སུ་གཏོང་བ་དང་། ས་གནས་དེ་གའི་ཐོན་ཁུངས་ཀྱི་ལེགས་ཆར་གཞི་ཚུགས་པ། ཇུས་འགོད་ཧུར་ཐག་བྱེད་པ། ཡུལ་བབ་དང་བསྟུན་པ། ཞིང་ལས་འཚམ་སར་ཞིང་ལས་དང་། འབྲོག་ལས་འཚམ་སར་འབྲོག་ལས། ཡུལ་སྐོར་འཚམ་སར་ཡུལ་སྐོར། ནགས་ལས་འཚམ་སར་ནགས་ལས་བཅས་བོད་ཀྱི་ཁྱད་ཆོས་དང་འཚམ་པའི་ནུས་ཆེའི་ཁྱད་ལྡན་ཐོན་ལས་འཕེལ་རྒྱས་བཏང་ནས་ཐོན་ལས་དབུལ་སྐྱོར་གྱི་ཕན་འབྲས་རྒྱུན་ཆད་མེད་པར་སྲ་བརྟན་དུ་གཏོང་དགོས།
དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་རྒྱལ་ཁ་ཕུ་ཐག་གཅོད་པར“བསམ་པའི་ཞན་ཆ”གསབ་དགོས། དངོས་གཙོའི་རྩོད་སྒྲུབ་རིག་པས་ང་ཚོར་ཕྱི་རྐྱེན་ནི་འགྱུར་ལྡོག་ཡོང་བའི་ཆ་རྐྱེན་ཡིན་པ་དང་ནང་རྐྱེན་ནི་འགྱུར་ལྡོག་གི་རྩ་བ་ཡིན་པ་བསྟན་ཡོད། ད་ཆ་དབུལ་སྒྲོལ་འགག་སྒྲོལ་དེ“མཐའ་མའི་གདོང་མཆོང་གི་དུས་སྐབས”སུ་སླེབས་པས་དབུལ་སྒྲོལ་
འགག་སྒྲོལ་གྱི་དམག་འཐབ་ལ་རྒྱལ་ཁ་ཐོབ་དགོས་ན་ངེས་པར་དུ“ཐོན་ལས་ཀྱི་ཞན་ཆ”གསབ་པའི་རྨང་གཞིའི་ཐོག་དབུལ་སྒྲོལ་འགག་སྒྲོལ་ཁྲོད་གནས་པའི“བསམ་པའི་ཞན་ཆའི”གནད་དོན་ཐག་གཅོད་གང་ལེགས་བྱེད་དགོས། དབུལ་སྐྱོར་བྱེད་སྐབས་བློ་སྐྱོར་དང་རིག་སྐྱོར་ཟུང་འབྲེལ་བྱ་རྒྱུར་མཐོང་ཆེན་བྱས་ཏེ། དབུལ་སྒྲོལ་ཐད“གཞན་བརྟེན”ནས“རང་བརྟེན”དུ་འགྱུར་བ་དང་། གཞན་སྐུལ་ནས་རང་འགུལ་དུ་འགྱུར་བ་བྱས་ནས་དབུལ་ཕོངས་མང་ཚོགས་ཀྱི་ཧུར་སེམས་དང་རང་འགུལ་བྱེད་སེམས་སྐུལ་སློང་གང་ལེགས་བྱས་ཏེ། མང་ཚོགས་དབུལ་སྒྲོལ་གྱི་ནང་གནས་སྒུལ་ཤུགས་ཆེ་རུ་བཏང་ནས་དབུལ་པོའི་རྣམ་པར་འགྱུར་ལྡོག་གཏོང་བའི་ངར་ཤུགས་དང་ཆོད་སེམས་འཕེལ་བ་བྱེད་དགོས། ས་གནས་ཁག་གིས་དབུལ་ཕོངས་མང་ཚོགས་ལ་གནས་པའི“བསམ་པའི་དབུལ་ཕོངས”གནད་དོན་ལ་དབྱེ་ཞིབ་ནན་ཏན་བྱས་ནས་ནད་བསྟུན་སྨན་བཅོས་དང་ལས་དོན་གསར་གཏོད་ཀྱིས་མང་ཚོགས་ལ་རང་ཉིད་ཀྱི་ལས་ལ་བརྩོན་པའི་ལག་གཉིས་ལ་བརྟེན་ནས་དབུལ་པོའི་ལས་དབང་ལ་འགྱུར་ལྡོག་གཏོང་རྒྱུའི་སློབ་གསོ་དང་ཁྲིད་སྟོན་ཏན་ཏིག་བྱེད་དགོས།
2020ནི་དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་རྩིས་སྤྲོད་ལོ་ཞིག་ཡིན་ལ། ཕྱོགས་ཡོངས་ནས་འབྱོར་འབྲིང་སྤྱི་ཚོགས་སྐྲུན་ཐུབ་པ་དང“ལོ་ལྔ་ཚན་བཅུ་གསུམ་པའི”འཆར་འགོད་ཀྱི་མཇུག་བསྡུའི་ལོ་ཞིག་ཀྱང་ཡིན། བོད་ཀྱིས་རྒྱུན་རྣམ་ཅན་གྱི་རིམས་ནད་སྔོན་འགོག་ཚོད་འཛིན་ལ་དམ་འཛིན་ཡག་པོ་བྱེད་པ་དང་དུས་མཚུངས། དབུལ་སྒྲོལ་གྱི་གྲུབ་འབྲས་སྲ་བརྟན་དུ་གཏོང་རྒྱུ་དེ་ལས་འགན་ཨང་དང་པོར་བརྩིས་ནས་སྟོབས་ཤུགས་བསྡུ་སྒྲིལ་དང་ཞན་ཆ་གསབ་ནས་དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་སྤུས་ཚད་དེ་བས་ལེགས་པའི་འགན་ལེན་བྱེད་དགོས།
