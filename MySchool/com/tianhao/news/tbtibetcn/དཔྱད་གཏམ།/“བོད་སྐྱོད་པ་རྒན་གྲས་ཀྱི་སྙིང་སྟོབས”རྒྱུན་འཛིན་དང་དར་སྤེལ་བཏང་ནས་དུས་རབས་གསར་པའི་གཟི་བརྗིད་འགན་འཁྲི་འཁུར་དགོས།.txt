“བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས”རྒྱུན་འཛིན་དང་དར་སྤེལ་བཏང་ནས་དུས་རབས་གསར་པའི་གཟི་བརྗིད་འགན་འཁྲི་འཁུར་དགོས།

སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིན་ཕིང་གིས་ཀྲུང་དབྱང་གི་བོད་ཀྱི་ལས་དོན་སྐོར་གྱི་བཞུགས་མོལ་ཚོགས་འདུ་ཐེངས་བདུན་པའི་ཐོག་གནང་བའི་གསུང་བཤད་གལ་ཆེན་གྱི་དགོངས་དོན་སློབ་སྦྱོང་དང་ལག་བསྟར་བྱེད་རྒྱུའི་སྐོར་གྱི་གླེང་ཐེངས་བཅུ་བཞི་པ།
མི་ལ་སྙིང་སྟོབས་མེད་ན་དོན་མི་སྒྲུབ། རྒྱལ་ཁབ་ལ་སྙིང་སྟོབས་མེད་ན་སྟོབས་འབྱོར་ཡོང་མི་སྲིད། “རྒྱ་ཆེའི་ལས་བྱེད་པ་དང་ལྷག་པར་དུ་བོད་ཀྱི་ལས་བྱེད་པས‘བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས’དར་སྤེལ་བཏང་ནས་གསོ་རླུང་དཀོན་ཡང་སྙིང་སྟོབས་མི་ཉམས་པ་དང་། དཀའ་ཚེགས་ཆེ་ཡང་དཀའ་སྡུག་ལ་མི་སྐྲག་པ། ས་བབ་མཐོ་སར་གོ་རྟོགས་དེ་བས་མཐོ་བ་བཅས་བྱས་ཏེ། ལས་དོན་ཁྲོད་འགན་འཁྲིའི་བསམ་པ་དང་འོས་འགན་སྒྲུབ་སེམས་ཟམ་མི་ཆད་པར་ཟབ་ཏུ་གཏོང་བ་དང་། འཇོན་ཐང་ཆེ་རུ་གཏོང་བ། སྤྱོད་ཚུལ་ལ་ཞུན་བརྡར་ཐུབ་པ་བཅས་བྱེད་དགོས” ཀྲུང་དབྱང་གི་བོད་ཀྱི་ལས་དོན་སྐོར་གྱི་བཞུགས་མོལ་ཚོགས་འདུ་ཐེངས་བདུན་པའི་ཐོག་སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིན་ཕིང་གིས་དུས་རབས་གསར་པའི་བོད་ཀྱི་ལས་དོན་ཡག་པོ་སྒྲུབ་པའི་བསམ་པའི་འདེགས་སྐྱོར་གསལ་སྟོན་གཏིང་ཟབ་གནང་ཡོད། ང་ཚོས་སྤྱི་ཚོགས་ཡོངས་སུ“བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས”དར་སྤེལ་ཤུགས་ཆེན་གཏོང་བ་དང་། གསར་བརྗེའི་ས་བོན་རྒྱུན་འཛིན། འཐབ་རྩོད་ཀྱི་འཇོན་ཐང་ཆེ་རུ་གཏོང་བ། དུས་རབས་ཀྱི་ནང་དོན་སྣོན་པ་བཅས་བྱེད་རྒྱུ་དེ་དུས་རབས་གསར་པའི་བོད་ཡུན་རིང་བདེ་འཁོད་དང་སྤུས་ཚད་མཐོ་བའི་འཕེལ་རྒྱས་ལ་སྐུལ་འདེད་གཏོང་བའི་སྟོབས་ཤུགས་ཆེན་པོར་འགྱུར་བ་བྱེད་དགོས།
རླབས་ཆེན་འཐབ་རྩོད་ཀྱིས་རླབས་ཆེན་གྱི་སྙིང་སྟོབས་འཕེལ་བར་བྱེད། ས་མཐོར་ལས་དོན་སྒྲུབ་སྐབས་དཀོན་ཤོས་ནི་གསོ་རླུང་ཡིན་པ་དང་། རྩ་ཆེ་ཤོས་ནི་སྙིང་སྟོབས་ཡིན། “སྡུག་བསྲན་ཁྱད་པར་ཐེག་པ་དང་། བཟོད་བསྲན་ཁྱད་པར་ཐུབ་པ། མཐུན་སྒྲིལ་ཁྱད་པར་ཐུབ་པ། དམག་འཐབ་ཁྱད་པར་འཇོན་པ། ལེགས་སྐྱེས་ཁྱད་པར་འབུལ་བ”བཅས་ནང་དོན་གཙོ་བོར་བྱས་པའི“བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས”ནི་རང་རེའི་ཏང་གིས་བོད་ཀྱི་མི་རིགས་ཁག་གི་མི་དམངས་ཀྱི་སྣེ་ཁྲིད་དེ་བཙན་རྒྱལ་རིང་ལུགས་ཀྱི་སྟོབས་ཤུགས་མཐར་སྐྲོད་གཏོང་བ་དང་། བཀས་བཀོད་རྒྱུད་འཛིན་ཞིང་བྲན་ལམ་ལུགས་མེད་པ་བཟོ་བ། མི་དམངས་རང་ཁྱིམ་རང་བདག་གི་དུས་རབས་གསར་པ་གཏོད་པ་བཅས་ཀྱི་རླབས་ཆེན་འཐབ་རྩོད་ཁྲོད་སྐྱེད་སྲིང་བྱུང་བ་དང་། རང་རེའི་ཏང་གིས་བོད་ཀྱི་མི་རིགས་ཁག་གི་མི་དམངས་ཀྱི་སྣེ་ཁྲིད་དེ་སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་བོད་ལྗོངས་གསར་པ་འཛུགས་སྐྲུན་བྱེད་པའི་རླབས་ཆེན་ལག་ལེན་ཁྲོད་འཕེལ་རྒྱས་བྱུང་བ། རྒྱ་ཆེའི་བོད་སྡོད་དཔུང་སྡེའི་དཔོན་དམག་དང་། ལས་བྱེད་ལས་བཟོ། མི་རིགས་ཁག་གི་མི་དམངས་མང་ཚོགས་བཅས་ཀྱིས་བོད་དུ་གཞི་ཚུགས་པ་དང་། དཀའ་སྤྱད་འབད་འཐབ། ཚེ་སྲོག་དང་ཁྲག་དྲོན་བཅས་ལས་བྱུང་བ་ཞིག་རེད།
“བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས”དེ“ཡུན་རིང་དཀའ་སྤྱད་འབད་འཐབ་ཀྱི་གསར་བརྗེའི་ལག་ལེན་ཁྲོད་རང་རེའི་ཏང་གིས་བཏོད་པའི“ཅིན་ཀང་ཧྲན་རི་བོའི་སྙིང་སྟོབས”དང་། “རྒྱང་སྐྱོད་ཀྱི་སྙིང་སྟོབས”“ཡན་ཨན་གྱི་སྙིང་སྟོབས” “ཤིས་པའེ་ཕོའི་སྙིང་སྟོབས”བཅས་དང་གཅིག་རྒྱུན་གཅིག་འཛིན་བྱས་པ་ཞིག་རེད། དེ་ནི་བོད་དུ་ཀྲུང་ཧྭ་མི་རིགས་ཀྱི་རིག་གནས་སྙིང་བཅུད་རྒྱུན་འཛིན་དང་དར་སྤེལ་བཏང་བ་ཞིག་དང་། ཀྲུང་ཧྭ་མི་རིགས་ཀྱི་ཕུལ་བྱུང་གཤིས་རྒྱུད་བོད་དུ་རྒྱུན་འཛིན་དང་དར་སྤེལ་བཏང་བ་ཞིག བོད་དུ་གུང་ཁྲན་ཏང་མིའི་གསར་བརྗེའི་སྙིང་སྟོབས་རྒྱུན་འཛིན་དང་དར་སྤེལ་བཏང་བ་ཞིག ཀྲུང་གོའི་སྙིང་སྟོབས་ཀྱི་གྲུབ་ཆ་གལ་ཆེན་ཞིག་བཅས་ཡིན་པ་དེར་ཚན་རིག་དང་མཐུན་པའི་ནང་དོན་དང་དུས་རབས་ཀྱི་རིན་ཐང་ཕུན་སུམ་ཚོགས་པོ་ལྡན་ཡོད།
རླབས་ཆེན་ལག་ལེན་གྱིས་རླབས་ཆེན་གྱི་སྙིང་སྟོབས་ལ་ཞུན་བརྡར་བྱེད། བོད་ཀྱི་ལོ་རྒྱུས་རང་བཞིན་གྱི་འཕོ་འགྱུར་ཆེན་པོའི་བརྗིད་ཆེའི་འཕེལ་རིམ་ཁྲོད། མི་རབས་ནས་མི་རབས་བར་གྱི་གུང་ཁྲན་ཏང་མིས་སྔ་འཛིན་ཕྱི་གཏོད་ཀྱིས“བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས”ཀྱི་གསོན་ཤུགས་དང་འབོད་སྐུལ་ནུས་པ་གསོན་ཉམས་དོད་པོས་མངོན་པར་བྱས་ཡོད། ཁོང་ཚོས་ཕུགས་བསམ་གྱི་འོད་ཟེར་ལ་བརྟེན་ནས་འབད་འཐབ་ཀྱི་ལམ་བུ་གསལ་བར་བྱས་པ་དང་། དད་མོས་ཀྱི་སྟོབས་ཤུགས་ལ་བརྟེན་ནས་མཛེས་སྡུག་ལྡན་པའི་འབྱུང་འགྱུར་བཏོད་ཡོད། ཏང་གི་ཚོགས་ཆེན་བཅོ་བརྒྱད་པ་འཚོགས་ཚུན། བློ་མཐུན་ཞི་ཅིན་ཕིང་གིས་སྲོག་ཤིང་གནང་བའི་ཏང་ཀྲུང་དབྱང་གི་མཁས་མཛངས་ལྡན་པའི་འགོ་ཁྲིད་དང་བྱམས་བརྩེའི་ཐུགས་ཁུར་འོག་བོད་ཀྱི་གུང་ཁྲན་ཏང་མི་ཚོས་མི་རིགས་ཁག་གི་ལས་བྱེད་པ་དང་མང་ཚོགས་ཀྱི་སྣེ་ཁྲིད་དེ་ཆབ་སྲིད་ཀྱི་འགན་འཁྲིའི་འཁུར་སེམས་དང་ལོ་རྒྱུས་ཀྱི་འོས་འགན་སྒྲུབ་སེམས་ཆེན་པོའི་ཐོག་ནས“བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས”སྲུང་སྐྱོང་མཁན་དང་། རྒྱུད་འཛིན་མཁན། ལག་ལེན་མཁན་བཅས་ཡག་པོ་ཞིག་བྱས་ནས། བརྟན་ལྷིང་དང་། འཕེལ་རྒྱས། སྣོད་བཅུད། མཐའ་མཚམས་སྟོབས་ཆེར་གཏོང་བ་བཅས་ཀྱི“དོན་ཆེན་བཞི”ཡི་འབད་འབུངས་ཡར་བརྩོན་ཁྲོད་རྔུལ་ཆུ་འདོན་པ་དང་སྒེར་སེམས་མེད་པར་ལེགས་སྐྱེས་ཕུལ་བ་དང་། དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་རྒྱལ་ཁ་ཕུ་ཐག་གཅོད་པ་དང་ཕྱོགས་ཡོངས་ནས་འབྱོར་འབྲིང་སྤྱི་ཚོགས་སྐྲུན་པའི་ཚ་ཚ་འུར་འུར་གྱི་ལག་ལེན་ཁྲོད་ཐོག་མའི་བསམ་འདུན་དང་འགན་འཁྲི་བསྒྲུབས་པ། བླ་ན་མེད་པའི་རྒྱལ་ཁབ་ཀྱི་ས་ཆ་སྲུང་མཁན་དང་བདེ་སྐྱིད་ལྡན་པའི་ཁྱིམ་གཞིས་སྐྲུན་མཁན་བྱེད་པའི་འབད་འཐབ་ལྷོད་མེད་ཁྲོད་མེས་རྒྱལ་དང་མི་དམངས་ལ་ལྷག་བསམ་བཅངས་བ། མེས་རྒྱལ་གཅིག་གྱུར་སྲུང་སྐྱོང་དང་མི་རིགས་མཐུན་སྒྲིལ་ལ་ཤུགས་སྣོན་རྒྱག་པའི་བློ་བརྟན་འགྱུར་མེད་ཀྱི་བྱ་སྤྱོད་ཁྲོད་རྒྱལ་གཅེས་རིང་ལུགས་ཀྱི་བརྩེ་སེམས་ཟབ་མོ་མངོན་པ་བཅས་ཀྱི་ཐད་ནས་བོད་ཀྱི་བྱ་གཞག་ཁག་ཕྱོགས་ཡོངས་ནས་ཡར་རྒྱས་དང་ལོ་རྒྱུས་རང་བཞིན་གྱི་གྲུབ་འབྲས་ཐོབ་རྒྱུར་སྐུལ་འདེད་བཏང་ཡོད། འདི་ནི་དུས་རབས་གསར་པར“བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས”གསོན་ཉམས་དོད་པོས་གསལ་འགྲེལ་དང་དེའི་རིན་ཐང་མངོན་པ་ཞིག་ཡིན།
རླབས་ཆེན་བྱ་གཞག་གིས་རླབས་ཆེན་གྱི་སྙིང་སྟོབས་ལ་རེ་འབོད་བྱེད།“བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས”ནི་བོད་ཀྱི་མི་རིགས་ཁག་གི་མི་དམངས་ཀྱི་བོད་སྐྱོང་དང་། བོད་བརྟན། བོད་དར་བཅས་ཀྱི་སྟོབས་ཤུགས་ཀྱི་འབྱུང་རྩ་དང་། བོད་ཀྱི་མི་རིགས་ཁག་གི་ལས་བྱེད་པ་དང་མང་ཚོགས་ཚོས་དཀའ་ངལ་ལས་རྣམ་པར་རྒྱལ་བ་དང་གསར་འབྱེད་ཡར་བརྩོན་ལ་སྐུལ་བའི་སྙིང་སྟོབས་ཀྱི་སྟོབས་ཤུགས་ཤིག་རེད། ང་ཚོས་ངེས་པར་དུ“བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས”དར་སྤེལ་ཤུགས་ཆེན་བཏང་ནས་གསོ་རླུང་དཀོན་ཡང་སྙིང་སྟོབས་མི་ཉམས་པ་དང་། བློ་བརྟན་འགྱུར་མེད་དང་། དཔའ་ངར་ཆེན་པོས་མདུན་སྐྱོད་བཅས་བྱས་ཏེ། བློ་མཐུན་ཞི་ཅིན་ཕིང་གིས་སྲོག་ཤིང་གནང་བའི་ཏང་ཀྲུང་དབྱང་གིས་ཇུས་བཀོད་གནང་བའི་འཆར་གཞི་བཟང་པོ་ཟམ་མི་ཆད་པར་དོན་དངོས་ལ་འགྱུར་བར་བྱས་ནས། དུས་རབས་གསར་པའི་བོད་ཀྱི་ལས་དོན་གོམ་ཞེང་ཆེན་པོས་མདུན་ཕྱོགས་སུ་བསྐྱོད་རྒྱུར་སྐུལ་འདེད་གཏོང་དགོས། ངེས་པར་དུ“བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས”དར་སྤེལ་ཤུགས་ཆེན་བཏང་ནས་དཀའ་ཚེགས་ཆེ་ཡང་དཀའ་སྡུག་ལ་མི་སྐྲག་པ་དང་། དཀའ་ངལ་མཐའ་དག་སེལ་བ། མཁྲེགས་སར་གཏུག་ཕོད་པ་བཅས་ཀྱི་བློ་སྟོབས་དང་ངར་ཤུགས་ལ་བརྟེན་ནས། དཀའ་ངལ་མཐའ་དག་སེལ་བ་དང་ཉེན་ཁ་མཐའ་དག་ལས་རྣམ་པར་རྒྱལ་ཏེ། རྒྱལ་ཁ་ཡང་ཡང་ལེན་དགོས། ངེས་པར་དུ“བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས”དར་སྤེལ་ཤུགས་ཆེན་བཏང་ནས་ས་བབ་མཐོ་ན་གོ་རྟོགས་དེ་བས་མཐོ་བ་བྱས་ནས། ཐོག་མཐའ་བར་གསུམ་དུ་མེས་རྒྱལ་གྱི་ཁེ་ཕན་དེ་ཐམས་ཅད་ལས་ལྷག་པར་བརྩི་རྒྱུ་མཐའ་འཁྱོངས་བྱེད་པ་དང་། ཐོག་མཐའ་བར་གསུམ་དུ་མི་དམངས་ཀྱི་ཁེ་ཕན་དེ་ཐམས་ཅད་ལས་ལྷག་པར་བརྩི་རྒྱུ་མཐའ་འཁྱོངས། མཐུན་སྒྲིལ་བྱེད་ཐུབ་པའི་སྟོབས་ཤུགས་མཐའ་དག་ལ་མཐུན་སྒྲིལ་བཅས་བྱས་ནས། མི་རིགས་ཁག་གི་མི་དམངས་སེ་འབྲུའི་རྡོག་ལྟར་གོང་བུ་གཅིག་ཏུ་སྒྲིལ་ཏེ། ཀྲུང་ཧྭ་མི་རིགས་རླབས་ཆེན་བསྐྱར་དར་གྱི་བོད་ཀྱི་ལེའུ་རྩོམ་ཆེད་སྟོབས་ཤུགས་གཅིག་སྒྲིལ་བྱེད་དགོས།
གསར་བརྗེའི་སྲོལ་རྒྱུན་རྒྱུན་འཛིན་བྱས་ཏེ“བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས”ལ་ནམ་ཡང་འོད་ཟེར་འཕྲོ་བར་བྱེད། “ལོ་བརྒྱ་གཉིས”ཀྱི་འབད་འཐབ་དམིགས་ཚད་ཀྱི་ལོ་རྒྱུས་འབྲེལ་མཚམས་སུ། ང་ཚོས་བློ་མཐུན་ཞི་ཅིན་ཕིང་གིས་སྲོག་ཤིང་གནང་བའི་ཏང་ཀྲུང་དབྱང་གི་མཐའ་འཁོར་དུ་དེ་བས་མཐུན་སྒྲིལ་དམ་ཟབ་བྱེད་པ་དང་། ཀྲུང་གོའི་ཁྱད་ཆོས་ལྡན་པའི་སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་རླབས་ཆེན་དར་ཆ་མཐོན་པོར་སྒྲེང་བ། དཔའ་བོའི་བློ་སྟོབས་ཀྱི་གསར་བརྗེའི་སྙིང་སྟོབས་དང་གསར་བརྗེའི་ཕུགས་བསམ་མཐའ་དག་ལས་ཡངས་བའི་བླ་ན་མེད་པའི་བསམ་པའི་ཉམས་འགྱུར་གྱི་སྒོ་ནས་དུས་རབས་གསར་པའི་བོད་ལྗོངས་ཡུན་རིང་བདེ་འཁོད་དང་སྤུས་ཚད་ལེགས་པའི་འཕེལ་རྒྱས་ལ་སྐུལ་འདེད་གཏོང་བའི་རྣམ་པ་གསར་པ་ཟམ་མི་ཆད་པར་བཏོད་དེ། མཐུན་སྒྲིལ་དང་། འབྱོར་ཕྱུག ཤེས་དཔལ། ཞི་མཐུན། མཛེས་སྡུག་བཅས་ལྡན་པའི་སྤྱི་ཚོགས་རིང་ལུགས་དེང་རབས་ཅན་གྱི་བོད་གསར་འཛུགས་སྐྲུན་བྱེད་པའི་རྒྱལ་ཁ་གསར་པ་ཟམ་མི་ཆད་པར་ལེན་དགོས།
རྩོམ་སྒྲིག་པ། བདེ་སྒྲོལ།
