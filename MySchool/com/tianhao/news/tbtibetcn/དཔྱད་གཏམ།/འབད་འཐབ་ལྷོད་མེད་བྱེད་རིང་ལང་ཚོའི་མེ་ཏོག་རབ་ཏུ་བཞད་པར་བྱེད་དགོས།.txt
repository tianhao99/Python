འབད་འཐབ་ལྷོད་མེད་བྱེད་རིང་ལང་ཚོའི་མེ་ཏོག་རབ་ཏུ་བཞད་པར་བྱེད་དགོས།

གཞོན་རབས་ཚོར་ཕུགས་བསམ་དང་། འཇོན་ཐང་། འགན་འཁྲི་བཅས་ཡོད་ན། རྒྱལ་ཁབ་ལ་མདུན་ལམ་དང་མི་རིགས་ལ་རེ་བ་ཡོད། དེ་རིང་ནི་ལྔ་བཞི་གཞོན་ནུའི་དུས་ཆེན་རེད་ལ་ལྔ་བཞིའི་ལས་འགུལ་སྤེལ་ནས་ལོ102འཁོར་བའི་དྲན་གསོའི་ཉིན་མོ་ཡང་རེད། ལང་ཚོའི་དཔལ་འབར་བའི་ཉིན་མོ་འདིར་ང་ཚོས་རྒྱ་ཆེའི་གཞོན་ནུ་རྣམས་ལ་དུས་ཆེན་གྱི་སྨོན་འདུན་ཞུ་རྒྱུ་ཡིན་ལ། རྒྱལ་གཅེས་གཞོན་ནུའི་འོད་སྟོང་འཕྲོ་བའི་མཛད་རྗེས་ལ་རྗེས་དྲན་ཟབ་མོ་ཞུ་རྒྱུ་དང་། ལྔ་བཞིའི་སྔོན་བྱོན་པའི་ལོ་རྒྱུས་ཀྱི་མཛད་རྗེས་ལ་ཡང་རྗེས་དྲན་ཟབ་མོ་ཞུ་རྒྱུ་ཡིན།
ལྔ་བཞི་ནི་དེང་རབས་ཀྲུང་གོའི་ལང་ཚོའི་དྲན་ཤེས་རེད། ལྔ་བཞིའི་ལས་འགུལ་གྱི་ཉམས་སད་ཐེག་པ་བརྒྱུད། ཀྲུང་གོའི་སྔོན་ཐོན་པ་དེ་བས་མང་པོ་མར་ཁེ་སི་རིང་ལུགས་ཀྱི་དར་ཆའི་འོག་འཛོམས་པ་དང་། 1921ལོར་ཀྲུང་གོ་གུང་ཁྲན་ཏང་དངོས་སུ་དབུ་བརྙེས་པ་ཁྱབ་བསྒྲགས་བྱས་པ་ནས་བཟུང་། ཀྲུང་གོའི་ལོ་རྒྱུས་ཀྱི་ཤོག་ལྷེ་གསར་རྐྱང་ཞིག་ཕྱེ་ཡོད། ལྔ་བཞིའི་ལས་འགུལ་སྤེལ་བ་ནས་བཟུང་། ཀྲུང་གོ་གུང་ཁྲན་ཏང་གི་འགོ་ཁྲིད་འོག ཀྲུང་གོས་དཔའ་ངར་ཆེར་བསྐྱེད་ཀྱིས་འཛམ་གླིང་སྐྱེ་བོ་ཀུན་གྱིས་དོ་སྣང་བྱེད་པའི་རླབས་ཆེན་གྲུབ་འབྲས་བཏོད་ཅིང་། ང་ལང་ཚོའི་སྟོབས་ཤུགས་ལ་བརྟེན་ནས་ལང་ཚོའི་ཀྲུང་གོ་བཏོད་ཡོད། ཀྲུང་གོ་གུང་ཁྲན་ཏང་གི་འགོ་ཁྲིད་འོག་གཞོན་ནུ་མི་རབས་ནས་མི་རབས་ཀྱིས་འབོད་སྐུལ་དང་ལེན་ཧུར་ཐག་ཞུས་ནས་གངས་ལྗོངས་ཀྱི་མཐའ་མཚམས་སུ་གཞི་ཚུགས་ཏེ་བོད་ལ་བཅིངས་འགྲོལ་དང་། བོད་ལྗོངས་འཛུགས་སྐྲུན། བོད་ལྗོངས་འཕེལ་རྒྱས་བཅས་ཀྱི་རླབས་ཆེན་བྱ་གཞག་ལ་དཔའ་ངར་ཆེན་པོས་ཞུགས་པ་དང་། ལང་ཚོ་དང་། ཁྲག་དྲོན། དེ་བཞིན་སྤྲོ་སེམས་དང་ལེགས་སྐྱེས་འབུལ་བར་བརྟེན་ཏེ་ཡར་ངོའི་ཟླ་ལྟར་འཕེལ་བའི་འབད་འཐབ་ཀྱི་ལེའུ་བརྩམས་ནས། འཛམ་གླིང་ཡང་རྩེར་འཛམ་གླིང་སྐྱེ་བོ་ཀུན་གྱིས་དོ་སྣང་བྱེད་པའི“ཀྲུང་གོའི་ངོ་མཚར”མང་པོ་བཏོད་ཡོད།
འབད་འཐབ་ལྷོད་མེད་ཀྱི་ལང་ཚོ་ནི་ཧ་ཅང་རྣམ་པར་བཀྲ་བ་ཞིག་ཡིན་པས། ང་ཚོས་ལང་ཚོའི་སྟོབས་ཤུགས་ལ་གུས་འདུད་ཞུ་དགོས། འབྱོར་འབྲིང་སྤྱི་ཚོགས་ཕྱོགས་ཡོངས་ནས་སྐྲུན་པའི་རྒྱང་སྐྱོད་ལམ་དུ་འབད་འབུངས་ཡར་བརྩོན་དང་འབད་འཐབ་ལྷོད་མེད་བྱས་ཏེ། ལང་ཚོའི་འགན་འཁྲི་འཁུར་སེམས་མངོན་ཡོད། གྲོང་གསེབ་དར་རྒྱས་ཀྱི་དམག་འཐབ་ཁྲོད་དུ་སེམས་ནུས་ཡོད་རྒུ་བཏོན་ནས་རང་ཉིད་ཀྱི་ཡོད་ཚད་ལེགས་སྐྱེས་ཕུལ་ཏེ་དུས་རབས་གསར་པའི་ལང་ཚོའི་གླུ་དབྱངས་རྩོམ་བཞིན་ཡོད། ཁ་ཕྲལ་ལ་ངོ་རྒོལ་བྱེད་པའི་འཐབ་རྩོད་ཀྱི་འཐབ་ས་དང་པོར་འགན་འཁྲི་དང་དུ་བླངས་ཤིང་ཁྲག་དྲོན་གཏོར་ཏེ་ལང་ཚོའི་སྟོབས་ཤུགས་ལ་བརྟེན་ནས། གཅིག་གྱུར་སྲུང་སྐྱོང་དང་མཐུན་སྒྲིལ་ལ་ཤུགས་སྣོན་རྒྱག་པའི་ཟངས་གྱང་ལྕགས་རྩིག་སྐྲུན་བཞིན་ཡོད། གཞི་རིམ་གྱི་ལས་གནས་སུ་མིག་རྐྱང་རིང་པོ་འཆང་བ་དང་ཏན་ཏན་ཏིག་ཏིག་ངང་མི་སྒེར་གྱི་ཕུགས་བསམ་དོན་གཉེར་བྱེད་པ་ཏང་དང་རྒྱལ་ཁབ་ཀྱི་བྱ་གཞག་ཁྲོད་མཉམ་འདྲེས་ཐུབ་པར་བྱས་ཡོད། ལྗོངས་ཡོངས་ཀྱི་མི་རིགས་ཁག་གི་རྒྱ་ཆེའི་གཞོན་ནུ་ཚོས་ལང་ཚོར་གནོང་འགྱོད་མེད་པར་ལྷག་བསམ་ཟོལ་མེད་ཀྱིས་ལས་རིགས་ཁག་ཏུ་བྱ་སྤྱོད་དངོས་ལ་བརྟེན་ནས་ལང་ཚོའི་རང་མདངས་མངོན་ཡོད་ལ། དཀའ་སྤྱད་འབད་འཐབ་ལ་བརྟེན་ནས་དུས་རབས་ཀྱི་ཉམས་འགྱུར་མངོན་དང་མངོན་མུས་རེད།
འབད་འཐབ་ལྷོད་མེད་ཀྱི་ལང་ཚོ་ནི་ཧ་ཅང་རྣམ་པར་བཀྲ་བ་ཞིག་ཡིན་པས། ང་ཚོས་ལང་ཚོའི་ཕུགས་འདུན་མངོན་འགྱུར་ལ་གཞོལ་དགོས།
དུས་རབས་གསར་པའི་ཀྲུང་གོའི་གཞོན་ནུ་རྣམས་ཀྲུང་ཧྭ་མི་རིགས་འཕེལ་རྒྱས་ཀྱི་དུས་སྐབས་བཟང་ཤོས་སུ་གནས་པ་དང་། ཀྲུང་ཧྭ་མི་རིགས་རླབས་ཆེན་བསྐྱར་དར་མངོན་འགྱུར་བྱེད་པའི་འགག་རྩའི་དུས་སྐབས་སུ་གནས་ཡོད་པས། འཐོབ་དཀའ་བའི་བྱས་རྗེས་འཇོག་པའི་མི་ཚེའི་གོ་སྐབས་ཕྲད་ཡོད་ལ། “འགན་འཁྲི་ཆེན་པོ་མི་ལ་སྤྲད་པའི”དུས་རབས་ཀྱི་ལས་འགན་ཡང་ཕྲད་ཡོད། ཞི་བས་བཅིངས་འགྲོལ་བཏང་ནས་ལོ་ངོ་70འཁོར་བའི་འཕེལ་རྒྱས་ཀྱི་འཕོ་འགྱུར་ཆེན་པོ་བརྒྱུད། བོད་ཀྱིས་སྤྱི་ཚོགས་རིང་ལུགས་དེང་རབས་ཅན་གྱི་བོད་ལྗོངས་གསར་པ་འཛུགས་སྐྲུན་བྱེད་པའི་རྒྱང་སྐྱོད་ལམ་གསར་པ་ཕྱོགས་ཡོངས་ནས་བཏོད་པ་དང་། དུས་རབས་ཀྱི་འགན་འཁྲི་གཞོན་ནུར་སྤྲད་པ་དང་དུས་རབས་ཀྱི་གཟི་བརྗིད་གཞོན་ནུར་དབང་ཡོད། ལྗོངས་ཡོངས་ཀྱི་མི་རིགས་ཁག་གི་རྒྱ་ཆེའི་གཞོན་ནུ་ཚོས་སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིན་ཕིང་གི་མངགས་བཅོལ་ཟབ་མོ་བློ་ལ་བརྟན་པོར་བཅངས་ནས། ལོ་རྒྱུས་ཀྱི་འགན་འཁྲི་ཕྲག་ཏུ་འཁུར་བ་དང་མདུན་སྐྱོད་ཀྱི་ཡིད་ཆེས་བརྟན་དུ་བཏང་སྟེ། ཀུན་སྤྱོད་ཆེན་པོ་འཛུགས་སྐྱོང་དང་། སྤྱོད་བཟང་རྟོགས་པ། མཁས་པར་འགྱུར་བ། འགན་འཁྲི་ཆེན་པོ་འཁུར་བ་བཅས་བྱས་ཏེ་མི་རིགས་བསྐྱར་དར་གྱི་ལས་འགན་ལྕི་པོ་འཁུར་ཕོད་པའི་དུས་རབས་ཀྱི་མི་གསར་ཞིག་ཏུ་ཧུར་ཐག་གྱུར་པར་བྱས་ནས། ཀྲུང་ཧྭ་མི་རིགས་རླབས་ཆེན་བསྐྱར་དར་གྱི་ཀྲུང་གོའི་ཕུགས་འདུན་དང་བོད་ཀྱི་ལེའུ་རྩོམ་པའི་གསོན་ཉམས་ལྡན་པའི་ལག་ལེན་ཁྲོད་དུ་ལང་ཚོའི་ཕུགས་འདུན་མངོན་འགྱུར་བྱས་ཏེ་མེས་རྒྱལ་དང་། མི་རིགས། མི་དམངས། མིའི་རིགས་བཅས་ལ་འབད་འཐབ་བྱེད་རིང་ལང་ཚོའི་རྣམ་པར་བཀྲ་བའི་མེ་ཏོག་རབ་ཏུ་བཞད་པར་བྱས་ཡོད།
འབད་འཐབ་ལྷོད་མེད་ཀྱི་ལང་ཚོ་ནི་ཧ་ཅང་རྣམ་པར་བཀྲ་བ་ཞིག་ཡིན་པས། ང་ཚོས་སླར་ཡང་ལང་ཚོའི་འོད་སྟོང་གཏོད་དགོས། ལྗོངས་ཡོངས་ཀྱི་རྒྱ་ཆེའི་མི་རིགས་ཁག་གི་གཞོན་ནུ་ཚོས་ཏང་གི་ལོ་རྒྱུས་སློབ་སྦྱོང་བྱེད་རིང་དད་སེམས་འཕེལ་བ་དང་། བློ་སྒོ་འབྱེད་པ། སྟོབས་ཤུགས་ལེན་པ་བཅས་བྱས་ཏེ“གདེང་ཚོད་བཞི”རྒྱུན་ཆད་མེད་པར་བརྟན་དུ་བཏང་ནས་མེས་རྒྱལ་དང་མི་དམངས་ཀྱི་ཆེད་དུ་ནམ་ཡང་འབད་འཐབ་དང་ལྷག་བསམ་ཟོལ་མེད་ངང་ལེགས་སྐྱེས་འབུལ་བའི་ཕུགས་བསམ་བརྟན་པོ་འཛུགས་དགོས། བློ་བརྟན་འགྱུར་མེད་ཀྱི་སྒོ་ནས་མེས་རྒྱལ་གཅིག་གྱུར་སྲུང་སྐྱོང་དང་། དར་ཆ་གསལ་སྟོན་གྱི་སྒོ་ནས་ཁ་ཕྲལ་ལ་ངོ་རྒོལ། བདེ་ཐང་དང་ཤེས་དཔལ་ལྡན་པའི་འཚོ་བ་སྐྱེལ་སྟངས་དོན་གཉེར། ཏཱ་ལ་སྐྱེ་ཕྲེང་བཅུ་བཞི་པ་དང་ཏཱ་ལའི་རུ་ཚོགས་ལ་མཐའ་གཅིག་ཏུ་འཐབ་རྩོད་བྱེད་པ། བྱ་སྤྱོད་དངོས་ལ་བརྟེན་ནས་རྒྱལ་གཅེས་ཀྱི་སྙིང་སྟོབས་རྒྱུན་འཛིན་དང་གཞོན་ནུར་མིག་དཔེ་འདོན་དགོས། བློ་བརྟན་འགྱུར་མེད་ཀྱི་སྒོ་ནས་སྤུས་ཚད་མཐོ་བའི་འཕེལ་རྒྱས་ལ་སྐུལ་འདེད་བཏང་སྟེ་གཞི་རིམ་ལ་ལེགས་སྐྱེས་འབུལ་བ་དང་། ཕ་ཡུལ་འཛུགས་སྐྲུན། མང་ཚོགས་ལ་ཞབས་འདེགས་ཞུ་བ་བཅས་བྱེད་རིང་ང་ཚོའི་ཉམས་འགྱུར་མངོན་དགོས། བློ་བརྟན་འགྱུར་མེད་ཀྱི་སྒོ་ནས་སྐྱེ་ཁམས་ཤེས་དཔལ་འཛུགས་སྐྱོང་ལ་སྐུལ་འདེད་བཏང་སྟེ་ས་མཐོའི་སྲོག་ཆགས་དང་། རྩི་ཤིང་། རི་ཆུ་ཀླུང་གསུམ་བཅས་སྲུང་སྐྱོང་ཡག་པོ་བྱས་ཏེ་མཛེས་སྡུག་ལྡན་པའི་བོད་ལྗོངས་འཛུགས་སྐྲུན་བྱེད་རིང་མཛེས་སྡུག་ལྡན་པའི་ལང་ཚོ་གཏོད་དགོས། བློ་བརྟན་འགྱུར་མེད་ཀྱི་སྒོ་ནས་མཐའ་མཚམས་སྟོབས་ཆེར་གཏོང་བ་དང་བརྟན་པོར་བསྲུངས་ནས། བླ་ན་མེད་པའི་རྒྱལ་ཁབ་ཀྱི་ས་ཆ་སྲུང་མཁན་དང་བདེ་སྐྱིད་ལྡན་པའི་ཁྱིམ་གཞིས་སྐྲུན་མཁན་ཞིག་གང་ཐུབ་བྱས་ཏེ། མེས་རྒྱལ་ལ་བཅངས་པའི་བརྩེ་བ་སྤྲོ་སེམས་རབ་ཏུ་འཕེལ་བའི་ལང་ཚོའི་ལོ་ཟླ་དང་མཉམ་འདྲེས་ཐུབ་པ་དང་། ཚེ་སྲོག་ལ་བརྟེན་ནས“བརྩེ་བ་ཟབ་མོ་ཀྲུང་གོ་ཁོ་ནར་འཆང་གི་ཡིན”ཞེས་བརྗོད་དགོས།
ང་ཚོས“ལྔ་བཞིའི་སྙིང་སྟོབས”དང་། “བོད་སྐྱོད་པ་རྒན་གྲས་ཀྱི་སྙིང་སྟོབས” “ལམ་གཉིས་ཀྱི་སྙིང་སྟོབས” ཁུང་ཧྥན་སེན་གྱི་སྙིང་སྟོབས་བཅས་དར་སྤེལ་ཤུགས་ཆེན་བཏང་ནས་འབད་འཐབ་ལྷོད་མེད་བྱེད་རིང་ལང་ཚོའི་མེ་ཏོག་རབ་ཏུ་བཞད་པར་བྱས་ཏེ། མཐུན་སྒྲིལ་དང་། འབྱོར་ཕྱུག ཤེས་དཔལ། ཞི་མཐུན། མཛེས་སྡུག་བཅས་ལྡན་པའི་སྤྱི་ཚོགས་རིང་ལུགས་དེང་རབས་ཅན་གྱི་བོད་ལྗོངས་གསར་པ་འཛུགས་སྐྲུན་བྱ་ཆེད་ལང་ཚོ་སྐྱེས་སུ་འབུལ་བ་དང་སྟོབས་ཤུགས་ལེགས་སྐྱེས་འབུལ་དགོས།
རྩོམ་སྒྲིག་པ། བདེ་སྒྲོལ།
