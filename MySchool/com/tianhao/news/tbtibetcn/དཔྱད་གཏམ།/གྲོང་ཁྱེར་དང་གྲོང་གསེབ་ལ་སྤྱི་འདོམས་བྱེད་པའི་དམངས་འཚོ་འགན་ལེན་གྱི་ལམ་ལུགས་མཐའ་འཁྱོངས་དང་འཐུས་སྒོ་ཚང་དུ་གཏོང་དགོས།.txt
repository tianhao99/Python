གྲོང་ཁྱེར་དང་གྲོང་གསེབ་ལ་སྤྱི་འདོམས་བྱེད་པའི་དམངས་འཚོ་འགན་ལེན་གྱི་ལམ་ལུགས་མཐའ་འཁྱོངས་དང་འཐུས་སྒོ་ཚང་དུ་གཏོང་དགོས།

ཀྲུང་གོ་གུང་ཁྲན་ཏང་གིས་ཐོག་མཐའ་བར་གསུམ་དུ་དམངས་འཚོ་འགན་ལེན་དང་ལེགས་གཏོང་ལ་མཐོང་ཧ་ཅང་ཆེན་པོ་གནང་གི་ཡོད། ལྷག་དོན་དུ་ཏང་གི་ཚོགས་ཆེན་བཅོ་བརྒྱད་པ་འཚོགས་ཚུན་དུ། བློ་མཐུན་ཞི་ཅིན་ཕིང་གིས་སྲོག་ཤིང་གནང་བའི་ཏང་ཀྲུང་དབྱང་གིས་ཐོག་མཐའ་བར་གསུམ་དུ་མི་དམངས་ལྟེ་བར་འཛིན་པའི་འཕེལ་རྒྱས་ཀྱི་བསམ་བློ་མཐའ་འཁྱོངས་བྱས་ནས་མཛེས་སྡུག་ལྡན་པའི་འཚོ་བར་བཅངས་པའི་མི་དམངས་མང་ཚོགས་ཀྱི་རེ་འདོད་ང་ཚོའི་རྩ་དོན་དང་དམིགས་ཚད་ལ་བརྩིས་ནས་དམངས་ཕན་གྱི་བྱ་ཐབས་འབོར་ཆེན་ལག་བསྟར་དོན་འཁྱོལ་བྱས་པ་བརྒྱུད། དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་ལས་དོན་ལ་ཐག་གཅོད་རང་བཞིན་གྱི་འཕེལ་རྒྱས་བྱུང་བ་དང་། སློབ་གསོའི་བྱ་གཞག་ཕྱོགས་ཡོངས་ནས་འཕེལ་རྒྱས་ཕྱིན་པ། ལས་ཞུགས་ཀྱི་གནས་ཚུལ་མུ་མཐུད་ལེགས་སུ་ཕྱིན་པ། གྲོང་ཁྱེར་དང་གྲོང་གསེབ་ཀྱི་གྲོང་མིའི་ཡོང་འབབ་འཕར་ཚད་དཔལ་འབྱོར་འཕར་ཚད་ལས་མཐོ་བ། ཡོང་འབབ་འབྲིང་གྲས་ཀྱི་ཚོགས་སྤྱི་རྒྱུན་མཐུད་མང་དུ་ཕྱིན་པ། གྲོང་ཁྱེར་དང་གྲོང་གསེབ་ཏུ་ཁྱབ་པའི་ཡུལ་མིའི་སྤྱི་ཚོགས་འགན་སྲུང་མ་ལག་གཞི་རྩའི་ཆ་ནས་བཙུགས་ཟིན་པ། མི་དམངས་ཀྱི་གཟུགས་གཞི་བདེ་ཐང་དང་སྨན་བཅོས་འཕྲོད་བསྟེན་གྱི་ཆུ་ཚད་ཧ་ཅང་མཐོ་རུ་ཕྱིན་པ་བཅས་བྱུང་ནས་མི་རིགས་ཁག་གི་མང་ཚོགས་ཀྱི་ཐོབ་ཚོར་དང་། བདེ་སྐྱིད་ཚོར་སེམས། བདེ་འཇགས་ཚོར་སེམས་བཅས་ཆེ་རུ་ཕྱིན་ཡོད།
ཏང་གི་སྐབས་བཅུ་དགུ་པའི་ཀྲུང་ཨུ་ཚང་འཛོམས་གྲོས་ཚོགས་ཐེངས་བཞི་པའི་དགོངས་དོན་སློབ་སྦྱོང་ལག་བསྟར་བྱེད་པར། ངེས་པར་དུ་གྲོང་ཁྱེར་དང་གྲོང་གསེབ་ལ་སྤྱི་འདོམས་བྱེད་པའི་དམངས་འཚོ་འགན་ལེན་གྱི་ལམ་ལུགས་མཐའ་འཁྱོངས་དང་འཐུས་སྒོ་ཚང་དུ་གཏོང་རྒྱུའི་སྐོར་གྱི་བླང་བྱ་གསར་པ་དང་། བསམ་ཕྱོགས་གསར་པ། བྱེད་ཐབས་གསར་པ་བཅས་ལ་གོ་བ་གཏིང་ཟབ་ལེན་དགོས།
ཉེ་བའི་ལོ་ཤས་རིང་། རང་སྐྱོང་ལྗོངས་ཏང་ཨུད་ཀྱིས་དམངས་འཚོ་འགན་ལེན་དང་ལེགས་སུ་གཏོང་རྒྱུའི་སྐོར་གྱི་སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིན་ཕིང་གི་མཛུབ་སྟོན་གལ་ཆེན་གྱི་དགོངས་དོན་ལག་བསྟར་དོན་འཁྱོལ་གཏིང་ཟབ་བྱས་ནས་ཀུན་ཕན་རང་བཞིན་དང་། རྨང་གཞིའི་རང་བཞིན། ཐ་སྐྱོང་རང་བཞིན་བཅས་ཀྱི་དམངས་འཚོ་འཛུགས་སྐྲུན་ལ་མཐོང་ཆེན་བྱས་ཏེ་མང་ཚོགས་ཀྱི་གཞི་རྩའི་འཚོ་བ་འགན་ལེན་དང་། ཐབས་བརྒྱ་ཇུས་སྟོང་གིས་དམངས་འཚོའི་བདེ་རྩ་ཆེ་རུ་གཏོང་བ། དམངས་ཕན་བྱ་ཐབས་གལ་ཆེན་རབ་དང་རིམ་པ་འདོན་པ་བཅས་བྱས་པ་བརྒྱུད། ལས་ཞུགས་ཀྱི་གནས་ཚུལ་ལེགས་སུ་ཕྱིན་པ་དང་། བདེ་ཐང་བོད་ལྗོངས་འཛུགས་སྐྲུན་ལ་གཡུར་ཟའི་གྲུབ་འབྲས་ཐོབ་པ། སྤྱི་ཚོགས་འགན་སྲུང་ནུས་པ་མུ་མཐུད་ཆེ་རུ་ཕྱིན་པ། སྤྱི་ཚོགས་ཀྱི་བྱ་གཞག་ཁག་མགྱོགས་མྱུར་འཕེལ་རྒྱས་ཕྱིན་པ་བཅས་བྱུང་ཡོད།
ལྗོངས་ཡོངས་ཀྱི་རིམ་པ་ཁག་དང་སྡེ་ཚན་ཁག་གིས་གྲོང་ཁྱེར་དང་གྲོང་གསེབ་ལ་སྤྱི་འདོམས་བྱེད་པའི་དམངས་འཚོ་འགན་ལེན་ལམ་ལུགས་མཐའ་འཁྱོངས་དང་འཐུས་སྒོ་ཚང་དུ་གཏོང་བ་དང་། མི་དམངས་ཀྱི་ཉིན་བཞིན་ཆེ་རུ་འགྲོ་བའི་མཛེས་སྡུག་ལྡན་པའི་འཚོ་བའི་དགོས་མཁོ་སྐོང་བའི་སྐོར་གྱི་ཏང་གི་སྐབས་བཅུ་དགུ་པའི་ཀྲུང་ཨུ་ཚང་འཛོམས་གྲོས་ཚོགས་ཐེངས་བཞི་པའི་བཀོད་སྒྲིག་གལ་ཆེན་རབ་དང་རིམ་པ་དང་གཙོ་གནད་ལས་འགན་སྟངས་འཛིན་གནད་ལ་འཁེལ་བ་བྱས་ནས་རང་ལྗོངས་ཀྱི་དོན་དངོས་དང་ཟུང་འབྲེལ་གྱིས་ལག་བསྟར་དོན་འཁྱོལ་ཡོང་བར་དམ་འཛིན་ནན་པོ་བྱེད་དགོས། ལས་ཞུགས་ནི་དམངས་འཚོའི་རྩ་བ་ཡིན་པ་མཐའ་འཁྱོངས་དང་། ལས་ཞུགས་སྔོན་ལ་འཇོག་པའི་སྲིད་ཇུས་ལག་ལེན་བསྟར་བ། སྤྱི་པའི་ལས་ཞུགས་ཞབས་ཞུ་དང་ཚེ་གང་ལས་རིགས་རྩལ་ནུས་གསོ་སྦྱོང་གི་ལམ་ལུགས་འཐུས་སྒོ་ཚང་དུ་གཏོང་བ་བཅས་བྱེད་དགོས། ལས་གཏོད་ལ་བརྟེན་ནས་ལས་ཞུགས་ལ་སྐུལ་འདེད་གཏོང་བ་དང་ཐབས་ལམ་མང་བའི་སྟབས་བསྟུན་གྱི་ལས་ཞུགས་ལམ་སྲོལ་འཛུགས་པ་དང་། འདྲ་མཉམ་ལས་ཞུགས་ཀྱི་ཁོར་ཡུག་གཏོད་པ། ངལ་རྩོལ་འབྲེལ་བའི་མཐུན་སྦྱོར་ལམ་སྲོལ་འཐུས་སྒོ་ཚང་དུ་གཏོང་བ། རྒྱ་ཆེའི་ངལ་རྩོལ་པར་ངོ་སོའི་ངལ་རྩོལ་ཡོད་པ་དང་ཕྱོགས་ཡོངས་ནས་འཕེལ་རྒྱས་ཡོང་བར་སྐུལ་འདེད་གཏོང་དགོས། ཏང་གི་སློབ་གསོའི་མཛད་ཕྱོགས་ཕྱོགས་ཡོངས་ནས་ལག་ལེན་བསྟར་ཏེ་སློབ་གསོ་སྔོན་ལ་འཕེལ་རྒྱས་གཏོང་རྒྱུ་མཐའ་འཁྱོངས་དང་། ཀུན་སྤྱོད་མི་སྐྱོང་གི་སྒྲིག་སྲོལ་ལམ་སྲོལ་འཐུས་སྒོ་ཚང་དུ་གཏོང་བ། ཀུན་སྤྱོད་དང་། ཤེས་རིག ལུས་རྩལ། མཛེས་རིག ངལ་རྩོལ་བཅས་ཕྱོགས་ཡོངས་ནས་འཕེལ་རྒྱས་ཀྱི་སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་འཛུགས་སྐྲུན་པ་དང་ཤུལ་འཛིན་པ་གསོ་སྐྱོང་བྱེད་དགོས། གྲོང་ཁྱེར་དང་གྲོང་གསེབ་ཀྱི་འོས་འགན་སློབ་གསོ་གཞི་གཅིག་ཅན་འཕེལ་རྒྱས་ལ་སྐུལ་འདེད་བཏང་ནས་སློབ་ཞུགས་གོང་གི་སློབ་གསོ་དང་དམིགས་བསལ་སློབ་གསོ་དང་། མཐོ་རིམ་སློབ་འབྲིང་དུས་མཚམས་ཀྱི་སློབ་གསོ་ཁྱབ་གདལ་བཅས་ཀྱི་འགན་ལེན་གྱི་ལམ་སྲོལ་འཐུས་སྒོ་ཚང་དུ་གཏོང་བ་དང་། ལས་རིགས་ལག་རྩལ་སློབ་གསོ་དང་། མཐོ་རིམ་སློབ་གསོ། རྒྱུན་མཐུད་སློབ་གསོ་བཅས་ཀྱི་སྤྱི་འདོམས་མཐུན་སྦྱོར་འཕེལ་རྒྱས་ཀྱི་ལམ་སྲོལ་འཐུས་སྒོ་ཚང་དུ་གཏོང་དགོས། གྲོང་ཁྱེར་དང་གྲོང་གསེབ་ལ་ཁྱབ་པའི་ཁྱིམ་ཚང་སློབ་གསོའི་ཞབས་ཞུའི་མ་ལག་བསྐྲུན་པ་དང་། སློབ་གསོ་དང་སློབ་སྦྱོང་བྱེད་སྟངས་གསར་གཏོད་བྱེད་པ། མི་ཀུན་ལ་ཕྱོགས་པ་དང་། ཀུན་ལ་ཕན་པ། དེ་བས་སྒོ་དབྱེ་སྟབས་བསྟུན་བཅས་ཀྱི་སློབ་གསོའི་མ་ལག་མགྱོགས་མྱུར་འཕེལ་རྒྱས་གཏོང་དགོས། འགན་ལེན་ཐུབ་རིགས་འགན་ལེན་བྱེད་པའི་རྩ་དོན་མཐའ་འཁྱོངས་བྱས་ཏེ་གྲོང་ཁྱེར་དང་གྲོང་གསེབ་ལ་སྤྱི་འདོམས་བྱེད་པ་དང་། རྒྱུན་མཐུད་ཆོག་པའི་གཞི་རྩའི་རྒན་གསོ་འགན་བཅོལ་གྱི་ལམ་ལུགས་དང་གཞི་རྩའི་སྨན་བཅོས་འགན་བཅོལ་ལམ་ལུགས་འཐུས་སྒོ་ཚང་དུ་བཏང་ནས་འགན་ལེན་གྱི་ཆུ་ཚད་བརྟན་པོའི་ངང་མཐོ་རུ་གཏོང་དགོས། སྤྱི་ཚོགས་འགན་བཅོལ་སྤོ་སྒྱུར་འབྲེལ་མཐུད་དང་ཡུལ་གཞན་དུ་སྨན་བཅོས་བྱས་པའི་སྨན་རིན་རྩིས་རྒྱག་ལམ་ལུགས་གང་མགྱོགས་དོན་འཁྱོལ་བྱས་ནས་སྤྱི་ཚོགས་འགན་བཅོལ་གྱི་ཐེབས་རྩ་དོ་དམ་ཚད་ལྡན་དུ་གཏོང་བ་དང་ཚོང་ལས་འགན་བཅོལ་འཕེལ་རྒྱས་གཏོང་དགོས། སྤྱི་ཚོགས་རོགས་སྐྱོབ་དང་། སྤྱི་ཚོགས་ཕན་བདེ། དགེ་སྒྲུབ་བྱ་གཞག གཟིགས་སྐྱོང་བཀོད་སྒྲིག་སོགས་ཀྱི་ལམ་ལུགས་སྤྱི་འདོམས་དང་འཐུས་སྒོ་ཚང་དུ་གཏོང་བ་དང་། དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་དམག་འཐབ་ལ་རྒྱལ་ཁ་མཐའ་གཅིག་ཏུ་ལེན་པ། ལོ་འདིའི་ནང་དབུལ་པོ་དབུལ་རྐྱང་གཞི་རྩའི་ཆ་ནས་སེལ་ཐུབ་པའི་འགན་ལེན་བཅས་བྱས་ནས་བལྟོས་བཅས་དབུལ་ཕོངས་སེལ་བའི་ཕུགས་ཕན་ལམ་སྲོལ་འཛུགས་དགོས། ཚེ་སྲོག་གི་དུས་འཁོར་ཧྲིལ་པོ་དང་བདེ་ཐང་གི་བརྒྱུད་རིམ་ཧྲིལ་པོར་དོ་ཁུར་བྱེད་རྒྱུ་མཐའ་འཁྱོངས་བྱས་ཏེ་རྒྱལ་དམངས་ཀྱི་གཟུགས་གཞི་བདེ་ཐང་གི་སྲིད་ཇུས་འཐུས་སྒོ་ཚང་དུ་བཏང་ནས་རྒྱ་ཆེའི་མི་དམངས་མང་ཚོགས་ལ་དྲང་གཞག་དང་གོ་རིམ་རྒྱུན་མཐུད་ལྡན་པའི་བདེ་ཐང་ཞབས་ཞུ་ལོངས་སུ་སྤྱོད་ཐུབ་པ་བྱེད་དགོས། སྨན་བཅོས་འཕྲོད་བསྟེན་སྒྲིག་གཞི་སྒྱུར་བཅོས་གཏིང་ཟབ་ཏུ་བཏང་ནས་གཞི་རྩའི་སྨན་བཅོས་འཕྲོད་བསྟེན་ལམ་ལུགས་འཐུས་སྒོ་ཚང་དུ་གཏོང་བ་དང་སྤྱི་པའི་འཕྲོད་བསྟེན་ཞབས་ཞུ་དང་། སྨན་བཅོས་ཞབས་ཞུ། སྨན་བཅོས་འགན་ལེན། སྨན་རྫས་མཁོ་འདོན་བཅས་ཀྱི་འགན་ལེན་ཆུ་ཚད་མཐོ་རུ་གཏོང་དགོས། སྤྱི་པའི་འཕྲོད་བསྟེན་རིམས་འགོག་དང་འགོ་ནད་ཆེ་རིགས་སྔོན་འགོག་ཚོད་འཛིན་བྱེད་ཤུགས་ཆེ་རུ་བཏང་ནས་ནད་གཞི་ཚབས་ཆེན་དང་ཆེ་རིགས་ཀྱི་སྨན་བཅོས་འགན་བཅོལ་དང་རོགས་སྐྱོབ་ལམ་ལུགས་འཐུས་སྒོ་ཚང་དུ་གཏོང་བ། ཁྱིམ་སྡོད་དང་སྡེ་ཁུལ་གྱི་སྒྲིག་གཞི་ཕན་ཚུན་མཐུན་སྦྱོར་དང་སྨན་བཅོས་དང་ལུས་གསོ་ཟུང་འབྲེལ་གྱི་རྒན་གསོ་ཞབས་ཞུའི་མ་ལག་སྐྲུན་མགྱོགས་སུ་གཏོང་བ། མི་དམངས་ཀྱི་གཟུགས་གཞི་བདེ་ཐང་ཡོང་བར་དམིགས་པ། དམངས་ཡོངས་ལུས་སྦྱོང་ལ་སྐུལ་འདེད་གཏོང་བའི་ལམ་ལུགས་རང་བཞིན་གྱི་བྱ་ཐབས་འཐུས་སྒོ་ཚང་དུ་གཏོང་བ་བཅས་བྱེད་དགོས།
ང་ཚོས་ཀྲུང་གོ་གསར་པ་དབུ་བརྙེས་ནས་ལོ་ངོ་70འཁོར་བར་རྟེན་འབྲེལ་གཟབ་རྒྱས་ཞུས་མ་ཐག་ཡིན་པ་དང་། “ཐོག་མའི་བསམ་འདུན་མ་བརྗེད་པར་འགན་འཁྲི་བློ་ལ་བརྟན་པོ་འཆང་བའི”བརྗོད་དོན་གཙོ་བོའི་སློབ་གསོར་སྐུལ་འདེད་གཏིང་ཟབ་གཏོང་བཞིན་པ། ཕྱི་ལོ་ནི་ཕྱོགས་ཡོངས་ནས་འབྱོར་འབྲིང་སྤྱི་ཚོགས་བསྐྲུན་པའི་མཇུག་བསྡུའི་ལོ་ཡིན་པས་ང་ཚོས་ངེས་པར་དུ་ཏང་གི་སྐབས་བཅུ་དགུ་པའི་ཀྲུང་ཨུ་ཚང་འཛོམས་གྲོས་ཚོགས་ཐེངས་བཞི་པའི་བཀོད་སྒྲིག་གཞིར་བཟུང་། འཕེལ་རྒྱས་ཀྱི་དོན་དངོས་དང་ཁ་མི་འབྲལ་བའི་སྔོན་འགྲོའི་ཆ་རྐྱེན་འོག་དམངས་འཚོ་འགན་ལེན་དང་ལེགས་བཅོས་དང་སྤུས་ཚད་མཐོ་བའི་འཕེལ་རྒྱས་བརྒྱུད། དམངས་འཚོ་ལེགས་གཏོང་ལ་སྐུལ་ཁྲིད་བྱེད་རྒྱུ་ཐོག་མཐའ་བར་གསུམ་དུ་མཐའ་འཁྱོངས་བྱས་ནས་གོམ་སྟབས་རེ་རེ་བཞིན་བརྟན་པོ་སྤོ་བ་དང་། དོན་དག་རེ་རེ་བཞིན་མུ་མཐུད་ཡག་པོ་སྒྲུབ་པ། དམངས་འཚོའི་ཁེ་སྤོགས་དོན་ཐོག་ཏུ་ཏན་ཏིག་འཁྱོལ་བར་བྱེད་པ་བཅས་བྱས་ཏེ་དམངས་འཚོ་ཡུན་རིང་འགན་ལེན་དང་བོད་ཀྱི་མི་རིགས་ཁག་གི་མི་དམངས་མང་ཚོགས་ཀྱི་འཚོ་བ་ཉིན་བཞིན་ལེགས་སུ་འགྲོ་ཐུབ་པ་བྱེད་དགོས།
རྩོམ་སྒྲིག་པ།　བདེ་སྒྲོལ།
