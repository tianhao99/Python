བོད་ལྗོངས་རྒྱ་བོད་སྐད་གཉིས་ཡིག་སྒྱུར་རྐང་འཛིན་གསོ་སྦྱོང་འཛིན་གྲྭ་འགོ་ཚུགས།
ཟླ་4ཚེས་16ཉིན་གྱི་སྔ་དྲོ། བོད་ལྗོངས་རྒྱ་བོད་སྐད་གཉིས་ཡིག་སྒྱུར་རྐང་འཛིན་གསོ་སྦྱོང་འཛིན་གྲྭ་ལྷོ་ནུབ་མི་རིགས་སློབ་ཆེན་དུ་དངོས་སུ་འགོ་ཚུགས།  རྒྱལ་ཁབ་མི་རིགས་དོན་གཅོད་ཨུ་ལྷན་གྱི་ཀྲུའུ་རེན་གཞོན་པ་དོན་འགྲུབ་དབང་འབུམ་དང་བོད་རང་སྐྱོང་ལྗོངས་མི་དམངས་སྲིད་གཞུང་གི་ཀྲུའུ་ཞི་གཞོན་པ། ལྗོངས་ཏང་ཨུད་འཐབ་ཕྱོགས་གཅིག་གྱུར་པུའུ་རྒྱུན་ལས་པུའུ་ཀྲང་གཞོན་པ། ལྗོངས་མི་རིགས་ཆོས་ལུགས་ཨུ་ལྷན་གྱི་ཏང་ཙུའུ་ཧྲུའུ་ཅི་སྐལ་བཟང་ཚེ་རིང་། སི་ཁྲོན་ཞིང་ཆེན་མི་དམངས་སྲིད་གཞུང་གི་ཧྲིན་ཀྲང་གཞོན་པ། ཞིང་ཆེན་མི་རིགས་ལས་དོན་ཨུ་ལྷན་གྱི་ཧྲུའུ་ཅི་གཞོན་པ་མུའུ་ཧྲི་ཧྭ། ཀྲུང་དབྱང་འཐབ་ཕྱོགས་གཅིག་གྱུར་པུའི་ཅུས་བདུན་པའི་སྐོར་གཟིགས་པ་ཡའོ་མའོ་ཁྲིན། རྒྱལ་ཁབ་མི་རིགས་དོན་གཅོད་ཨུ་ལྷན་མི་རིགས་གཞུང་ལུགས་དང་སྲིད་ཇུས་ཞིབ་འཇུག་ཁང་གི་ཀྲུའུ་རེན་གཞོན་པ་ལིས་ཧུང་ཅེ། ཀྲུང་གོའི་བོད་རིག་པ་ཞིབ་འཇུག་ལྟེ་གནས་ཀྱི་སྤྱི་ཁྱབ་ལས་འཛིན་གཞོན་པ་དགྲ་འདུལ། ལྷོ་ནུབ་མི་རིགས་སློབ་གྲྭ་ཆེན་མོའི་ཏང་ཨུད་རྒྱུན་ལས། སློབ་གཙོ་གཞོན་པ་ཝང་ཡོན་ཀྲིན། བོད་རང་སྐྱོང་ལྗོངས་བོད་ཀྱི་སྐད་ཡིག་ཨུ་ལྷན་གཞུང་སྒྲུབ་ཁང་གི་ཀྲུའུ་རེན་ནོར་བུ། ཀྲུའུ་རེན་གཞོན་པ་ཆོས་གྲགས་སོགས་འགོ་ཁྲིད་དང་སྐུ་མགྲོན་རྣམས་འཛིན་གྲྭའི་འགོ་ཚུགས་མཛད་སྒོར་ཞུགས།  ལྷོ་ནུབ་མི་རིགས་སློབ་གྲྭ་ཆེན་མོའི་ཏང་ཨུད་ཧྲུའུ་ཅི་སྤེན་ཅིན་གྱིས་མཛད་སྒོ་གཙོ་སྐྱོང་གནང་།
          གསོ་སྦྱོང་འཛིན་གྲྭའི་འགོ་ཚུགས་མཛད་སྒོའི་སྟེང་ཀྲུའུ་རེན་གཞོན་པ་ལིས་ཧུང་ཅེས་གསུང་རྒྱུར། སྐད་གཉིས་ཤེས་ལྡན་མི་སྣ་གསོ་སྦྱོང་ལྟེ་གནས་བཙུགས་པ་བརྒྱུད་ལྟེ་གནས་ཀྱི་ལམ་ཕྱོགས་དང་འཐབ་ར་གཙོ་བོའི་ནུས་པ་བཏོན་ཏེ། གཞི་རིམ་མི་རིགས་སྐད་ཡིག་ཡིག་སྒྱུར་བའི་ལས་སྒོའི་བྱང་ཚད་དང་གཞུང་ལུགས་ཀྱི་ཆུ་ཚད་མཐོར་འདེགས་ཐུབ་པ། མི་རིགས་སྐད་ཡིག་ཡིག་སྒྱུར་བའི་ཕུལ་བྱུང་དཔུང་སྡེ་འཛུགས་པར་སྐུལ་འདེད་ཐོན་པ་མ་ཟད། མི་རིགས་ས་ཁུལ་ཀྱི་ལས་རིགས་ཁག་ལ་ཚད་ལྡན་གྱི་མི་རིགས་སྐད་ཡིག་ཡིག་སྒྱུར་བའི་ཤེས་ལྡན་མི་སྣ་མཁོ་སྤྲོད་གནང་ཡོད།
 
          ནོར་བུ་ཅུས་ཀྲང་གིས་སློབ་མ་རྣམས་ཀྱིས་སློབ་སྦྱོང་གི་གོ་སྐབས་དཀོན་པོ་འདི་དམ་བཟུང་གིས་སྤྱོད་སྟངས་ཡག་པོ་དང་སེམས་ཤུགས་གཅིག་སྡུད། སློབ་སྦྱོང་ལ་བརྩོན་པ། བསམ་གཞིགས་མང་ཙམ་བཏང་སྟེ་ཕན་ཚུན་བར་བསམ་བློ་བརྗེ་རེས་ཀྱིས་རང་ངོས་ཀྱི་ནུས་པ་དང་ཆུ་ཚད་མཐོ་རུ་གཏོང་དགོས། མ་འོངས་པའི་བྱ་བར་རྨང་གཞི་ལེགས་པོ་བཏིང་དགོས། ཐུན་མོང་གི་འདུ་ཤེས་དང་ཚོགས་པའི་འདུ་ཤེས། རྩ་འཛུགས་ཀྱི་འདུ་ཤེས། དུས་ཚོད་ཀྱི་འདུ་ཤེས་བཅས་ཇེ་དྲག་ཏུ་གཏོང་བ་དང་། རང་འགུལ་གྱིས་དོ་དམ་དང་བཀོད་སྒྲིག་ལ་བརྩི་སྲུང་། དགེ་རྒན་ལ་བཀུར་ཞིང་ཕན་ཚུན་ལ་གཞོགས་འདེགས་ཀྱིས་“ནན་གསུམ་དྲང་གསུམ་”གྱི་རང་རྟོགས་ཀྱི་བསམ་བློ་ལག་ལེན་བྱ་ཐུབ་དགོས་པ་དང་། བོད་ལྗོངས་མི་རིགས་ལས་བྱེད་པའི་བྱང་ཚད་དང་ཀུན་སྤྱོད། སློབ་རྒྱུན་བཟང་པོ་མངོན་དགོས་པའི་བླང་བྱ་བཏོན།
          ལྷོ་ནུབ་མི་རིགས་སློབ་གྲྭ་ཆེན་མོའི་ཏང་ཨུད་ཧྲུའུ་ཅི་སྤེན་ཅིན་གྱིས་བརྗོད་རྒྱུར། སློབ་གྲྭས་རྒྱལ་ཁབ་མི་རིགས་དོན་གཅོད་ཨུ་ལྷན་གྱི་སྐད་གཉིས་ཤེས་ལྡན་མི་སྣ་གསོ་སྦྱོང་ལྟེ་གནས་རྟེན་གཞིར་བྱས་ཏེ། སི་ཁྲོན་དང་ཡུན་ནན། ཀོའུ་ཀྲིག བོད་ལྗོངས་བཅས་ལ་གསོ་སྦྱོང་འཛིན་གྲྭ་ཐེངས་མང་སྒྲུབ་པ་དང་། གྲུབ་འབྲས་མངོན་གསལ་བླངས་ཡོད། སློབ་གྲྭས་ལག་ལེན་སྟེང་ནས་“གྲངས་ཉུང་མི་རིགས་དང་མི་རིགས་ས་ཁུལ་ལ་ཞབས་ཞུ། རྒྱལ་ཁབ་ཀྱི་འཕེལ་རྒྱས་འཐབ་ཇུས་དང་མི་རིགས་བྱ་བར་ཞབས་ཞུ་བྱེད་པའི་”སློབ་གྲྭ་འཛུགས་གཉེར་གྱི་དམིགས་ཡུལ་མངོན་འགྱུར་བྱུང་ཡོད།
          ཐེངས་འདིའི་གསོ་སྦྱོང་འཛིན་གྲྭའི་སློབ་ཡུན་ཉིན་15ཡིན། ལྷོ་ནུབ་མི་རིགས་སློབ་གྲྭ་ཆེན་མོ་དང་བོད་རང་སྐྱོང་ལྗོངས་བོད་སྐད་ཡིག་ཨུ་ལྷན་（རྩོམ་སྒྱུར་ཅུས）གྱིས་གཙོ་གཉེར་བྱས་པ་དང་། ལྷོ་ནུབ་མི་རིགས་སློབ་གྲྭ་ཆེན་མོའི་མུ་མཐུད་སློབ་གསོའི་སློབ་གླིང་གིས་སྒྲུབ་པ་ཡིན། བོད་ལྗོངས་ཀྱི་ཆབ་མདོ་དང་མངའ་རིས། ནག་ཆུ། ལྷོ་ཁ། ཉིང་ཁྲི་སོགས་ས་གནས་སྡེ་ཁག་ནས་སླེབས་པའི་གཞི་རིམ་ལས་བྱེད་པ་བསྡོམས་པའི་སློབ་མ་60ཡོད། གསོ་སྦྱོང་འཛིན་གྲྭས་རྒྱལ་ཁབ་མི་རིགས་དོན་གཅོད་ཨུ་ལྷན་དང་ཀྲུང་གོའི་བོད་རིག་པ་ལྟེ་གནས། ཀྲུང་གོའི་སྤྱི་ཚོགས་ཚན་རིག་ཞིབ་འཇུག་ཁང་། ཀྲུང་དབྱང་མི་རིགས་སློབ་ཆེན། ལྷོ་ནུབ་མི་རིགས་སློབ་ཆེན། ནུབ་བྱང་མི་རིགས་སློབ་ཆེན། བོད་ལྗོངས་སློབ་ཆེན། བོད་ལྗོངས་རླུང་འཕྲིན་བརྙན་འཕྲིན་ཁང་། བོད་ལྗོངས་བོད་ཀྱི་སྐད་ཡིག་ཨུ་ལྷན་གཞུང་སྒྲུབ་ཁང་། སི་ཁྲོན་ཁམས་པའི་བརྙན་འཕྲིན་ཁང་བཅས་ཀྱི་ཆེད་མཁས་དགེ་རྒན་ཆེན་མོ་19གདན་འདྲེན་བྱས་ཏེ། སྐབས་དོན་ཆབ་སྲིད། མི་རིགས་སྲིད་ཇུས་དང་གཞུང་ལུགས། བོད་རྒྱ་རིག་གནས་ཀྱི་བྱུང་བ། བོད་རྒྱ་ཡིག་སྒྱུར་གྱི་བྱུང་བ། ཡིག་སྒྱུར་གཞུང་ལུགས་དང་ཐབས་ལམ་སོགས་ཀྱི་ནང་དོན་འཆད་འགྱུར་ཡོད།བོད་ལྗོངས་རྒྱ་བོད་སྐད་གཉིས་ཡིག་སྒྱུར་རྐང་འཛིན་གསོ་སྦྱོང་འཛིན་གྲྭ་འགོ་ཚུགས།
ཟླ་4ཚེས་16ཉིན་གྱི་སྔ་དྲོ། བོད་ལྗོངས་རྒྱ་བོད་སྐད་གཉིས་ཡིག་སྒྱུར་རྐང་འཛིན་གསོ་སྦྱོང་འཛིན་གྲྭ་ལྷོ་ནུབ་མི་རིགས་སློབ་ཆེན་དུ་དངོས་སུ་འགོ་ཚུགས།  རྒྱལ་ཁབ་མི་རིགས་དོན་གཅོད་ཨུ་ལྷན་གྱི་ཀྲུའུ་རེན་གཞོན་པ་དོན་འགྲུབ་དབང་འབུམ་དང་བོད་རང་སྐྱོང་ལྗོངས་མི་དམངས་སྲིད་གཞུང་གི་ཀྲུའུ་ཞི་གཞོན་པ། ལྗོངས་ཏང་ཨུད་འཐབ་ཕྱོགས་གཅིག་གྱུར་པུའུ་རྒྱུན་ལས་པུའུ་ཀྲང་གཞོན་པ། ལྗོངས་མི་རིགས་ཆོས་ལུགས་ཨུ་ལྷན་གྱི་ཏང་ཙུའུ་ཧྲུའུ་ཅི་སྐལ་བཟང་ཚེ་རིང་། སི་ཁྲོན་ཞིང་ཆེན་མི་དམངས་སྲིད་གཞུང་གི་ཧྲིན་ཀྲང་གཞོན་པ། ཞིང་ཆེན་མི་རིགས་ལས་དོན་ཨུ་ལྷན་གྱི་ཧྲུའུ་ཅི་གཞོན་པ་མུའུ་ཧྲི་ཧྭ། ཀྲུང་དབྱང་འཐབ་ཕྱོགས་གཅིག་གྱུར་པུའི་ཅུས་བདུན་པའི་སྐོར་གཟིགས་པ་ཡའོ་མའོ་ཁྲིན། རྒྱལ་ཁབ་མི་རིགས་དོན་གཅོད་ཨུ་ལྷན་མི་རིགས་གཞུང་ལུགས་དང་སྲིད་ཇུས་ཞིབ་འཇུག་ཁང་གི་ཀྲུའུ་རེན་གཞོན་པ་ལིས་ཧུང་ཅེ། ཀྲུང་གོའི་བོད་རིག་པ་ཞིབ་འཇུག་ལྟེ་གནས་ཀྱི་སྤྱི་ཁྱབ་ལས་འཛིན་གཞོན་པ་དགྲ་འདུལ། ལྷོ་ནུབ་མི་རིགས་སློབ་གྲྭ་ཆེན་མོའི་ཏང་ཨུད་རྒྱུན་ལས། སློབ་གཙོ་གཞོན་པ་ཝང་ཡོན་ཀྲིན། བོད་རང་སྐྱོང་ལྗོངས་བོད་ཀྱི་སྐད་ཡིག་ཨུ་ལྷན་གཞུང་སྒྲུབ་ཁང་གི་ཀྲུའུ་རེན་ནོར་བུ། ཀྲུའུ་རེན་གཞོན་པ་ཆོས་གྲགས་སོགས་འགོ་ཁྲིད་དང་སྐུ་མགྲོན་རྣམས་འཛིན་གྲྭའི་འགོ་ཚུགས་མཛད་སྒོར་ཞུགས།  ལྷོ་ནུབ་མི་རིགས་སློབ་གྲྭ་ཆེན་མོའི་ཏང་ཨུད་ཧྲུའུ་ཅི་སྤེན་ཅིན་གྱིས་མཛད་སྒོ་གཙོ་སྐྱོང་གནང་།
          གསོ་སྦྱོང་འཛིན་གྲྭའི་འགོ་ཚུགས་མཛད་སྒོའི་སྟེང་ཀྲུའུ་རེན་གཞོན་པ་ལིས་ཧུང་ཅེས་གསུང་རྒྱུར། སྐད་གཉིས་ཤེས་ལྡན་མི་སྣ་གསོ་སྦྱོང་ལྟེ་གནས་བཙུགས་པ་བརྒྱུད་ལྟེ་གནས་ཀྱི་ལམ་ཕྱོགས་དང་འཐབ་ར་གཙོ་བོའི་ནུས་པ་བཏོན་ཏེ། གཞི་རིམ་མི་རིགས་སྐད་ཡིག་ཡིག་སྒྱུར་བའི་ལས་སྒོའི་བྱང་ཚད་དང་གཞུང་ལུགས་ཀྱི་ཆུ་ཚད་མཐོར་འདེགས་ཐུབ་པ། མི་རིགས་སྐད་ཡིག་ཡིག་སྒྱུར་བའི་ཕུལ་བྱུང་དཔུང་སྡེ་འཛུགས་པར་སྐུལ་འདེད་ཐོན་པ་མ་ཟད། མི་རིགས་ས་ཁུལ་ཀྱི་ལས་རིགས་ཁག་ལ་ཚད་ལྡན་གྱི་མི་རིགས་སྐད་ཡིག་ཡིག་སྒྱུར་བའི་ཤེས་ལྡན་མི་སྣ་མཁོ་སྤྲོད་གནང་ཡོད།
 
          ནོར་བུ་ཅུས་ཀྲང་གིས་སློབ་མ་རྣམས་ཀྱིས་སློབ་སྦྱོང་གི་གོ་སྐབས་དཀོན་པོ་འདི་དམ་བཟུང་གིས་སྤྱོད་སྟངས་ཡག་པོ་དང་སེམས་ཤུགས་གཅིག་སྡུད། སློབ་སྦྱོང་ལ་བརྩོན་པ། བསམ་གཞིགས་མང་ཙམ་བཏང་སྟེ་ཕན་ཚུན་བར་བསམ་བློ་བརྗེ་རེས་ཀྱིས་རང་ངོས་ཀྱི་ནུས་པ་དང་ཆུ་ཚད་མཐོ་རུ་གཏོང་དགོས། མ་འོངས་པའི་བྱ་བར་རྨང་གཞི་ལེགས་པོ་བཏིང་དགོས། ཐུན་མོང་གི་འདུ་ཤེས་དང་ཚོགས་པའི་འདུ་ཤེས། རྩ་འཛུགས་ཀྱི་འདུ་ཤེས། དུས་ཚོད་ཀྱི་འདུ་ཤེས་བཅས་ཇེ་དྲག་ཏུ་གཏོང་བ་དང་། རང་འགུལ་གྱིས་དོ་དམ་དང་བཀོད་སྒྲིག་ལ་བརྩི་སྲུང་། དགེ་རྒན་ལ་བཀུར་ཞིང་ཕན་ཚུན་ལ་གཞོགས་འདེགས་ཀྱིས་“ནན་གསུམ་དྲང་གསུམ་”གྱི་རང་རྟོགས་ཀྱི་བསམ་བློ་ལག་ལེན་བྱ་ཐུབ་དགོས་པ་དང་། བོད་ལྗོངས་མི་རིགས་ལས་བྱེད་པའི་བྱང་ཚད་དང་ཀུན་སྤྱོད། སློབ་རྒྱུན་བཟང་པོ་མངོན་དགོས་པའི་བླང་བྱ་བཏོན།
          ལྷོ་ནུབ་མི་རིགས་སློབ་གྲྭ་ཆེན་མོའི་ཏང་ཨུད་ཧྲུའུ་ཅི་སྤེན་ཅིན་གྱིས་བརྗོད་རྒྱུར། སློབ་གྲྭས་རྒྱལ་ཁབ་མི་རིགས་དོན་གཅོད་ཨུ་ལྷན་གྱི་སྐད་གཉིས་ཤེས་ལྡན་མི་སྣ་གསོ་སྦྱོང་ལྟེ་གནས་རྟེན་གཞིར་བྱས་ཏེ། སི་ཁྲོན་དང་ཡུན་ནན། ཀོའུ་ཀྲིག བོད་ལྗོངས་བཅས་ལ་གསོ་སྦྱོང་འཛིན་གྲྭ་ཐེངས་མང་སྒྲུབ་པ་དང་། གྲུབ་འབྲས་མངོན་གསལ་བླངས་ཡོད། སློབ་གྲྭས་ལག་ལེན་སྟེང་ནས་“གྲངས་ཉུང་མི་རིགས་དང་མི་རིགས་ས་ཁུལ་ལ་ཞབས་ཞུ། རྒྱལ་ཁབ་ཀྱི་འཕེལ་རྒྱས་འཐབ་ཇུས་དང་མི་རིགས་བྱ་བར་ཞབས་ཞུ་བྱེད་པའི་”སློབ་གྲྭ་འཛུགས་གཉེར་གྱི་དམིགས་ཡུལ་མངོན་འགྱུར་བྱུང་ཡོད།
          ཐེངས་འདིའི་གསོ་སྦྱོང་འཛིན་གྲྭའི་སློབ་ཡུན་ཉིན་15ཡིན། ལྷོ་ནུབ་མི་རིགས་སློབ་གྲྭ་ཆེན་མོ་དང་བོད་རང་སྐྱོང་ལྗོངས་བོད་སྐད་ཡིག་ཨུ་ལྷན་（རྩོམ་སྒྱུར་ཅུས）གྱིས་གཙོ་གཉེར་བྱས་པ་དང་། ལྷོ་ནུབ་མི་རིགས་སློབ་གྲྭ་ཆེན་མོའི་མུ་མཐུད་སློབ་གསོའི་སློབ་གླིང་གིས་སྒྲུབ་པ་ཡིན། བོད་ལྗོངས་ཀྱི་ཆབ་མདོ་དང་མངའ་རིས། ནག་ཆུ། ལྷོ་ཁ། ཉིང་ཁྲི་སོགས་ས་གནས་སྡེ་ཁག་ནས་སླེབས་པའི་གཞི་རིམ་ལས་བྱེད་པ་བསྡོམས་པའི་སློབ་མ་60ཡོད། གསོ་སྦྱོང་འཛིན་གྲྭས་རྒྱལ་ཁབ་མི་རིགས་དོན་གཅོད་ཨུ་ལྷན་དང་ཀྲུང་གོའི་བོད་རིག་པ་ལྟེ་གནས། ཀྲུང་གོའི་སྤྱི་ཚོགས་ཚན་རིག་ཞིབ་འཇུག་ཁང་། ཀྲུང་དབྱང་མི་རིགས་སློབ་ཆེན། ལྷོ་ནུབ་མི་རིགས་སློབ་ཆེན། ནུབ་བྱང་མི་རིགས་སློབ་ཆེན། བོད་ལྗོངས་སློབ་ཆེན། བོད་ལྗོངས་རླུང་འཕྲིན་བརྙན་འཕྲིན་ཁང་། བོད་ལྗོངས་བོད་ཀྱི་སྐད་ཡིག་ཨུ་ལྷན་གཞུང་སྒྲུབ་ཁང་། སི་ཁྲོན་ཁམས་པའི་བརྙན་འཕྲིན་ཁང་བཅས་ཀྱི་ཆེད་མཁས་དགེ་རྒན་ཆེན་མོ་19གདན་འདྲེན་བྱས་ཏེ། སྐབས་དོན་ཆབ་སྲིད། མི་རིགས་སྲིད་ཇུས་དང་གཞུང་ལུགས། བོད་རྒྱ་རིག་གནས་ཀྱི་བྱུང་བ། བོད་རྒྱ་ཡིག་སྒྱུར་གྱི་བྱུང་བ། ཡིག་སྒྱུར་གཞུང་ལུགས་དང་ཐབས་ལམ་སོགས་ཀྱི་ནང་དོན་འཆད་འགྱུར་ཡོད།
