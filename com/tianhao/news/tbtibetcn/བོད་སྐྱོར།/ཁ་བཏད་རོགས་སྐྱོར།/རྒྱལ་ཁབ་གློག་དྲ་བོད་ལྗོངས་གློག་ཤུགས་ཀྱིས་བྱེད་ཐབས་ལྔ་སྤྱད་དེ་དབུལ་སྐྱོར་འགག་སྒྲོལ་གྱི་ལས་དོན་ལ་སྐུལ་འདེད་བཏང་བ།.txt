རྒྱལ་ཁབ་གློག་དྲ་བོད་ལྗོངས་གློག་ཤུགས་ཀྱིས་བྱེད་ཐབས་ལྔ་སྤྱད་དེ་དབུལ་སྐྱོར་འགག་སྒྲོལ་གྱི་ལས་དོན་ལ་སྐུལ་འདེད་བཏང་བ།
ཀྲུང་གོ་བོད་ལྗོངས་གསར་འགྱུར་དྲ་བའི་ཐོག་གནས་ཚུལ་སྤེལ་བར་གཞིགས་ན། "ལོ་ལྔ་ཚན་བཅུ་གསུམ་པའི"རིང་བོད་ཀྱི་དུད་ཚང་ཁྲི་6.4དང་མི་འབོར་དབུལ་པོ་ཁྲི་26.3ཡུལ་གཞན་དུ་གནས་སྤོ་བྱ་རྒྱུ་ཡིན་པ་དང་རྒྱལ་ཁབ་གློག་དྲ་བོད་ལྗོངས་གློག་ཤུགས་ཚད་ཡོད་ཀུང་སིས་དེ་མཚུངས་ཀྱི་གློག་ཤུགས་སྒྲིག་ཆས་སྒྲིག་སྦྱོར་བྱ་རྒྱུ་མ་ཟད། ཆན་ཕུའུ་10ཡི་གློག་སྐུད་སྤྱི་ལེ་920འཁྲིད་རྒྱུ་དང་། གློག་གནོན་སྒྱུར་ཆས་223སྒྲིག་སྦྱོར་བྱ་རྒྱུར་གློག་གསོག་ཚད་ཆན་ཕུའུ་ཨན་ཁྲི་2.71ཟིན་རྒྱུ། གློག་གནོན་དམའ་བའི་གློག་སྐུད་སྤྱི་ལེ་660འཁྲིད་རྒྱུ། དུད་ཚང་ཁག་ལ་གློག་གི་ཆུ་ཚོད་11852བསྒར་རྒྱུ་བཅས་ལ་ཁྱོན་མ་དངུལ་སྒོར་དུང་ཕྱུར་3.17གཏོང་རྒྱུ་ཡིན་པ་རེད།
          གསར་འགོད་པས་རྒྱལ་ཁབ་གློག་དྲ་བོད་ལྗོངས་གློག་ཤུགས་ཚད་ཡོད་ཀུང་སི་ནས་ཤེས་རྟོགས་བྱས་པར། བོད་རང་སྐྱོང་ལྗོངས་ཀྱིས་2016ལོར་དབུལ་སྒྲོལ་རྫོང་ཁག་དང་པོའི་གྲས་ཡིན་པའི་ལྷོ་ཁ་ཆུ་གསུམ་རྫོང་གི་འདི་ལོ་དུད་ཚང་165ཡི་གྲོང་མི་རྒྱ་མཚོའི་ངོས་ལས་མཐོ་ཚད་སྨི་4500ཡན་ཡོད་པའི་ཆུ་གསུམ་གྲོང་རྡལ་མཚོ་སྟོད་གྲོང་ཚོ་ནས་ཆུ་གསུམ་རྫོང་གྲོང་གི་ཉེ་འགྲམ་དུ་སྤོ་རྒྱུ་ཡིན་ཞིང་། འདུ་འགོད་འདི་ལོ་མཇུག་ཏུ་ལག་བསྟར་བྱ་རྒྱུ་དང་འབྲེལ་ཆན་ཕུའུ་ཨན་315ཟིན་པའི་གློག་གནོན་སྒྱུར་ཆས་གཅིག་སྒྲིག་སྦྱོར་བྱ་རྒྱུ་དང་། ཆན་ཕུའུ་བཅུ་ཡི་གློག་སྐུད་སྨི་500ཙམ་འཁྲིད་རྒྱུ། ཆན་ཕུའུ་380ཡི་གློག་སྐུད་སྨི་3000ཙམ་འཁྲིད་རྒྱུ་བཅས་ལ་སྔོན་རྩིས་བྱས་ན་མ་དངུལ་སྒོར་ཁྲི་46གཏོང་རྒྱུ་ཡིན་པ་རེད། རྫོང་འདིའི་ཡུལ་གཞན་དུ་གནས་སྤོས་དབུལ་སྐྱོར་ལས་གྲྭའི་གློག་ཆས་སྒྲིག་སྦྱོར་ལས་གྲྭའི་འཛུགས་སྐྲུན་ལས་དོན་ཡག་པོ་བསྒྲུབ་ཆེད་ཆུ་གསུམ་རྫོང་གློག་གཏོང་ཚད་ཡོད་ཀུང་སིས་རང་འགུལ་ངང་རྫོང་དབུལ་སྐྱོར་གཞུང་དོན་ཁང་དང་འབྲེལ་བ་བྱས་ཏེ་རྫོང་སྲིད་གཞུང་གི་ལས་འགན་ཁག་བགོས་འགོ་ཁྲིད་ལ་དུས་ཐོག་ཏུ་སྙན་སེང་ཞུས་ནས་ཆུ་གསུམ་གྲོང་རྡལ་ཆུ་གསུམ་གྲོང་ཚོའི་གྲོང་མི་ཡུལ་གཞན་དུ་གནས་སྤོ་བཀོད་སྒྲིག་བྱ་ཡུལ་གྱི་གློག་སྤྱོད་ལས་གྲྭའི་ཆེད་དོན་གློག་གཏོང་ཇུས་གཞི་སྔ་སྣུར་དུ་བཟོས་ཏེ་ཆུ་གསུམ་རྫོང་གི་ཡུལ་གཞན་དུ་གནས་སྤོས་དབུལ་སྐྱོར་གྱི"གྲོང་མིའི་འདུ་འགོད"གློག་སྒྱུར་རྣམ་གྲངས་འཛུགས་སྐྲུན་ལ་སྐུལ་འདེད་ཧུར་ཐག་བཏང་བ་རེད།
          རྒྱལ་ཁབ་གློག་དྲ་ལྷོ་ཁ་གློག་གཏོང་ཀུང་སིས་ཆུ་གསུམ་རྫོང་གི་ཡུལ་གཞན་དུ་གནས་སྤོས་དབུལ་སྐྱོར་གྱི་གློག་ལས་རྣམ་གྲངས་ཀྱི་འདུ་འགོད་འདི་ཆུ་གསུམ་རྫོང་གི་2016ལོའི་གྲོང་གསེབ་གློག་དྲའི་ཆེད་དོན་ལས་གྲྭའི་འཆར་འགོད་ནང་བཞག་ནས་གློག་སྒྱུར་ལས་གྲྭ་འཛུགས་སྐྲུན་གྱི་མ་དངུལ་བརྩོན་ལེན་དང་ཐག་གཅོད་ཧུར་ཐག་བྱས་པ་མ་ཟད། ས་གནས་སྲིད་གཞུང་ལ་གཞོགས་འདེགས་ཧུར་ཐག་བྱས་ཏེ་ཡུལ་གཞན་དུ་གནས་སྤོས་བཀོད་སྒྲིག་བྱ་ཡུལ་གྱི་གློག་ཁུངས་འཛུགས་སྐྲུན་ལས་དོན་གྱི་དུས་འགོའི་གྲ་སྒྲིག་ལས་དོན་ཡག་པོ་བསྒྲུབས་ནས། ཡུལ་གཞན་དུ་གནས་སྤོའི་ལས་གྲྭའི་འཛུགས་སྐྲུན་གྱི་གློག་དང་གནས་སྤོས་མང་ཚོགས་ཀྱི་ཐོན་སྐྱེད་དང་འཚོ་བའི་གློག་གི་གནད་དོན་ཐག་གཅོད་ཏན་ཏིག་བྱས་པ་རེད། ཤེས་རྟོགས་བྱུང་བར་གཞིགས་ན། "ལོ་ལྔ་ཚན་བཅུ་གསུམ་པའི"རིང་བོད་ཀྱི་དུད་ཚང་ཁྲི་6.4དང་མི་འབོར་དབུལ་པོ་ཁྲི་26.3ཡུལ་གཞན་དུ་གནས་སྤོ་རྒྱུ་ཡིན་པས། རྒྱལ་ཁབ་གློག་དྲ་བོད་ལྗོངས་གློག་ཤུགས་ཚད་ཡོད་ཀུང་སིས་དེ་མཚུངས་ཀྱི་གློག་ཤུགས་སྒྲིག་ཆས་སྒྲིག་སྦྱོར་བྱ་རྒྱུ་ཡིན་པ་རེད།
          དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་དམག་འཐབ་ལ་རྒྱལ་ཁ་ལེན་རྒྱུའི་སྐོར་གྱི་རང་སྐྱོང་ལྗོངས་ཏང་ཨུད་དང་སྲིད་གཞུང་གི་བླང་བྱ་གཞིར་བཟུང་། རྒྱལ་ཁབ་གློག་དྲ་བོད་ལྗོངས་གློག་ཤུགས་ཚད་ཡོད་ཀུང་སིས་སྒྲིག་འཛུགས་དང་འགོ་ཁྲིད་ལ་ཤུགས་སྣོན་བརྒྱབ་སྟེ་ཡུལ་གཞན་དུ་གནས་སྤོ་དབུལ་སྐྱོར་གྱི་གློག་ཤུགས་རྨང་གཞིའི་སྒྲིག་བཀོད་འཛུགས་སྐྲུན་ལ་མགྱོགས་མྱུར་སྐུལ་འདེད་བཏང་བའི་ཁར་ཀུང་སིའི་ཐོན་ཁུངས་དང་། ལག་རྩལ། དོ་དམ། ཞབས་ཞུའི་ལེགས་ཆ་བཅས་འདོན་སྤེལ་གང་ལེགས་བྱས་ཏེ། སྒྲིག་འཛུགས་མཐུན་སྦྱོར་ལ་ཤུགས་སྣོན་དང་། སྲིད་ཇུས་ཀྱི་རྒྱབ་སྐྱོར་བརྩོན་ལེན། ལག་བསྟར་སྒྲིག་འཛུགས་ཞིབ་ཚགས། ཀྲུང་དབྱང་ཁེ་ལས་ཀྱི་སྤྱི་ཚོགས་འགན་འཁྲི་ཧུར་ཐག་བསྒྲུབས་པ་བཅས་བྱས་ཏེ་ཕྱོགས་ཡོངས་ནས་དབུལ་སྒྲོལ་འགག་སྒྲོལ་དམག་འཐབ་ཀྱི་ལས་འགན་ལེགས་འགྲུབ་ཡོང་བར་འགན་ལེན་གྱིས་དབུལ་སྒྲོལ་འགག་སྒྲོལ་ཆེད་ལེགས་སྐྱེས་ཧུར་ཐག་ཕུལ་བ་སྟེ། གཅིག་ནས་འཆར་འགོད་ཁྲིད་སྟོན་དུ་བྱས་ཏེ་ནུས་ཤུགས་ཡོད་རྒུས"ལོ་ལྔ་ཚན་བཅུ་གསུམ་པའི"རིང་གི་གློག་ཤུགས་སྒྲིག་བཀོད་འཛུགས་སྐྲུན་ལ་སྐུལ་འདེད་བཏང་ནས་གྲོང་གསེབ་གློག་དྲའི་གློག་གཏོང་ནུས་པ་དང་གློག་གཏོང་སྤུས་ཚད་མཐོ་རུ་བཏང་བ་དང་། གཉིས་ནས་ཡུལ་གཞན་དུ་གནས་སྤོས་དབུལ་སྒྲོལ་སྲིད་ཇུས་ལ་གཞོགས་འདེགས་ཧུར་ཐག་བྱས་ཏེ་ཡུལ་གཞན་དུ་གནས་སྤོས་དབུལ་སྒྲོལ་བཀོད་སྒྲིག་བྱ་ཡུལ་དུ་གློག་ཁྲིད་རྒྱུ་དེ་གྲོང་གསེབ་གློག་དྲའི་འཛུགས་སྐྲུན་འཆར་འགོད་ནང་བཞག་སྟེ་གནས་སྤོས་དུད་ཚང་གི་སྡོད་གནས་གསར་པར་གློག་གཏོང་རྒྱུའི་ནུས་པ་དང་གློག་གཏོང་ཞབས་ཞུ་འཕྲལ་དུ་རྗེས་བསྙེགས་བྱ་རྒྱུ་འགན་ལེན་བྱས་པ། གསུམ་ནས་ཁྱད་ལྡན་ཐོན་ལས་དབུལ་སྒྲོལ་གྱི་གློག་གཏོང་བའི་འགན་ལེན་ལས་དོན་ཡག་པོ་བསྒྲུབས་ནས་གྲོང་ཚོ་རིམ་པའི་མཉམ་སྡེབ་དཔལ་འབྱོར་དང་ཁེ་ལས་འབྲིང་ཆུང་དང་ཆེས་ཆུང་གྲས་འཕེལ་རྒྱས་འགྲོ་བར་དུས་ཐོག་དང་། ནུས་ལྡན། བརྟན་པོ་བཅས་ཀྱི་གློག་གཏོང་འགན་ལེན་བྱས་པ། བཞི་ནས་འཛུགས་སྐྲུན་རྣམ་གྲངས་ཐད་ངལ་ནུས་ནང་ཞུགས་པ་སོགས་ཀྱི་བྱེད་ཐབས་སྣ་ཚོགས་སྤྱད་ནས་ལས་གྲྭའི་འཛུགས་སྐྲུན་ཚན་པས་རྣམ་གྲངས་གནས་ཡུལ་ལ་ཁྲིད་སྟོན་དང་རྒྱབ་སྐྱོར་བྱས་པའི་ཁར་ངལ་ནུས་འགན་གཙང་ཁག་བགོས་ཀྱི་བྱེད་ཐབས་སྤྱད་པ། ལྔ་ནས་གྲོང་ཚོར་བཅའ་སྡོད་གནས་22ཀྱི་དབུལ་སྐྱོར་འགག་སྒྲོལ་ལས་དོན་ལ་སྐུལ་འདེད་བཏང་སྟེ་དབུལ་སྒྲོལ་འགག་སྒྲོལ་དམག་འཐབ་ལ་རྒྱལ་ཁ་ཐོབ་པ་བྱས་ནས་2020ལོར་དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་དམིགས་ཚད་མངོན་འགྱུར་བྱ་ཆེད་ལེགས་སྐྱེས་ཧུར་ཐག་ཕུལ་བ་བཅས་རེད།
