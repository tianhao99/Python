དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་གྲུབ་འབྲས་སྤུས་ཚད་མཐོ་པོས་སྲ་བརྟན་དུ་གཏོང་དགོས།
ཕྱོགས་ཡོངས་ནས་འབྱོར་འབྲིང་སྤྱི་ཚོགས་བསྐྲུན་པའི་འགག་རྩའི་ལོར། རྒྱལ་ཡོངས་ཀྱི་ཞིང་ཆེན་རིམ་པའི་གཅིག་སྡུད་ཡུག་འབྲེལ་གྱི་ས་ཁུལ་དབུལ་པོ་འབའ་ཞིག་ཡིན་པའི་བོད་ཀྱི་རྫོང་དང་ཆུས་74ཚང་མ་དབུལ་སྒྲོལ་བྱུང་བ་རེད།
          ང་ཚོས་ཆབ་སྲིད་ཀྱི་འགན་འཁྲི་འཁུར་སེམས་དང་ལས་འགན་སྒྲུབ་སེམས་སྔར་བས་ཆེ་རུ་གཏོང་བ་དང་། འགན་འཁྲི་དོན་འཁྱོལ་དང་། སྲིད་ཇུས་དོན་འཁྱོལ། ལས་དོན་དོན་འཁྱོལ་བཅས་ལ་དམ་འཛིན་ཡག་པོ་བྱེད་པ། དམིགས་ཚད་ལ་དམ་པོར་གཞིགས་ནས་ལས་དོན་ཧུར་ཐག་སྒྲུབ་པ། དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་གྲུབ་འབྲས་མཐའ་གཅིག་ཏུ་སྲ་བརྟན་དུ་གཏོང་བ་བཅས་བྱས་ནས་དབུལ་སྒྲོལ་འགག་སྒྲོལ་ལ་ཕྱོགས་ཡོངས་ནས་རྒྱལ་ཁ་ཐོབ་རྒྱུར་ཧུར་ཐག་གཞོལ་དགོས།
          དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་གྲུབ་འབྲས་སྲ་བརྟན་དུ་གཏོང་བའི་གལ་ཆེའི་དོན་སྙིང་ལ་ངོས་འཛིན་གཏིང་ཟབ་བྱེད་དགོས་ཤིང་། དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་གྲུབ་འབྲས་སྲ་བརྟན་དུ་གཏོང་རྒྱུ་ནི་དུས་བཀག་ལྟར་ཕྱོགས་ཡོངས་ནས་འབྱོར་འབྲིང་སྤྱི་ཚོགས་བསྐྲུན་པའི་ངོས་ནས་བརྗོད་ན་གལ་ཆེའི་དོན་སྙིང་ལྡན་ཡོད། ཆབ་སྲིད་ཀྱི་ཚུགས་གནས་མཐོ་རུ་ཏན་ཏིག་གཏོང་བ་དང་། ཆབ་སྲིད་ཀྱི་འགན་འཁྲི་ཕྲག་ཏུ་འཁུར་བ། དབུལ་སྒྲོལ་འགག་སྒྲོལ་ལས་དོན་གཏིང་ཟབ་ཕྱོགས་སུ་སྤེལ་བ་དང་། དམ་འཛིན་ཏན་ཏིག་བྱེད་པ། དོན་འཁྱོལ་ཞིབ་ཚགས་བྱེད་པ་བཅས་ཀྱི་ཐོག་ནས་དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་གྲུབ་འབྲས་མུ་མཐུད་སྲ་བརྟན་དུ་གཏོང་དགོས།
          བློ་བརྟན་འགྱུར་མེད་ངང“མཚམས་འཇོག་མི་བྱེད་པ་བཞིའི”བླང་བྱ་དོན་འཁྱོལ་བྱེད་དགོས་ཤིང་། མུ་མཐུད་དབུལ་འགོག་ལམ་སྲོལ་འཛུགས་པ་དང་འཐུས་སྒོ་ཚང་དུ་གཏོང་བ་དང་། མི་འབོར་དབུལ་པོར་སྲིས་ཇུས་ཀྱི་རོགས་སྐྱོར་བྱེད་ཤུགས་ཆེ་རུ་གཏོང་བ། མི་འབོར་དབུལ་པོའི་ནང་སྐྱེས་སྒུལ་ཤུཌ་འཕེལ་བར་བྱེད་པ་སོཌ་ཐད་འབད་པ་ལྷུར་ལེན་དང་ཕན་འབྲས་དངོས་དོན་གཉེར་བྱས་ཏེ། དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་སྤུས་ཚད་དང་ཆུ་ཚད་ཕྱོཌ་ཡོངས་ནས་མཐོ་རུ་གཏོང་དགོས།
          “ཕྱིར་མིག་ལྟ་བའི”གྲལ་བཤེར་དག་ཐེར་ལ་སྐུལ་འདེད་གཏིང་ཟབ་གཏོང་དགོས་ཤིང་། དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་འགན་འཁྲི་དང་། སྲིད་ཇུས། ལས་དོན་བཅས་དོན་འཁྱོལ་བྱས་ཚུལ་ཕྱོགས་ཡོངས་ནས་གྲལ་བཤེར་བྱས་ཏེ། གྲལ་བཤེར་བྱས་ནས་ཤེས་རྟོགས་བྱུང་བའི་གནད་དོན་རེ་རེ་བཞིན་དག་ཐེར་དམིགས་སར་འབྱོར་བར་བྱེད་དགོས།
          དབུལ་སྒྲོལ་འགག་སྒྲོལ་དང་གྲོང་གསེབ་དར་རྒྱས་སྦྲེལ་མཐུད་སྤྱི་འདོམས་ལ་ཤུགས་སྣོན་རྒྱག་དགོས་ཤིང་། གསར་སྤེལ་རྣམ་པའི་དབུལ་སྐྱོར་དང་འགན་སྲུང་རང་བཞིན་གྱི་དབུལ་སྐྱོར་ཟུང་འབྲེལ་བྱ་རྒྱུར་སྐུལ་འདེད་བཏང་སྟེ། དབུལ་སྒྲོལ་འགག་སྒྲོལ་གྱི་གྲུབ་འབྲས་སྲ་བརྟན་དང་རྒྱ་ཆེར་གཏོང་བའི་རྨང་གཞིའི་ཐོག ཞིང་འབྲོག་པར་རྒྱུན་མཐུད་འབབ་འཕར་དང་གྲོང་ཚོའི་ཁོར་ཡུག་རྒྱུན་མཐུད་ལེགས་སུ་འགྲོ་བར་བྱས་ནས་མི་རིགས་ཁག་གི་མང་ཚོགས་ཀྱི་བདེ་སྐྱིད་ཚོར་སེམས་དང་ཐོབ་ཚོར་ཆེ་རུ་གཏོང་དགོས།
རྩོམ་སྒྲིག་པ། བདེ་སྒྲོལ།
