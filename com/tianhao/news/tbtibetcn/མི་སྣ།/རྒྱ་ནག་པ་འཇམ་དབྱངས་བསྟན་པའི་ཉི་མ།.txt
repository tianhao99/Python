རྒྱ་ནག་པ་འཇམ་དབྱངས་བསྟན་པའི་ཉི་མ།
རྒྱ་ནག་པ་འཇམ་དབྱངས་བསྟན་པའི་ཉི་མ་ནི་ཧོར་ཡུལ་ནས་འཁྲུངས་ཤིང་བོད་ཡུལ་དུ་སྐུ་ཡོན་དང་མཛད་འཕྲིན་རྒྱས་པའི་མཁས་ཆེན་ཞིག་སྟེ། ཁོང་གི་འཁྲུངས་ཡུལ་ནི། ཁོང་གིས་མཛད་པའི་རང་རྣམ༼འཇམ་དབྱངས་བསྟན་པའི་ཉི་མའི་རྣམ་ཐར་བཀའ་རྩོམ་མདོར་བསྡུས༽སུ། རོང་འབྲོག་ལོངས་སྤྱོད་འཛོམས་པ་ཧོར་གྱི་ཡུལ་། དབུལ་མིན་ཕྱུག་མིན་བར་མའི་ཁྱིམ་དུ་སྐྱེས་ཞེས་བཀོད་པ་ལྟར། ཀུན་མཁྱེན་འཇིགས་མེད་དབང་པོ་སོགས་བོད་ཀྱི་སྐྱེས་ཆེན་དམ་པ་དུ་མས་ཞབས་ཀྱིས་བཅགས་པ་སོག་པོའི་ཨ་རུ་ཧོར་ཆེན་ཏེ། ད་ལྟ་རྒྱ་སྐད་དུ阿鲁科尔沁ཟེར་བ་དེ་ཡིན་ཞིང་། བོད་རབ་བྱུང་བཅུ་གསུམ་པའི་མེ་སྟག་སྟེ་སྤྱི་ལོ༡༨༠༦ལོར་འཁྲུངས། དགུང་ལོ་བདུན་པར་བླ་བྲང་བཀྲ་ཤིས་འཁྱིལ་གྱི་གསེར་ཁྱི་རྣམ་བཞིའི་ཡ་གྱལ་སྡེ་ཁྲི་སྤྲུལ་བའི་སྐུ་རིན་པོ་ཆེ་འཇམ་དབྱངས་ཐུབ་བསྟན་ཉི་མས༼༡༧༧༩ནས༡༨༦༢༽བླ་བྲང་དུས་འཁོར་པ་རྗེ་ངག་དབང་ཕུན་ཚོགས་ཀྱི་ཡང་སྤྲུལ་དུ་ཐག་བཅད་དེ་རབ་ཏུ་བྱུང་།
    སྡེ་ཁྲི་རིན་པོ་ཆེ་འཇམ་དབྱངས་ཐུབ་བསྟན་ཉི་མའི་བཀའ་གསུང་ལྟར། དགུང་གྲངས་བཅུ་གཅིག་པ་བཞེས་པ་མེ་བྱི༼༡༨༡༦༽ལོར་བླ་བྲང་བཀྲ་ཤིས་འཁྱིལ་དུ་བྱོན། དགུང་ལོ་བཅུ་གཉིས་པ་ལ། ཀུན་མཁྱེན་སྐུ་ཕྲེང་གསུམ་པ་རྗེ་འཇིགས་མེད་རྒྱ་མཚོ༼༡༧༩༢ནས༡༨༥༣༽དགུང་གྲངས་ཉེར་དྲུག་བཞེས་པ་མེ་གླང༼༡༨༡༧༽ལོར་བླ་བྲང་བཀྲ་ཤིས་འཁྱིལ་གྱི་ཚོགས་ཆན་གྱི་ཁྲི་ཞབས་སོར་བཀོད་ནས་བརྩི་བཞག་ཐོག་མར་གནང་སྐབས་ཆོས་གྲྭར་ཞུགས། ལོ་དེར་རྗེ་གུང་ཐང་བསྟན་པའི་སྒྲོན་མེའི༼༡༧༥༦ནས༡༨༢༣༽དྲུང་དུ་བྱམས་ཆོས་ལྔ་དང་རིགས་ཚོགས་དྲུག་སོགས་ཀྱི་ཚིག་རྒྱུགས་ཕུལ་པར་གཟེངས་བསྟོད་ཐོབ། དེ་ནས་བཟུང་སྟེ་དགུང་ལོ་ཉེར་ལྔ་པའི་བར་དུ། ཀུན་མཁྱེན་ཡབ་སྲས་རྣམས་ཚུལ་བཞིན་དུ་བསྟེན་ནས་བཀའ་པོད་ལྔའི་གཞུང་ལ་སློབ་གཉེར་ཧུར་ཐག་མཛད་ཅིང་། དབང་ལུང་རྗེས་གནང་མན་ངག་གི་རིགས་གང་བཞུགས་ཆེད་གཉེར་གྱིས་ཞུས། ཆོས་གྲྭའི་ཞོར་ལ་ཐ་སྙད་རིག་གནས་ཀྱི་སྐོར་ལའང་སྦྱོང་བ་གནད་སྨིན་མཛད། དགུང་ལོ་ཉེར་ལྔ་པ་དལ་དྲན་དབང་བྱེ་བའི་དབུས་སུ་རྡོ་རམས་པའི་དམ་བཅའ་བཞག དེའི་ཚུལ༼རང་རྣམ༽ལས་མདོར་ན་མཁས་གྲུབ་དམ་པའི་མགོན་ལ་བསྟེན། ཆོས་མཐུན་ཚངས་པར་སྤྱོད་པའི་གྲོགས་དང་བཅས། མཐུན་རྐྱེན་ལོག་འཚོ་བྲལ་བའི་ཟས་གོས་འཛོམས། གནས་སྐབས་དེ་ནི་ལྷ་ཡུལ་ལས་ཀྱང་སྐྱིད་ཅེས་སོ། །
    ཨ་རུ་ཧོར་ཆེན་དུ་ལན་དུ་མར་ཕེབས་ཏེ་འགྲོ་དོན་བསྐྱངས། ཧལ་ཧ་གཡོན་རུའི་སར་ཧུ་བེ་ཅུན་ཝང་ཟེར་བས་སྦྱིན་བདག་བྱས་ཏེ། བདེ་མཆོག་བློས་བསླང་ལྟར་གྱི་གཙུགལག་ཁང་བཞེངས། དེའི་སྒོ་བཞི་གར་རྒྱུད་པ་དང། དུས་འཁོར། སྨན་པ། རིག་གནས་བཅས་ཀྱི་གྲྭ་ཚང་བཞི་བཙུགས་ཏེ་བོད་ཀྱི་ཆོས་དང་རིག་གནས་སོག་ཡུལ་དུ་ཆེ་ཆེར་དར་བར་མཛད། སྔ་ཕྱིར་ཨ་མཆོག་བདེ་མོ་ཐང་དགའ་ལྡན་ཆོས་འཁོར་གླིང་དང་། བླ་བྲང་བཀྲ་ཤིས་འཁྱིལ་གྱི་དུས་འཁོར་གྲྭ་ཚང་གི་ཁྲིར་བཞུགས་ཏེ་མདོ་སྔགས་ཀྱི་འཆད་ཉན་ཡར་ལྡན་དུ་སྤེལ། བླ་བྲང་བཀྲ་ཤིས་འཁྱིལ་གྱི་དུས་འཁོར་གྲྭ་ཚང་གི་ཁྲིར་བཞུགས་ཏེ་མདོ་སྔགས་ཀྱི་འཆད་ཉན་ཡར་ལྡན་དུ་སྤེལ། བླ་བྲང་དུ་རྗེ་གཟིགས་པ་ལྔ་ལྡན་རྟེན་གཙོར་བྱས་པའི་ཙོང་ཁ་བརྒྱད་ཅུའི་ལྷ་ཁང་གཡུ་ཐོག་ཅན་བདེ་ཡངས་དང་བཅས་གསར་དུ་བསྐྲུན། སྐུ་ཚེ་སྨད་ལ་རི་བོ་རྩེ་ལྔར་ཚེ་འདིའི་འཁྲི་བ་བཙན་ཆོད་དུ་སྤངས་ཏེ་སྒྲུབ་པ་ཉམས་ལེན་ལ་ཐུགས་རྩེ་གཅིག་ཏུ་གཞོལ། མཐར་དགུང་ལོ་ང་གསུམ་པ་རབ་བྱུང་བཅུ་བཞི་པའི་ས་རྟ༼༡༨༥༨ལོར་ཨུ་ཅུ་མུ་ཆིན་གྱི་གཡོན་རུའི་ཤོག་ཨེ་རིན་གྱི་ས་ཆར་སྐུ་གཤེགས།
    ཁོང་གི་འཁྲུངས་ལོ་རང་རྣམ་དུ་མི་གསལ་ཞིང་། ༼མདོ་སྨད་ཆོས་འབྱུང༽དུ་མེ་སྟག་ལ་འཁྲུངས་ཞེས་པ་ཙམ་ལས་རབ་བྱུང་གང་ཡིན་གསལ་བོར་བཀོད་མེད་པས། དཔྱད་བཟོད་བྲལ་པའི་བཤད་སྒྲོས་མ་དག་རྒྱུན་འབྱམས་སུ་འགྲོ་བཞིན་པར་དམིགས་ཏེ། ཕྲན་གྱིས་༡༩༩༣ལོའི༼མཚོ་སྔོན་སློབ་གསོ༽འདོན་ཐེངས་གཉིས་པའི་སྟེང་དུ༼སྤྲིན་གྱི་བཞོན་པའི་རྟོགས་བརྗོད་ཀྱི་རྩོམ་པ་པོའི་ལོ་རྒྱུས་ལ་ཅུང་ཟད་བརྟག་པ༽ཞེས་པའི་རྩོམ་ཐུང་ཞིག་སྤེལ་ཡོད་པ་སྐབས་འདིར་ཟུར་ལྟ་བྱེད་ཆོག
དམ་པ་འདི་བས་མཛད་པའི་བསྟན་བཅོས༼ཟབ་མོ་ལྟ་བའི་དམིགས་ཁྲིད་ཐར་བའི་སྒོ་འབྱེད༽དང་། ༼དབུས་མཐའི་འགྲེལ་བ༽སོགས་ཆོས་ཚན་སོ་གསུམ་པོད་གཉིས་སུ་དྲིལ་བ་བླ་བྲང་བཀྲ་ཤིས་འཁྱིལ་དུ་ཤིང་པར་བསྐྲུན་ཡོད་དོ། །
