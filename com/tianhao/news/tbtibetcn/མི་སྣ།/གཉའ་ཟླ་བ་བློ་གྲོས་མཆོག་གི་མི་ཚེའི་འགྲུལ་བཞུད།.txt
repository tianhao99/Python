གཉའ་ཟླ་བ་བློ་གྲོས་མཆོག་གི་མི་ཚེའི་འགྲུལ་བཞུད།
གཉའ་ཟླ་བ་བློ་གྲོས།
དང་པོ། ཕུགས་བསམ་རུམ་ལ་བཅུག་པའི་ལེའུ།
          མཁས་པའི་དབང་པོ་གཉའ་ཟླ་བ་བློ་གྲོས་རྒྱལ་མཚན་མཆོག་ནི་མཚོ་སྔོན་ཞིང་ཆེན་དཔའ་ལུང་རྫོང་རྩ་བ་ཡུལ་ཚོ་ཉིན་མགོ་སྡེ་བའི་གཉའ་ཚང་ཞེས་པའི་བོད་ཁྱིམ་ཞིག་ཏུ་འཁྲུངས། ཆུང་དུས་ཐེམ་ཆེན་དང་དཔའ་ལུང་། རྐང་ཚ་སོགས་ནས་གཉོམ་ལག་རྒྱས་ཤིང་། དགུང་ལོ་བཅུ་བཞི་སྟེ་1964ལོར་ལས་ཀྱི་བསྐལ་བ་བཟང་པོ་ཞིག་གིས་རྐང་ཚ་རྫོང་སློབ་ཆུང་ནས་ཤེས་རིག་པདྨའི་དགའ་ཚལ་ལ་སྤྲོ་ལྡན་བུང་བའི་ཚུལ་བཞིན་རྒྱུ་བའི་གོ་སྐབས་རྙེད། སྐྱེས་སྟོབས་ཀྱི་ཤེས་རབ་ཀྱིས་ཕྱུག་ཅིང་སྦྱངས་བརྩོན་གྱི་སྤོབས་པ་རྒོད་པས་ཕོ་ལོ་བཅོ་བརྒྱད་སྟེ་1968ལོར་ཕུལ་བྱུང་སྦྱངས་འབྲས་ཁྱེར་དེ་ཟི་ལིང་མི་རིགས་སློབ་འབྲིང་དུ་ཞུགས། ཁོང་ཉིད་ཀྱིས་གསུངས་པར་དེ་དུས་ཀྱི་སློབ་སྦྱོང་གི་ཆ་རྐྱེན་ཤིན་ཏུ་ཞན་ལ། ལྷག་པར་སློབ་འགྲིམ་བྱ་རྒྱུ་དང་རིམ་པ་བཞིན་བར་ཆད་མེད་པར་འདོན་རྒྱུར་དཀའ་ཁག་ཆེ་སྟབས། བརྩོན་པ་འགྲུས་ཀྱི་འབད་པ་ནམ་ཡང་ལྷོད་མ་མྱོང། སློབ་འབྲིང་མཐར་ཕྱིན་ལ་ཉེ་བའི་དུས། ལས་སྨོན་ཞིག་གིས་ཆེས་ཐོག་མའི་ཚོད་ལྟའི་རང་བཞིན་གྱི་བཟོ་ཞིང་དམག་གི་སློབ་མ་བྱས་ཏེ་རྒྱལ་ས་པེ་ཅིན་དུ་སློབ་གྲྭར་ཆས། དེ་དུས་ཁོང་ཉིད་དགུང་ལོ་ཉེར་གཅིག་ཏུ་འཁེལ་ཡོད་ལ། 1971ལོར་ཕ་ཡུལ་དང་ཁྱིམ་མི། དེ་ལས་ལྷག་པའི་མི་རིགས་ཤིག་གི་རེ་བཞག་རྒྱབ་ལ་ཁུར་ཏེ་ཀྲུང་དབྱང་མི་རིགས་སློབ་གྲྭ་ཆེན་མོའི་སྒོ་ཐེམ་རྡོག་པས་བཅག་ཅིང་། ལོ་ངོ་བཞིའི་རིང་ཤེས་རིག་གི་ལྡིངས་ཀ་འདི་ནས་འབད་འབུངས་རྒྱུན་མ་ཆད་པར་གནའ་དེང་ཕྱི་ནང་ཀུན་གྱི་རིག་གཞུང་ལ་མྱོང་རོལ་བྱས། ལྷག་པར་དུ་བོད་ཀྱི་ཉེ་རབས་མཁས་པའི་ཁྱུ་མཆོག་ཨ་ལགས་ཚེ་ཏན་ཞབས་དྲུང་གི་སྐུ་མདུན་ནས་བསྟུད་མར་ལོ་ངོ་གཉིས་ཀྱི་རིང་ལ་སུམ་རྟགས་དག་གསུམ་དང་སྒྲ་སྙན་ངག་གིས་གཙོས་པའི་བོད་ཀྱི་རིག་གནས་སྤྱི་དང་བྱེ་བྲག་ཏུ་མི་རིགས་གཞན་དང་གཞན་གྱི་བརྡ་སྤྲོད་རིག་པ་དང་། གཏན་ཚིགས་རིག་པ། ཚིག་རྒྱན་རིག་པ། སྐད་བརྡ་རིག་པ་སོགས་ཤེས་བྱ་བའི་རིག་གནས་མཐའ་དག་ལ་ཐོས་དོན་རྒྱུད་ལ་སྦྱོར་བའི་ཐོས་བསམ་རྣལ་མ་མཛད། སྐྱེས་སྟོབས་ཀྱི་ཤེས་རབ་དང་སྦྱངས་སྟོབས་ཀྱི་བརྩོན་འགྲུས་ལ་བརྟེན་ནས་རང་གཞན༌རིག་པའི་གནས་མཐའ་དག་གི་མཐར་ཐུག་ལྟ་བའི་འདུས་མདོ་རོ་གཅིག་ཏུ་མཐུན་པར་ངེས་པ་རྙེད།
          མི་ལོ་ཉི་ཤུ་ཁ་གྲངས་ཀྱི་འཚར་ལོངས་དང་སློབ་གཉེར་གྱི་བརྒྱུད་རིམ་འདི་ཁོང་ཉིད་ཀྱི་མི་ཚེ་འདིའི་རྒྱུ་ནོར་དུ་བརྩི་བཞིན་ཡོད་ལ། ལྷག་པར་ཁོང་རང་ད་ལྟ་དགུང་ལོ་བགྲེས་པའི་དུས་སུ་ཕྱིར་དྲན་གྱི་ཚུལ་དུ་གསུངས་པ་ལྟར། སྔོན་བསགས་ལས་བསྐལ་བཟང་པོ་ཞིག་གིས་རྒྱལ་ཁབ་ཕྱི་ནང་དང་ལྷག་པར་རང་མི་རིགས་ཀྱི་མཁས་པ་དགུང་གི་ཉི་མ་འདྲ་བ་དག་གི་གསུངས་ལ་བདུད་རྩི་མངར་མོ་མྱོང་བའི་གོ་སྐབས་བྱུང་བ་ཚེ་བསོད་དུ་འདོད་ཡོད་ལ། ད་ལྟ་རྒས་རུང་ཚེ་འདིར་མཁས་པ་མང་པོ་མཇལ་ཞིང་མཁས་པའི་གསུངས་ཉན་པས་དགའ་པོ་བྱུང་ཡོད། བསོད་ནམས་ཆེ་བར་རློམ་ཡོད་ཟེར། ལོ་ངོ་ཉི་ཤུ་ཁ་གྲངས་ཀྱི་སློབ་གཉེར་འདི་ལ་མཚོན་ན་ཁོ་བོའི་མི་ཚེ་འདིའི་དབུ་འཁྱུད་དུ་གྱུར་ཡོད་ལ། ད་རག་པར་དུ་སློབ་གཉེར་གྱི་གོམ་པ་འཇགས་མ་མྱོང་ཟེར། ཁོང་ཉིད་ཀྱི་གསུངས་དྲོན་པོ་འདི་དག་ཐོས་མ་ཐག་བོད་ཀྱི་ལེགས་བཤད་ལས་གསུངས་པའི་རིག་པ་ནངས་པར་འཆི་ཡང་བསླབ་ཅེས་པ་བློ་ལ་རང་ཤུགས་ཀྱིས་འཁོར་ངེས། གཞོན་པ་གཉོམ་ལག་རྒྱས་པའི་ལེའུ་སྟེ་དང་པོའི་ཚུལ་དུ་བྱའོ། །
གཉིས་པ། བསྐལ་ལྡན་སློབ་བུ་བསྐྱངས་པའི་ལེའུ།
          དགུང་ལོ་ཉེར་བཞི་སྟེ་1974ལོར་ཁོང་ཀྲུང་དབྱང་མི་རིགས་སློབ་གྲྭ་ཆེན་མོ་ནས་སློབ་མཐར་ཕྱིན་རྗེས། གོང་རིམ་གྱི་མངག་བཀོད་ལྟར་མཚོ་སྔོན་ཞིང་ཆེན་མཚོ་ནུབ་མི་རིགས་དགེ་ཐོན་སློབ་གྲྭར་ཕེབས་ནས། བློ་གསལ་ངང་མོ་འབུམ་ཕྲག་རིག་གནས་བཅུད་ཀྱིས་གསོ་བའི་དགེ་རྒན་གྱི་འགན་ཕྲག་ཏུ་བཞེས་སྟེ། མཁས་པའི་བྱ་བ་རྣམ་གསུམ་གྱི་འཆད་པའི་སྐོར་ཞུགས་ཤིང་། བགྲང་བྱ་ལྔ་ལྷག་གི་སློབ་ཁྲིད་ཀྱི་བྱ་བའི་ཡུན་ལ་ཁོང་གིས་བྱ་བ་ཐབས་ཅད་ལེགས་འགྲུབ་ཡོང་བའི་རྟག་སྦྱོར་གྱི་བརྩོན་པ་ནམ་ཡང་དོར་མ་མྱོང་ལ། མཁས་པའི་བྱ་བ་གསུམ་གྱིས་ལོ་དང་ཟླ་བའི་ཚེས་ཞག་རེ་རེ་བཞིན་ཕར་བསྐྱལ་བས་ལས་རོགས་དགེ་རྒན་གཞན་པ་དང་སྤྱི་ཚོགས་ཀྱིས་རྗེས་སུ་ཡི་རང་གི་བསྟོད་ཚིག་དྲི་མེད་ལྷ་རྫས་དར་དཀར་རིང་མོ་ཁོང་ལ་ཕུལ། ཕུལ་དུ་བྱུང་བའི་འབྲས་བུ་ལ་དམིགས་ནས་ཁོང་ལ་དགེ་རྒན་རྐང་འཛིན་པའི་མཚན་སྙན་བསྩལ། སློབ་གྲྭ་དེ་གར་སྔ་གཞུག་ཏུ་སློབ་རིམ་གསུམ་ལྷག་ཏུ་བོད་ཡིག་འཛིན་གྲྭའི་སློབ་ཁྲིད་ཀྱི་འགན་བཞེས་ཏེ། དགེ་རྒན་གཞན་དག་དང་མཉམ་དུ་ལོ་བཞི་ལྔ་ཙམ་གྱི་རིང་ལ་ཚད་མཐུན་ཤེས་ལྡན་པ་རབ་དང་རིམ་པར་སྐྱེད་སྲིང་བྱས་ནས་སྤྱི་ཚོགས་སྟེང་དུ་ཡོན་ཏན་གྱིས་ཕྱུག་པའི་ན་གཞོན་དགེ་རྒན་བརྒྱ་ལ་ཉེ་བ་གསོ་སྐྱོང་གནང་།
རྒྱལ་ཕྱི་ནང་གི་སློབ་གཉེར་པར་འཆད་ཁྲིད་གནང་བཞིན་པ། 
          ཁོང་གིས་སློབ་གྲྭ་དེར་བོད་ཡིག་གི་སློབ་དེབ་མེད་པའི་གནས་ཚུལ་ལ་དམིགས་ནས་བྲེལ་ཟིང་ཆེ་བའི་སློབ་ཁྲིད་ཀྱི་ཐུན་གསེང་གི་ཁོམ་པ་དག་བེད་སྤྱད་ནས་ངལ་བ་ཁྱད་བསད་དང་དཀའ་བ་དང་དུ་བླངས་ནས་སློབ་མ་རྣམས་ཀྱི་བློ་ཚོད་དང་བསྟུན་པའི་སློབ་དེབ་《བོད་ཀྱི་བརྡ་སྤྲོད་རིག་པ།》སོགས་རྩོམ་སྒྲིག་ཀྱང་མཛད། དགུང་ལོ་ཉེར་བདུན་ཏེ་1977ལོར་མཚོ་སྔོན་ཞིང་ཆེན་སྐབས་ལྔ་པའི་མི་དམངས་འཐུས་མི་ཚོགས་ཆེན་གྱི་འཐུས་མིར་བདམས་ཤིང་། དགུང་ལོ་ཉེར་བརྒྱད་པ་སྟེ་1978ལོར་མཚོ་ནུབ་མི་རིགས་དགེ་ཐོན་སློབ་གྲྭའི་དགེ་རྒན་ཕུལ་བྱུང་ཅན་དང་། མཚོ་ནུབ་ཁུལ་སློབ་གསོ་འཐབ་ཕྱོགས་སྟེང་གི་དགེ་རྒན་སྔོན་ཐོན་ཅན་ལ་བདམས། དགུང་ལོ་ཉེར་དགུ་པ་1979ལོར་ཟླ9པར་ནུབ་བྱང་མི་རིགས་སློབ་གྲྭ་ཆེན་མོའི་གྲངས་ཉུང་མི་རིགས་སྐད་ཡིག་ཚན་ཁག་གི་བོད་ཡིག་ཞིབ་འཇུག་སློབ་མར་རྒྱུགས་འཕྲོད་དེ་བོད་ཀྱི་བརྡ་སྤྲོད་རིག་པ་དང་ཚིག་རྒྱན་རིག་པ་ལ་ཆེད་གཉེར་གྱིས་ཞིབ་འཇུག་གནང་ནས། དགུང་ལོ་སོ་གསུམ་པ་1983ལོར་རབ་འབྱམས་པའི་དམ་བཅའ་བཞག་མཐར་ཁོང་གི་སྐྱེས་སྦྱངས་རིག་པའི་ཟབ་ཚད་ལ་དམིགས་ཏེ་སློབ་གྲྭ་དེ་གར་དགེ་རྒན་དུ་བསྐོས་ནས། བོད་ཀྱི་སྐད་ཡིག་ཚན་ཁག་ཏུ་དངོས་གཞི་ཚན་ཁག་གི་སློབ་མ་རྣམས་ལ་བོད་ཀྱི་སྙན་ངག་དང་ཚིག་རྒྱན་རིག་པའི་དཔེ་ཁྲིད་གནང་ཞོར་དུ་《བོད་ཀྱི་སྐད་ཡིག་གི་ཨ་མདོའི་བཀླག་གདངས་ཆེ་ལོང་ཙམ་བརྗོད་པ།》ཞེས་པ་སོགས་དཔྱད་རྩོམ་དུ་མ་《མཚོ་སྔོན་སློབ་གསོ།》སོགས་དུས་དེབ་ཁག་ཏུ་སྤེལ་ཞིང་།
          བརྩམས་ཆོས་《སྙན་ངག་གི་རང་མདངས་གསར་པ།》ཞེས་པ་ཀན་སུའུ་མི་རིགས་དཔེ་སྐྲུན་ཁང་གིས་པར་དུ་བསྐྲུན། བརྩམས་དཔེ་འདིས་བོད་ཀྱི་སྲོལ་རྒྱུན་སྙན་གཞུང་ལ་ལྟ་བ་གསར་པ་དང་། རིགས་ལམ་གསར་པ། ཐབས་ལམ་གསར་པ་བཅས་ཀྱི་སྒོ་ནས་ལེགས་པར་བརྟག་ཅིང་དཔྱད་དེ། བོད་ཀྱི་ཚིག་རྒྱན་རིག་པའི་གཞུང་ལུགས་ཀྱི་སྟོང་ཆ་མང་པོ་ཞིག་ཡོངས་སུ་བསྐངས། དགུང་ལོ་སོ་ལྔ་བ་སྟེ་1985ལོར་བྱ་བའི་ལས་གནས་ལྷོ་ནུབ་མི་རིགས་སློབ་གླིང་དུ་སྤར་ཏེ་མི་རིགས་ཚན་ཁག་ཏུ་དགེ་རྒན་གྱི་བྱ་བ་བསྒྲུབས། ལོ་ངོ་དྲུག་ལྷག་གི་རིང་ལ་སློབ་མ་རྣམས་ལ་སྙན་ངག་དང་ཚིག་རྒྱན་རིག་པ། བརྡ་སྤྲོད་རིག་པ་སོགས་ཀྱི་དཔེ་ཁྲིད་གནང་ནས་སྤྱི་ཚོགས་ལས་རིགས་སོ་སོའི་སྟེང་དུ་སློབ་གྲྭ་ཆེན་མོ་བ་འབོར་ཆེན་ཞིག་སྐྱེད་སྲིང་བྱས། 1987ལོའི་ཟླ9པ་ནས1988ལོའི་ཟླ1པོའི་བར་ལ་ཕྱི་གླིང་སློབ་གཉེར་བ་སྐོར་ཞིག་ལ་བོད་ཡིག་དཔེ་ཁྲིད་གནང་སྟེ་རྒྱལ་ཁབ་ཕྱིའི་བོད་རིག་པའི་ཤེས་ལྡན་པ་སྐོར་ཞིག་གསོ་སྐྱོང་གནང་། དེའི་བར་ལ་ལས་ཞོར་གྱི་ཁོམ་པ་བེད་སྤྱད་ནས་བྲིས་པའི་《བོད་ཅེས་པའི་ཐ་སྙད་ལ་དཔྱད་པ།》དང་《བོད་སྐད་ཀྱི་ཚིག་ཕྲད་ཀྱང་འང་ཡང་གསུམ་གྱི་བཀོལ་སྟངས།》 བོད་སྐད་ཀྱི་སྒྲ་གདངས་ཀྱི་གནད་དོན་འགའ་བཤད་པ་《མེ་ལོང་མའི་ལེའུ་དང་པོར་ཅུང་ཙམ་དཔྱད་པ།》བཅས་ཀྱིས་མཚོན་པའི་དཔྱད་རྩོམ་རྣམས་དང་། ལྷུག་རྩོམ་《ཁྱི།》ཞེས་པ་དང་སྙན་ངག་རིང་མོ་《ཁ་བའི་རྡུལ་མང་བརྩེགས་པ་ལ་ཁོང་ནང་སྙིང་བྱས་ཀྱང་གཅེས་པར་འཛིན་པའི་གདོང་ལྔའི་དབང་པོ་ཞེས་བྱ་བའི་འདས་ཚིགས་ཀྱི་ཞུ་མཆིད།》ཅེས་པ་དང་《ཤིས་ཚིག་བཀྲ་ཤིས་འབོད་པའི་དུང་དཀར་གཡས་སུ་འཁྱིལ་བའི་སྒྲ་དབྱངས།》སོགས་ཀྱིས་མཚོན་པའི་བརྩམས་ཆོས་རྣམས་སོ་སོར་《བོད་ལྗོངས་ཞིབ་འཇུག》དང་《མཚོ་སྔོན་སློབ་གསོ།》 《བོད་ལྗོངས་སྒྱུ་རྩལ།》 《མི་རིགས་སྐད་ཡིག་བྱ་བ།》 《རྩེར་སྙེག》སོགས་དུས་དེབ་ཁག་དང་དཔལ་དོན་གྲུབ་རྒྱལ་གྱི་གསུང་རྩོམ་ཕྱོགས་བསྒྲིགས་ཀྱི་མཇུག་རྣམས་སུ་བཀོད་ཡོད། ད་དུང་དགེ་གྲོགས་འགའ་དང་མཉམ་དུ་《སྙན་ངག་གི་དཔེར་བརྗོད་ཕྱོགས་བསྒྲིགས།》ཞེས་པ་རྩོམ་སྒྲིག་གནང་ཞིང་གླེང་གཞི་ཡང་ཡིག་འབྲུ་ཆིག་ཁྲི་ལྷག་ལ་ཉེ་བ་མཛད། 《མིག་རྒྱང་།》དང་《ནོར།》ཞེས་པའི་ལྷུག་རྩོམ་རྣམས་《ཉི་གཞོན།》དུས་དེབ་དང་ཀྲུང་དབྱང་མི་རིགས་དཔེ་སྐྲུན་ཁང་གིས་དཔར་བསྐྲུན་བྱས་པའི་《རྩྭ་ཐང་གི་ཞོགས་པའི་གླུ་དབྱངས།》སོགས་དུས་དེབ་སྟེང་སྤེལ་ཞིང་། དེ་ལས་《མིག་རྒྱང་།》ཞེས་པ་སློབ་གྲྭ་ཆེན་མོ་དག་གི་བོད་ཀྱི་དུས་རབས་རིམ་བྱུང་གི་རྩོམ་ཡིག་ཅེས་པའི་སློབ་དེབ་ཁྲོད་དུ་བདམས་བཀོད་བྱས།
དགེ་ཟླ་བ་བློ་གྲོས་དང་དགེ་བན་ཁོ།  དགེ་བདེ་སྐྱིད་འཚོ་བཅས་སིང་གྷ་ལ་རུ་ཕེབས་པའི་མཉམ་པར།  
          སྐབས་དེར་《བརྡ་སྤྲོད་ཉི་ཤུ་པ་རྩ་འགྲེལ།》སོགས་གནའ་དཔེ་སྐོར་ལ་ཞུ་དག་དང་གཏན་འབེབས། གླེང་གཞི་མཛད་པ་སོགས་ཀྱི་འགན་ཡང་ཁུར། དགུང་ལོ་སོ་བདུན་པ་སྟེ་1987ལོར་ལྷོ་ནུབ་མི་རིགས་སློབ་གླིང་དུ་དམིགས་བསལ་གྱིས་སློབ་དཔོན་གཞོན་པའི་ལས་ཐོབ་བླངས་ཤིང་། ལོ་དེར་གྲོང་ཁྱེར་ཁྲེན་ཏུའུ་ཡི་མཐུན་སྒྲིལ་དཔེ་བཟང་མི་སྣར་བདམས། 1988ལོར་ལྷོ་ནུབ་མི་རིགས་སློབ་གླིང་གིས་ཁོང་ལ་སློབ་ཁྲིད་འབྲས་བུ་ཕུལ་བྱུང་ཅན་གྱི་མཚན་སྙན་གནང་ཞིང་། 1989ལོར་རྒྱལ་ཁབ་སློབ་གསོ་ཨུ་ཡོན་ལྷན་ཁང་དང་། མི་དོན་ཅུས། ཀྲུང་གོ་སློབ་གསོའི་བཟོ་ཚོགས། རྒྱལ་ཡོངས་ཨུ་ཡོན་ལྷན་ཁང་བཅས་ཀྱིས་རྒྱལ་ཡོངས་དགེ་རྒན་ཕུལ་བྱུང་ཅན་གྱི་མཚན་སྙན་དཔང་ཡིག་དང་བཅས་པ་གནང་ནས་ཁོང་གི་གྲུབ་འབྲས་གཟེངས་སུ་བསྟོད། དགུང་ལོ་ཞེ་གཅིག་པ་སྟེ་1991ལོར་ལྷོ་ནུབ་མི་རིགས་སློབ་གླིང་གིས་བརྩམས་དཔེ་ཕུལ་བྱུང་ཅན་གྱི་བྱ་དགའ་གནང་། ལྷག་པར་སློབ་གྲྭ་འདི་ནས་ཕྱག་ལས་གནང་བའི་དུས་སུ་རྩོམ་རིག་ལ་དགའ་ཞེན་ཡོད་པའི་སློབ་བུ་འགའ་ལ་རོགས་སྐྱོར་དང་ཇུས་འགོད་བྱས་ནས་སྒྲོན་སྟེགས་བྱས་པའི་རྩོམ་རིག་དུས་དེབ་ཅིག་སྒྲུབ་པ་དང་། ད་ལྟའི་བར་རྒྱུན་མ་ཆད་པར་པར་ཐེངས་སུམ་ཅུ་འབུད་ལ་ཉེ།
          མི་ལོ་བཅུ་དྲུག་ལྷག་གི་དགེ་མིང་ཁུར་བའི་དུས་ཚོད་ཁོང་གི་མི་ཚེ་འདིའི་ཆེས་དོན་སྙིང་ལྡན་པའི་བར་སྐབས་ཤིག་ཏུ་བརྩི་བཞིན་ཡོད་ལ། ལྷག་པར་ཁོང་གིས་གསུངས་པར། མི་ལོ་བཅུ་དྲུག་གི་རིང་ལ་བོད་སྐད་ཡིག་གི་འཇོན་ཐང་ཅན་གྱི་མི་སྣ་བརྒྱ་ཕྲག་གསོ་སྐྱོང་བྱས་ཡོད་ལ། འདི་དག་ཐམས་ཅད་རང་ཉིད་ཀྱི་མི་ཚེའི་རིན་ཐང་དུ་རློམ་ཡོད། དེ་མིན་མི་ལོ་བཅུ་དྲུག་ཁྲོད་དགེ་སློབ་ཀྱི་འབྲེལ་བ་ཡོད་པའི་སློབ་མ་དག་བདག་ལ་བཀའ་དྲིན་ཆེའོ། དགེ་རྒན་གྱི་བྱ་བ་གཉེར་བའི་ལོ་ངོ་དེ་དག་ལ་སློབ་ཕྲུག་དག་གིས་དགེ་རྒན་ཞིག་ཡིན་ན་མཁྱེན་བརྩེ་ནུས་གསུམ་གྱི་བདག་ཉིད་ཅན་ལྟ་བུ་ཞིག་ཅིས་ཀྱང་ཡིན་དགོས་པར་ཤེས་སུ་བཅུག དེ་དག་གིས་ངའི་ཀུན་སྤྱོད་དང་སློབ་གཉེར། ཐ་ན་འཚོ་བ་སོགས་སུ་བཀའ་དྲིན་བླ་ན་མེད་ཟེར། བློ་གསལ་སློབ་བུ་ཐུགས་ཀྱིས་བསྐྱངས་པའི་ལེའུ་སྟེ་གཉིས་པའི་ཚུལ་དུ་བྱའོ། །
གསུམ་པ། རིག་པ་སྨྲ་བའི་ལེའུ།
          དགུང་ལོ་ཞེ་གཅིག་སྟེ་1991ལོའི་ཟླ་7པར་ལྷོ་ནུབ་མི་རིགས་སློབ་གླིང་ནས་བྱ་བའི་ལས་གནས་མཚོ་སྔོན་ཞིང་ཆེན་སྤྱི་ཚོགས་ཚན་རིག་ཁང་དུ་སྤོས་ཏེ། བོད་ཀྱི་ཤེས་རིག་ཞིབ་འཇུག་གི་བྱ་བའི་འགན་ཁུར། དེའི་རིང་ལ་《བོད་སྤྱི་འགྲོས་ཀྱི་སྐད་གདངས་དང་ཐ་སྙད་ཀྱི་གཞི་རྩའི་མ་ལག་ལ་རྣམ་པར་དཔྱད་པ།》དང་《བརྡ་སྤྲོད་ཐད་ཀྱི་གནད་དོན་འགའ་ཞིག་གི་རྙོག་མ་བགྲུང་བྱེད་ཀེ་ཏ་ཀའི་ནོར་བུ།》 《སྙན་ངག་ལ་འཇུག་པའི་མྱུར་ལམ།》 《ས་པཎ་གྱི་རྩོམ་གཤིས་རྣམ་བཤད།》ཅེས་པ་སོགས་ཡིག་འབྲུ་དགུ་ཁྲི་ལྷག་ལ་ཉེ་བའི་དཔྱད་རྩོམ་རྣམས་《རྩེར་སྙེག》དང་《ཀྲུང་གོའི་བོད་ཀྱི་ཤེས་རིག》 《མཚོ་སྔོན་སློབ་གསོ།》སོགས་རྒྱལ་ཁབ་དང་ཞིང་ཆེན་རིམ་པའི་རིགས་གཞུང་དུས་དེབ་དག་ཏུ་སྤེལ་ནས། བོད་ཀྱི་ཤེས་རིག་ཞིབ་འཇུག་གླིང་གི་དཔྱོད་ལྡན་པ་རྣམས་ལ་ལྟ་བའི་བག་ཆགས་ཟབ་མོ་བཞག དགུང་ལོ་ཞེ་གསུམ་པ་སྟེ་1993ལོར་མཚོ་སྔོན་སྤྱི་ཚོགས་ཚན་རིག་ཁང་གིས་ཁོང་ལ་ཞིབ་འཇུག་པ་གཞོན་པའི་ལས་ཐོབ་གནང་ཞིང་། དེ་དང་བསྟུན་ནས་མཚོ་སྔོན་ཞིང་ཆེན་མི་དམངས་སྲིད་གཞུང་གིས་ཀྱང་མཚོ་སྔོན་ཞིང་ཆེན་ཆེད་གཉེར་ལག་རྩལ་པ་ཕུལ་བྱུང་ཅན་གྱི་མཚན་སྙན་དཔང་ཡིག་གནང་། 1993ལོའི་ཟླ་12པར་རྒྱལ་སྲིད་སྤྱི་ཁྱབ་ཁང་གིས་ཁོང་ལ་སྤྱི་ཚོགས་ཚན་རིག་ཞིབ་འཇུག་བྱ་བའི་ཕྱག་རྗེས་མངོན་གསལ་བླངས་པར་ཆེད་དུ་གཟེངས་བསྟོད་གནང་ཆེད་སྲིད་གཞུང་གིས་གནང་བའི་དམིགས་བསལ་ཕོགས་སྣོན་གྱི་བདག་རྐྱེན་ཡང་མཛད། ལོ་དེའི་ལོ་མཇུག་ཏུ་སི་ཁྲོན་མི་རིགས་དཔེ་སྐྲུན་ཁང་གིས་གནས་སྐབས་གདན་ཞུ་ལྟར་དེ་གར་ཕེབས་ནས་རྩོམ་སྒྲིག་གི་བྱ་བ་ལོ་ངོ་གསུམ་བསྒྲུབས་པའི་རིང་ལ། 《གནས་ཡིག་ཕྱོགས་བསྒྲིགས།》དང་《རིན་གཏེར་སྨན་ཡིག་གཅེས་བཏུས།》 《བོད་ཀྱི་ལོ་རྒྱུས་སྤྱི་དོན་པད་མ་ར་གའི་ལྡེ་མིག》 《རྩ་རླུང་འཕྲུལ་འཁོར།》 《སྟོན་པ་ཤཱཀྱ་ཐུབ་པའི་རྣམ་ཐར།》 《བོང་སྟག་ལོ་རྒྱུས་མཐོང་བ་དོན་ལྡན་འཇུག་ངོགས།》 《དཔལ་དོན་གྲུབ་རྒྱལ་གྱི་གསུང་རྩོམ་པོད་དང་པོ།》བཅས་ཡིག་འབྲུ་ཁྲི་ཚོ་སུམ་བརྒྱ་ལ་ཉེ་བའི་དཔེ་ཆའི་མ་ཕྱི་པོད་བཅུ་གསུམ་ལྷག་རྩོམ་སྒྲིག་གནང་ཞིང་《མཚོ་བྱུང་ངག་གི་ཕོ་ཉ་བྱོན་པའི་ཚུལ།》སོགས་ཡིག་འབྲུ་ལྔ་ཁྲི་ལོངས་པའི་དཔེ་ཆའི་དཔྱད་གཏམ་ཡང་མཛད།
ཟླ་བ་བློ་གྲོས་དང་ལྷོ་ནུབ་མི་རིགས་སློབ་ཆེན་གྱི་88ལོའི་བོད་རྒྱ་སྐད་གཉིས་དྲུང་ཡིག་འཛིན་གྲྭའི་མཉམ་པར།

          1998ལོའི་ཟླ་3པར་ཁོང་གིས་ཞང་ཀང་བོད་ཀྱི་ཤེས་རིག་ཞིབ་འཇུག་ཚོགས་པའི་ཚོགས་གཙོ་གཞོན་པའི་འགན་བཞེས་གནང་བ་སོགས་ཀྱིས་མཚོན་པའི་མི་ཚེ་ཧྲིལ་པོར་བོད་ཀྱི་རིག་གནས་སློབ་གསོའི་བྱ་གཞག་ལ་རྩེ་གཅིག་ཏུ་གཞོལ་བས། གངས་ཅན་ཤེས་རིག་དབྱིག་གི་གན་མཛོད་དུ་རིག་པའི་ནོར་གྱི་མཐུན་རྐྱེན་གྱ་ནོམ་པ་བསྐྲུན་ཞིང་ཡོན་ཏན་དྲི་ཞིམ་འདབ་བརྒྱ་ལྷུག་པར་བཀྲ་ལ། བཅུད་ལྡན་རོ་བརྒྱའི་འབྲས་བུ་ཁུར་དུ་ལྕི་བས་ཤེས་ལྡན་འདབ་ཆགས་མ་བོས་མགྲོན་དུ་ཁུག་བཞིན་པ་འདི་ལགས། ཁོང་གི་བརྩམས་ཆོས་ལ་རྩོམ་སྒྲིག་ཞུ་དག་གི་སྐོར་ཕུད་བརྡ་སྤྲོད་རིག་པའི་སྐོར་དང་། སྐད་བརྡ་རིག་པའི་སྐོར། སྙན་ངག་དང་ཚིག་རྒྱན་རིག་པའི་སྐོར། ཡུལ་སྲོལ་གོམས་གཤིས་ཀྱི་སྐོར། དཔེ་ཆའི་དཔྱད་གཏམ་གྱི་སྐོར་བཅས་སྐོར་ཚོ་ཆེན་པོ་ལྔ་འདུས་ཤིང་བསྡོམས་པས་ཡིག་འབྲུ་ཁྲི་བདུན་སྟོང་ལྷག་ཡོད། ད་ལྟ་མི་རིགས་དཔེ་སྐྲུན་ཁང་གིས་མཁས་དབང་ཁོང་ཉིད་ཀྱི་བརྩམས་ཆོས་དུ་མ་ཕྱོགས་སྒྲིག་བྱས་ཏེ་《གཉའ་བློ་གྲོས་རྒྱལ་མཚན་གྱི་རྩོམ་ཡིག་ཕྱོགས་བསྡུས།》ཞེས་པའི་པོད་ཏི་གཉིས་ཀྱི་བདག་ཉིད་ཅན་བསྒྲིགས་འདུག དེ་ལྟར་རྩོམ་སྒྲིག་པའི་ཕྱག་ལས་གནང་ཞོར་བརྩམས་ཆོས་གྱ་ནོམ་པ་སྤེལ་བའི་ལེའུ་སྟེ་གསུམ་པའི་ཚུལ་དུ་བྱའོ།
བཞི་བ། མཇུག་བསྡུ་བའི་གཏམ།
          མདོར་ན་གཉའ་བློ་གྲོས་རྒྱལ་མཚན་མཆོག་ནི་འཇུག་པར་བྱ་བའི་ལས་གང་ལའང་ལྟ་བ་རྣམ་དག་གིས་གཞི་བཟུང་ཞིང་ཀུན་སློང་གཙང་མས་རྒྱུད་བསྐུལ་ཏེ། བདེན་དྲང་གཏམ་གྱིས་བླང་བྱ་དང་དོར་བྱའི་འཇུག་ལྡོག་གི་གནས་མ་འདྲེས་པར་སྟོན་པར་དགའ་ཞིང་། བརྟག་པར་བྱ་བའི་གནས་གང་ལའང་རིགས་པས་དཔྱད་བཟོད་དུ་གྲུབ་པའི་བདེན་ཚིག་རེ་འཇོག་པ་ཡིན་པས། ཁོང་གི་ཞིབ་འཇུག་བྱེད་སྟངས་ལ་མཚོན་ནའང་གཟུར་གནས་དྲང་བདེན་དང་འཆར་འགོད་དམིགས་གཏད་ཀྱིས་དོན་གྱི་ཟབ་གནད་འབྱིན་པར་སྤྲོ་ཞིང་། བརྩམས་ཆོས་ཀྱི་ངོས་ནས་བཤད་ནའང་། དོན་གྱི་ཟབ་གནད་དབྱུང་ཞིང་བྱེ་བྲག་གི་གསལ་ཆ་ཐོན་པའི་ཁྱད་ཆོས་མང་པོ་ཡོད། ཁོང་གི་བོད་ཀྱི་རིག་གནས་སློབ་གསོའི་བྱ་གཞག་ལ་ཉམ་ང་བ་དང་བག་ཚ་བ་ཅུང་ཟད་ཙམ་ཡང་མེད་པར་རང་དོར་གཞན་ཕན་གྱི་ལྷག་བསམ་རྣམ་པར་དག་པའི་སྒོ་ནས་ཉིན་ཞག་འཁོར་མོ་ཡུག་ཏུ་བརྩོན་པའི་སྦྱོར་བ་དང་དུ་བླངས་ཤིང་། མདོར་ན་ཁོང་གིས་བོད་ཀྱི་ཤེས་རིག་གི་བྱ་གཞག་ཐམས་ཅད་མི་ཉམས་ཡར་ངོའི་ཟླ་ལྟར་འཕེལ་བ་ལ་ནམ་ཡང་ཕྱར་བ་གྲུ་འདེགས་ཀྱི་ནུས་པ་བླ་ན་མེད་པ་ཐོན་བཞིན་མཆིས་སོ་ཞེས་པའི་མཇུག་བསྡུ་བ་ལེའུ་བཞི་པའི་ཚུལ་དུ་ཡོངས་སུ་རྫོགས་སོ། །
