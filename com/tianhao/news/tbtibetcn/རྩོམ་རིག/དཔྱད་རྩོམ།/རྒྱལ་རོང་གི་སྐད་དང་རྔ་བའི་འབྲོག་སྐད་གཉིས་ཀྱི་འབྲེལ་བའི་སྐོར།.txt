རྒྱལ་རོང་གི་སྐད་དང་རྔ་བའི་འབྲོག་སྐད་གཉིས་ཀྱི་འབྲེལ་བའི་སྐོར།
དེ་ཡང་སྤྱི་ཚོགས་རྙིང་བའི་སྐབས་སུ་འགྲིམ་འགྲུལ་སོགས་ཀྱི་ཆ་རྐྱེན་ཤིན་ཏུ་ཞན་པའི་སྟབས་ཀྱི་ཞིང་སྡེ་མི་འདྲ་བ་དང་ཡུལ་ལུང་མི་འདྲ་བ་རྣམས་སུ་སྐད་ཀྱི་ཁྱད་པར་སྣ་ཚོགས་ཡོད་པས། སྒྲ་གདངས་འདོན་ཚུལ་སོགས་ལ་ཁྱད་པར་བྱུང་ནས་ཡུལ་སྐད་དང་ལུང་སྐད་མང་བོར་གྱུར་སོང་བ་དོན་དངོས་དང་མཐུན་པར་འདོད་མོད། འོན་ཀྱང་དེའི་དོན་ནི་མིང་ཚིག་ཉུང་ཤས་ཀྱི་སྒྲ་གདངས་འགའ་ཅུང་ཟད་འགྱུར་སོང་བའི་གནད་དོན་རེ་མ་གཏོགས་བོད་མི་རིགས་ཀྱི་སྐད་དེ་གཞན་མི་རིགས་ཤིག་གམ་རྒྱལ་ཁབ་ཤིག་གི་སྐད་རིགས་སུ་འགྱུར་སོང་བ་གཏན་ནས་མིན།
གླེང་གཞི།
          ཐོག་མར་རང་རེ་བོད་མི་རིགས་འདི་ལ་གཟི་བརྗིད་འོད་སྟོང་འབར་བའི་ལོ་རྒྱུས་རིང་པོ་རིན་ཐང་བྲལ་བ་ཞིག་ཡོད་པས་འཛམ་གླིང་མིའི་རིགས་ཀྱི་རིག་གནས་གར་སྟེགས་སྟེང་དུ་ཡང་གལ་ཆེ་བའི་གོ་གནས་ཆེན་པོ་ཞིག་བཟུང་ཡོད་ཅིང༌། ལྷག་པར་དུ་དེང་རབས་ཚན་རིག་གི་ལག་རྩལ་གསར་དར་གྱི་རྦ་རླབས་མཁའ་ལ་འཕྱོ་བའི་དུས་རབས་ཉེར་གཅིག་པའི་ནང་འདི་ལ། རྒྱ་ཆེ་བའི་འཛམ་གླིང་གི་རྒྱལ་ཁབ་ཁག་དང་ས་གནས་སོ་སོ་ན་བོད་རིག་པར་ཞིབ་འཇུག་བྱེད་པའི་ཚ་རླབས་ཆེན་པོ་དྲག་ཏུ་འཕྱུར་བཞིན་ཡོད་པའི་སྐབས་སུ་ཡང་སྔོན་བྱོན་གྱིབཀའ་དྲིན་ཅན་མཁས་དབང་དམ་པ་གསེར་རིའི་ཕྲེང་བས་ཤུལ་དུ་བཞག་པའི་རྩ་ཆེན་གྱི་རིག་གནས་དག་ལ་རིས་མེད་ཀྱི་མཁས་དབང་དང༌། བློ་གཟུར་གནས་ཀྱི་འཛམ་གླིང་གི་ཞིབ་འཇུག་པ་མང་པོས་བོད་མི་རིགས་ཀྱི་རིག་གནས་ལོ་རྒྱུས་ལ་གདེངས་འཇོག་ཚད་མེད་གནང་སྟེ་འབྲས་བུ་བླ་ལྷག་ལེན་བཞིན་པ་འདི་ལ་ང་ཚོས་ཅིས་ཀྱང་དགའ་བའི་བསྟོད་དབྱངས་དང༌། མགུ་བའི་ཡི་རངས་བྱེད་དགོས་པའི་གནས་ཤིག་རེད། འོན་ཀྱང དེང་སྐབས་མི་རིགས་སོ་སོའི་རིག་གནས་བརྗེ་རེས་དང་ཕན་ཚུན་བར་འབྲེལ་འདྲེས་ཇེ་མང་དུ་འགྲོ་བཞིན་པའི་སྐབས་འདིར་རིག་གནས་ཀྱི་ངོ་བོ་འགྱུར་སླ་ལ་བསྒྱུར་ཉེན་ཀྱང་ཆེ་བའི་གནད་དོན་དེར་བསམ་བློ་གཏོང་འོས་བ་དེའོ།།
          དེ་ཡང་༼གངས་རྒྱལ་ཏི་སེའི་དཀར་ཆག༽ཞེས་པ་ལས། གངས་ཀྱི་རྒྱལ་པོ་ཏི་སེ་ཞེས་བྱ་བ། །ཆུ་བོ་བཞི་ཡང་དེ་ཡི་རྨང་ནས་འབབ། །ལན་གསུམ་བསྐོར་ན་མཚམས་མེད་ལྔ་ཡང་དག །གཏན་ཚིགས་དེའི་ཕྱིར་གངས་ཀྱི་རྒྱལ་པོ་ཞེས། །ཐོག་མར་ཞང་ཞུང་སྒོ་ཕུག་བར་གསུམ་ཞེས། །ཕུག་ཏེ་ཞང་ཞུང་རྒྱལ་གཤེན་བཅོ་བརྒྱད་དེ། །བར་ལ་དབུས་གཙང་ལྷོ་བྱང་ཀུན་ཏུ་ཤེས། །སྒོ་དེ་སུམ་པ་གླིང་གི་གྱིམ་ཤོད་དུ། །སྦས་པའི་ཤོད་ཆེན་བཅོ་བརྒྱད་རྒྱལ་མོ་རོང་། །ས་ཀུན་ཞང་ཞུང་མི་རྣམས་ཞང་ཞུང་གྲགས། །ཞེས་པའི་ཚིག་འདིས་ཤར་རྒྱལ་མོ་ཚ་བ་རོང་དེ་ཡ་ཐོག་གི་དུས་ནས་ཞང་ཞུང་གི་མངའ་ཁོངས་སུ་གཏོགས་ཡོད་པ་གསལ་པོ་ཤེས་ཐུབ།
          སྤྱི་ལོའི་སྔོན་གྱི་དུས་རབས་ཉི་ཤུ་བའི་ལོ་རབས་གཉིས་པའི་ནང(སྤྱི་ལོའི་སྔོན་གྱི་1917ལོ)ཤིང་བྱི་གཅིག་དཀར་གྱི་ལོའི་ཟླ་བ་དང་པོའི་ཚེས་བཅོ་ལྔའི་ཉིན་བོན་གྱི་སྟོན་པ་གཤེན་རང་མི་བོ་ཆེ་དེ་ཉིད་ཞང་ཞུང་ཡུལ་དུ་སྐུ་འཁྲུངས་པ་ནི༼  མདོ་གཟེར་མིག ༽  དང་། ༼   ལེགས་བཤད་རིན་པོ་ཆེའི་གཏེར་མཛོད  ༽   ༼   ཞང་བོད་ལོ་རྒྱུས་ཏི་སེའི་འོད་སྣང   ༽    ༼   བོད་ཀྱི་ལོ་རྒྱུས་པད་མའི་རྭ་གའི་ལྡེ་མིག  ༽  སོགས་ཀྱི་འདོད་པ་གཅིག་མཐུན་ཡིན། སྐབས་དེར་ཞང་ཞུང་རྒྱལ་པོ་ཁྲི་ཨེར་གསེར་གྱི་བྱ་རུ་ཅན་ཟེར་བ་དེས་ཞང་ཞུང་སྒོ་ཕུག་བར་གསུམ་སྟེ་གངས་རི་ར་བས་བསྐོར་བའི་ལྗོངས་ཀུན་དབང་བསྒྱུར་བྱས་ནས་རིམ་བཞིན་རྒྱལ་རབས་བཅོ་བརྒྱད་བྱུང་ཡོད་པ་དང་། དུས་མཚུངས་སུ་གཤེན་རབ་མི་བོ་ཆེས་གཡུང་དྲུང་བོན་དང་རིག་གནས་སྤེལ་ཏེ་ཡུལ་འདི་བོན་གྱི་ཤེས་རིག་གིས་དབང་བསྒྱུར་བྱེད་པའི་སྲོལ་རྒྱུན་གྱི་བརྒྱུད་པ་ལ་བོན་གྱི་བླ་རབས་བཅོ་བརྒྱད་དང་རིང་བརྒྱུད་བླ་རབས་ཉེར་བཞི་སོགས་བྱུང་ཡོད་ཅེས་པའི་བཤད་སྲོལ་ཡོད། རྐྱེན་དེའི་རང་ཉིད་ནི་འབར་ཁམས་རྒྱལ་རོང་གི་སྐྱེས་ཁུངས་སུ་རྟོགས་ཞིང་།  ཆུང་དུས་ནས་རྔ་བའི་འབྲོག་ཁུལ་ནས་འཚར་ལོངས་བྱུང་སྟབས་སྐད་དེ་གཉིས་གང་ཡིན་ཀྱང་།  གཞན་གྱི་བཤད་པ་སླ་མོར་གོ་ཐུབ་ལ་རང་གིས་ཀྱང་བཤད་ཤེས།  དེ་མིན་གཞན་ངོས་ནས་བཤད་ན་དེང་སྐབས་རྒྱ་ཡིག་སོགས་ལ་ཅུང་ཟད་མཁས་འདོད་དུ་སྣང་བའི་མི་ཁ་ཤས་ཀྱི་སྐད་བརྡ་རིག་པ་དང་ལོ་རྒྱུས་ཡིག་ཆ་ཕྱོགས་ཞེན་ཅན་འགའ་རེ་ཚད་མར་བཟུང་ཏེ། རྒྱལ་རོང་སྐད་དང་འབྲོག་སྐད་བར་ལ་འབྲེལ་བ་མེད་པ་སོགས་འོལ་ཚོད་ཀྱི་གཏམ་དེའི་ཕྱིར་རང་ཉིད་ཀྱི་དེའི་ཕྱོགས་ལ་ཅུང་ཟད་བཤད་འདོད་ཅིང་།ལྷག་པར་རྔ་བའི་འབྲོག་སྐད་ནང་གི་བརྡ་རྙིང་ཕལ་ཆེལ་ཞིག་དང་འབར་ཁམས་ཕྱོགས་ཀྱི་རྒྱུན་སྤྱོད་མིང་ཚིག་མང་བོའི་བར་ལ་འབྲེལ་བ་ཡོད་པ་དང་།  དེ་མིན་ཁ་སྐད་ཁྲོད་ཀྱི་བརྡ་སྤྲོད་རྣམ་པའི་སྐད་ཆའི་མཚོན་སྟངས་མང་བོ་ཞིག་རྒྱལ་རོང་བའི་སྐད་ཀྱི་ཚིག་གི་གྲུབ་སྟངས་དང་འདྲ་ས་མང་བས།  འབར་ཁམས་རྒྱལ་རོང་སྐད་དམ་རྔ་བའི་འབྲོག་སྐད་གང་ཡིན་ཀྱང་། བོད་ཀྱི་བརྡ་རྙིང་དེ་གནའ་བོའི་སྐད་དམ་སྐད་གསར་བཅད་མ་བྱས་གོང་གི་བོད་སྐད་ཡིན་པ་ནི་ཐེ་ཚོམ་གྱི་གནས་སུ་གྱུར་མེད།  དེང་སྐབས་དེའི་སྐོར་ལ་ཞིབ་འཇུག་བྱས་པ་ཅུང་ཁ་གསལ་ལ་ཕྱོགས་ཡོངས་ནས་འགྲེལ་བཤད་བྱས་ཡོད་པ་ནི།   མཁས་དབང་བཙན་ལྷ་ངག་དབང་ཚུལ་ཁྲིམས་ཚང་གི་  ༼    བརྡ་དཀྲོལ་གསེར་གྱི་མེ་ལོང་  ༽   དང་  ༼   རྒྱལ་རོང་ཞིབ་འཇུག་དཔྱད་ཡིག་ཕྱོགས་བསྒྲིགས   ༽   ། ད་དུང་ ༼     རྒྱལ་རོང་ཁ་སྐད་ཚིག་མཛོད་   ༽   དང་ ༼     རྒྱལ་རོང་སྐད་ཀྱི་ནང་གི་བརྡ་རྙིང་  ༽  ཞེས་པ་དང་།  དེ་མིན་  ༼   ཨ་མདོའི་སྐད་དང་རྒྱལ་རོང་སྐད་དཔྱད་བསྡུར་དབྱེ་ཞིབ    ༽   ཅེས་པ་སོགས་དང་།  གཞན་མཁས་དབང་ཚེ་རིང་ཐར་དང་ཐུབ་བསྟན་ཕུན་ཚོགས།  ཤེར་གྲགས། མཁས་དབང་ཉན་ཤུལ་མཁྱེན་རབ་འོད་གསལ་དང་དཀོན་མཆོག་རྡོ་རྗེ་སོགས་ཀྱི་དེའི་སྐོར་ལ་དཔྱད་རྩོམ་མང་པོ་སྤེལ་ཡོད་པ་དེ་བཞིན་རེད།  དེའི་ཕྱིར་འབར་ཁམས་རྒྱལ་རོང་གི་སྐད་དང་རྔ་བའི་འབྲོག་སྐད་གཉིས་ཀྱི་འདྲ་འབྲེལ་གླེང་བའི་ཁྲོད་ནས་གོ་རྟོགས་བྱུང་དགོས་བ་ནི་ཡུལ་སྐད་སོ་སོའི་འབྲེལ་བ་ཟབ་ཅིང་དབྱེ་བ་མེད་པ་མཚོན་སྟེ།བོད་སྐད་ཡིག་རྒྱ་ཆེར་བཀོལ་སྤྱོད་དང་ཁྱབ་གདལ་གཏོང་རྒྱུ་ནི་བོད་སྐད་དང་བོད་ཡིག་གི་རིན་ཐང་ངོ་མ་མངོན་པརམཚོན་བྱེད་ཀྱི་ཐབས་ལམ་ཉག་ཅིག་ཏུ་གྱུར་ཅིང་།   སྤྱི་སྐད་གཏན་འབེབས་ཀྱི་རྨང་གཞིའམ་སྔོན་འགྲོའི་ཆ་རྐྱེན་ལེགས་པོ་ཞིག་ཀྱང་ཡིན་པར་འདོད་པས། ང་ཚོའི་གནས་བབས་འདིར་སྲིད་ཅུས་འཐུས་ཚང་ཞིག་དང་ཁྲིམས་སྲོལ་ནན་མོ་ཞིག་ལ་བརྟེན་ནས་ཐབས་བརྒྱ་ཅུས་སྟོང་སྒོས་ལག་ལེན་བྱེད་གང་ཐུབ་ལ་འབད་དགོས་པ་ཁ་ཚ་དགོས་གཏུག་ཀྱི་ལས་ཤིག་ཡིན་པ་དེར་དམིགས་ནས་དེའི་སྐོར་ནས་ཅུང་ཟད་ཅིག་གླེང་བར་འདོད་པ་དེའོ།
          དང་བོ། འབར་ཁམས་རྒྱལ་རོང་གི་སྐད་བཤད་པ།     
          ༡  འབར་ཁམས་རྒྱལ་རོང་གི་མི་འབོར་དང་ས་བབ་ཆགས་ཚུལ་སོགས་བཤད་པ།                   
          འབར་ཁམས་ནི་རྔ་བ་བོད་རིགས་རང་སྐྱོང་ཁུལ་གྱི་ལྟེ་གནས་ལ་ཡོད་ཅིང་། ས་རྒྱ་སྤྱི་ལེ་གྲུ་བཞི་མ་༦༦༣༣་ཡོད་པ་དང་།འབར་ཁམས་ཀྱི་ལོ་རྒྱུས་རྒྱུན་རིང་ཞིང་།ལྷ་ཤུགས་གནའ་ཤུལ་དང་མཁོ་ནོ་གནའ་ཤུལ་ལ་བརྟག་ཞིབ་བྱས་པ་ལྟར་ན། འབར་ཁམས་ཀྱི་ལོ་རྒྱུས་ནི་གནའ་རབས་ཀྱི་རྡོ་ཆས་གསར་བའི་དུས་རབས་སུ་ཁུངས་འདེད་བྱེད་བྱེད་ཐུབ་ཅེས་བཤད་ཡོད་ཅིང་། འབར་ཁམས་རྫོང་གི་གྲོང་མི་མང་ཆེ་བ་ནི་རྒྱལ་རོང་བ་ཡིན་ལ། ༢༠༠༦ལོར་སྡོམ་རྩིས་བྱས་བ་ལྟར་ན་མི་གྲངས་ཁྲི་༥.༥་ཙམ་ཡོད་པ་དེའི་ནང་བོད་རིགས་ཁྲི་༤ལྷག་ཙམ་ཡོད་པས་མི་འབོར་ཡོངས་ཀྱི་བརྒྱ་ཆ་༧༡.༩༡ཙམ་ཟིན་ཡོད།  དེ་ཡང་རྒྱལ་རོང་ཞེས་པ་ནི་རྒྱལ་མོ་ཚ་བ་རོང་གི་བསྡུས་མིང་ཡིན་ལ། རྒྱ་མོ་ཚ་བ་རོང་ནི་བོད་རོང་ཆེན་བཞིའི་ཡ་གྱལ་ཏུ་འདྲེན་པའི་བོད་ཀྱི་ཤར་ཕྱོགས་ཀྱི་རོང་ཆེན་དག་གི་སུལ་དང་ཆུ་ཆེན་རྣམས་ཀྱི་འགྲམ་དུ་ཏུ་ཆགས་པའི་ཡུལ་གྲུ་དེར་ཟེར་ཞིང་།བོད་གདོང་རྒྱུད་བསྙོན་དུ་མེད་པ་ནི་མིག་མཐོང་ལག་ཟིན་དུ་སྣང་ལ།ལྷག་པར་དུ་བོད་ཀྱི་རིག་གནས་ལ་ཆ་རྒྱུས་ཅུང་ཟད་ཡོད་མཁན་ཅིག་ཡིན་ན་དེ་ལས་ཀྱང་ཐེ་ཚོམ་གྱི་གནས་སུ་འགྱུར་མ་སྲིད་དོ།
          ༢  འབར་ཁམས་རྒྱལ་རོང་གི་སྐད་དང་འབྲེལ་བའི་ཡུལ་སྲོལ་གོམས་གཤིས་སོགས་གླེང་བ།
          དེ་ཡང་སྤྱིར་སྐད་ཡིག་དང་ཆོས་ལུགས་གང་འདྲ་ཞིག་ཡིན་རུང་མི་རིགས་དེ་དང་མ་འབྲེལ་བའི་སྐད་ཡིག་དང་ཆོས་ལུགས་ཅིག་གཏན་ནས་ཡོད་མི་སྲིད་པས །ང་ཚོས་དེ་ལས་ཀྱང་རྒྱལ་རོང་བོད་རིགས་གི་སྐད་དང་ཡི་གེ་གཙོ་བྱས་པའི་ཆོས་ལུགས་དད་མོས་དང་། ཡུལ་སྲོལ་གོམས་གཤིས་སོགས་བཤད་དགོས་ལ་བཤད་རིན་ཀྱང་ཡོད་པ་ཞིག་རེད་འདོད་གི་ཡོད། ལྷག་པར་དུ་ལོ་ངོ་བརྒྱ་སྟོང་རིང་ཕྱི་ནང་གང་དུའང་རྒྱལ་མོ་རོང་གི་མི་རྣམས་ནི་བོད་རིགས་ཡིན་མིན་སྐོར་ལ་ལབ་གླེང་ཙམ་བྱུང་མྱོང་མེད་ཀྱང་།བར་སྐབས་ཤིག་ནས་དགོས་བ་དང་དགོས་གཞི་ཅི་འདྲ་ཞིག་ཡོད་མེད་མི་ཤེས་ཀྱང་མཁས་མིང་འཛིན་པ་རེ་འགའི་རྒྱལ་རོང་བ་རྣམས་ནི་ཧ་ལམ་བོད་རིགས་མིན་པར་མི་རིགས་གཞན་ཞིག་ཡིན་ཚུལ་གྱི་གླེང་ཕྱོགས་བྱུང་བ་རེད། གོ་ཐོས་སུ་མཁས་པ་དེ་ཚོས་རྒྱལ་རོང་གི་ཡུལ་སྐད་དང་བོད་ཀྱི་སྤྱི་སྐད་གཉིས་གཏན་ནས་མི་མཁྱེན་ནའང་རྒྱལ་རོང་གི་ཡུལ་སྐད་དང་བོད་ཀྱི་སྤྱི་སྐད་གཉིས་ཅུང་ཟད་མི་འདྲ་བ་ཙམ་གོ་ཐོས་སུ་བྱུང་བ་དེ་རྒྱུ་མཚན་དུ་བཀོད་དེ་ཁ་གཡར་ས་ཙམ་ཞིག་བྱེད་པ་ལས་གཞན་རྒྱུ་མཚན་དང་ལུང་རིགས་གང་ཡང་དྲངས་རྒྱུ་མེད།   དེ་ཡང་སྐད་བརྡ་མ་འཕྲོད་པ་ཙམ་གྱི་མི་རིགས་ཐ་དད་འདོད་པ་ནི་བྱིས་བློ་ཙམ་དང་མི་རིགས་གང་དེའི་སྐད་དང་ཡི་གེ   ཆོས་ལུགས་དད་མོས།ཡུལ་སྲོལ་གོམས་གཤིས་སོགས་ལ་ཆ་རྒྱུས་དང་ཞིབ་འཇུག་མེད་པའི་ཧོལ་གཏམ་ཙམ་དུ་འདོད།ཡང་གཅིག་བཤད་ན། བོད་རིགས་ལ་ཆ་རྒྱུས་ཡོད་པའི་མི་སུ་ཞིག་ཡིན་རུང་རྒྱལ་མོ་རོང་དུ་སོང་ན།  དེ་གའི་མི་དམངས་ཚོའི་བཟའ་བཏུང་གཙོ་བོ་མར་དང་།   རྩམ་པ།  ནོར་ལུག་གི་ཤ   བཏུང་བྱའི་གཙོ་བོ་ནས་ཆང་སོགས་ཡིན་པ་དང་། ཞྭ་གོས་ལྷམ་སོགས་གྱོན་ཆས་རིགས་དང་། གཡུ་བྱུར་སྤོས་ཤེལ་གསེར་དངུལ་སོགས་ལུས་ཀྱི་རྒྱན་ཆ་དང་། དུས་ཆེན་གྱི་སྐབས་ཕོ་མོ་རྒན་གཞོན་མེད་པར་གཟབ་སྤྲོས་དང་བཅས་མཉམ་དུ་འཛོམས་ཤིང་ཐང་དཀྱིལ་དུ་ནས་ཆང་རྫ་མ་མང་པོའི་ཁར་ཁ་བཏགས་ཀྱི་དཀྲིས་ཤིང་། ནང་དུ་སོག་མ་ཁ་ཤས་བཅུག་ཅིང་ཆང་མཆོད་ཀྱི་བཤད་པ་དང་ཕོ་མོ་རྒན་གཞོན་གྱི་ལག་རྡང་སྦྲེལ་དེ་སྐོར་བྲོ་འཁྲབ་ཞོར་དང་ཆང་འཐུང་ཞོར་བྱེད་པ་སོགས་རྒྱལ་རོང་མ་ཡིན་པའི་བོད་ཀྱི་ས་ཁུལ་གཞན་དག་ན་ཡང་ཧ་ཅང་རྒྱུགས་ཆེ་ཞིང་།  གཞན་མི་ཤི་རྗེས་སུ་ཨ་མཆོད་རྟེན་པ་དང་།  མཆོད་མེ་སྒྲོན་པ།   ཇོ་དར་འཐེན་པ་སོགས་ཆོས་ལུགས་ཀྱི་ཆོ་ག་སྒྲུབ་སྟེང་ལའང་བོད་ཁུལ་གཞན་དང་མི་མཐུན་པ་མེད། དེ་ལས་གཞན་རྒྱལ་རོང་གི་མི་རྣམས་ཀྱི་གདོང་དབྱིབས་ཆགས་ཚུལ་དང་པྭགས་མདོག་སོགས་མགོ་ལུས་ཕུང་གསུམ་གང་ལ་བལྟས་ཀྱང་བོད་རིགས་ཡིན་མིན་ལ་ཐེ་ཚོམ་ཟ་དགོས་བའི་གནས་སུ་གཏན་ནས་འགྱུར་མེད། འོན་ཀྱང་ཞིབ་འཇུག་བ་དང་མཁས་མིང་གི་ཐོབ་ཐང་འཛིན་འདོད་ཀྱི་མི་སྐོར་ཞིག་གི་དགོས་བ་དང་དགོངས་གཞི་ཁྱད་པར་བ་རེ་ལ་བརྟེན་དེ་ཐབས་ལམ་སྣ་ཚོགས་ཀྱིས་བསྙོན་དུ་མེད་པའི་བོད་རིགས་འདི་དག།བོད་རིགས་བྱིངས་ལས་ཁ་བྲལ་བཏང་སྟེ་ཕྱི་ཚུལ་དུ་མི་རིགས་གཞན་ཞིག་གམ་མི་རིགས་ཆེ་གེ་མོ་ཞིག་གི་མིང་དངོས་སུ་འདོགས་རྩིས་ཡོད་ཀྱང་། དེ་ནི་ཕྱི་ཚུལ་ཙམ་དང་གནས་སྐབས་ཙམ་ལས་རྒྱལ་མོ་རོང་གི་མི་དམངས་ཚུད་པའི་བོད་མི་ཕལ་མོ་ཆེའི་བློ་ལ་མི་འཐད་ངེས་པ་ཡིན་ལ།སེམས་ཀྱི་གཏིང་ནས་ནམ་ཡང་ཁས་ལེན་ཐབས་བྲལ་བ་ཡིན་ཞེས་མཁས་དབང་ཆེན་མོ་ཉན་ཤུལ་མཁྱེན་རབ་འོད་གསལ་དང་།བཙན་ལྷ་ངག་དབང་ཚུལ་ཁྲིམས་སོགས་ཀྱི་མགྲིན་གཅིག་དུ་འདོད་ཚུལ་དང་དགོངས་བ་སོགས་མཐུན་པར་བསྟན་ཡོད།   
          ༣  འབར་ཁམས་རྒྱལ་རོང་སྐད་ཀྱི་གྲུབ་ཚུལ་དང་གོ་བདེའི་ཁྱད་ཆོས།
          སྤྱིར་བོད་སྤྱིའི་སྐད་ཀྱི་དབྱེ་བ་ལས་འབར་ཁམས་རྒྱལ་རོང་གི་སྐད་དང་རྔ་བའི་འབྲོག་སྐད་གང་ཡིན་ཀྱང་། རང་ཉིད་ལ་གྲུབ་ཚུལ་དང་ལོ་རྒྱུས་ཀྱི་རྒྱབ་ལྗོངས་གསལ་ཐག་ཆོད་པ་རེ་ཡོད་ནི་གདོན་ཟ་མི་དགོས་དེ། དེ་ཡང་སྤྱི་ཚོགས་རྙིང་བའི་སྐབས་སུ་འགྲིམ་འགྲུལ་སོགས་ཀྱི་ཆ་རྐྱེན་ཤིན་ཏུ་ཞན་པའི་སྟབས་ཀྱི་ཞིང་སྡེ་མི་འདྲ་བ་དང་ཡུལ་ལུང་མི་འདྲ་བ་རྣམས་སུ་སྐད་ཀྱི་ཁྱད་པར་སྣ་ཚོགས་ཡོད་པས། སྒྲ་གདངས་འདོན་ཚུལ་སོགས་ལ་ཁྱད་པར་བྱུང་ནས་ཡུལ་སྐད་དང་ལུང་སྐད་མང་བོར་གྱུར་སོང་བ་དོན་དངོས་དང་མཐུན་པར་འདོད་མོད། འོན་ཀྱང་དེའི་དོན་ནི་མིང་ཚིག་ཉུང་ཤས་ཀྱི་སྒྲ་གདངས་འགའ་ཅུང་ཟད་འགྱུར་སོང་བའི་གནད་དོན་རེ་མ་གཏོགས་བོད་མི་རིགས་ཀྱི་སྐད་དེ་གཞན་མི་རིགས་ཤིག་གམ་རྒྱལ་ཁབ་ཤིག་གི་སྐད་རིགས་སུ་འགྱུར་སོང་བ་གཏན་ནས་མིན། དེ་ལ་དཔེ་བརྗོད་ཞིག་བཞག་ན་སྔོན་ཆད་གསང་སྔགས་རྙིང་མའི་གདན་ས་རྡོར་སྨིན་དཔལ་གསུམ་དང་ཀ་དཔལ་རྫོགས་གསུམ།དེ་བཞིན་དུ་སྔགས་གསར་མའི་གྭ་ས་ཆེན་པོ་སེ་འབྲས་དགེ་གསུམ་སོགས་དང་།གཞན་ཡང་བོད་ཀྱི་གྲུབ་མཐའ་ཁག་སོ་སོའི་གྲྭ་ས་ལ་ཕྱོགས་བཞི་མཚམས་བརྒྱད་ནས་ཡོང་བའི་དཔེ་ཆ་སློབ་གཉེར་བྱེད་མཁན་དགེ་འདུན་པ་འབུམ་ཕྲག་མང་བོ་འདུས་ཡོད་པ་དེ་ཚོས་བར་དུ་སྐད་བརྡ་མ་འཕྲོད་པར་ཕན་ཚུན་བར་དུ་ལོ་ཙཱ་བ་ཞིག་བཞག་རབས་ལོ་རྒྱུས་ཐོག་བྱུང་མེད། དེ་ཡང་རང་གི་ཉེ་འགྲམ་གྱི་དཔེ་བརྗོད་ཅིག་བཞག་ན་རང་སློབ་གྲྭའི་(སི་ཁྲོན་མི་རིགས་སློབ་གྲྭ་ཆེན་མོ)བོད་ཡིག་སྡེ་ཁག་དུ་གཙང་དང་།   ལྷ་ས།   ཀོང་པོ།  ལྷོ་ཁ། ནག་ཆུ།  ཆབ་མདོ།   མཚོ་སྔོན། ཀན་ལྷོ། རྔ་བ། སྡེ་དགེ འབའ་བ།   རྒྱལ་རོང་སོགས་བོད་ཀྱི་ས་ཁུལ་མང་པོ་ནས་ཡོང་བའི་སློབ་མ་ཡོད་ཅིང་། ང་ཚོ་ཐོག་མར་འཕྲད་སྐབས་ཕན་ཚུན་བར་བརྡ་འཕྲོད་པར་ཅུང་དཀའ་བའི་སྣང་བ་གཏེར་རུང་ཡུན་རིང་འགོར་མི་དགོས་བར་སློབ་གྲོགས་ཕན་ཚུན་བར་བོད་སྐད་ཀྱི་ཐོག་ནས་གླེང་མོལ་ཅི་འདོད་དུ་བྱེད་ཐུབ་པ་མ་ཟད།  རང་གི་སྣང་བ་རུ་བོད་ཀྱི་ཡུལ་སྐད་དེ་དག་ནི་སྔོན་ཆད་རང་གི་བཤད་མྱོང་ལ་བར་སྐངས་ཤིག་ནས་བརྗེད་སོང་བའི་སྐད་ཞིག་ཡང་བསྐྱར་བཤད་པའི་ཚོར་བ་ཞིག་ངང་ངམ་ཤུགས་ཀྱི་གཏེར་བ་ལས།   མི་རིགས་གཞན་ཞིག་གི་སྐད་ཀྱི་གླེང་མོལ་བྱེད་པ་དང་གཏན་ནས་མི་འདྲ་བ་སོ་སོར་སྣང་།དེ་མ་ཟད་རང་སློབ་གྲྭའི་དགེ་རྒན་རྣམས་ཀྱི་མདོ་དབུས་ཁམས་གསུམ་གྱི་སྐད་གང་གི་ཐོག་ནས་སློབ་ཁྲིད་བྱས་ཀྱང་ང་ཚོའི་སླ་མོར་གོ་ཐུབ་པ་དང་རྟོགས་ཐུབ་པ་དེ་བཞིན་ཡིན།
          གཉིས།  རྔ་བའི་འབྲོག་སྐད་སྐོར་རགས་ཙམ་གླེང་བ།
          ༡ འབྲོག་སྐད་བེད་སྤྱོད་བྱེད་པའི་ས་ཁུལ་དང་རྫོང་།
          སྤྱིར་ས་ཁུལ་གྱི་སྐད་དམ་ཡུལ་སྐད་ཅེས་པའི་ཆེས་གལ་ཆེ་བའི་བསྟན་དོན་ནི་ས་ཁུལ་ལམ་ཡུལ་ཁམས་ཀྱི་འགྱུར་བ་ལས་བྱུང་བའི་སྐད་ཀྱི་སྣང་ཚུལ་མི་འདྲ་བ་དེ་ཡིན་པས།  རྔ་བའི་འབྲོག་ཁུལ་ཀྱི་ཡུལ་སྐད་བཤད་པ་ལ།  ཐོག་མར་རྔ་བའི་འབྲོག་ཁུལ་གྱི་མངའ་ཁོངས་དག་གསལ་བོ་ཤེས་དགོས །དེར་འབྲོག་ལས་གཉེར་ཞིང་འབྲོག་སྐད་བཤད་བཞིན་པའི་རྫོང་ཁག་བཞི་ཡོད་དེ །  སོ་སོ་ནི་རྔ་བ་རྫོང་（阿坝县）དང་ཁྱུང་མཆུ་རྫོང་（红原县）གཞན་ད་དུང་མཛོད་དགེ་རྫོང་（若尔盖县）ཛམ་ཐང་རྫོང་（壤塘县）སོགས་ཡོད་ཅིང་།   མཁས་དབང་ཐུབ་བསྟན་ཕུན་ཚོགས་ཚང་གི་《བོད་རིག་པ་སྤྱི་བཤད》ཅེས་པར་གཞིག་ན།  ཕལ་ཆེར་ས་ཁུལ་གྱི་གནས་ཕྱོགས་ནི་ཤར་ཐིག་དཔག་ཚད་༡༠༠དང་ཆ་ཕྲན་༣༠དང་དཔག་ཚད་༡༠༤དང་ཆ་ཕྲན་༢༧བར་གནས་ཤིང་།  བྱང་ཐིག་དཔག་ཚད་༣༠དང་ཆ་ཕྲན་༣༥་ནས་དཔག་ཚད་༣༤དང་ཆ་ཕྲན་༡༩བར་གནས་ཡོད ། རྫོང་དེ་དག་གི་རྒྱ་ཁྱོན་སྤྱི་ལེ་གྲུ་བཞི་མ་ཁྲི་གཉིས་ལྷག་ཟིན་ཡོད་ལ།མི་འབོར་འབུམ་གཅིག་ཙམ་འཚོ་སྡོད་བྱས་ཡོད་ཅིང་། ཕྱི་སྡོད་མི་སྣ་རེ་འགའ་མ་གཏོགས་ཚང་མ་བོད་མི་རྐྱང་རྐྱང་ཡིན།
          ༢ འབྲོག་སྐད་ཀྱི་་ཁྱད་ཆོས་སོགས་གླེང་བ།
          རྔ་བའི་འབྲོག་སྐད་ཅེས་གླེང་དགོས་བའི་རྒྱུ་མཚན་ཡང་གོང་གསལ་གྱི་འབྲོག་རྫོང་དག་གི་བེད་སྤྱོད་བྱེད་བཞིན་པའི་སྐད་ཡིན་ལ། སྤྱིར་འབྲོག་ཅེས་པའི་ནང་དོན་ནི་གྲོང་ཁྱེར་དང་གྲོང་སྡེ་ཆགས་མེད་པའི་ས་ཆ་སྟེ། འབྲོག་སྟོང་དང་འབྲོག་དགོན་པ་སོགས་ཟེར་སྲོལ་ཡོད་མོད། འོན་ཀྱང་འདིར་བརྗོད་བཞིན་པའི་འབྲོག་ནི་ཕྱུགས་འཚོ་སའི་རྩྭ་ས་དང་ཕྱུགས་ལས་གཙོ་བྱེད་པའི་ས་ཆའི་མིང་ལ་གོ་དགོས།  དེ་བཞིན་འབྲོག་པ་ནི་ཕྱུགས་ལས་གཙོ་བྱེད་པ་ལ་བརྟེན་ནས་འཚོ་བའི་མི་ལ་ཟེར་ཞིང་་་་། འབྲོག་སྐད་ནི་མི་དེ་དག་གི་བེད་སྤྱོད་བྱེད་བཞིན་པའི་སྐད་ལ་ཟེར་བ་ཡིན། དེ་ཡང་རང་གི་འདོད་ཚུལ་དུ་སྤྱིར་བོད་ཀྱི་སྐད་ལ་རོང་འབྲོག་གི་སྐད་ཅེས་དབྱེ་རྒྱུ་ཡོད་པ་ལས།  མདོ་དབུས་ཁམས་གསུམ་གྱི་སྐད་ཅེས་དབྱེ་བ་བྱེད་པ་ལ་འཐད་པ་ཆེར་མེད།  རྒྱུ་མཚན་ནི་བོད་ཀྱི་ལོ་རྒྱུས་སུ་རྒོད་གཡུང་གཉིས་དེ་དམག་མི་དང་འབངས་མི་་གཉིས་འཚོ་སྡོད་བྱེད་པའི་ཁུལ་དབྱེ་བ་ལས། ལྷ་ས་ནི་གཙོ་བོ་བོད་འབངས་འཚོ་སྡོད་བྱེད་པའི་གནས་གཙོ་བོ་ཡིན་ལ། བཀོལ་སྤྱོད་བྱེད་བཞིན་པའི་་སྐད་ལ་ཡང་ཞི་ཞིང་གཉོམ་པ།  ཞེ་ས་མང་བའི་ཁྱད་ཆོས་མངོན་གསལ་དོད་པོའི་རང་བཞིན་ལྡན་ལ། རྔ་བ་ལ་སོགས་པའི་མཐའ་ཁུལ་ཀྱི་སྐད་ནི་སྔོན་ཆད་མཐའ་ཁུལ་སྲོང་བའི་དམག་མིའི་རྒྱུད་ནས་ཆད་པ་ཡིན་ཅིང་། བཤད་པའི་སྐད་ལ་དྲགས་ཅིང་རྩུབ་རེགས་ཆེ་བའི་ཁྱད་ཆོས་ཀྱི་ཕྱུགས་པས། དེ་སྔོན་ཆད་དམག་སྐད་ཡིན་པར་ཐེ་ཚོམ་་དགོས་བའི་གནས་སུ་འགྱུར་མེད །  
          སྤྱིར་བོད་སྐད་ནི་བོད་ཡུལ་དུ་བེད་སྤྱོད་བྱེད་པའི་སྐད་དེ། རོང་འབྲོག་གཉིས་ལས་འབྲོག་པའི་སྐད་ནི་འབྲོག་སྡེ་སྟོད་སྨད་བར་གསུམ་ཀུན་ཏུ་ཁྱད་པར་མེད་ཅིང་། སྒྲ་གདངས་དྲག་ཅིང་གྱོང་ཉམས་ཆེ་ལ་སྔོན་གྱི་སྒྲ་སྦྱོར་གྱི་ཟུར་མ་ཉམས་པར་ཐོན་པ་དང་། ཞིང་སྡེའི་ཡུལ་ལུང་རྣམས་སུ་སྐད་ཀྱི་ཁྱད་པར་སྣ་ཚོགས་ཡོད་པས་གདངས་འདོན་ཚུལ་སོགས་གཅིག་ཏུ་མ་ངེས་ཏེ།   འབར་ཁམས་རྒྱལ་རོང་གི་སྐད་ལ་ཡང་དེའི་ཁྱད་ཆོས་མངོན་གསལ་དུ་སྣང་ཏེ།  ལུང་བ་རེ་ལ་སྐད་ཆ་རེའི་སྣང་ཚུལ་ཡོད་པར་སྣང་ཡང་།  དོན་དངོས་སུ་སྐད་དེ་དག་ཀྱང་ཕན་ཚུན་བར་གོ་བརྡ་མི་འཕྲོད་པའི་དཀའ་ངལ་མེད་པ་མ་ཟད། དེ་དག་ནི་གཅིག་ལ་གཅིག་བརྟེན་གྱི་ཚུལ་ཏུ་གནས་ཡོད། ད་ཆར་ཡོངས་གྲགས་སུ་བོད་ཀྱི་ཡུལ་གྲུ་སྟོད་སྨད་བར་གསུམ་ལ་ལྟོས་ཏེ་ནང་གསེས་རིགས་གསུམ་དུ་ཕྱེ་བ་སྟེ། སྟོད་དབུས་གཙང་གི་སྐད་དང་། བར་ཁམས་པའི་སྐད་དང་། སྨད་ཨ་མདོའི་སྐད་རྣམས་སུ་བཤད་མོད།འོན་ཀྱང་གོང་གསལ་བཤད་ཟིན་པ་ནང་བཞིན། བཤད་ཚུལ་འདིའི་རིགས་ལ་ཚད་མའི་གྲུབ་པའི་ཁོངས་ལུང་གི་ཆ་ཉུང་བར་འདོད་པ་སོགས་རྒྱལ་རོང་གི་སྐད་དང་རྔ་བའི་འབྲོག་སྐད་བཤད་པའི་ཞོར་ནས་གླེང་བ་ཙམ་ཡིན། རྒྱུ་མཚན་ནི་ས་ཆ་གཞན་ན་བེད་སྤྱོད་བྱེད་བཞིན་ཡོད་པའི་བོད་སྐད་ལ་རོང་འབྲོག་གཉིས་ཀྱི་སྐད་ལས་མདོ་དབུས་ཁམས་གསུམ་སོགས་ཀྱི་སྐད་ཅེས་སོ་སོར་མེད་པ།   དཔེར་ན། བོད་སྐད་ནི་ཧོར་སོག་ལ་སོགས་་གྲངས་ཉུང་མི་རིགས་གཞན་པ་ཁག་ཅིག་གིས་བེད་སྤྱོད་གཏོང་བཞིན་ཡོད་པར་མ་ཟད། བལ་པོ་དང་། འབྲས་ལྗོངས། འབྲུག་ཡུལ་བཅས་ཀྱི་རྒྱལ་ཁབ་རྣམས་སུའང་བེད་སྤྱོད་གཏོང་བཞིན་ཡོད་པའི་བོད་སྐད་དེ་ལ་བལྟས་ན་ཤེས་བར་འདོད། 
          གསུམ།  འབར་ཁམས་རྒྱལ་རོང་གི་སྐད་དང་རྔ་བའི་འབྲོག་སྐད་གཉིས་ཀྱི་འབྲེལ་བ་བཤད་པ།
          སྤྱིར་བོད་སྐད་དེ་ཡང་རོང་སྐད་དང་འབྲོག་སྐད་གཉིས་སུ་དབྱེ་ཡོད་ལ། དེ་ཡང་གོང་ནས་བཤད་ཟིན་པ་བཞིན་རྒྱལ་རོང་ནི་བོད་ནས་ཆད་སྔ་ཞིང་བར་སྐབས་སུ་འབྲེལ་འདྲེས་ཧ་ཅང་ཉུང་བ་སོགས་ཀྱི་དབང་གི་བོད་བྱིངས་གི་སྐད་དང་མི་འདྲ་བར་གྱུར་པ་ནི་བོད་སྐད་ཀྱི་བརྡ་རྙིང་ཤིན་ཏུ་མང་བ་དག་སོར་གནས་དང་རང་ས་ཁོ་ན་ནས་རིམ་གྱི་འགྱུར་ཡིན་ཤས་ཆེ་བས། འབར་ཁམས་རྒྱལ་རོང་གི་སྐད་ཡང་རོང་ཆེན་གཞན་དག་གི་སྐད་ནང་བཞིན་འགྱུར་ཡོད་པ་མ་ཟད།  རྔ་བའི་འབྲོག་སྐད་ནང་གི་བརྡ་གསར་རྙིང་གི་ཐ་སྙད་མང་བོ་ཞིག་དང་འབྲེལ་བ་ཡོད་ཅིང་།   དེ་ཇི་ལྟར་འགྱུར་ཚུལ་སྒྲ་གདངས་དྲག་ཆེ་བའི་དབང་གི་འགྱུར་བར་མངོན་ཏེ། དཔེར་ན།མིག་ལ་དུ་རུ་མྱག ཁ་ལ་དུ་རུ་ཁ། མཆུ་ལ་དུ་རུ་མཆུ། རྣ་ལ་དུ་རུ་རྣ། སྣ་ལ་དུ་རུ་སྣ། འཛུབ་གུ་ལ་དུ་འཛུས་རོས།   ལག་པ་ལ་དུ་ཡག སྐེ་ལ་དུ་རུ་ཀེ་སོགས་དང་། གཞན་ཡང་ཕག་ལ་པག་དང་ཁྱི་ལ་ཁི་ཟེར་བ་སོགས་ལྟ་བུ་ལ་ཞིབ་མོར་བརྟག་ན་སྒྲ་གདངས་གི་དྲག་ཞན་དང་ཡིག་གེའིགཟུགས་ཀྱི་འགྱུར་ཚུལ།  གཞན་ད་དུང་བོད་ཀྱི་བརྡ་སྤྲོད་གྲུབ་ཚུལ་སོགས་ལ་ཞིབ་མོར་དཔྱད་ན་མ་གཞིའི་འགྱུར་གཞི་གཅིག་པ་དེ་ང་ཚོའི་ཤེས་ཐུབ། དེ་མིན་ད་དུང་འབར་ཁམས་རྒྱལ་རོང་གི་ཡུལ་སྐད་དམ་ལུང་སྐད་མང་བོ་ཞིག་ཏུ་མགོ་ཅན་སོ་སོའི་སྒྲ་གདངས་དང་། སྔོན་འཇུག་ལྔའི་སྒྲ་གདངས་སོགས་ཕན་ཚུན་མ་འདྲེས་བར་གསལ་བར་ཐོན་ཡོད་པ་དེའི་རྔ་བའི་འབྲོག་སྐད་ཀྱི་ཁ་སྐད་རང་བཞིན་དེ་མངོན་པར་མཚོན་ནུས་པའི་ཁྱད་ཆོས་འདྲ་མཚུངས་དུ་སྣང་། དཔེར་ན། རྒྱལ་རོང་སྐད་དུ་འབྲུག་སྨན་ཞེས་འབོད་སྐབས་སྨན་གྱི་ས་མགོ་དེ་གསལ་བར་ཐོན་པས་དབང་གི་མིང་དེ་རྒྱ་ཡིག་གི་བྲིས་དེ་ཀློག་དུས་དེའབྲུག་སི་སྨན་ལྟ་བུར་འགྱུར་བ་དང་།   དེ་བཞིན་དུ་ཨ་སྨན་དང་ཨ་སྐྱབས་སོགས་སོ་སོར་ཨ་སི་སྨན་དང་ཨ་སི་ཅའ་ལྟ་བུའི་སྒྲ་རུ་འབོད་བཞིན་ཡོད།  ཡང་སི་ཏུ་པཎ་་ཆེན་གྱི་ཞལ་ནས་“གཞན་ཡང་བོད་ཡིག་སྡེབ་སྦྱོར་ཇི་ལྟར་སྦྱོར་བའི་ཚུལ་ནི་སྔོན་གྱི་སྐད་དག་པ་ཉིད་ལ་བརྗོད་པའི་སྒྲ་ཇི་ལྟར་སྦྱོརའབྱུང་བ་དེ་ལྟར་མཁན་པོ་འདིས་སྦྱར་བ་ཡིན་པས་སྐད་ལ་འཕུལ་རྟེན་སོགས་ཀྱི་ཟུར་གསལ་བ་ མེད་ན་ཡི་གེ་ལ་དེ་ལྟར་ཐོབ་པ་རྒྱུ་མཚན་གང་ཞིག་གིས་རྟོགས་པར་ནུས”○༡ཞེས་པའི་དོན་ལ་ཞིབ་མོར་བརྟག་ན།  དེ་ཡང་། མཁས་དབང་བཙན་ལྷ་ངག་དབང་དང་རྡོ་རྗེ་སོགས་ཀྱི་བཞག་པའི་དཔེ་བརྗོད་ལྟར་“དཔེར་ན་། ས་སྐྱ་བའི་བོད་ལ་དབང་སྒྱུར་བྱེད་པའི་སྐབས་འགྲོ་མགོན་ཆོས་རྒྱལ་འཕགས་པའི་མཚན་ལ་རྒྱ་སྐད་དུ་བསྒྱུར་བར་“帕思巴”་ཞེས་པ་བོད་་སྐད་ཀྱི་འཕགས་པ་ཞེས་པའི་ཚེག་ཁྱིམ་གཉིས་ལྡན་གྱི་མིང་སྒྲ་སྒྱུར་བྱས་པར་རྒྱ་ཡིག་ཏུ་ཡིག་འབྲུ་གསུམ་དུ་འགྱུར་བའི་རྒྱུ་མཚན་ནི་།སྐབས་དེར་བོད་སྐད་ཀྱི་སྒྲ་ལ་སྔོན་འཇུག་དང་རྗེས་འཇུག་་མགོ་ཅན་དང་འདོགས་ཅན་སོགས་ཀྱི་སྒྲ་གསལ་བོ་ཡོད་པ་མ་ཟད་ཡང་འཇུག་གི་སྒྲའང་གསལ་པོ་ཡོད་པ་མངོན་སུམ་གྱི་གྲུབ་པ་ཡིན།  གལ་ཏེ་དེ་ལྟར་མིན་ན་དེང་སྐབས་འཕགས་པ་ཞེས་པའི་མིང་དེ་རྒྱ་ཡིག་དུ་སྒྲ་སྒྱུར་བྱེད་སྐབས་“帕巴”ཞེས་ཡིག་འབྲུ་གཉིས་ལས་མི་འབྱུང་བའི་རྒྱུ་མཚན་ཅི་ཞིག་ཡོད་”○༢ཅེས་དཔེ་བཀོད་པ་བཞིན་ཡར་སྔོན་གྱི་བོད་ཡིག་དང་སྔར་གྱི་བོད་སྐད་ནི་དེ་ལྟ་བུ་ཞིག་ཡིན་པ་མ་ཟད།   དེའི་སྣང་ཚུལ་མང་བོ་ད་ལྟ་ཡང་བོད་ཀྱི་ས་ཁུལ་སོ་སོའི་ཡུལ་སྐད་དང་ལུང་སྐད་མང་བོའི་ནང་སྔོན་གྱི་སྒྲ་གདངས་དེ་ཁྱབ་བརྡལ་དུ་སོང་ཡོད་པ་ང་ཚོའི་བློ་ཞིབ་མོའི་དཔྱད་པ་བྱས་དེ་ཤེས་སུ་འཇུག་རུང་བ་ཡིན །
          དེ་ཡང་ཁོང་ཚོས་བོད་ཡིག་གི་འབྲི་འདེད་བྱེད་ལུགས་དང་ཀློག་སྟངས་སྐོར་ལ་དབྱངས་གསལ་དང་མགོ་ཅན་འདོགས་འདོགས་ཅན་འཕུལ་རྟེན་བཅས་ཀྱི་སྒྲ་ཟུར་ཤིན་དུ་དག་པོ་ཡོད་ཅེས་སི་ཏུའི་༼སུམ་རྟགས་འགྲེལ་ཆེན་ ༽  ནང་དུ་གསུངས་ཡོད་པ་དེ་བཞིན་དུ། མགོ་སྐྲ་ལ་ས་ཀ་ར།  མཁྲིས་པ་ལ་དེ་མ་ཁ་རིས།  སོགས་ད་ལྟའི་བོད་སྤྱིའི་ཡི་གེ་ལྟར་ན་སོ་སོར་འཕྲལ་ནས་འདོན་རྒྱུ་མིན་པར་ཚེག་བར་གཅིག་གི་ནང་དུ་འཕུལ་རྟེན་མགོ་འདོགས་སོགས་ཀྱི་སྒྲ་ཟུར་ཙམ་ཐོན་པ་ཞིག་ཀློག་གིན་ཡོད་དེ། ཡང་ཕྱིས་ཀྱི་སྐདགསར་བཅད་ཀྱི་ཁྲོད་དུ་བསྒྱུར་བར་འདོད་དེ། རྒྱ་རོང་བོད་མིའི་སྐད་ནི་ཆེས་སྔ་མོའི་བོད་ཀྱི་སྐད་ཡིན་ཅིང་། གོང་དུ་བཤད་པ་བཞིན་ཕྱིས་སུ་འབྲེལ་འདྲེས་སོགས་ཕྱི་ནང་གི་རྒྱུ་རྐྱེན་མང་བོའི་དབང་གི་རྒྱལ་རོང་སྐད་ལ་སྐད་གསལ་བཅད་ཀྱི་ཁྲོད་བཟོ་བཅོས་མ་བྱས་པའི་སྣང་ཚུལ་དང་།  བཟོ་བཅོས་བྱས་ཀྱང་ཁྱབ་རྡལ་དུ་མ་སོང་བ་དེ་གསལ་བར་མངོན་འདུག ། དེ་ཡང་དཔེར་ན་དུ་སྔེ་དང་དུ་མྱག དུ་ཀེ་དང་དུ་ཛུ་རུས་སོགས་ལྟ་བུར་མཚོན་ན་སོ་སོར་བོད་སྤྱིའི་སྐད་དུ་ངོ་དང་མིག་སྐེ་དང་འཛུབ་གུ་སོགས་ཀྱི་དོན་ཡིན་ཅིང་། དེ་དག་གི་མིང་ཚིག་གྲུབ་སྟངས་ཀྱི་ཁྱད་ཆོས་ཡང་བོད་ཀྱི་ཡི་གེ་ལ་ཆ་རྒྱུས་ཅུང་ཟད་ཡོད་མཁན་ཅིག་ཡིན་ན་བལྟས་བ་ཙམ་གྱི་མཐོང་ཐུབ་ལ། དེ་དག་སྔ་ཕྱིའི་མིང་ཚིག་སོ་སོའི་མིང་གཞི་ལ་འགྱུར་བ་བཏང་མེད་པ་དེར་བལྟས་ན་ཡང་དེ་དག་གི་འགྱུར་གཞི་གཅིག་པ་དེ་ཡང་བསྐྱར་བདེན་དཔང་བྱེད་ཐུབ།            
          གཅིག དཔྱད་གཞིའི་ཡིག་ཆ།
༡ ༼   ཞང་ཞུང་ཤར་མཐའ་རྒྱལ་རོང་གནའ་བོའི་ལོ་རྒྱུས་ཀུན་གསལ་མེ་ལོང་།   ༽   རྩོམ་པ་པོ། རྒྱལ་རོང་བ་བསྟན་པའི་རྒྱལ་མཚན། དཔེ་སྒྲིག་མཁན། རྒྱལ་རོང་གནས་འགོ་དགོན་གྱི་རིག་གནས་ཀྱི་ཕྲེང་སྟར་དཔེ་ཚོགས་རྩོམ་སྒྲིག་ཚོགས་པའི་བསྒྲིགས།
༢  ༼    མདོ་སྨད་ཞིབ་འཇུག    ༽  ཀན་སུའུ་བོད་ཀྱི་ཤེས་རིག་ཞིབ་འཇུག་ཁང་གིས་བསྒྲིགས། མི་རིགས་དཔེ་སྐྲུན་ཁང་པར་དུ་སྐྲུན་བྱས།
༣ ༼    རྒྱལ་རོང་ཞིབ་འཇུག་དཔྱད་ཡིག་ཕྱོགས་བསྒྲིགས    ༽    ཀྲུང་གོའི་བོད་རིག་པ་དཔེ་སྐྲུན་ཁང་།  བཙན་ལྷ་ངག་དབང་དང་རྡོ་རྗེ།  ཏཱ་རེ་སྐྱིད་བཅས་ཀྱིས་བསྒྲིགས།
༤ ༼    དག་ཡིག་གསལ་བསྒྲིགས    ༽   མི་རིགས་དཔེ་སྐྲུན་ཁང་།   ཤོག་ངོས་༥༦༨སྟེང་དུ་གསལ།
          གཉིས།  ཟུར་ལྟའི་ཡིག་ཆ།
༡ ༼      དམུ་དགེ་བསམ་གཏན་གྱི་དཔྱད་རྩོམ་ཕྱོགས་བསྒྲིགས     ༽ 
༢ ༼     བོད་ཀྱི་ཡུལ་སྐད་ཕྱོགས་བསྒྲིགས     ༽
༣  ༼     ནམ་མཁའི་ནོར་བུའི་གསུང་རྩོམ་ཕྱོས་བསྒྲིགས    ༽
༤  ༼     བོད་རིགས་པ་སྤྱི་བཤད    ༽
༥  ༼     ཀྲུང་གོའི་བོད་རིགས་པ    ༽
༦ ༼     ཁམས་པའི་འཛུམ་རིས     ༽ 
༧ ༼     ཉན་ཤུལ་མཁྱེན་རབ་འོད་གསལ་གྱི་གསུང་རྩོམ་ཕྱོས་བསྒྲིགས    ༽
