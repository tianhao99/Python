བོད་ཁམས་ཀྱི་རྩྭ་དང་རྩི་ཤིང་།
༡ དབྱར་རྩྭ་དགུན་འབུ། 冬虫夏草 Cordyceps sinensis(Berk)Sassདབྱར་རྩྭ་དགུན་འབུ་ནི《མན་ངག་བྱེ་བ་རིང་བསྲེལ》ལས། “དབྱར་རྩྭ་དགུན་འབུ་ཞེས་བྱ་བ།རྩྭ་རི་སྒམ་པོ་ལ་སོགས་ཀྱི།རི་བོ་དབེན་པའི་ལྗོངས་སུ་སྐྱེ།།རི་བོ་དབེན་པའི་ལྗོངས་སུ་སྐྱེ།།རྩ་བ་འབུལ་ལ་ལོ་མ་རྩྭ།།”ཞེས་གསུངས་པ་ལྟར།མཚོ་ངོས་ལས་མཐོ་ཚད་སྨི་༤༣༠༠ ནས་སྨི་༥༠༠༠ བར་གྱི་འབྲོག་ཁུལ་གང་དུའང་ཐར་ཐོར་ཡོད་ཅིང་།མང་ཆེ་བ་ཁ་མཐོའི་སྲིབ་རིའི་སྤང་དུ་འཁྲུངས་ཤིང་ལོ་གཅིག་སྐྱེ་བའི་རྩྭ་རིགས་ཏེ།དགུན་དུས་སུ་ས་གཏིང་དུ་ཡོད་པའི་འབུ་མེ་ལྕེབ་ཕྱེ་མ་ལེབ་རིགས་ཀྱི་འབུ་ཕྲུག་སྟེ།དེའི་ལུས་ལ་འབུ་ཕྲ་ཞུགས་ནས་ཤི་བའི་ལུས་སླར་ཡང་སྐྱེ་དངོས་ཀྱི་རྩྭ་རུ་གྱུར་ནས་སྐྱེས་པ་ཡིན་པས།ཤ་རྒྱུའི་རྩ་བ་ཕྱི་སེར་ལ་ནང་དཀར་བ་འཕྲེད་གཉེར་དང་ལྡན་པ།མགོ་དང་ཁ་ཡོད་པར་མ་ཟད། ལྟོ་བར་རྐང་ལག་ཆ་བརྒྱད་ཡོད་པ་སོགས་འབུ་ཡི་ལུས་དང་འདྲ་ལ།ལྟག་པའི་རྒྱུད་དུ་སྤུ་ཐུང་ནག་ཞིབ་ཐར་ཐོར་ཡོད་པ།ལུས་ཚད་ཕལ་ཆེར་སྨི་༠.༠༥ ནས་སྨི༠.༠༦ ཙམ་ཟིན་པ་སའི་ནང་དུ་ཐད་ཀར་ཟུག་ཡོད།     སྤྱི་ཟླ་ལྔ་པ་ནས་དྲུག་པར་དེ་ལ“འབུ་རྭ”ཞེས་པ་སྐྱེ། འབུ་རྭ་ནི་དོན་དུ་ཤ་མོའི་རིགས་ཡིན་ལ་རྫ་སྨུག་གམ་ཁམ་ནག་སྟོད་སྦོམ་ཞིང་སྨད་ཕྲ་བ་ནང་ཤ་དཀར་པོ་ཅན་ཏེ།རིང་ཚན་ལ་སྨི༠.༠༣ ནས་སྨི་༠.༠༦ ཙམ་ཡོད། ཟླ་བ་༧ པ་ཙམ་ནས་རྩ་བའི་བཅུད་དམ་ནུས་པ་ཉམས་ནས་རུལ་ཏེ་མེད་པར་འགྲོ། རྩ་སྡོང་ཡོངས་རྫོགས་སྨན་དུ་སྦྱོར་ཏེ།ནུས་པའི་བཅུད་ལེན་ཟུང་འཕེལ། ཁྱད་པར་མཁལ་དྲོད་གསོ་ཞིང་མཁལ་སྟོབས་སྐྱེད།རོ་ཙ་སྐྱེད་ཅིང་ཐིག་ལེ་འཛག་གཅོད་སྲུང་།གློ་གཅོང་དང་བད་རླུང་གློ་བར་ཞུགས་པ།   དབུགས་མི་བདེ་བ་བཅས་སེལ་ཞིང་།རྒན་བྱིས་དང་ཟུངས་ཟད།རླུང་ལྡན་རྣམས་ལ་བདུད་རྩིར་མཚུངས་པར་གསུངས།སྨན་དུ་སྦྱོར་བ་ལས་གཞན་བྱ་ཤ་སོགས་སུ་གདུས་ཏེ་བསྟེན་པས་ཀྱང་ལུས་ཟུངས་གསོ་ཞིང་རོ་ཙ་ལ་ཕན་པར་བཤད།  དབྱར་རྩྭ་དགུན་འབུ་ནི་ཚོང་རར་བྲིན་ཆེ་ལ་གོང་ཡང་ཤིན་ཏུ་ཆེ་སྟེ།དུས་རབས་བདུན་ཅུ་པར་རྒྱ་མ་རེར་སྒོར་༡༠ ཡས་མས་དང་།ལོ་རབས་བརྒྱད་ཅུ་བར་དབྱར་རྩྭ་སྤུས་འབྲིང་རྒྱ་མ་གང་གི་རིན་གོང་སྐོར་༡༠༠ ཡན་ལ་སླེབས།ལོ་རབས་དགུ་བཅུ་བར་སྒོར་སྟོང་ལས་བརྒལ་ཞིང་། ༢༠༠༠ ལོ་ནས་བཟུང་བོད་ཁམས་ཚོང་རར་དབྱར་རྩྭའི་རིན་གོང་དམངས་སྒོར་ཁྲིའི་ཡན་ལ་བུད་ཅིང་། ཉེ་ལམ་དབྱར་རྩྭའི་བོངས་ཆེ་ཆུང་གི་དབང་གིས་རིན་གོང་དམངས་སྒོར་ཁྲི་གཉིས་ནས་ཁྲི་དྲུག་ལ་སླེབས་པས་རོང་འབྲོག་མང་ཚོགས་ཀྱི་ཡོངས་འབབ་གཙོ་བོ་ཞིག་ཏུ་གྱུར་ཡོད།
