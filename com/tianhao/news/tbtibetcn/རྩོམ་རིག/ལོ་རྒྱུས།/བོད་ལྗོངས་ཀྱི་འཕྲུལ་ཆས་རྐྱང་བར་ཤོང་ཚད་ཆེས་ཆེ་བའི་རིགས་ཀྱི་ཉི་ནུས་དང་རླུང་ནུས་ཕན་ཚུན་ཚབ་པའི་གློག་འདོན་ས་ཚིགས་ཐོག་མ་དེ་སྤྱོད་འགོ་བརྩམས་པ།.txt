བོད་ལྗོངས་ཀྱི་འཕྲུལ་ཆས་རྐྱང་བར་ཤོང་ཚད་ཆེས་ཆེ་བའི་རིགས་ཀྱི་ཉི་ནུས་དང་རླུང་ནུས་ཕན་ཚུན་ཚབ་པའི་གློག་འདོན་ས་ཚིགས་ཐོག་མ་དེ་སྤྱོད་འགོ་བརྩམས་པ།
གསར་འགོད་པས་ཉེ་ལམ་བོད་ལྗོངས་ཀྱི་ནུས་ཁུངས་དཔེ་སྟོན་ཞིབ་འཇུག་ལྟེ་གནས་ལས་ཤེས་རྟོགས་བྱུང་བ་ལྟར་ན། བོད་ལྗོངས་ཀྱི་འཕྲུལ་ཆས་རྐྱང་བར་ཤོང་ཚད་ཆེས་ཆེ་བའི་རིགས་ཀྱི་ཉི་ནུས་དང་རླུང་ནུས་ཕན་ཚུན་ཚབ་རེས་བྱེད་པའི་གློག་འདོན་ས་ཚིགས་ཐོག་མ་དེ་ནག་ཆུ་ས་ཁུལ་ཤོ་མོང་ཤང་གི་གྲོང་ཚོ་དང་པོར་ལེགས་གྲུབ་བྱུང་ཞིང་དངོས་སུ་སྤྱོད་འགོ་བརྩམས་ཡོད་པ་རེད།  
མཚེའུ་གཉིས་དམིགས་བསལ་ཁུལ་གྱི་འབྲོག་མོ་འགས་ཉི་ནུས་དང་རླུང་ནུས་ཕན་ཚུན་ཚབ་རེས་ཀྱི་གློག་འདོན་ས་ཚིགས་འགྲམ་དུ་འོ་མ་འཇོ་བཞིན་པ། 
    གློག་འདོན་ས་ཚིགས་འདིའི་ལས་གཞི་ནི་བོད་རང་སྐྱོང་ལྗོངས་ཚན་རྩལ་ཐིང་གིས་གྲོས་འགོ་བཏོན་ཞིང་ཆོག་མཆན་བཀོད་པ་བརྒྱུད། བོད་ལྗོངས་ནུས་ཁུངས་དཔེ་སྟོན་ཞིབ་འཇུག་ལྟེ་གནས་ཀྱིས་འགན་ཁུར་ནས་བསྐྲུན་པ་དང་། སྤྱིའི་མ་དངུལ་ལ་སྒོར་ཁྲི་༢༡༠བཏང་བ་རེད། ལས་གཞི་འདིའི་ལག་རྩལ་པས་ངོ་སྤྲོད་གནང་བ་ལྟར་ན། གློག་འདོན་ས་ཚིགས་འདི་ནི་ཤོང་ཚད་ལ་སྟོང་ཝ་༡ཅན་གྱི་ཁྱིམ་སྤྱོད་རླུང་ནུས་གློག་འདོན་མ་ལག་ཆ་༢༠དང་། ཤོང་ཚད་ལ་སྟོང་ཝ་༢༠ཅན་གྱི་ཉི་ནུས་དང་རླུང་ནུས་ཚབ་རེས་རང་བཞིན་གྱི་གཅིག་འདུས་གློག་འདོན་མ་ལག་བཅས་ཀྱིས་གྲུབ་ཅིང་། སྤྱིའི་རྩོལ་ནུས་དེ་སྟོང་ཝ་༤༠ཡིན་པ་དང་། རླུང་ནུས་གློག་འདོན་འཕྲུལ་ཆས་རྐྱང་མའི་རྩོལ་ནུས་དེ་སྟོང་ཝ་༡༠ནས་སྟོང་ཝ་༥བར་ཡིན་ལ། དེ་ནི་བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་ད་སྔའི་རྩོལ་ནུས་ཆེས་ཆེ་བའི་རླུང་བརྟེན་གློག་འདོན་འཕྲུལ་ཆས་དེ་ཡིན་པ་དང་། གློག་འདོན་ས་ཚིགས་དེས་གྲོང་ཚོ་དེའི་དུད་ཁྱིམ་༥༨ཙམ་གྱི་ མི་༤༠༠ལྷག་གི་གློག་གི་དགོས་མཁོ་སྐོང་ཐུབ་པ་རེད།
