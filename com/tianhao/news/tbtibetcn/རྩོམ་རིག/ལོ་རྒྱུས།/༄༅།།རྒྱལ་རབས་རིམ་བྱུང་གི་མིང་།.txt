༄༅།།རྒྱལ་རབས་རིམ་བྱུང་གི་མིང་།
ལྷ་གནམ་གྱི་སྟེང་ནས་གཤེགས་པ། གནམ་ལྷབ་ཀྱི་བླ་ན། ཡབ་ལྷ་བདག་དྲུག་བཞུགས་པའི་སྲས། གཅེན་གསུམ་གཅུང་གསུམ་ན། ཁྲི་འི་བདུན་ཚིགས་དང་བདུན། ཁྲི་འི་བདུན་ཚིགས་ཀྱི་སྲས། ཁྲི་ཉག་ཁྲི་བཙན་པོ། ། ས་དོག་ལ་ཡུལ་ཡབ་ཀྱི་རྗེ། དོག་ཡབ་ཀྱི་ཆར་དུ་གཤེགས་སོ། ལྷ་རི་གྱང་དོར་གཤེགས་ན། རི་རབ་ལྷུན་པོ་ཡང་དགུའ་དུད་དུད་(398)། ཤིང་སྡོང་པོ་ཡང་བང་ཐང་ཐང་(399)། ཆབ་ལུ་མ་(400)ཡང་དངོ་སིལ་སིལ་(401)། གོར་ཕ་བོང་ལ་སྩོགས་པ་ཡང་མཉེད་ཁྲུང་ཁྲུང་(402)གིས་ཕྱག་འཚལ་ལོ་བོད་ཀ་གཡག་དྲུག་གི་རྗེར་གཤེགས་སོ། ཐོག་མ་ས་ལ་གཤེགས་པ་ཡང་། གནམ་མཐའ་འོག་གི་རྗེར་གཤེགས་པས། གནམ་གྱི་ནི་དབུས། སའི་ནི་དཀྱིལ། གླིང་གི་ནི་སྙིང་པོ། གངས་ཀྱི་ནི་ར་བ། ཆུ་བོ་ཀུན་གྱི་ནི་མགོ་བོ། རི་མཐོ་ས་གཙང་། ཡུལ་བཟང་། མྱི་འཛངས་ཤེས་དཔའ་རུ་སྐྱེ། ཆོས་བཟང་དུ་བྱེད། རྟ་མགྱོགས་སུ་འཕེལ་བའི་གནསུ། ཚུལ་ཀྱིས་བདམས་(403)སྟེ་གཤེགསསོ། ། རྒྱལ་པོ་གཞན་དང་མྱི་མཚུངས་པའི་ཆོས། པྱག་མཉེད་ཁྲུང་ཁྲུང་གིས་འཚལ་ཞིང་། སླེ་བ་(404)འོད་(405)ཀྱིས་ལེན། མྱི་བཙུན་སོན་པའི་རྣམས། རྗེ་ས་(406)འི་གོས་གྱིན་ཞིང་། སྐྱེས་པ་ཆེད་པོ་(407)རྣམས་བཙུན་པའི་རྣམ་པ་ཡང་དེ་ནང་བྱུང་བ་ཡིན་ནོ། ཤིང་རབས་ཀྱི་ནང་ན་ནི་ཐང་ཤིང་རིང་ནོ་རིང་། ཆུ་རབས་ཀྱི་ནང་ན་ནི་ཡར་ཆུ་སྔོན་པོ་ཆེ། ཡར་ལྷ་ཤམ་པོ་ནི་གཙུག་གི་ལྷ་འོ། གནམ་ལྷབ་(408)ཀྱི་བླ་ན་ཡབ་བླ་བདག་དྲུག་བཞུགས་པའི་སྲས། གཅེན་གསུམ་གཅུང་གསུམ་ན། ཁྲི་འི་བདུན་ཚིགས་དང་བདུན། ཁྲི་འི་བདུན་ཚིགས་ཀྱི་སྲས། ལྡེ་ཉག་ཁྲི་བཙན་པོ། ས་དོག་ལ་ཡུལ་ཡབ་ཀྱི་རྗེ་དོག་ཡབ་ཀྱི་ཆར་དུ་གཤེགས་ནས། ལྷ་སྲས་མྱི་ཡུལ་གྱི་རྒྱལ་མཛད་ཅིང་བཞུགས་པ་ལས། མངོན་དུ་ཐལ་བྱུང་དགུང་དུ་གཤེགས་པ། ལྡེ་ཉག་ཁྲི་བཙན་བོ་དང་། གནམ་མུག་མུག་དུ་བཤོས་(409)པའི་སྲས། མུ་ཁྲི་བཙན་པོ། མུག་ཁྲི་བཙན་པོ་དང་ས་དིང་དིང་དུ་བཤོས་པའི་སྲས། དིང་ཁྲི་བཙན་པོ། དིང་ཁྲི་བཙན་པོ་དང་། སོ་ཐམ་ཐམ་དུ་བཤོས་པའི་སྲས། སོ་ཁྲི་བཙན་པོ། སོ་ཁྲི་བཙན་པོའི་སྲས། དེ་ཁྲི་བཙན་པོ། དེ་ཁྲི་བཙན་པོའི་སྲས། ཁྲི་སྤེ་བཙན་པོ། འདི་ཡན་ཆད་འདྲའ་སྟེ། སྲས་ཆིབས་ལ་ཐུབ་ན། ཡབ་དགུང་དུ་གཤེགས་སོ། །ཁྲི་སྤེ་བཙན་པོ་འི་སྲས། དྲི་གུམ་བཙན་པོ། དྲི་གུམ་བཙན་པོའི་སྲས། སྤུ་དེ་གུང་རྒྱལ། གནམ་ལ་དྲི་བདུན། ས་ལེ་ལེགས་དྲུག་བཤོས་ན། སྤུ་དེ་གོང་རྒྱལ་གྲོངས་ན། གྲང་མོ་གནམ་གསེར་བརྩིག གསེར་བརྩིག་གི་སྲས། ཐོ་ལེག་བཙན་པོ། ཐོ་ལེག་བཙན་པོའི་སྲས། ཤོ་ལེགས་བཙན་པོའི་སྲས། འབྲོང་ཞི་ལེགས་བཙན་པོ། འབྲོང་ཞི་ལེགས་བཙན་པོའི་སྲས། ཐི་ཤོ་ལེག་བཙན་པོའ། ཐི་ཤོ་ལེག་གི་སྲས། ཨི་ཤོ་ལེགས་བཙན་པོ། ཨི་ཤོ་ལེག་གི་སྲས། ཟྭ་གནམ་ཟིན་ཏེ། ཟྭ་གནམ་ཟིན་ཏེའི་སྲས། ལྡེ་པྲུ་བོ་གནམ་གཞུང་བཙན། གནམ་གཞུང་བརྩན་གྱི་སྲས། ལྡེ་གོལ། ལྡེ་གོལ་གྱི་སྲས།
