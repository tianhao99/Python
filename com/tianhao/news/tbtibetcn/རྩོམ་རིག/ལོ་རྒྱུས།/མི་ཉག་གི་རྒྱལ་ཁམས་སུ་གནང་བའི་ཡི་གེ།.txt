མི་ཉག་གི་རྒྱལ་ཁམས་སུ་གནང་བའི་ཡི་གེ།
༄༅། །ཨོཾ་སྭསྟི་སི་ངྔྷཾ།མི་ཉག་གི་རྒྱལ་ཁམས་ན་ཆོས་རྒྱལ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་བླ་མའི་མཆོད་གནས་སངས་རྒྱས་ཀྱི་བསྟན་པ་རིན་པོ་ཆེའི་སྲོག་ཤིང་ཆེན་པོར་གྱུར་པ།དཔལ་སྡེ་ཆེན་ལྷུན་གྱིས་གྲུབ་པའི་གཙུག་ལག་ཁང་ཆེན་པོར་གཉུག་མར་བཞུགས་པའི་མཁན་སློབ་ཞལ་བཞུགས་པ་རྣམ་པས་དབུ་མཛད།དགེ་འདུན་རྣམས་ཀྱི་ཕྱག་ཏུ་ཕུལ་བ།རྣམ་དག་ཆོས་རྒྱལ་འབྱོར་བའི་ཆབ་ཀྱིས་ཀུན་ནས་ཐ་གྲུར་ལེགས་བལྟམས་ཤིང་། །མི་མཚུངས་བློ་ཅན་མཁས་པའི་མེ་ཏོག་ཕྱོགས་ཀུན་ཁྱབ་པའི་འདབ་མ་ཅན། །སྡོམ་བརྩོན་དམ་པ་བུང་བའི་ཚོགས་ཀྱིས་ལེགས་བཤད་སྦྲང་རྩིའི་ཕྱིར་བསྟེན་པ། །སྡེ་ཆེན་བསྟན་རྩ་པདྨོའི་འབྱུང་གནས་ཆུ་གཏེར་ཆེན་པོ་རྒྱལ་གྱུར་ཅིག།དེ་ཡི་དགེ་འདུན་བློ་གྲོས་གསལ་ལྡན་པ། །ཚུལ་ཁྲིམས་དག་པའི་དྲི་ངད་ཕྱོགས་རྣམས་སུ། །རབ་ལྡང་མཁྱེན་བརྩེའི་མཆེ་བ་རྣོ་བ་ཅན། །བདག་ཉིད་ཆེན་པོ་སྤོས་ཀྱི་གླང་པོ་ཁྱོད། །འཇིག་ཚོགས་ལྟ་བའི་རལ་པའི་ཚོགས་མང་རབ་འཁྲུགས་བག་ལ་ཉལ་བའི་མིག་དམར་པ། །འཛིན་པའི་སྤར་མོ་རྡོ་རྗེའི་རྩེ་ལྟར་རྣོ་བ་དང་བཅས་ཀུན་ནས་ལྡང་བའི་ཡན་ལག་ཅན། །ཕྲ་རྒྱས་ཚོགས་ཀྱིས་མཆེ་བས་རབ་གཏམས་བདག་ཏུ་འཛིན་པའི་ཁ་སྒོ་ཀུན་ནས་ཕྱེ། །སྲིད་པའི་རི་དྭགས་རབ་བསྙེགས་མི་མཐུན་ཕྱོགས་ཀྱི་ཕྱུགས་བདག་ཉེ་བར་མ་གྱུར་ཅིག།ཁྱེད་ཀྱི་རྣམ་ཐར་ཆེན་པོའི་ཆེན་པོ་ཕྱོགས་ཀུན་ཁྱབ་པའི་སྙན་པ་ཡི། །གྲགས་པ་རྒྱ་ཆེན་ཕུལ་དུ་བྱུང་བ་སྐྱེ་བོའི་ཚོགས་ཀྱིས་རབ་བསྒྲགས་པ། །རིང་ན་གནས་ཀྱང་མངོན་སུམ་ཇི་བཞིན་ཉེ་བར་ཁུམས་པས་ཁོ་བོ་ཡི། །དགའ་བའི་རྒྱ་མཚོ་དབུར་གྱི་ཆུ་བཞིན་གོང་ནས་གོང་དུ་འཕེལ་བར་གྱུར། །དེ་སླད་མཆོད་འོས་སྡོམ་བརྩོན་ཁྱེད་རྣམས་ལ། །དགའ་བའི་ཡིད་ཀྱིས་ཚིག་གི་མེ་ཏོག་གི།ཕྲེང་བ་སྤེལ་ལེགས་ངག་གིས་རབ་བརྒྱུས་པ། །འདི་ན་དགེ་བ་སྤེལ་ཕྱིར་བདག་འབུལ་ལོ། །བདེ་བར་གཤེགས་པའི་བསྟན་པ་རིན་ཆེན་གཏེར། །འདུལ་བའི་རྒྱ་མཚོའི་ནང་ན་ལེགས་གནས་པ། །དགེ་འདུན་དགེ་ལེགས་ཚོགས་སྒོར་འཇུག་རྣམས་ལ། །རྗེས་སུ་སྟོན་པའི་བཤེས་གཉེན་དེད་དཔོན་མཆོག།ཡན་ལག་བཞི་ལྡན་དཔུང་གིས་ཞབས་བརྟན་ཞིང་།གདུང་རྒྱུད་དག་པའི་སྤྱི་གཙུག་རབ་མཐོ་བ། །ཕས་རྒོལ་འཇོམས་པའི་རྒྱལ་པོ་གླང་པོ་ཆེ། །ཡང་དག་ལམ་འཛུད་ལྕགས་ཀྱུ་འཛིན་པ་པོ། །དགེ་ལེགས་ཆར་རྒྱུན་ཕྱོགས་རྣམས་ཀུན་ཁྱབ་ཅིང་། །ཟབ་ཡངས་ཆོས་ཀྱི་འབྲུག་སྒྲ་སྟོང་སྒྲོགས་ཤིང་། །གདུལ་བྱའི་ལོ་ཏོག་ཤིན་ཏུ་རྒྱས་མཛད་པ། །སྡོམ་བརྩོན་ཆེ་ཁྱེད་ཆར་སྤྲིན་ཆེན་པོ་ལགས། །མཁས་པའི་སློབ་གྲྭའི་ཚལ་དུ་རབ་འཁྲུངས་ཤིང་། །དམ་པའི་ལེགས་བཤད་ཆབ་ཀྱིས་ལེགས་བསྐྲུན་པ། །ཡོན་ཏན་ཚོགས་ཀྱི་མེ་ཏོག་འདབ་མ་ཅན། །ཕན་བདེའི་འབྲས་འབྱིན་དཔག་བསམ་ལྗོན་ཤིང་ཁྱེད། །སྡེ་བ་ཀུན་གྱི་བསྟན་པའི་རྩ་བ་སྟེ། །ཆོས་རྒྱལ་ཆེན་པོའི་གཙུག་གི་ནོར་བུར་གྱུར། །དབང་ཕྱུག་རྣམས་ཀྱི་མཆོད་བྱའི་དམ་པ་སྟེ། །མགོན་མེད་རྣམས་ཀྱི་་དཔག་བསམ་ལྗོན་པའི་ཤིང་། །དེ་སླད་ཁྱེད་ཀྱི་མཛད་པར་བགྱི་བ་ནི། །ད་དུང་སྡོམ་མརྩོན་འབུམ་ཕྲག་ལྷག་པ་དང་། །མཁས་པའི་སློབ་གྲྭ་བརྒྱ་ཕྲག་དུ་མ་དང་། །ལེགས་བཤད་ཚོལ་（འཚོལ）བ་བྱེ་བ་ཕྱག་འགའ་ཞིག།བདེ་བར་གཤེགས་པའི་བསྟནཔ་སྤེལ་བ་དང་། །ཡུན་རིང་རྙེད་དང་བཀུར་སྟི་གྲགས་ལྡན་པས། །སྐྱེ་བོ་ཀུན་ལ་སྨན་པར་བྱ་བའི་ཕྱིར། །བདག་ཉིད་ཆེན་པོ་ཁྱེད་ཀྱིས་བརྩོན་པར་ཞུ། །སྡོམ་བརྩོན་བདེ་ལ་ལེགས་པར་གནས་བྱའི་ཕྱིར། །ཁྲིམས་སུ་བྱ་བ་ཇི་སྙེད་ཡོད་པ་ཀུན། །དགག་སྒྲུབ་ཚུལ་ལ་གཟུ་བོའི་ལམ་ཉིད་ཀྱིས། །རྣམ་པར་དཔྱད་ན་བསྟན་པ་འཕེལ་ལགས་གྲང་། །མཚན་མ་ཅུང་ཟད་མངོན་རྟགས་ཉིད་བྱས་ནས། །དགེ་སྦྱོང་དག་པ་བསྙེལ་པར་མཛད་ལགས་ན། །རྣམ་ཐར་དག་པའི་གྲངས་སུ་བགྲང་མོད་ཀྱི། །འོན་ཀྱང་འཐད་པ་རིགས་པའི་སྤྱན་གྱིས་གཟིགས། །དགེ་སྦྱོང་ལྷོད་པའི་ལུས་ངག་མ་བསྡམས་ཤིང་། །བློ་ངན་ཐོས་པའི་ནོར་དང་མི་ལྡན་པ། །འགའ་ཞིག་བརྩེ་བས་རྗེས་སུ་འཛིན་པ་ཡང་། །བསྟན་པའི་ཚེར་མར་འགྱུར་དུ་དོགས་པ་མཆི། །སློབ་པར་འདོད་པའི་དགེ་སློང་བློ་གྲོས་ཅན། །ཁྲིམས་ཀྱིས་བསྡམས་པ་ཡུལ་ནས་བགྲོད་མིན་པ། །དེ་དག་འཁོར་དང་འབྱོར་བ་ལྷན་ཅིག་ཏུ། །མཁས་པའི་མདུན་སར་བཏང་ན་ཅིས་མི་ལེགས། །ཤིན་ཏུ་མཁས་ཀྱང་གྲགས་པ་མ་ཐོབ་ལ། །སྦྱིན་བདག་ཉུང་ཞིང་འཁོར་དང་མི་ལྡན་པ། །དེ་དག་གཙུག་ལག་ཁང་བཟང་སྡེ་ཆེན་གྱི། །སོ་སོའི་འཆད་པོར་སྩལ་ན་ཅིར་འགྱུར་ལགས། །དབང་པོ་དུལ་བ་བསླབ་པའི་ཁུར་འཛིན་པ། །བསོད་སྙོམས་རྒྱུ་བས་ཁ་ཟས་མི་འཕྲོད་ཅིང་། །གནས་མལ་གོས་དང་སྨན་གྱིས་ཕོངས་པ་ཡི། །སྡོམ་བརྩོན་དེ་དག་བདག་གིར་མཛད་པར་གསོལ། །མཁས་པ་མཁས་པའི་ཚོགས་ཀྱིས་ཡོངས་བསྐོར་པ། །ཆེ་བའི་ཆེ་བ་བདག་ཉིད་ཆེ་རྣམས་ལ། །དེ་སྐད་གསོལ་བ་ཀུན་དགའི་རྒྱལ་མཚན་དཔལ། །བཟང་པོ་ཞེས་བྱར་གྲངས་པ་དེ་ཉིད་ལགས། །ད་དུང་ཞུ་བར་གསོལ་བ་མཆིས་མོད་ཀྱི། །དོན་མཐུན་འདི་དག་འགྲོ་བར་འདོད་པ་དང་། །བདག་ཀྱང་རྣམ་པར་གཡེང་བ་མང་བས་ན། །རེ་ཞིག་གུས་པའི་མཚན་མ་འདི་འབུལ་གྱིས། །བརྩེ་བས་བདག་དང་བདག་གི་རྗེས་འབྲང་བ། །ཡུལ་དང་ཡུལ་གཞན་དག་ཏུ་བགྲོད་འགྱུར་རམ། །དུས་དང་དུས་མིན་ཇི་ལྟར་གནས་ཀྱང་རུང་། །ཐུགས་རྗེའི་ཐུགས་ཀྱིས་རྗེས་སུ་འཛིན་པར་ཞུ། །དཔལ་སྡེ་ཆེན་གྱི་གཙུག་ལག་ཁང་ཆེན་པོར་བཞུགས་པའི་མཁན་སློབ་རིན་པོ་ཆེས་དབུ་མཛད་དགེ་འདུན་རྣམས་ཀྱི་ཕྱག་ཏུ་ཕུལ་བ་འདི་ནི། །དཔལ་ལྡན་ས་སྐྱའི་དགོན་པར་གནས་པ།ཤཱཀྱའི་དགེ་སློང་ཀུན་དགའ་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ཞེས་བྱ་བས་གསོལ་བ་ལགས།ཡི་གེ་མི་འཁྱར་བའི་རྟེན་ཁ་ཆེའི་གུར་ཀུམ་ཞོ་བཅུ་གསུམ་འབུལ། །མི་ཉག་གི་རྒྱལ་ཁམས་སུ་གནང་བའི་ཡི་གེ།
ཨོཾ་སྭསྟི་སིདྡྷཾ།མི་ཉག་གི་རྒྱལ་ཁམས་ན་ཆོས་རྒྱལ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་བླ་མའི་མཆོད་གནས་སངས་རྒྱས་ཀྱི་བསྟན་པ་རིན་པོ་ཆེའི་སྲོག་ཤིང་ཆེན་པོར་གྱུར་པ། དཔལ་སྡེ་ཆེན་ལྷུན་གྱིས་གྲུབ་པའི་གཙུག་ལག་ཁང་ཆེན་པོར་གཉུག་མར་བཞུགས་པའི་མཁན་སློབ་ཞལ་བཞུགས་པ་རྣམ་པས་དབུ་མཛད། དགེ་འདུན་རྣམས་ཀྱི་ཕྱག་ཏུ་ཕུལ་བ། 
　　རྣམ་དག་ཆོས་རྒྱལ་འབྱོར་བའི་ཆབ་ཀྱིས་ཀུན་ནས་ཐ་གྲུར་ལེགས་བལྟམས་ཤིང་། །མི་མཚུངས་བློ་ཅན་མཁས་པའི་མེ་ཏོག་ཕྱོགས་ཀུན་ཁྱབ་པའི་འདབ་མ་ཅན། །སྡོམ་བརྩོན་དམ་པ་བུང་བའི་ཚོགས་ཀྱིས་ལེགས་བཤད་སྦྲང་རྩིའི་ཕྱིར་བསྟེན་པ། །སྡེ་ཆེན་བསྟན་རྩ་པདྨོའི་འབྱུང་གནས་ཆུ་གཏེར་ཆེན་པོ་རྒྱལ་གྱུར་ཅིག །དེ་ཡི་དགེ་འདུན་བློ་གྲོས་གསལ་ལྡན་པ། །ཚུལ་ཁྲིམས་དག་པའི་དྲི་ངད་ཕྱོགས་རྣམས་སུ། །རབ་ལྡང་མཁྱེན་བརྩེའི་མཆེ་བ་རྣོ་བ་ཅན། །བདག་ཉིད་ཆེན་པོ་སྤོས་ཀྱི་གླང་པོ་ཁྱོད། །འཇིག་ཚོགས་ལྟ་བའི་རལ་པའི་ཚོགས་མང་རབ་འཁྲུགས་བག་ལ་ཉལ་བའི་མིག་དམར་པ། །འཛིན་པའི་སྤར་མོ་རྡོ་རྗེའི་རྩེ་ལྟར་རྣོ་བ་དང་བཅས་ཀུན་ནས་ལྡང་བའི་ཡན་ལག་ཅན། །ཕྲ་རྒྱས་ཚོགས་ཀྱིས་མཆེ་བས་རབ་གཏམས་བདག་ཏུ་འཛིན་པའི་ཁ་སྒོ་ཀུན་ནས་ཕྱེ། །སྲིད་པའི་རི་དྭགས་རབ་བསྙེགས་མི་མཐུན་ཕྱོགས་ཀྱི་ཕྱུགས་བདག་ཉེ་བར་མ་གྱུར་ཅིག  ཁྱེད་ཀྱི་རྣམ་ཐར་ཆེན་པོའི་ཆེན་པོ་ཕྱོགས་ཀུན་ཁྱབ་པའི་སྙན་པ་ཡི། །གྲགས་པ་རྒྱ་ཆེན་ཕུལ་དུ་བྱུང་བ་སྐྱེ་བོའི་ཚོགས་ཀྱིས་རབ་བསྒྲགས་པ། །རིང་ན་གནས་ཀྱང་མངོན་བཞིན་གོང་ནས་གོང་དུ་འཕེལ་བར་གྱུར། ། དེ་སླད་མཆོད་འོས་སྡོམ་བརྩོན་ཁྱེད་རྣམས་ལ། །དགའ་བའི་ཡིད་ཀྱིས་ཚིག་གི་མེ་ཏོག་གི །ཕྲེང་བ་སྤེལ་ལེགས་ངག་གིས་རབ་བརྒྱུས་པ། །འདི་ན་དགེ་བ་སྤེལ་ཕྱིར་བདག་འབུལ་ལོ། །བདེ་བར་གཤེགས་པའི་བསྟན་པ་རིན་ཆེན་གཏེར། །འདུལ་བའི་རྒྱ་མཚོའི་ནང་ན་ལེགས་གནས་པ། །དགེ་འདུན་དགེ་ལེགས་ཚོགས་སྒོར་འཇུག་རྣམས་ལ། །རྗེས་སུ་སྟོན་པའི་བཤེས་གཉེན་དེད་དཔོན་མཆོག །ཡན་ལག་བཞི་ལྡན་དཔུང་གིས་ཞབས་བརྟན་ཅིང་། །གདུང་རྒྱུད་དག་པའི་སྤྱི་གཙུག་རབ་མཐོ་བ། །ཕས་རྒོལ་འཇོམས་པའི་རྒྱལ་པོ་གླང་པོ་ཆེ། །ཡང་དག་ལམ་འཛུད་ལྕགས་ཀྱུ་འཛིན་པ་པོ། །དགེ་ལེགས་ཆར་རྒྱུན་ཕྱོགས་རྣམས་ཀུན་ཁྱབ་ཅིང་། །ཟབ་ཡངས་ཆོས་ཀྱི་འབྲུག་སྒྲ་སྟོང་སྒྲོགས་ཤིང་། །གདུལ་བྱའི་ལོ་ཏོག་ཤིན་ཏུ་རྒྱས་མཛད་པ། །སྡོམ་བརྩོན་ཆེ་ཁྱེད་ཆར་སྤྲིན་ཆེན་པོ་ལགས། །མཁས་པའི་སློབ་གྲྭའི་ཚལ་དུ་རབ་འཁྲུངས་ཤིང་། །དམ་པའི་ལེགས་བཤད་ཆབ་ཀྱིས་ལེགས་བསྐྲུན་པ། །ཡོན་ཏན་ཚོགས་ཀྱི་མེ་ཏོག་འདབ་མ་ཅན། །ཕན་བདེའི་འབྲས་འབྱིན་དཔག་བསམ་ལྗོན་ཤིང་ཁྱེད། །སྡེ་བ་ཀུན་གྱི་བསྟན་པའི་རྩ་བ་སྟེ། །ཆོས་རྒྱལ་ཆེན་པོའི་གཙུག་གི་ནོར་བུར་གྱུར། །དབང་ཕྱུག་རྣམས་ཀྱི་མཆོད་བྱའི་དམ་པ་སྟེ། །མགོན་མེད་རྣམས་ཀྱི་དཔག་བསམ་ལྗོན་པའི་ཤིང་། །དེ་སླད་ཁྱེད་ཀྱི་མཛད་པར་བགྱི་བ་ནི། །ད་དུང་སྡོམ་བརྩོན་འབུམ་ཕྲག་ལྷག་པ་དང་། །མཁས་པའི་སློབ་གྲྭ་བརྒྱ་ཕྲག་དུ་མ་དང་། །ལེགས་བཤད་འཚོལ་བ་བྱེ་བ་ཕྲག་འགའ་ཞིག  །བདེ་བར་གཤེགས་པའི་བསྟན་པ་སྤེལ་བ་དང་། །ཡུན་རིང་རྙེད་དང་བཀུར་སྟི་གྲགས་ལྡན་པས། །སྐྱེ་བོ་ཀུན་ལ་སྨན་པར་བྱ་བའི་ཕྱིར། །བདག་ཉིད་ཆེན་པོ་ཁྱེད་ཀྱིས་བརྩོན་པར་ཞུ། །སྡོམ་བརྩོན་བདེ་ལ་ལེགས་པར་གནས་བྱའི་ཕྱིར། །ཁྲིམས་སུ་བྱ་བ་ཇི་སྙེད་ཡོད་པ་ཀུན། །དགག་སྒྲུབ་ཚུལ་ལ་གཟུ་བོའི་ལམ་ཉིད་ཀྱིས། །རྣམ་པར་དཔྱད་ན་བསྟན་པ་འཕེལ་ལགས་གྲང་། །མཚན་མ་ཅུང་ཟད་མངོན་རྟགས་ཉིད་བྱས་ནས། །དགེ་སྦྱོང་དག་པ་བསྙེལ་པར་མཛད་ལགས་ན། །རྣམ་ཐར་དག་པའི་གྲངས་སུ་བགྲང་མོད་ཀྱི། །འོན་ཀྱང་འཐད་པ་རིགས་པའི་སྤྱན་གྱིས་གཟིགས། །དགེ་སྦྱོང་ལྷོད་པའི་ལུས་ངག་མ་བསྡམས་ཤིང་། །བློ་ངན་ཐོས་པའི་ནོར་དང་མི་ལྡན་པ། །འགའ་ཞིག་བརྩེ་བས་རྗེས་སུ་འཛིན་པ་ཡང་། །བསྟན་པའི་ཚེར་མར་འགྱུར་དུ་དོག་པ་མཆི། །སློབ་པར་འདོད་པའི་དགེ་སློང་བློ་གྲོས་ཅན། །ཁྲིམས་ཀྱིས་བསྡམས་པ་ཡུལ་ནས་བགྲོད་མིན་པ། །དེ་དག་འཁོར་དང་འབྱོར་བ་ལྷན་ཅིག་ཏུ། །མཁས་པའི་མདུན་སར་བཏང་ན་ཅིས་མི་ལེགས། །ཤིན་ཏུ་མཁས་ཀྱང་གྲགས་པ་མ་ཐོབ་ལ། །སྦྱིན་བདག་ཉུང་ཞིང་འཁོར་དང་མི་ལྡན་པ། །དེ་དག་གཙུག་ལག་ཁང་བཟང་སྡེ་ཆེན་གྱི། །སོ་སོའི་འཆད་པོར་སྩལ་ན་ཅིར་འགྱུར་ལགས། །དབང་པོ་དུལ་བ་བསླབ་པའི་ཁུར་འཛིན་པ། །བསོད་སྙོམས་རྒྱུ་བས་ཁ་ཟས་མི་འཕྲོད་ཅིང་། །གནས་མལ་གོས་དང་སྨན་གྱིས་ཕོངས་པ་ཡི། །སྡོམ་བརྩོན་དེ་དག་བདག་གིར་མཛད་པར་གསོལ། །མཁས་པ་མཁས་པའི་ཚོགས་ཀྱིས་ཡོངས་བསྐོར་པ། །ཆེ་བའི་ཆེ་བ་བདག་ཉིད་ཆེ་རྣམས་ལ། །དེ་སྐད་གསོལ་བ་ཀུན་དགའི་རྒྱལ་མཚན་དཔལ། །བཟང་པོ་ཞེས་བྱར་གྲངས་པ་དེ་ཉིད་ལགས། །ད་དུང་ཞུ་བར་གསོལ་བ་མཆིས་མོད་ཀྱི། །དོན་མཐུན་འདི་དག་འགྲོ་བར་འདོད་པ་དང་། །བདག་ཀྱང་རྣམ་པར་གཡེང་བ་མང་བས་ན། །རེ་ཞིག་གུས་པའི་མཚན་མ་འདི་འབུལ་གྱིས། །བརྩེ་བས་བདག་དང་བདག་གི་རྗེས་འབྲང་བ། །ཡུལ་དང་ཡུལ་གཞན་དག་ཏུ་བགྲོད་འགྱུར་རམ། །དུས་དང་དུས་མིན་ཇི་ལྟར་གནས་ཀྱང་རུང་། །ཐུགས་རྗེའི་ཐུགས་ཀྱིས་རྗེས་སུ་འཛིན་པར་ཞུ། །དཔལ་སྡེ་ཆེན་གྱི་གཙུག་ལག་ཁང་ཆེན་པོར་བཞུགས་པའི་མཁན་སློབ་རིན་པོ་ཆེས་དབུ་མཛད་དགེ་འདུན་རྣམས་ཀྱི་ཕྱག་ཏུ་ཕུལ་བ་འདི་ནི། དཔལ་ལྡན་ས་སྐྱའི་དགོན་པར་གནས་པ། ཤཱཀྱའི་དགེ་སློབ་ཀུན་དགའ་རྒྱལ་མཚན་དཔལ་བཟང་པོ་ཞེས་བྱ་བས་གསོལ་བ་ལགས། ཡི་གེ་མི་འཁྱར་བའི་རྟེན་ཁ་ཆེའི་གུར་ཀུམ་ཞོ་བཅུ་གསུམ་འབུལ།
