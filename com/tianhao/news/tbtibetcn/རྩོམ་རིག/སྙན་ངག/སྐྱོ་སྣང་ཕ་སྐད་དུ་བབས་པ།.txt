སྐྱོ་སྣང་ཕ་སྐད་དུ་བབས་པ།
དེ་རིང་ཞོགས་པར་མལ་ལས་ལྡང་དུས།སྐྱ་ཀ་ཞིག་གིས་ཅག་ཅག་བྱས་ཏེ།རྣམ་རྟོག་ཨེན་ཙམ་འཁྲུགས་མདོག་བྱས་ཀྱང་།སྙིང་བོ་མེད་ལ་དོགས་བློ་ཅི་ཡོད།བྱིས་པའི་ཞོ་གཉེར་གདོང་ནས་དག་ཅིང་།གཞོན་པའི་དཔུང་ཤ་ཡལ་བར་ཕྱོགས་ནས།རྒས་པའི་སྤུ་ཁ་གཟེངས་ནས་ཡོང་ཚེ།སྐྱོ་ཡང་སྐྱོ་ལ་སྐྲག་ཀྱང་སྐྲག་འོང་།སྲིད་པ་འཁོར་བའི་ཐང་ཆེན་འདི་ན།ལ་ལ་མདུང་འཛིན་ལ་ལ་གྲི་འཛིན།རིགས་མི་གཅིག་པའི་མགྲོན་པོ་མང་བོ།འདུ་བ་མང་ལ་བྲལ་བ་མང་ཨང་།ང་ནི་ས་ཡི་རྩི་ཏོག་ཙམ་མོ།རླུང་བུ་གཡུག་ཚེ་འདར་བ་མང་ཞིང་།བ་མོ་ཆགས་དུས་མགོ་བོ་སྒུར་ནས།ཉམ་ཆུང་རང་མདོག་གཞི་ནས་བསྟན་འོང་།རྟེན་འབྲེལ་སྟོན་པའི་དུས་ཀྱི་འགྱུར་ལྡོག།འགྱུར་བ་མེད་ལ་བསྒྱུར་ཐབས་དཀའ་བས།རྨི་ལམ་འདྲ་བའི་མི་ཚེ་འདི་ལ།ཐོབ་པ་མང་ལ་ཤོར་པ་མང་བ།རྣམ་ཤེས་ནགས་ཀྱི་སྤྲེའུ་འདྲ་བོ།དྲན་ཤེས་ཀྱོ་བ་རྣོན་པོས་མ་ཐུབ། གཏད་སོ་མེད་ལ་སྡོད་ལོང་མེད་པར།དགོས་མེད་གསོག་པའི་མགྲོན་པོ་ཡིན་འདུག།བྱར་མེད་འཛིན་མེད་སྟོང་བའི་ངང་ལས།ཅི་ཡང་སྣང་ལ་ཅི་ཡང་ཤར་ཚེ།སྲིད་པའི་ལྕེ་དམར་རྐང་ལ་རྡེབ་བཞིན།མི་ཚེ་དེ་ག་རང་དུ་ཟད་སོང་།ད་ཡང་དགུན་གྱི་གྲང་རླུང་འོང་།ལུས་ཀྱི་བ་སྤུ་གཞོགས་ཉལ་བྱས་འགྲོ།གྱོན་པ་མཐུག་ཙམ་རྒྱབ་ལ་གོན་ནས།ཇི་ལྟ་ཇི་ལྟར་འདུག་དགོས་བསམ་ཨང་། རྨི་ལམ་འཛིན་པ།སྨུག་པས་རི་ལུང་བཏུམས་པའི་ཞོགས་པ།སྐྱོ་བ་སྣོན་པའི་འཇིག་རྟེན་བྱེད་པོ།ཟིན་མེད་ཚེ་ཡི་བལ་སྐུད་ཕྲ་མོ།མེ་ལོང་ངོས་སུ་ཟིལ་བ་ཐོར་འདྲ།དྲན་དྲན་འདྲ་བའི་མཛའ་བཤེས་མང་བོ།བྱིའུ་བཞིན་དུ་རང་སར་འཕུར་ཅིང་།ཐོས་འདོད་ཆེ་བའི་སྙིང་གཏམ་གང་མང་།ཡུལ་གྱིས་བསྐལ་པའི་ས་རུ་ལུས་ཡོད།བྱིས་པའི་འཚོ་བ་ཞེ་ལ་བཞག་ན།དགོང་གའི་སྐར་མ་འཛིན་འདོད་ཆེ་ཡང་།ལུས་ཀྱི་གྲིབ་སོ་ཚུར་ལ་འགུགས་དུས།འཕྲུལ་བའི་མེ་ཏོག་དབྱིངས་སུ་བཏང་འགྲོ།སྤྲིན་པ་འདི་ཡང་ལུས་ཀྱི་གོས་ཡིན།གཙང་བོ་འདི་ཡང་བརྩེ་བའི་སྲོག་ཡིན།ས་གཞི་འདི་ཡང་སྐྱོང་བའི་གཉེན་ཡིན།ཞེས་པའི་ལས་འཕྲོ་ད་དུང་མང་ཨང་།ལམ་ཕྲན་སྐྱ་མོའི་ཕ་གིའི་སྣེ་ན།བོར་བའི་ལུས་པོ་རྗེས་སུ་བསྐྱུར་ཅིང་།བསགས་པའི་ལས་རླུང་སྒོམ་དུ་བྱེད་པའི།བར་དོའི་མགྲོན་རྣམས་གང་དུ་ཆས་སམ།དེ་རིང་སྐྱི་ལ་སང་ཡང་སྐྱོ་བའི།འདུ་བྱེད་ལུས་རྙི་ཐག་གཟེབ་ཏུ།འགུལ་ཙམ་ཉིད་ནས་ཞེས་དེ་འོངས་པས།གང་ལ་སེམས་ཀྱི་རྟེན་ཕུར་བྱ་དགོས།རྟེན་འབྲེལ་སྟོན་པའི་གླེགས་བམ་གསེབ་ཏུ།ང་ཡང་སྟོང་ལ་ཁྱེད་ཀྱང་སྟོང་ཡང་།ཆགས་སྡང་ཚ་བ་འཁོར་བའི་ཐང་ཆེན།ཁྱེད་ཀྱིས་བགྲོད་དཀའ་ངས་ཀྱང་བགྲོད་དཀའ།བདག་གི་འཆར་ཡན།ཡང་བའི་སེམས་ལ་རླུང་སྤུར་ནས།མིག་གི་ཆུ་རིས་ཡོམ་པའི་ཚེ།ཡིད་ལ་འགུལ་བའི་ལྗོན་པ་ནི།ལོ་མ་ཞིག་གི་འཕུར་ས་ཡིན།གྱ་གྱུའི་ཤུགས་ལྡན་གཙང་བོ་ཞིག།བྱེ་ཐང་བརྒལ་ནས་དུབ་པའི་ཚེ།ངག་གི་སྐྱོ་དབྱངས་གསིལ་བ་ནི།རྣམ་ཤེས་གང་གི་འོད་ཟེར་རོ།མི་སྨྲའི་བརྟུལ་ཞུགས་བཟུང་བ་ཡི།མེ་ཏོག་ལྟ་བུའི་རལ་གྲི་གང་།ཡིད་ལ་བསྣུན་པའི་བརྡ་ཐབས་སུ།དཀོན་མཆོག་འབོད་པའི་ཕོ་ཉར་བཏང་།གློ་བ་ཟིལ་བས་དཀར་བ་དང།གདོང་ལ་དྲི་མས་ནོངས་བྱས་པའི།སྐྱིད་ཆུའི་འགྲམ་གྱི་ཚངས་མགུར་ཞིག།ཤེལ་ཕོར་གཏིང་ལ་ལྕེབས་ཏེ་ཐལ།བདག་གི་མགྲིན་པ་འདར་བ་དང་།ཁོང་གི་སྙིང་ཡང་འདར་བ་འདིས།ཕ་གིའི་རྩིག་ངོས་ནག་པོ་ལ།སངས་རྒྱས་སྟོང་གི་བྱིན་རླབས་འབྱུག།ཞབས་གདུབ་ཕུད་པའི་རྗེ་བཙུན་མ།ལམ་གྱི་དཀར་ནག་མཚམས་སུ་ནི།ཡོན་པོའི་སྤྱན་ལ་འཐེང་བོའི་འགྲོས།མཉམ་དུ་བསྡེབས་ནས་རྫི་གསང་དོད།ཨེ་མ་ལུས་ཀྱི་རྩ་འདི་དང་།རླུང་གི་བྱེད་ནུས་བསམ་མི་ཁྱབ་།བསམ་མི་ཁྱབ་པའི་ངང་ཉིད་ལས།ཡིད་ཀྱི་འཆར་སྣང་བསམ་མི་ཁྱབ།
