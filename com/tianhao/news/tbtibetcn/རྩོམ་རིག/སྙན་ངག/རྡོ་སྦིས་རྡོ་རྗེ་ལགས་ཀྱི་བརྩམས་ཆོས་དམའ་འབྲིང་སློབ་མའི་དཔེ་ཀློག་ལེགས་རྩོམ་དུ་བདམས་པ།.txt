རྡོ་སྦིས་རྡོ་རྗེ་ལགས་ཀྱི་བརྩམས་ཆོས་དམའ་འབྲིང་སློབ་མའི་དཔེ་ཀློག་ལེགས་རྩོམ་དུ་བདམས་པ།
རྩོམ་པ་པོ་རྡོ་སྦིས་རྡོ་རྗེ་ལགས། 
　རྩོམ་པ་པོ་གྲགས་ཅན་རྡོ་སྦིས་རྡོ་རྗེ་ལགས་ཀྱིས་རྒྱ་ཡིག་གི་ལམ་ནས་བརྩམས་པའི་རིག་གནས་འབོལ་རྩོམ་སྟེ《གངས་རི་ཁ་བའི་ལྗོངས། ནོར་འཛིན་སྒེག་མོའི་ཁྱོན་དུ་འཕྲོས་པའི་འོད་ཐིག་ཅིག》ཅེས་པས་ཁྱད་དུ་འཕགས་པའི་རྩོམ་རིག་གི་ཉམས་འགྱུར་དང་། འཕྱོར་སྒེག་གི་ཉམས་ཀྱིས་ཕྱུག་པའི་ཚིག་སྦྱོར། ངོ་མཚར་རྨད་དུ་བྱུང་བའི་ཚིག་གི་ཕྲེང་བར་སྤེལ་བ་བཅས་ཀྱིས་རྒྱལ་ཁབ་རིམ་པའི་གཙོ་སྙིང་དུས་དེབ་སྟེ《ཀྲུང་ཧྭའི་རྩོམ་རིག་འདེམ་སྒྲིག དམའ་འབྲིང་རིམ་པའི་པར་གཞི》ཞེས་པའི་གྲས་སུ་བདམས་ཤིང་། དམའ་འབྲིང་སློབ་མའི་དཔེ་ཀློག་གི་ལེགས་རྩོམ་དུ་བདམས་པ་རེད། 
　རྡོ་སྦིས་རྡོ་རྗེ་ལགས་ཀྱིས་སྔ་རྗེས་སུ་རྒྱ་ཡིག་གི་ལམ་ནས《མཚོ་ལྷོར་བསྐྱོད་པ》དང《ལྷ་ལམ་དུ་མཐོང་བའི་གངས་ལྗོངས》། 《སེམས་ཁོངས་སུ་དྲག་ཏུ་འཕྱོ་བའི་མཚོ་སྔོན་པོ》སོགས་ཆེད་རྩོམ་མང་དུ་བརྩམས་པར་མ་ཟད། 《ཁྲི་ཀའི་རྨ་ཀླུང་དྭངས་མོ》དང་། 《སུམ་རྩེན་ཡུལ་གྱི་ལྷ་མཚོ། མཚོ་སྔོན་ཁྲོ་ཤོག་རྒྱལ་མོ》སོགས་དངོས་ཕབ་གློག་བརྙན་མང་དུ་བསྒྲིགས། ཁོང་གིས་གཙོ་བོ་དུས་དེབ་དང་ཚགས་པར་གྱི་རྩོམ་སྒྲིག་དང་། གནས་སྐོར་ཁྱབ་སྒྲོག་ཇུས་འགོད། གློག་བརྙན་འཕྲིན་འཁྲབ་ཁྲིད། རྩོམ་རིག་གསར་རྩོམ། རིག་གནས་ཁྱབ་སྒྲོག་སོགས་ཀྱི་ལས་ཀ་གཉེར་བཞིན་ཡོད།རྡོ་སྦིས་རྡོ་རྗེ་ལགས་ཀྱི་བརྩམས་ཆོས་དམའ་འབྲིང་སློབ་མའི་དཔེ་ཀློག་ལེགས་རྩོམ་དུ་བདམས་པ།
རྩོམ་པ་པོ་རྡོ་སྦིས་རྡོ་རྗེ་ལགས། 
　རྩོམ་པ་པོ་གྲགས་ཅན་རྡོ་སྦིས་རྡོ་རྗེ་ལགས་ཀྱིས་རྒྱ་ཡིག་གི་ལམ་ནས་བརྩམས་པའི་རིག་གནས་འབོལ་རྩོམ་སྟེ《གངས་རི་ཁ་བའི་ལྗོངས། ནོར་འཛིན་སྒེག་མོའི་ཁྱོན་དུ་འཕྲོས་པའི་འོད་ཐིག་ཅིག》ཅེས་པས་ཁྱད་དུ་འཕགས་པའི་རྩོམ་རིག་གི་ཉམས་འགྱུར་དང་། འཕྱོར་སྒེག་གི་ཉམས་ཀྱིས་ཕྱུག་པའི་ཚིག་སྦྱོར། ངོ་མཚར་རྨད་དུ་བྱུང་བའི་ཚིག་གི་ཕྲེང་བར་སྤེལ་བ་བཅས་ཀྱིས་རྒྱལ་ཁབ་རིམ་པའི་གཙོ་སྙིང་དུས་དེབ་སྟེ《ཀྲུང་ཧྭའི་རྩོམ་རིག་འདེམ་སྒྲིག དམའ་འབྲིང་རིམ་པའི་པར་གཞི》ཞེས་པའི་གྲས་སུ་བདམས་ཤིང་། དམའ་འབྲིང་སློབ་མའི་དཔེ་ཀློག་གི་ལེགས་རྩོམ་དུ་བདམས་པ་རེད། 
　རྡོ་སྦིས་རྡོ་རྗེ་ལགས་ཀྱིས་སྔ་རྗེས་སུ་རྒྱ་ཡིག་གི་ལམ་ནས《མཚོ་ལྷོར་བསྐྱོད་པ》དང《ལྷ་ལམ་དུ་མཐོང་བའི་གངས་ལྗོངས》། 《སེམས་ཁོངས་སུ་དྲག་ཏུ་འཕྱོ་བའི་མཚོ་སྔོན་པོ》སོགས་ཆེད་རྩོམ་མང་དུ་བརྩམས་པར་མ་ཟད། 《ཁྲི་ཀའི་རྨ་ཀླུང་དྭངས་མོ》དང་། 《སུམ་རྩེན་ཡུལ་གྱི་ལྷ་མཚོ། མཚོ་སྔོན་ཁྲོ་ཤོག་རྒྱལ་མོ》སོགས་དངོས་ཕབ་གློག་བརྙན་མང་དུ་བསྒྲིགས། ཁོང་གིས་གཙོ་བོ་དུས་དེབ་དང་ཚགས་པར་གྱི་རྩོམ་སྒྲིག་དང་། གནས་སྐོར་ཁྱབ་སྒྲོག་ཇུས་འགོད། གློག་བརྙན་འཕྲིན་འཁྲབ་ཁྲིད། རྩོམ་རིག་གསར་རྩོམ། རིག་གནས་ཁྱབ་སྒྲོག་སོགས་ཀྱི་ལས་ཀ་གཉེར་བཞིན་ཡོད།
