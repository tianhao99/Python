རྩོམ་པ་པོ་སྟག་འབུམ་རྒྱལ་གྱི་བརྩམས་སྒྲུང་གཅེས་བཏུས《ཁ་བ་མེད་པའི་དགུན་ཁ》།
རྩོམ་པ་པོ་སྟག་འབུམ་རྒྱལ་གྱི་བརྩམས་སྒྲུང་གཅེས་བཏུས《ཁ་བ་མེད་པའི་དགུན་ཁ》།
                                   
                                                                             དཀར་ཆག
 
 
 
         རྩོམ་པ་པོ་སྟག་འབུམ་རྒྱལ།  
          ཁོང་ནི1966ལོར་མཚོ་སྔོན་ཞིང་ཆེན་མཚོ་ལྷོ་ཁུལ་མང་རྫོང་རྩྭ་ཐང་དུ་སྐྱེས་ཤིང་། 1986ལོར་མཚོ་ལྷོ་མི་རིགས་དགེ་ཐོན་སློབ་གྲྭ་ནས་སློབ་མཐར་ཕྱིན།  1988ལོར་ནུབ་བྱང་མི་རིགས་སློབ་གྲྭ་ཆེན་མོའི་གྲངས་ཉུང་མི་རིགས་སྐད་ཡིག་ཚན་ཁག་ཏུ་ཟབ་སྦྱོང་བྱས།  
          དུས་རབས་སྔོན་མའི་ལོ་རབས་བརྒྱད་ཅུ་བའི་མཇུག་ཏུ་གསར་རྩོམ་གྱི་ལས་ལ་ཞུགས་ཤིང་། སྔ་གཞུག་ཏུ《མཚོ་སྔོན་བོད་ཡིག་གསར་འགྱུར》དང《སྦྲང་ཆར》། 《བོད་ཀྱི་རྩོམ་རིག་སྒྱུ་རྩལ》། 《གངས་རྒྱན་མེ་ཏོག》སོགས་ཞིང་ཆེན་ཕྱི་ནང་གི་དུས་དེབ་དང་ཚགས་པར་སྟེང་དུ་སྒྲུང་གཏམ་དྲུག་ཅུ་ལྷག་སྤེལ་ཞིང་།  དེའི་ཁྲོད་ཀྱི《ཧཱ་པ་གསོས་པའི་ཟིན་བྲིས》།  《སྒོ་ཁྱི》། 《ཁྱི་རྒན》། 《མི་ཚེའི་གླུ་དབྱངས》།  《ཁྱི་དང་བདག་པོ།  ད་དུང་གཉེན་ཚན་དག》སོགས་སྒྲུང་གཏམ་སྐོར་ཞིག་རྒྱ་ཡིག་ཏུ་བསྒྱུར་ནས《མི་རིགས་རྩོམ་རིག》སོགས་ཀྱི་སྟེང་དུ་བཀོད་ཡོད་ལ།  འགའ་རེ་སློབ་གྲྭ་ཆེན་མོ་ཁག་གི་བསླབ་དེབ་ཏུ་བདམས་འདུག    
          སྔ་གཞུག་ཏུ་སྐབས་གཉིས་པ་དང་བཞི་བ་དང་ལྔ་བའི“སྦྲང་ཆར་བྱ་དགའ”དང་།  མཚོ་སྔོན་ཞིང་ཆེན་གྱི་སྐབས་བཞི་བ་དང་ལྔ་བའི་ཕུལ་བྱུང་རྩོམ་རིག་བྱ་དགའ།  རྒྱལ་ཡོངས་སྐབས་གཉིས་བའི་གངས་རྒྱན་བུམ་པའི་རྩོམ་རིག་བྱ་དགའ།  མཚོ་སྔོན་པོ་རྩོམ་རིག་བྱ་དགའ(རྒྱ་ཡིག)སོགས་ཐོབ་མྱོང་།  འཐུས་ཚབ་བརྩམས་ཆོས་ལ་སྒྲུབ་རིང《ལྷིང་འཇགས་ཀྱི་རྩྭ་ཐང》དང《རྒུད》།   སྒྲུང་འབྲིང《སྡེ་དཔོན》དང《མེས་པོ་དང་ཁོས་གསོས་པའི་ཁྱུ་གུ་དག》།  《ནོར་རྫི་བྱིས་པ》།   སྒྲུང་ཐུང་ཧྲག་བསྡུས《མི་རབས་གསུམ་གྱི་རྨི་ལམ》སོགས་ཡོད།  
          ད་ལྟ་མཚོ་སྔོན་ཞིང་ཆེན་མང་རྫོང་མི་རིགས་སློབ་འབྲིང་གི་དགེ་རྒན་ཡིན་ལ།  ཀྲུང་གོའི་རྩོམ་པ་པོ་མཐུན་ཚོགས་ཀྱི་ཚོགས་མི་དང་། མཚོ་ལྷོ་ཁུལ་རྩོམ་པ་པོ་མཐུན་ཚོགས་ཀྱི་རྒྱུན་ལས་ལས་འཛིན་པའི་འགན་ཁུར་ཡོད།
