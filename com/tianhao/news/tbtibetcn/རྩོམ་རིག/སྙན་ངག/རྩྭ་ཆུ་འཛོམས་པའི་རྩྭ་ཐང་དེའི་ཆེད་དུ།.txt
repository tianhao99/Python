རྩྭ་ཆུ་འཛོམས་པའི་རྩྭ་ཐང་དེའི་ཆེད་དུ།
བཅར་འདྲི་ཚན་ཁག་རྨ་ཆུའི་བཞུར་རྒྱུད་ཀྱི་རྫོང་གཉིས་པ་སྟེ་དར་ལག་རྫོང་ལ་ཐོན་དུས་དགུང་མོའི་དུས་ཚོད་དྲུག་ལ་ཐོན་པ་དང་། ལམ་ཐག་རིང་བས་ཧ་ཅང་ཐང་ཆད་ནའང་ཆར་ཤུལ་གྱི་མཇའ་ཚོན་རྣམ་པར་བཀྲ་བའི་མཛེས་སྡུག་གིས་ང་ཚའི་མིག་དབང་འཕྲོག་ཅིང་། མཚོ་སྔོན་ལ་རོགས་སྐྱོར་དུ་ཡོང་བའི་ལས་བྱེད་པ། མཚོ་སྔོན་ཞིང་ཆེན་ཁོར་ཡུག་སྲུང་སྐྱོབ་ཐིན་གྱི་ཐིན་ཀྲང་གཞོན་པ་ཀྲིག་ཀྲི་ཆང་གི་བཏོན་པའི་གྲོས་གཞི་ལྟར། ང་ཚོ་ཚང་མ་དར་ལག་རྫོང་ཐོག་གི་ནུབ་རིར་འགོས་པ་དང་རི་མགོ་ནས་རྨ་ཆུའི་བཞུར་རྒྱུད་ལ་བལྟས་དུས། རྨ་ཆུ་ནི་དར་སྐུད་རིང་མོ་ཞིག་དང་འདྲ་བར་ཀྱག་ཀྱོག་གི་བཞུད་ལམ་དེད་ཅིང་། ཆུ་ལག་ཆེ་ཆུང་སྣ་ཚོགས་མིའི་ལུས་སྟེང་གི་ཁྲག་རྩ་དང་འདྲ་བར་ས་གཞི་དང་རི་ལུང་གི་སྐྱེས་དངོས་ཡོད་ཚད་གསོས་ཏེ་ཡོད་ཚད་ལ་གསོན་ཤུགས་སྩོལ་བཞིན་མཆིས། 
    རྨ་ཆུར་མཚོན་ནས་བཤད་ན། དེའི་མགོ་ཁུངས་ནི་བ་ཡན་ཧ་རིའི་བྱང་རྒྱུད་ཡིན་ལ་མཚོ་སྔོན་ནང་ཁུལ་གྱི་རྫོང་དང་གྲོང་ཁྱེར་སོ་གཅིག་བརྒྱུད་ནས་འབབ་བཞིན་ཡོད་པ་དང་། བཞུར་རྒྱུད་དུ་མི་གྲངས་ས་ཡ་གསུམ་དང་འབུམ་དགུ་ཁྲི་དགུ་ཡོད་ཅིང་། རྒྱལ་དམངས་ཐོན་སྐྱེད་ཀྱི་སྤྱིའི་རིན་ཐང་མཚོ་སྔོན་ཞིང་ཆེན་ཡོངས་ཀྱི་བརྒྱ་ཆའི་དོན་བརྒྱད་དང་གྲངས་ཆུང་ལྔ་ཟིན་ལ། རྨོ་ཞིང་གི་རྒྱ་ཁྱོན་ཀུང་ཆིན་འབུམ་ལྔ་དང་ཁྲི་གཅིག་ཡོད་པས། མཚོ་སྔོན་ཞིང་ཆེན་ཡོངས་ཀྱི་སྤྱིའི་རྨོ་ཞིང་གི་བརྒྱ་ཆའི་གྱ་དྲུག་ཟིན་ཡོད་ཀྱང་། སྔོན་ཆད་ས་བརྐོ་རྡོ་སློག་བྱས་མཐར་རྨ་ཆུའི་མགོ་ཁོངས་ཀྱི་སའི་བཅུད་ཉམས་ནས་ས་ཆུ་ཤོར་བ་དང་། རི་ཉིལ་བ། ཆུ་ལོག་དང་རྡོ་ཟེགས་ཉིལ་བ་ལ་སོགས་ཀྱི་རང་བྱུང་གནོད་འཚེ་རྒྱུན་མི་ཆད་པས། རྨ་ཆུའི་བཞུར་རྒྱུད་ཀྱི་མི་དམངས་མང་ཚོགས་ཀྱི་ཐོན་སྐྱེད་འཚོ་བར་གནོད་པ་ཚབས་ཆེན་བཟོ་བཞིན་ཡོད།
    ཆུ་དང་གཙང་བོས་མིའི་རིགས་ལ་བདེ་སྐྱིད་བསྐྲུན་དགོས་ན། དེའི་འབྱུང་ཁུངས་གཙང་གསུམ་འབྱུང་ཡུལ་ལ་སྲུང་སྐྱོབ་བྱེད་དགོས་པས། སྡོང་འཛུགས་རྩྭ་འདེབས་དང་ས་གཞི་སྔོ་ལྗང་དུ་སྒྱུར་ནས་བྱེ་བྲག་གི་ས་ཁུལ་དེའི་ནམ་ཟླར་ལེགས་བཅོས་དང། སྐྱེ་ཁམས་དཔལ་འབྱོར་འཕེལ་རྒྱས་སུ་གཏོང་བ། ཆུའི་ཐོན་ཁུངས་གསོ་སྐྱོང་། ས་ཆུ་སྲུང་འཛིན་བྱེད་པ། ས་འོག་གི་ཆུ་ཇེ་མང་དུ་གཏོང་བ་བཅས་བྱེད་དགོས།
    དར་ལག་རྫོང་ཏང་ཨུའི་ཧྲུའུ་ཅི་ཨོ་རྒྱན་གྱིས་བཅར་འདྲི་དང་ལེན་སྐབས་བཤད་རྒྱུར། བ་ཡན་ཧ་རིའི་རི་རྒྱུད་ཀྱི་དར་ལག་རྫོང་འཕྲེད་དུ་བཅད་དེ། དར་ལ་རྫོང་དེ་རྨ་ཆུ་དང་འབྲི་ཆུའི་ཆུ་རྒྱུད་ཆེན་པོ་གཉིས་སུ་བགོས་པས། སྐྱེས་ཁམས་གོ་གནས་ཧ་ཅང་གལ་ཆེ་ཤིང་། ང་ཚོར་རྨ་ཆུ་དང་འབྲི་ཆུ་ལ་སྲུང་སྐྱོབ་གང་ལེགས་བྱེད་པའི་འོས་འགན་ཡོད་ཟེར།
   རྒྱལ་ཁབ་ཀྱིས་གཙང་གསུམ་འབྱུང་ཡུལ་སྲུང་སྐྱོབ་དང་དེ་འཛུགས་སྐྲུན་བྱེད་པའི་ལས་གཞི་ལག་ལེན་དུ་བསྟར་བ་ནས་བཟུང་། དར་ལག་རྫོང་གི་ལས་བྱེད་པ་དང་མང་ཚོགས་ཀྱིས་ཧུར་ཐག་སྒོས་བྱ་སྤྱོད་སྤེལ་ནས། ཕྱུགས་འཚོ་མཚམས་བཞག་ནས་རྩྭ་འདེབས་པ་དང་ལྕགས་ར་སྐོར་བ། ཨ་བྲ་དང་གནོད་འབུ་སྔོན་འགོག རི་བཀག་ནགས་གསོ་སོགས་ཕྱོགས་བསྡུས་བཅོས་སྐྱོང་བཟོ་སྐྲུན་བརྒྱུད་དེ་ཕྱོགས་ཡོངས་ནས་བཅོས་སྐྱོང་བྱས།
   དར་ལག་རྫོང་རྒྱུད་སྨད་གྲོང་རྡལ་དང་སྡེར་སྣང་ཞང་། སྟག་ཐོག་ཞང་། འཛུགས་སྐྲུན་ཞང་བཅས་ནས་ཁྱོན་བསྡོམས་པས་ས་མུའུ་འབུམ་གཅིག་དང་ཁྲི་བཞི་སྟོང་ལྔ་བདུན་བརྒྱ་ལ་རི་བཀག་ནགས་གསོའི་ལས་གཞི་ལག་བསྟར་བྱས་པ་དང་། བར་བཞི་ཞང་རྟ་སྒ་འབྲོག་དམངས་ཨུ་ཡོན་ལྷན་ཁང་དང་དཔལ་ཡུལ་ཞང་། སོག་རུ་མ་ཞང་བཅས་སུ་རྩྭ་འདེབས་པ་གཙོར་བཟུང་ནས་རྩྭ་བཏབས་པས་ས་མུའུ་འབུམ་གཅིག་དང་སྟོང་གཉིས་ལ་རྩྭ་བཏབ་པ་དང་། དར་ལག་རྫོང་གི་འོད་ཟེར་ཞང་གྲགས་མཆོག་འབྲོག་དམངས་ཨུ་ཡོན་ལྷན་ཁང་གི་གྲམ་ཆུའི་འགྲམ་དུ། གསར་འགོད་པས་ས་དེའི་མང་ཚོགས་ཀྱིས་བཏབས་པའི་རྩྭ་སྐྱེས་པ་ཧ་ཅང་བཟང་བའི་གནས་ཚུལ་ཡུལ་དངོས་ནས་མཐོང་བ་དང་། དར་ལག་རྫོང་གཙང་གསུམ་འབྱུང་ཡུལ་གཞུང་སྒྲུབ་ཁང་གི་འབྲེལ་ཡོད་འགན་ཁུར་བས་མཚམས་སྦྱོར་བྱས་པ་ལྟར་ན། 2006ལོ་ནས་བཟུང་མུའུ་བཞི་བཅུ་ལྷག་ཡོད་པའི་ཆུ་འགྲམ་གྱི་ས་འདི་ཡོངས་ལ་མིའི་ཐབས་ཀྱི་རྩྭ་བཏབས་པ་དང་། ལོ་བདུན་གྱི་རིང་ལ། སྔོན་ཆད་ཀྱི་གྲམ་གཞུང་འདི་ད་ལྟ་སྔོ་ལྗང་གི་རྩྭ་ས་ཆེན་པོ་ཞིག་ཏུ་གྱུར་ཡོད་ཟེར།
    ཨ་བྲའི་གནོད་པ་ལ་དམིགས་ནས་དར་ལག་རྫོང་གིས་མ་དངུལ་སྒོར་མོ་བྱེ་བ་གཅིག་དང་ས་ཡ་གསུམ་འབུམ་གཉིས་ཁྲི་བདུན་བཏང་སྟེ། ཞང་དང་གྲོང་རྡལ་བཅུ་ནས་ཨ་བྲ་དང་གནོད་འབུ་སྔོན་འགོག་བཅོས་སྐྱོང་བྱེད་པའི་བཟོ་སྐྲུན་ལག་བསྟར་བྱས་ནས། ཨ་བྲ་དང་གནོད་འབུ་སྔོན་འགོག་བཅོས་སྐྱོང་བྱས་པའི་རྒྱ་ཁྱོན་མུའུ་བྱེ་བ་གཅིག་དང་ས་ཡ་གསུམ་འབུམ་བཞི་ཁྲི་བདུན་ལ་བསླེབས་པ་དང་། སྔོན་ཆད་ཨ་བྲ་དང་གནོད་འབུས་གནོད་པ་ཆེ་བའི་རྩྭ་ས་མུའུ་བརྒྱད་བརྒྱ་དང་ལྔ་བཅུ་དེ་ད་ལྟ་མུའུ་སུམ་བརྒྱ་དང་སོ་བརྒྱད་དུ་ཇེ་ཉུང་ལ་སོང་ཡོད་ལ། མ་དངུལ་སྒོར་མོ་དུང་ཕྱུར་གཅིག་དང་བྱེ་བ་གསུམ་བཏང་ནས་སྟག་ཐོག་ཞང་དང་། དཔོན་སྐོར་སྟོད་མ་ཞང་སོགས་ཞང་དང་གྲོང་རྡལ་བརྒྱད་ནས་ཕྱུགས་འཚོ་མཚམས་བཞག་ནས་རྩྭ་བཏབས་པ་དང་། རྩྭ་ར་བསྐོར་ནས་ཕྱུགས་འཚོ་མི་ཆོག་པའི་བྱེད་ཐབས་ལག་བསྟར་བྱས་ཏེ། འབྲོག་ཁྱིམ་སྟོང་གསུམ་དང་བཞི་བརྒྱ་ཞེ་བདུན་གྱི་འབྲོག་པ་མང་ཚོགས་ལ་རྒྱལ་ཁབ་ཀྱིས་གཟན་ཆས་རོགས་སྐྱོར་མ་དངུལ་དང་། མེ་ཤིང་གི་རོགས་སྐྱོར་མ་དངུལ་སོགས་ཀྱི་སྒོར་མོ་བྱིན་པ་དང་། རྩྭ་ས་མུའུ་ས་ཡ་བཞི་དང་འབུམ་དྲུག་ཁྲི་བཞི་སྟོང་ལྔ་སུམ་བརྒྱ་ལ་ར་སྐོར་ཡོད་པ་དང་། འབྲོག་ཁྱིམ་གནས་སྤར་ཏེ་ཕྱུགས་འཚོས་མི་ཆོག་པའི་རྩྭ་ས་མུའུ་ས་ཡ་གསུམ་དང་འབུམ་བཞི་ཁྲི་དྲུག་སྟོང་གཅིག་བཞི་བརྒྱ་ལ་བུད་ཡོད་ཅིང་། འབྲོག་པ་ཁྱིམ་ཚང་ལྔ་བརྒྱ་དང་བརྒྱད་གནས་སྤོར་སྡོད་དམངས་བཀོད་སྒྲིག་བྱེད་སར་གནས་སྤར་བ་དང་། ད་ལྟ་དར་ལག་རྫོང་ཡོངས་སུ་ཕྱུགས་ཟོག་འབུམ་གསུམ་ལྷག་ཇེ་ཉུང་ལ་སོང་ཡོད།
