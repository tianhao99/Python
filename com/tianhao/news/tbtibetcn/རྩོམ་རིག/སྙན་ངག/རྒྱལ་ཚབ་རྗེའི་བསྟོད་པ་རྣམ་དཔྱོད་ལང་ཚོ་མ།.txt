རྒྱལ་ཚབ་རྗེའི་བསྟོད་པ་རྣམ་དཔྱོད་ལང་ཚོ་མ།
རྣམ་དཔྱོད་ལང་ཚོ་དར་མའི་ལུས། ། ཡོན་ཏན་རིན་ཆེན་དུ་མས་མཛེས། ། ལེགས་བཤད་མིག་དཀྱུས་རིང་པོ་ཅན། ། མཁས་པའི་ཡིད་འཕྲོག་ཁྱོད་ལ་འདུད། ། ཡབ་རྗེ་དམ་པའི་བཀའ་ལུང་གིས། ། དྲི་མེད་རིགས་པའི་ཅོད་པན་བཅིངས། ། ཆོས་ཀྱིས་སྲིད་ལ་དབང་བསྐུར་བ། ། རྒྱལ་ཚབ་དམ་པ་རྒྱལ་གྱུར་ཅིག ། ཚངས་དང་སྐབས་གསུམ་དབང་པོ་སོགས། ། མཐོ་རིས་མགོན་པོའི་གཙུག་རྒྱན་གྱིས། ། ཞབས་ཀྱི་པདྨོ་ལ་བཏུད་པ། ། ཁྱོད་ལ་བདག་ཅག་མངོནཔར་དད། །  ཆེན་པོ་རྣམས་ཀྱང་མཆོད་པའི་གནས། ། ཚོགས་ཀྱི་ནང་ན་མཁས་པའི་མཆོག། སྐྱོན་མེད་ཡོན་ཏན་ཀུན་ལྡན་པ། ། ཁྱོད་ལ་སྙིང་ནས་སྐྱབས་སུ་མཆི། ། བསྐལ་པར་གོམས་པ་ཁྱོད་ཀྱི་ཐུགས། ། འགྲོ་ལ་བུ་གཅིག་ལྟར་དགོངས་པའི། །བྱམས་དང་སྙིང་རྗེའི་གཏིང་མཐའ་ནི། ། རྒྱ་མཚོའི་དཔེ་ཡིས་ཉེ་བར་མཚོན། ། འདི་ཀུན་བདག་གིས་བསྒྲལ་ལོ་ཞེས། ། འགྲོ་ལ་ཕན་པའི་དམ་བཅའ་ཁུར། ། ཤིན་ཏུ་ལྕི་བས་ནོན་ན་ཡང༌། ། དུབ་པར་མ་གྱུར་ཨེ་མ་མཚར། ། ཐབས་དང་ཤེས་རབ་གོམ་པའི་འགྲོས། ། དྲང་པོའི་ལམ་དུ་བགྲོད་པ་ལས། ། དམ་བཅའི་ཁུར་དེ་མཐར་ཕྱིན་པར། ། གཟིགས་ནས་རྒྱལ་སྲས་ལམ་ལ་འབད། ། ཡིད་འོང་ལུས་དང་ལོངས་སྤྱོད་རྣམས། ། སློང་བའི་འགྲོ་ལ་མ་ཞུམ་པར། ། སྦྱིན་པའི་ཕྱག་རིང་བརྐྱངས་ནས་ཀྱང༌། ། ཕོངས་པ་རྣམས་ཀྱི་དབུགས་དབྱུང་མཛད། །ཕྱི་དང་ནང་གི་དངོས་པོ་ཀུན། ། གཏོང་བ་འབའ་ཞིག་གོམས་པའི་བློས། ། སེར་སྣའི་དྲི་མ་བཅོམ་པ་ལས། ། ཕུན་ཚོགས་ཐམས་ཅད་སྤྲིན་བཞིན་འདུས། ། ལེགས་སྤྱད་འབྱོར་པ་མི་དམན་པའི། ། དལ་དང་འབྱོར་བའི་ཁང་བཟང་ལ། ། བསླབ་གསུམ་ནོར་བུས་རབ་མཚན་པའི། ། སྒྲུབ་པའི་རྒྱལ་མཚན་ལེགས་པར་གཙུགས། ། མཛེས་ཤིང་ཡིད་འཕྲོག་བརྗིད་པའི་སྐུ། ། ཧ་རི་ཙནྡན་ལྗོན་པའི་ཤིང༌། ། ཚུལ་ཁྲིམས་དྲི་ངད་ལྡང་བ་ཡིས། ། ཡིད་ཀྱི་གདུང་བ་མ་ལུས་འཕྲོག ། དཀའ་བ་བརྒྱ་ཕྲག་དུ་མས་ཀྱང༌། ། མི་འཁྲུགས་བཟོད་པའི་གོ་ཆ་གྱོན། ། ཁོང་ཁྲོའི་དགྲ་བོ་དྲུངས་ཕྱུང་ནས། ། དུལ་བའི་དཔལ་དེ་ཁྱོད་ཀྱིས་བརྙེས། ། ལེ་ལོའི་བུད་ཤིང་སྲེག་སླད་དུ། ། བརྩོན་པའི་མེ་ཆེན་སྦར་བ་ཡིས། ། ཉེས་པའི་ཚང་ཚིང་གཞན་དག་ཀྱང༌། ། ངེས་པར་བཅོམ་འདི་རྨད་དུ་བྱུང༌། ། བྱིང་རྒོད་རླུང་གིས་མི་གཡོ་བའི། ། ཏིང་འཛིན་མར་མེ་གསལ་བ་ལ། ། བརྟེན་ནས་ཆོས་ཀྱི་དེ་ཉིད་ནི། ། ཤེས་རབ་མིག་གིས་མངོན་གསུམ་གཟིགས། ། རྣོ་ཞིང་མྱུར་བ་ཟབ་པའི་བློས། །ཟབ་ཅིང་རྒྱ་ཆེའི་ཆོས་རྣམས་ལ། ། མཁས་པ་ཞེས་བྱ་ཁྱོད་ཉིད་དེ། ། གཞན་རྣམས་དེ་ལ་གཟུགས་བརྙན་ནོ། ། འཇིག་རྟེན་མ་ལུས་བྱེད་པ་ལ། ། བྱེད་པོ་མཁས་པར་གྲགས་ཀྱང་དེས། ། ཁྱོད་དང་མཚུངས་པའི་མཁས་པ་གཞན། ། བགྱིད་པར་ནུས་པ་མ་ལགས་སོ། ། དྲི་མེད་རིགས་པའི་བྱི་དོར་གྱིས། ། ཁྱོད་ཐུགས་དག་པ་མེ་ལོང་ལ། ། ཤེས་བྱའི་གཟུགས་བརྙན་ཅིག་ཅར་དུ། ། གསལ་ལ་མ་འདྲེས་རྫོགས་པར་ཤར། ། ཁྱོད་ཀྱི་ཡེ་ཤེས་དུས་མཐའི་མེ། ། ཆོས་དབྱིངས་མཁའ་ལ་འཕྱོ་བ་ན། །འདི་ཡིས་ངེས་པར་སྐེམས་སོ་ཞེས། ། སྡུག་བསྔལ་རྒྱ་མཚོ་འདར་རམ་སྙམ། ། མཁྱེན་པའི་སྐད་ཅིག་གཅིག་གིས་ཀྱང༌། ། ཆོས་ཀུན་ཁོང་དུ་ཆུད་པས་ན། ། ཤེས་བྱའི་ཁྱོན་ལས་ཉིད་ཀྱི་ཐུགས། ། ངེས་པར་ཡངས་པ་ལགས་སམ་སྙམ། ། རྒོལ་བའི་དཀྱིལ་འཁོར་ལགས་བཤམས་པའི། ། མཁས་པའི་མདུན་སར་མགོན་པོ་ཁྱོད། ། ཆོས་བཀའ་རྒྱ་ཆེན་སྒྲོག་པ་ལ། ། མི་འཇིགས་སེང་གེའི་ཁྲི་ལ་དབང༌། ། བློ་མིག་བྲལ་ཡང་མཁས་རློམ་པའི། ། ཕས་རྒོལ་མཐའ་དག་ཁྱོད་དྲུང་ན། ། ཚིག་གི་འཁར་བ་འཁྱོར་བཞིན་དུ། ། རྩོད་པའི་གོམ་པ་གྱ་ཚོམ་འདོར། ། རིགས་གཞུང་ཆུ་གཏེར་ལས་འཐོན་པའི། ། གང་གི་བློ་གྲོས་རི་དབང་ནི། ། ཕ་རོལ་རྒོལ་བའི་རླུང་ཤུགས་ཀྱིས། ། ཅུང་ཟད་ཙམ་ཡང་བསྐྱོད་པ་མིན། །མགོན་ཁྱོད་གདུལ་བྱའི་འཇིག་རྟེན་ལ། །ཉིན་བཞིན་ལེགས་བཤད་འདོམས་པ་ནི། ། སངས་རྒྱས་རྣམས་ཀྱི་འཕྲིན་ལས་བཞིན། །ནམ་ཡང་དུས་ལས་འདའ་མི་སྲིད། ། དཔལ་ལྡན་ཕྱོགས་ཀྱི་རི་མོ་མཁན། ། གདུལ་བྱའི་ཡིད་ཀྱི་རྩིག་ངོས་ལ། ། རྣམ་དཔྱོད་པིར་གྱི་རྩེ་མོ་ཡིས། ། ལེགས་བཤད་རི་མོ་ཅིར་ཡང་འབྲི། ། སངས་རྒྱས་ཉིད་དུ་བགྲོད་པ་ལ། ། གྲངས་མེད་གསུམ་ལ་མི་བལྟོས་པའི། ། རྡོ་རྗེ་ཐེག་པ་ཉེ་བའི་ལམ། ། རྒྱལ་བའི་དགོངས་པ་ཁྱོད་ཀྱིས་རྙེད། ། རྣམ་དག་དབང་གིས་རྒྱུད་སྨིན་ཅིང༌། ། དམ་ཚིག་སྡོམ་པ་ཚུལ་བཞིན་བསྲུང༌། །རིམ་གཉིས་ཟབ་མོ་མཐར་སོན་པ། ། གོ་འཕང་མཆོག་ཏུ་གྱུར་ལ་འདུད། ། དེ་ནས་སྐལ་བཟང་གདུལ་བྱ་ལ། །ཆོས་ཀྱི་བདུད་རྩིའི་བགོ་སྐལ་བྱིན། ། རྨོངས་པའི་དྲི་མ་མ་ལུས་བཀྲུས། ། མཚུངས་པ་མེད་པའི་བཀའ་དྲིན་མཛད། ། ཤེས་བྱ་ཀུན་ལ་མངའ་བསྒྱུར་ཀྱང༌། ། སྙིང་རྗེའི་གཞན་དབང་གྱུར་པ་ཁྱོད། ། སྲིད་པ་ཇི་སྲིད་གནས་ཀྱི་བར། ། གཞན་གྱི་དོན་ལ་བརྩོན་པ་ལགས། ། དེ་ཕྱིར་རྒྱལ་བའི་བསྟན་པ་ལ། ། བགྱི་པར་འོས་པ་ཀུན་མཛད་པ། ། མཉམ་མེད་ཐུགས་རྗེའི་གཏེར་ཆེན་ལ། །རྣམ་པ་ཀུན་ཏུ་ཕྱག་བགྱིའོ། ། གང་གི་ཡོན་ཏན་ཆུ་གཏེར་ལས། ། ཐིགས་པའི་ཆ་ཙམ་འཐུངས་པས་ཀྱང༌། ། དད་པའི་སྤུ་ལོང་ཆེར་གཡོས་ཏེ། །གུས་པའི་ཚིག་འདི་གསོལ་བ་ལགས། ། དྲི་མེད་སྟོན་ཀའི་ཟླ་ཟེར་ལྟར། །མཆོག་ཏུ་དཀར་བའི་དགེ་བ་འདིས། ། མགོན་པོ་ཁྱོད་ཀྱི་ཞབས་དྲུང་དུ། ། བདག་ཀྱང་སྐྱེ་བ་ཉིད་གྱུར་ཅིག ། ཁྱོད་ཀྱི་ཞལ་གྱི་བུམ་བཟང་ལས། ། ལེགས་འོང་ལེགས་བཤད་བདུད་རྩིའི་རྒྱུན། ། མ་ལུས་དད་པའི་སྙིམ་པ་ཡིས། ། ངོམས་པ་མེད་པར་འཐུང་བར་ཤོག ། རྒྱལ་བའི་གསུང་གི་གཅེས་པའི་མཛོད། ། མཐའ་དག་རྒྱལ་བའི་དགོངས་པ་བཞིན། ། ཉམས་པ་མེད་པར་འཛིན་པ་ལ། ། འཇམ་པའི་དབྱངས་སུ་བདག་གྱུར་ཅིག ། སྐུ་ལ་མཚན་དཔེའི་དཔལ་འབར་ཞིང༌། ། གསུང་དབྱངས་ཡན་ལག་དྲུག་ཅུས་བརྒྱན། ། ཐུགས་ནི་ཟབ་ཡངས་མཁྱེན་བརྩེའི་གཏེར། ། དཔལ་ལྡན་བླ་མའི་བཀྲ་ཤིས་ཤོག ། འདི་པཎ་ཆེན་དགེ་འདུན་གྲུབ་ཀྱིས་མཛད་པའོ།།།།
