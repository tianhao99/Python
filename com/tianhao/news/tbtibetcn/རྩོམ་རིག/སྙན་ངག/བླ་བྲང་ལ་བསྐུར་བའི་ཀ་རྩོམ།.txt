ཁ་ནས་ཅི་བཤད་མེད་པའི་ཨ་ཁུ་འགས། །
ག་ཡང་གནས་ཆུང་འཕྲིན་ལས་རྒྱལ་པོ་ཡིས། །
ང་རྒྱལ་ཆེ་བས་སྡོད་དུ་མ་བཅུག་ཟེར། །
 
ཅ་དག་བྱེད་པའི་ཆོས་སྐྱོང་ཞིག་ཡོད་ན། །
ཆ་རྒྱུས་ཡོད་མེད་ས་ཆ་ཀུན་འགྲིམས་ནས། །
ཇ་ཆང་སྐམ་ལུག་སོགས་ཀྱི་ཚོང་བྱེད་པའི། །
ཉ་ཉོག་དེ་ཚོ་སྡོད་དུ་ཅི་ལ་བཅུག །

ཐ་ཆད་ལག་ཆ་ལྕགས་གྲི་ཇ་ཤིང་བཟུང་། །
ད་ལྟ་དེ་ཚོ་བསྐྲད་ན་ཆོག་པ་ལ། །
ན་ནིང་དོ་ཚིགས་ཇེ་མང་ཇེ་མང་རེད། །
 
པ་སངས་ལྟ་བུའི་དད་པ་མེད་པས་ན། །

བ་ལང་མཛོ་མོ་བྱ་བྱིའུ་འབའ་འབུ་སོགས། །

ཙ་ཏུར་མཆེ་ལྡན་གནས་ཆུང་རྒྱལ་པོ་ཡིས། །

ཝ་མ་ཝོ་ལ་གཞན་དུ་སྐྲོད་དོན་མེད། །
 
ཞྭ་གོས་ལྷམ་སོགས་བཟང་བའི་བསྟན་བཤིག་དང་། །

འ་ཅག་རྣམས་ཀྱིས་བལྟས་ན་ཁྱད་ཆེ་ཡང་། །

རྭ་བསྡུས་བསེ་བསྡུས་ཤེས་པའི་ང་རྒྱལ་ཅན། །
ལ་ལུང་ཡུལ་གྲུ་གཞན་ལ་སྐྲོད་པ་ལས། །
ཤ་ཆང་དུད་ཚོང་བྱེད་པའི་ང་རྒྱལ་ཅན། །
ས་ཆ་གཞན་དུ་བསྐྲད་ན་ཅིས་མ་ལེགས། །
 
ཧ་ཧ་ཨེ་བདེན་རང་གིས་བསམ་བློ་ཐོངས། །
ཨ་ཁུ་དགེ་བཤེས་ཚོ་ལ་ཞིབ་ཏུ་དྲིས། །
དེ་སྐད་སྨྲ་བ་པོ་ཡང་ཐ་སྙད་པ། །
རིགས་པའི་སེང་གེ་སཾགྷ་ངྷརྨའོ། །བླ་བྲང་ལ་བསྐུར་བའི་ཀ་རྩོམ།
བླ་བྲང་ལ་བསྐུར་བའི་ཀ་རྩོམ།
 
  
ཀ་ཡེ་ཁོ་བོ་གཞན་དུ་སོང་རྗེས་སུ། །
ཁ་ནས་ཅི་བཤད་མེད་པའི་ཨ་ཁུ་འགས། །
ག་ཡང་གནས་ཆུང་འཕྲིན་ལས་རྒྱལ་པོ་ཡིས། །
ང་རྒྱལ་ཆེ་བས་སྡོད་དུ་མ་བཅུག་ཟེར། །
 
ཅ་དག་བྱེད་པའི་ཆོས་སྐྱོང་ཞིག་ཡོད་ན། །
ཆ་རྒྱུས་ཡོད་མེད་ས་ཆ་ཀུན་འགྲིམས་ནས། ། 
ཇ་ཆང་སྐམ་ལུག་སོགས་ཀྱི་ཚོང་བྱེད་པའི། །
ཉ་ཉོག་དེ་ཚོ་སྡོད་དུ་ཅི་ལ་བཅུག །
 
ཏ་ལའི་འདབ་བཞིན་ཤམ་ཐབས་རྡོག་ག་བརྒྱབ། །
ཐ་ཆད་ལག་ཆ་ལྕགས་གྲི་ཇ་ཤིང་བཟུང་། །
ད་ལྟ་དེ་ཚོ་བསྐྲད་ན་ཆོག་པ་ལ། །
ན་ནིང་དོ་ཚིགས་ཇེ་མང་ཇེ་མང་རེད། །
 
པ་སངས་ལྟ་བུའི་དད་པ་མེད་པས་ན། །
ཕ་རོལ་གཞན་དུ་བསྐྲད་ཅེས་ཁ་ཅིག་སྨྲ། །
བ་ལང་མཛོ་མོ་བྱ་བྱིའུ་འབའ་འབུ་སོགས། །
མ་དག་སེམས་ཅན་དེ་ཚོ་ཅིས་མ་བསྐྲད། །
 
ཙ་ཏུར་མཆེ་ལྡན་གནས་ཆུང་རྒྱལ་པོ་ཡིས། །
ཚ་གྲང་ངལ་དུབ་སྣ་ཚོགས་ཁྱད་བསད་ནས། །
ཛ་ཡའི་བསྟན་ལ་ཐོས་བསམ་བྱེད་མཁན་རྣམས། །
ཝ་མ་ཝོ་ལ་གཞན་དུ་སྐྲོད་དོན་མེད། །
 
ཞྭ་གོས་ལྷམ་སོགས་བཟང་བའི་བསྟན་བཤིག་དང་། །
ཟ་མ་ངན་ངོན་ཟ་བའི་བསྟན་བཤིག་གཉིས། །
འ་ཅག་རྣམས་ཀྱིས་བལྟས་ན་ཁྱད་ཆེ་ཡང་། །
ཡ་གིའི་རྒྱལ་པོས་གཟིགས་ན་ཁྱད་པར་མེད། །
 
རྭ་བསྡུས་བསེ་བསྡུས་ཤེས་པའི་ང་རྒྱལ་ཅན། །
ལ་ལུང་ཡུལ་གྲུ་གཞན་ལ་སྐྲོད་པ་ལས། །
ཤ་ཆང་དུད་ཚོང་བྱེད་པའི་ང་རྒྱལ་ཅན། །
ས་ཆ་གཞན་དུ་བསྐྲད་ན་ཅིས་མ་ལེགས། །
 
ཧ་ཧ་ཨེ་བདེན་རང་གིས་བསམ་བློ་ཐོངས། །
ཨ་ཁུ་དགེ་བཤེས་ཚོ་ལ་ཞིབ་ཏུ་དྲིས། །
དེ་སྐད་སྨྲ་བ་པོ་ཡང་ཐ་སྙད་པ། །
རིགས་པའི་སེང་གེ་སཾགྷ་ངྷརྨའོ། །བླ་བྲང་ལ་བསྐུར་བའི་ཀ་རྩོམ།
བླ་བྲང་ལ་བསྐུར་བའི་ཀ་རྩོམ།
 
  
ཀ་ཡེ་ཁོ་བོ་གཞན་དུ་སོང་རྗེས་སུ། །
ཁ་ནས་ཅི་བཤད་མེད་པའི་ཨ་ཁུ་འགས། །
ག་ཡང་གནས་ཆུང་འཕྲིན་ལས་རྒྱལ་པོ་ཡིས། །
ང་རྒྱལ་ཆེ་བས་སྡོད་དུ་མ་བཅུག་ཟེར། །
 
ཅ་དག་བྱེད་པའི་ཆོས་སྐྱོང་ཞིག་ཡོད་ན། །
ཆ་རྒྱུས་ཡོད་མེད་ས་ཆ་ཀུན་འགྲིམས་ནས། ། 
ཇ་ཆང་སྐམ་ལུག་སོགས་ཀྱི་ཚོང་བྱེད་པའི། །
ཉ་ཉོག་དེ་ཚོ་སྡོད་དུ་ཅི་ལ་བཅུག །
 
ཏ་ལའི་འདབ་བཞིན་ཤམ་ཐབས་རྡོག་ག་བརྒྱབ། །
ཐ་ཆད་ལག་ཆ་ལྕགས་གྲི་ཇ་ཤིང་བཟུང་། །
ད་ལྟ་དེ་ཚོ་བསྐྲད་ན་ཆོག་པ་ལ། །
ན་ནིང་དོ་ཚིགས་ཇེ་མང་ཇེ་མང་རེད། །
 
པ་སངས་ལྟ་བུའི་དད་པ་མེད་པས་ན། །
ཕ་རོལ་གཞན་དུ་བསྐྲད་ཅེས་ཁ་ཅིག་སྨྲ། །
བ་ལང་མཛོ་མོ་བྱ་བྱིའུ་འབའ་འབུ་སོགས། །
མ་དག་སེམས་ཅན་དེ་ཚོ་ཅིས་མ་བསྐྲད། །
 
ཙ་ཏུར་མཆེ་ལྡན་གནས་ཆུང་རྒྱལ་པོ་ཡིས། །
ཚ་གྲང་ངལ་དུབ་སྣ་ཚོགས་ཁྱད་བསད་ནས། །
ཛ་ཡའི་བསྟན་ལ་ཐོས་བསམ་བྱེད་མཁན་རྣམས། །
ཝ་མ་ཝོ་ལ་གཞན་དུ་སྐྲོད་དོན་མེད། །
 
ཞྭ་གོས་ལྷམ་སོགས་བཟང་བའི་བསྟན་བཤིག་དང་། །
ཟ་མ་ངན་ངོན་ཟ་བའི་བསྟན་བཤིག་གཉིས། །
འ་ཅག་རྣམས་ཀྱིས་བལྟས་ན་ཁྱད་ཆེ་ཡང་། །
ཡ་གིའི་རྒྱལ་པོས་གཟིགས་ན་ཁྱད་པར་མེད། །
 
རྭ་བསྡུས་བསེ་བསྡུས་ཤེས་པའི་ང་རྒྱལ་ཅན། །
ལ་ལུང་ཡུལ་གྲུ་གཞན་ལ་སྐྲོད་པ་ལས། །
ཤ་ཆང་དུད་ཚོང་བྱེད་པའི་ང་རྒྱལ་ཅན། །
ས་ཆ་གཞན་དུ་བསྐྲད་ན་ཅིས་མ་ལེགས། །
 
ཧ་ཧ་ཨེ་བདེན་རང་གིས་བསམ་བློ་ཐོངས། །
ཨ་ཁུ་དགེ་བཤེས་ཚོ་ལ་ཞིབ་ཏུ་དྲིས། །
དེ་སྐད་སྨྲ་བ་པོ་ཡང་ཐ་སྙད་པ། །
རིགས་པའི་སེང་གེ་སཾགྷ་ངྷརྨའོ། །
