བབ་བསྟུན་མདུན་སྐྱོད་ཀྱིས་རྒྱང་སྐྱོད་ཀྱི་ལམ་བུ་གསར་རྐྱང་འགོ་འཛུགས་དགོས།

རྒྱལ་ཡོངས་སྲིད་གྲོས་སྐབས་བཅུ་གསུམ་པའི་ཚོགས་འདུ་ཐེངས་བཞི་པ་རྒྱལ་ཁའི་ངང་གྲོལ་བར་སྤྲོ་སེམས་འཁོལ་བའི་རྟེན་འབྲེལ་ཞུ།
འབད་འབུངས་ཀྱི་སེམས་ཤུགས་རབ་ཏུ་འཕེལ་བའི་དཔྱིད་ཀྱི་དཔལ་ཡོན་བཞད་པའི་དུས་འདིར་རྒྱལ་ཡོངས་སྲིད་གྲོས་སྐབས་བཅུ་གསུམ་པའི་ཚོགས་འདུ་ཐེངས་བཞི་པས་མངགས་བཅོལ་ཟབ་མོ་ཆུད་ཟོས་སུ་མི་གཏོང་བ་དང་འགན་འཁྲིར་ངོ་གནོངས་མི་དགོས་པར་གྲོས་རིམ་ཁག་ཕུན་སུམ་ཚོགས་པོའི་ངང་ལེགས་གྲུབ་བྱུང་ནས་ཟླ་3ཚེས་10ཉིན་པེ་ཅིང་དུ་རྒྱལ་ཁའི་ངང་གྲོལ་བ་རེད། ང་ཚོས་ཚོགས་ཆེན་ཐུགས་གྲ་ལེགས་འགྲིག་བྱུང་བར་སྤྲོ་སེམས་འཁོལ་བའི་རྟེན་འབྲེལ་ཞུ་རྒྱུ་ཡིན།
འདི་ནི་དམངས་གཙོ་དང་། མཐུན་སྒྲིལ། བདེན་འཚོལ། འབད་འབུངས་བཅས་ཀྱི་ཚོགས་ཆེན་ཞིག་རེད། ཚོགས་འདུའི་རིང་ཀྲུང་གུང་ཀྲུང་དབྱང་གི་སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་དང་། རྒྱལ་ཁབ་ཀྱི་ཀྲུའུ་ཞི། ཀྲུང་དབྱང་དམག་ཨུད་ཀྱི་ཀྲུའུ་ཞི་བཅས་ཡིན་པའི་ཞི་ཅིན་ཕིང་སོགས་ཏང་དང་རྒྱལ་ཁབ་ཀྱི་འགོ་ཁྲིད་བློ་མཐུན་ཚོ་ཨུ་ཡོན་ཚོགས་ཆུང་ནང་ཞུགས་པ་དང་། མཉམ་འབྲེལ་ཚོགས་ཆུང་གི་ཚོགས་འདུར་ཞུགས་ནས་བསམ་འཆར་དང་གྲོས་འགོ་གསན་པ་མ་ཟད། ཨུ་ཡོན་ཚོ་དང་ལྷན་དུ་རྒྱལ་དོན་གླེང་བ་དང་འཕེལ་རྒྱས་དོན་གཉེར་གནང་ཡོད། རྒྱ་ཆེའི་སྲིད་གྲོས་ཀྱི་ཨུ་ཡོན་ཚོས་ལས་འགན་ཏན་ཏིག་སྒྲུབ་པ་དང་འགན་འཁྲི་ཧུར་ཐག་ཁུར་ཏེ། རྒྱལ་ཡོངས་སྲིད་གྲོས་རྒྱུན་ལས་ཨུ་ལྷན་གྱི་ལས་དོན་སྙན་སྒྲོན་དང་གྲོས་གཞིའི་ལས་དོན་གནས་ཚུལ་སྐོར་གྱི་སྙན་ཞུ་གསན་པ་དང་གྲོས་ཞིབ་གནང་བ་དང་། རྒྱལ་ཡོངས་མི་དམངས་འཐུས་ཚོགས་སྐབས་བཅུ་གསུམ་པའི་ཚོགས་འདུ་ཐེངས་བཞི་པར་ཟུར་ཞུགས་གནང་བ། སྲིད་གཞུང་གི་ལས་དོན་སྙན་སྒྲོན་དང་དེ་བཞིན་འབྲེལ་ཡོད་སྙན་སྒྲོན་གཞན་དག་གསན་པ་དང་གྲོས་བསྡུར་གནང་བ། རྒྱལ་དམངས་དཔལ་འབྱོར་དང་སྤྱི་ཚོགས་འཕེལ་རྒྱས་གཏོང་ཕྱོགས་ཀྱི་ལོ་ལྔ་ཚན་བཅུ་བཞི་པའི་འཆར་འགོད་དང་2035ལོའི་རྒྱང་རིང་དམིགས་ཚད་ཀྱི་རྩ་གནད་འཆར་ཟིན་ལ་གྲོས་བསྡུར་གནང་ནས་མོས་མཐུན་གནང་བ་མ་ཟད་རྩ་ཆེའི་བསམ་འཆར་དང་གྲོས་འགོ་ཡང་བཏོན་གནང་ཡོད། ཚང་མས་ཤང་ཀང་དམིགས་བསལ་སྲིད་འཛིན་ཁུལ་གྱི་འདེམས་བསྐོའི་ལམ་ལུགས་འཐུས་སྒོ་ཚང་དུ་གཏོང་རྒྱུར་བློ་བརྟན་འགྱུར་མེད་ངང་རྒྱབ་སྐྱོར་བྱ་རྒྱུའི་མོས་མཐུན་མཐའ་གཅིག་ཏུ་གནང་ཡོད། བྲེལ་བརྩོན་ཆེ་བ་དང་ལས་འགན་ལྕི་བའི་གནས་ཚུལ་འོག་ཨུ་ཡོན་ཚང་མས་ཚད་གཞི་མཐོ་བ་དང་། བླང་བྱ་མཐོ་བ། ལས་ཕྱོད་ཆེ་བ་བཅས་མཐའ་འཁྱོངས་གནང་བ་དང་། ཡིག་ཆར་གྲོས་ཞིབ་ནན་ཏན་དང་གྲོས་མོལ་དང་སྤེལ་རེས་གཏིང་ཟབ་གནང་བ། སྲིད་དོན་ཇུས་འབུལ་ཧུར་ཐག་དང་ཐུན་མོང་གི་ངོས་འཛིན་རྒྱ་ཁྱབ་ངང་གཅིག་སྡུད་གནང་བ་བཅས་ལས་རྒྱལ་ཁབ་བཅོས་སྐྱོང་མ་ལག་ཁྲོད་ཀྱི་ཆེད་དོན་གྲོས་མོལ་སྒྲིག་གཞིའི་གལ་ཆེའི་ནུས་པ་གང་ལེགས་མངོན་པ་དང་། ཀྲུང་གོའི་ཁྱད་ཆོས་ལྡན་པའི་སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་དམངས་གཙོ་ཆབ་སྲིད་ཀྱི་གསོན་ཤུགས་དང་སྐྱེ་སྟོབས་གང་ལེགས་མངོན་ཡོད།
འདི་ལོ་ནི་ཀྲུང་གོ་གུང་ཁྲན་ཏང་དབུ་བརྙེས་ནས་ལོ་ངོ་100འཁོར་བའི་ལོ་ཞིག་ཡིན་ལ། “ལོ་ལྔ་ཚན་བཅུ་བཞི་པའི”འཆར་འགོད་ལག་བསྟར་དང་ཕྱོགས་ཡོངས་ནས་སྤྱི་ཚོགས་རིང་ལུགས་དེང་རབས་ཅན་གྱི་རྒྱལ་ཁབ་འཛུགས་སྐྲུན་བྱེད་པའི་རྒྱང་སྐྱོད་ལམ་བུ་གསར་པ་འགོ་བཙུགས་པའི་ལོ་དང་པོའང་ཡིན། “ལོ་ལྔ་ཚན་བཅུ་བཞི་པའི”བརྗིད་ཆེའི་རྒྱང་སྐྱོད་ལམ་བུར་རྒྱང་ལྟ་བྱེད་སྐབས་དུས་སྐབས་དང་དུས་བབ་གཉིས་ཀ་ང་ཚོར་ཕྱོགས་ཡོད་པ་དང་། དེ་ནི་ང་ཚོའི་སྤོབས་པ་དང་གདེང་ཚོད་གནས་ས་རེད་ལ། ང་ཚོའི་ཡིད་ཆེས་དང་ཆོད་སེམས་གནས་སའང་རེད། དེ་དང་ཆབས་ཅིག་མིག་སྔ་དང་སླད་ཕྱིན་དུས་སྐབས་ཤིག་རིང་མ་གཞི་རང་རྒྱལ་གྱི་འཕེལ་རྒྱས་སྔར་བཞིན་འཐབ་ཇུས་ཀྱི་གོ་སྐབས་གལ་ཆེན་གྱི་དུས་མཚམས་སུ་གནས་ཀྱང་། གོ་སྐབས་དང་འགྲན་སློང་ཚང་མར་འཕེལ་རྒྱས་དང་འགྱུར་ལྡོག་གསར་པ་ཡོང་བ་ངེས་པར་དུ་གསལ་པོ་མཐོང་ཐུབ་པ་བྱེད་དགོས། མདུན་སྐྱོད་ལམ་བུའི་ཐོག་གི་ཉེན་ཁ་དང་འགྲན་སློང་ལས་རྣམ་པར་རྒྱལ་ཏེ“ལོ་ལྔ་ཚན་བཅུ་བཞི་པའི”འཆར་འགོད་དང་2035ལོའི་རྒྱང་རིང་དམིགས་ཚད་མངོན་འགྱུར་བྱེད་པར་ངེས་པར་དུ་ཏང་གིས་ཕྱོགས་ཡོངས་ལ་འགོ་ཁྲིད་བྱ་རྒྱུ་མཐའ་འཁྱོངས་བྱེད་པ་དང་། དགེ་མཚན་ལྡན་པའི་རྒྱུ་རྐྱེན་ཡོད་དོ་ཅོག་སྐུལ་སློང་གང་ལེགས་བྱེད་པ། མཐུན་སྒྲིལ་བྱེད་ཆོག་པའི་སྟོབས་ཤུགས་མཐའ་དག་ལ་མཐུན་སྒྲིལ་རྒྱ་ཁྱབ་ཏུ་བྱེད་པ་བཅས་བྱས་ཏེ་འཕེལ་རྒྱས་ལ་སྐུལ་འདེད་གཏོང་བའི་མཉམ་ཤུགས་ཆེན་པོ་ཞིག་ཆགས་པར་བྱེད་དགོས། ཏང་གིས་ལས་དོན་ཁག་ལ་འགོ་ཁྲིད་བྱ་རྒྱུ་མཐའ་འཁྱོངས་དང་ཤུགས་སྣོན་རྒྱག་པའི་འཐབ་ས་གལ་ཆེན་དང་། ཏང་གི་གསར་གཏོད་ཀྱི་རིགས་པའི་གཞུང་ལུགས་ལ་བརྟེན་ནས་མི་རིགས་ཁག་དང་ལས་རིགས་ཁག་གི་འཐུས་མིར་མཐུན་སྒྲིལ་དང་། སློབ་གསོ། ཁྲིད་སྟོན་བཅས་བྱེད་པའི་ལས་སྟེགས་གལ་ཆེན། ཐུན་མོང་གི་བསམ་བློ་ཆབ་སྲིད་ཀྱི་རྨང་གཞིའི་ཐོག་འགལ་ཟླ་སེལ་བ་དང་ཐུན་མོང་གི་ངོས་འཛིན་གཅིག་སྒྲིལ་བྱེད་པའི་ཐབས་ལམ་གལ་ཆེན་བཅས་ཡིན་པའི་ངོས་ནས་མི་དམངས་སྲིད་གྲོས་ཀྱིས་ངོ་བོ་དང་ངེས་གནས་ཁོང་དུ་ཡང་དག་ཆུད་པར་བྱེད་པ་དང་། བརྗོད་དོན་ཆེ་གྲས་གཉིས་ཡིན་པའི་མཐུན་སྒྲིལ་དང་དམངས་གཙོ་མཐའ་འཁྱོངས་བྱེད་པ། ཏང་དང་རྒྱལ་ཁབ་ཀྱི་ལྟེ་བའི་ལས་འགན་ལ་དམིགས་ནས་འགན་འཁྲི་གཙང་སྒྲུབ་བྱེད་པ། སྲིད་དོན་ཇུས་འབུལ་དང་ཐུན་མོང་ངོས་འཛིན་གཅིག་ཏུ་སྒྲིལ་བའི་ལས་དོན་ཡག་པོ་སྒྲུབ་པ། ཆུ་ཚད་མཐོ་བའི་འགན་སྒྲུབ་ལ་བརྟེན་ནས་སྤུས་ཚད་མཐོ་བའི་འཕེལ་རྒྱས་ལ་ཞབས་འདེགས་ཞུ་བ་བཅས་བྱས་ཏེ། “ལོ་ལྔ་ཚན་བཅུ་བཞི་པའི”འགོ་ཡག་པོ་འཛུགས་པ་དང་གོམ་པ་ཡག་པོ་སྤོ་བར་ཀུན་གྱི་བློ་གྲོས་གཅིག་སྡུད་དང་ཀུན་གྱི་སྟོབས་ཤུགས་གཅིག་སྒྲིལ་བྱེད་དགོས།
དུས་རབས་གསར་པར་ལས་འགན་གསར་པ་ཡོད་པ་དང་རྒྱང་སྐྱོད་ལམ་བུ་གསར་པར་བྱས་རྗེས་གསར་པ་འཇོག་དགོས། མི་དམངས་སྲིད་གྲོས་ཀྱིས་ཞི་ཅིན་ཕིང་གི་དུས་རབས་གསར་པའི་ཀྲུང་གོའི་ཁྱད་ཆོས་ལྡན་པའི་སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་དགོངས་པ་མཛུབ་ཁྲིད་དུ་བྱས་ཏེ། འཕེལ་རྒྱས་ཀྱི་དུས་མཚམས་གསར་པ་ཁོང་དུ་ཆུད་པ་དང་། འཕེལ་རྒྱས་ཀྱི་འདུ་ཤེས་གསར་པ་ལག་བསྟར་བྱེད་པ། འཕེལ་རྒྱས་ཀྱི་བཀོད་པ་གསར་པ་གཏོད་པ་བཅས་ལ་དམིགས་ནས་ཆབ་སྲིད་གྲོས་མོལ་དང་། དམངས་གཙོ་ལྟ་སྐུལ། སྲིད་ཞུགས་སྲིད་གླེང་བཅས་ཀྱི་ཆུ་ཚད་ཟམ་མི་ཆད་པར་མཐོ་རུ་གཏོང་བ་དང་། ཐུན་མོང་གི་ངོས་འཛིན་དེ་བས་ལེགས་པའི་སྒོ་ནས་གཅིག་སྒྲིལ་ཡོང་བ་བྱས་ཏེ་ཀྲུང་གུང་ཀྲུང་དབྱང་གི་ཐབས་ཇུས་བཀོད་སྒྲིག་དང་མི་དམངས་སྲིད་གྲོས་ཀྱི་ལས་དོན་ལ་བཏོན་པའི་བླང་བྱ་དོན་འཁྱོལ་བྱེད་པ། རྒྱལ་ཁབ་ཕྱི་ནང་གི་ཀྲུང་ཧྭའི་བུ་ཕྲུག་གི་བློ་གྲོས་དང་སྟོབས་ཤུགས་གཅིག་སྒྲིལ་བྱེད་པའི་ཆབ་སྲིད་ཀྱི་འགན་འཁྲི་ཏན་ཏིག་ཁུར་དགོས། རྒྱ་ཆེའི་སྲིད་གྲོས་ཨུ་ཡོན་ཚོས་རྒྱལ་ཁབ་ཆེད་དུ་འགན་ནུས་སྒྲུབ་པ་དང་མི་དམངས་ཆེད་དུ་འགན་ཁུར་བའི་བསམ་པ་མཐའ་འཁྱོངས་བྱས་ཏེ་ལས་ཀའི་འཇོན་ཐང་རྒྱུན་ཆད་མེད་པར་ཆེ་རུ་བཏང་ནས་འགན་འཁྲི་སེམས་ལ་འཆང་བ་དང་། ཕྲག་ཏུ་ཁུར་བ། ལག་ལེན་ཐོག་འཁྱོལ་བ་བཅས་བྱས་ནས་མི་དམངས་སྲིད་གྲོས་ལམ་ལུགས་ནང་ཞུགས་མཁན་དང་། ལག་བསྟར་བྱེད་མཁན། སྐུལ་འདེད་གཏོང་མཁན་བཅས་ཡག་པོ་ཞིག་བྱས་ཏེ་རྒྱལ་ཁབ་ལ་དྲིན་ལན་འཇལ་བའི་འདུན་པ་དང་། དམངས་ལ་ཞབས་འདེགས་ཞུ་བའི་བསམ་པ། འགན་སྒྲུབ་ཀྱི་ནུས་པ་བཅས་ཟུང་འབྲེལ་ཡག་པོ་བྱས་ནས་འགན་འཁྲི་གསར་པ་ཁུར་བ་དང་། གཟི་བརྗིད་གསར་པ་གཏོད་པ། ལེགས་སྐྱེས་གསར་པ་འབུལ་བ་བཅས་བྱེད་དགོས།
སེམས་མཐུན་ཕུགས་འདུན་མཉམ་དུ་བཏོད་དེ་བསྐྱར་དར་གྱི་རླབས་ཆེན་བྱ་གཞག་མཉམ་དུ་གླེང་བ་དང་བབ་བསྟུན་འཕྲལ་སྒྲུབ་བྱས་ནས་རྒྱང་སྐྱོད་ལམ་བུ་གསར་པ་འགོ་འཛུགས་དགོས། “ལོ་ལྔ་ཚན་བཅུ་བཞི་པའི”འཆར་འགོད་དང་2035ལོའི་རྒྱང་རིང་དམིགས་ཚད་མངོན་འགྱུར་བྱེད་པར་དོན་སྙིང་གལ་ཆེན་ལྡན་པ་དང་། ལས་འགན་དཀའ་ཚེགས་ཆེ་བ། མདུན་ལམ་ཡངས་པ་བཅས་ཡོད། རླབས་ཆེན་གྱི་དམིགས་ཚད་ལས་མཐུན་སྒྲིལ་འབད་འཐབ་བྱེད་དགོས་པའི་འབོད་སྐུལ་དང་གཟི་བརྗིད་ལྡན་པའི་འགན་འཁྲིས་གསར་འབྱེད་མདུན་སྐྱོད་བྱེད་དགོས་པའི་འབོད་སྐུལ་བྱས་ཡོད། ང་ཚོས་བློ་མཐུན་ཞི་ཅིན་ཕིང་གིས་སྲོག་ཤིང་གནང་བའི་ཏང་ཀྲུང་དབྱང་གི་མཐའ་འཁོར་དུ་དེ་བས་མཐུན་སྒྲིལ་དམ་ཟབ་བྱས་ནས“འདུ་ཤེས་བཞི”ཟབ་ཏུ་གཏོང་བ་དང་། “གདེང་ཚོད་བཞི”བརྟན་དུ་གཏོང་བ། “སྲུང་སྐྱོང་གཉིས”ཐུབ་པ་བཅས་བྱས་ཏེ་ཀུན་སེམས་གཅིག་སྒྲིལ་དང་། གསར་འབྱེད་ཡར་བརྩོན། ལས་དོན་ཏན་ཏིག་སྒྲུབ་པ་བཅས་བྱས་ཏེ“ལོ་ལྔ་ཚན་བཅུ་བཞི་པའི”རིང་གི་དཔལ་འབྱོར་དང་སྤྱི་ཚོགས་འཕེལ་རྒྱས་ཡོང་བའི་འགོ་ཡག་པོ་འཛུགས་པ་དང་། ཕྱོགས་ཡོངས་ནས་སྤྱི་ཚོགས་རིང་ལུགས་དེང་རབས་ཅན་གྱི་རྒྱལ་ཁབ་འཛུགས་སྐྲུན་གྱི་རྒྱལ་ཁ་གསར་པ་ལེན་པ། ཀྲུང་ཧྭ་མི་རིགས་རླབས་ཆེན་བསྐྱར་དར་གྱི་ཀྲུང་གོའི་ཕུགས་འདུན་མངོན་འགྱུར་བྱེད་པ་བཅས་ཀྱི་ཆེད་དུ་འབད་འཐབ་བྱེད་དགོས།
རྩོམ་སྒྲིག་པ། བདེ་སྒྲོལ།
