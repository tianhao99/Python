ཡིད་ཆེས་བརྟན་པོ་བཅངས་ནས་བློ་བརྟན་འགྱུར་མེད་ངང་རང་བརྟེན་གསར་གཏོད་བྱེད་དགོས།

ཀྲུང་གོ་ཚན་རིག་ཁང་གི་ཡོན་ཧྲིའི་ཚོགས་ཆེན་ཐེངས་བཅུ་དགུ་པ་དང་ཀྲུང་གོ་བཟོ་སྐྲུན་ཁང་གི་ཡོན་ཧྲིའི་ཚོགས་ཆེན་ཐེངས་བཅུ་བཞི་པའི་ཐོག་སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིན་ཕིང་གིས་འཛམ་གླིང་གི་ཚན་རྩལ་འཕེལ་ཕྱོགས་སྟངས་འཛིན་ཡང་དག་དང་ཚན་རྩལ་གསར་གཏོད་ཀྱི་གལ་ཆེའི་རང་བཞིན་དང་དགོས་ངེས་རང་བཞིན་ལ་རྒྱས་བཤད་གཏིང་ཟབ་གནང་སྟེ་རྒྱ་ཆེའི་ཚན་རྩལ་ལས་དོན་པས་བློ་ཁོག་ཆེན་པོས་དུས་རབས་གསར་པའི་ཚན་རྩལ་གསར་གཏོད་ཀྱི་དཔེ་དམག་བྱ་རྒྱུར་ཐུགས་རེ་ཟབ་མོ་གནང་བ་དེས་རྒྱ་ཆེའི་ཚན་རྩལ་ལས་དོན་པར་བློ་བརྟན་འགྱུར་མེད་ངང་རང་བརྟེན་གསར་གཏོད་བྱ་རྒྱུར་ལྕག་སྐུལ་ཆེན་པོ་ཐེབས་པ་མ་ཟད། ཏང་ཡོངས་དང་། རྒྱལ་ཡོངས། སྤྱི་ཚོགས་ཡོངས་བཅས་སྐུལ་སློང་བྱས་ནས་ཀུན་སེམས་གཅིག་སྒྲིལ་ངང་འཛམ་གླིང་གི་ཚན་རྩལ་སྟོབས་ཆེན་རྒྱལ་ཁབ་སྐྲུན་པའི་དམིགས་ཚད་རླབས་པོ་ཆེ་སྒྲུབ་ཕྱིར་བློ་ཁོག་ཆེན་པོས་མདུན་སྐྱོད་བྱ་རྒྱུར་སྐུལ་ལྕག་ཆེན་པོའང་ཐེབས་ཡོད།
དེང་གི་འཛམ་གླིང་འདིར་སུས་ཚན་རྩལ་གསར་གཏོད་ཀྱི་འགག་རྩ་དེ་བཟུང་བ་དང་སུས་ཚན་རྩལ་གསར་གཏོད་ཐད་སྔོན་ལ་བྱ་སྤྱོད་སྤེལ་ཚེ་དེར་སྔོན་ལ་གོ་སྐབས་དང་ལེགས་ཆ་ཐོབ་ཐུབ། ལོ་དེ་དག་རིང་རང་རྒྱལ་གྱི་ཚན་རྩལ་གསར་གཏོད་ཀྱི་གྲུབ་འབྲས་གལ་ཆེན་རྒྱུན་མི་ཆད་པར་ཐོབ་ནས་འཐབ་ས་མདུན་མའི་ཁ་ཕྱོགས་ཁག་གཅིག་མཉམ་བགྲོད་དང་མདུན་གྲལ་སྣེ་ཁྲིད་ཀྱི་དུས་མཚམས་སུ་སླེབས་ཡོད་ཀྱང་ཚན་རྩལ་ཁྱབ་ཁོངས་སུ་ཐག་གཅོད་བྱེད་དགོས་པའི་གནད་དོན་མངོན་གསལ་དོད་པོ་ཁག་གཅིག་སྔར་བཞིན་གནས་ཡོད། ཚན་རྩལ་གསར་གཏོད་ཀྱི་འགྲན་ར་ཆེ་མོར་ང་ཚོའི་དཔུང་ཁག་རྗེས་སུ་མི་ལུས་པ་བྱེད་པར་ངེས་པར་དུ་སྔོན་འགྲོའི་རྗེས་གང་མགྱོགས་ཟིན་པ་བྱེད་པ་དང་། མགྱོགས་མྱུར་རྗེས་ཟིན་ཐབས་བྱེད་པ། ཚད་བརྒལ་གང་ཐུབ་ཡོང་བ་བྱེད་དགོས། སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིན་ཕིང་གིས“ཀྲུང་གོ་འདི་ཉིད་སྟོབས་དང་ལྡན་པ་དང་བསྐྱར་དར་ཡོང་བ་བྱེད་པར་ངེས་པར་དུ་ཚན་རིག་ལག་རྩལ་འཕེལ་རྒྱས་ཤུགས་ཆེན་བཏང་ནས་འཛམ་གླིང་གི་ཚན་རིག་ལྟེ་གནས་གཙོ་བོ་དང་གསར་གཏོད་ཀྱི་མཐོ་སར་ཧུར་ཐག་འགྱུར་བ་བྱེད་དགོས” “ང་ཚོར་ལོ་རྒྱུས་ཐོག་གི་དུས་སྐབས་གང་རུང་ཞིག་ལས་འཛམ་གླིང་གི་ཚན་རྩལ་སྟོབས་ཆེན་རྒྱལ་ཁབ་བསྐྲུན་རྒྱུའི་དགོས་མཁོ་དེ་བས་ཆེ་བ་ཡོད”ཅེས་ནན་གསུངས་གནང་བ་ལྟར་རེད།
གོ་ལ་ཧྲིལ་པོའི་ཕྱོགས་བསྡུས་རྒྱལ་སྟོབས་ཀྱི་འགྲན་རྩོད་ཉིན་བཞིན་ཛ་དྲག་ཏུ་འགྲོ་བའི་སྐབས་ང་ཚོར་གདམ་ག་བྱ་ཡུལ་མང་པོ་མེད་པས་རང་བརྟེན་གསར་གཏོད་ཀྱི་འགྲོ་ལམ་ཐོག་མ་བསྐྱོད་ཀ་མེད་རེད། ལག་ལེན་ལས་འགག་རྩའི་ལྟེ་བའི་ལག་རྩལ་ལེན་ས་མེད་པ་དང་། ཉོ་ས་མེད་ལ་སློང་སའང་མེད་པ་ང་ཚོར་ཡང་ཡང་བསྟན་ཡོད། ཏང་གི་ཚོགས་ཆེན་བཅོ་བརྒྱད་པ་འཚོགས་ཚུན། སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིན་ཕིང་གིས་ཕེབས་ཡུལ་གལ་ཆེན་མང་པོ་ཞིག་ཏུ་ལྟེ་བའི་ལག་རྩལ་ནི་ང་ཚོའི་སྲོག་རྩ་ཡིན་པ་དང་ལྟེ་བའི་ལག་རྩལ་ལ་མིའི་ཚོད་འཛིན་ཕོག་པ་ནི་ང་ཚོའི་མངོན་མེད་རྐྱེན་ངན་ཆེ་ཤོས་དེ་རེད་ཅེས་གསུངས་མྱོང་ཡོད། འགག་རྩའི་ལྟེ་བའི་ལག་རྩལ་རང་ཉིད་ཀྱིས་སྟངས་འཛིན་ཐུབ་ཚེ་གཞི་ནས་རྒྱལ་ཁབ་ཀྱི་དཔལ་འབྱོར་བདེ་འཇགས་དང་། རྒྱལ་སྲུང་བདེ་འཇགས། བདེ་འཇགས་གཞན་དག་བཅས་རྩ་བ་ནས་འགན་ལེན་བྱེད་ཐུབ། འགག་རྩའི་ཐུན་མོང་གི་ལག་རྩལ་དང་། མདུན་ཕྱོགས་ཀྱི་སྣེ་ཁྲིད་ལག་རྩལ། དེང་རབས་ཀྱི་བཟོ་སྐྲུན་ལག་རྩལ། རྩ་སློག་རང་བཞིན་གྱི་ལག་རྩལ་བཅས་གསར་གཏོད་ཚད་བརྒལ་བྱ་གཞིར་བྱས་ནས་འགག་རྩའི་ལྟེ་བའི་ལག་རྩལ་ལ་རང་བརྟེན་ཚོད་འཛིན་ཧུར་ཐག་ཐུབ་ཚེ། གཞི་ནས་གསར་གཏོད་ཀྱི་རང་འགུལ་དབང་ཆ་དང་འཕེལ་རྒྱས་ཀྱི་རང་འགུལ་དབང་ཆ་རང་ཉིད་ཀྱི་ལག་ཏུ་ནན་པོ་འཛིན་ཐུབ།
གདེང་ཚོད་ལྡན་པའི་རྒྱལ་ཁབ་དང་མི་རིགས་ཀྱིས་གཞི་ནས་འབྱུང་འགྱུར་གྱི་འགྲོ་ལམ་ཐོག་བརྟན་པོའི་ངང་རྒྱང་རིང་དུ་བསྐྱོད་ཐུབ། སྤྱི་ཁྱབ་ཧྲུའུ་ཅི་ཞི་ཅིན་ཕིང་གིས་གསུངས་པ་བཞིན་གསར་གཏོད་ཚང་མ་ནི་ཕམ་ཉེས་ལས་ལེགས་གྲུབ་བྱུང་བ་ཞིག་ཡིན་ནའང་ང་ཚོས་ངེས་པར་དུ“དོན་གཉེར་བྱེད་དགོས་རིགས་སེམས་ཐག་བཅད་ཚེ་འཆི་སྲོག་ལ་ཐུག་ཀྱང་གནོང་འགྱོད་མེད་པའི་སྤོབས་སེམས་འཆང་དགོས་པ”བསྟན་ཡོད། འབྱུང་འགྱུར་ལ་ཁ་ཕྱོགས་ནས་རང་བརྟེན་གསར་གཏོད་ཀྱི་ནུས་པ་ཆེ་རུ་གཏོང་བར་བློ་བརྟན་འགྱུར་མེད་ཀྱི་སྒོ་ནས་ཀྲུང་གོའི་ཁྱད་ཆོས་ཀྱི་རང་བརྟེན་གསར་གཏོད་ཀྱི་འགྲོ་ལམ་གསར་པའི་ཐོག་བསྐྱོད་པ་དང་རང་བརྟེན་གསར་གཏོད་དང་། གཙོ་གནད་མཆོང་སྐྱོད། འཕེལ་རྒྱས་ལ་འདེགས་སྐྱོར། འབྱུང་འགྱུར་ལ་སྣེ་ཁྲིད་བཅས་ཀྱི་བྱེད་ཕྱོགས་མཐའ་འཁྱོངས་བྱས་ནས་གསར་གཏོད་རྣམ་པའི་རྒྱལ་ཁབ་སྐྲུན་པའི་གོམ་སྟབས་མགྱོགས་སུ་གཏོང་རྒྱུ་ནི་གལ་འགངས་ཆེ་ཤོས་ཡིན། གསར་གཏོད་བྱེད་རིང་། རང་གཉའ་རང་གིས་མི་གཅོག་ལ་རང་མཐོང་རང་ཆེའང་མི་བྱེད་པར་རང་བརྟེན་གསར་གཏོད་བྱ་རྒྱུ་སྒོ་དབྱེའི་ཁོར་ཡུག་ཁྲོད་ཀྱི་གསར་གཏོད་ཅིག་ཡིན་པས། ཕྱོགས་བཞི་མཚམས་བརྒྱད་ཀྱི་སྟོབས་ཤུགས་གཅིག་ཏུ་སྒྲིལ་དགོས་པ་ལས་སྒོ་བརྒྱབ་སྟེ་གཏན་ནས་སྡོད་མི་རུང་ལ། རྒྱལ་སྤྱིའི་ཚན་རྩལ་སྤེལ་རེས་ཀྱི་མཉམ་ལས་ཟབ་ཏུ་བཏང་ནས་གོ་ལ་ཧྲིལ་པོའི་ཚན་རྩལ་གསར་གཏོད་ཀྱི་དྲ་རྒྱ་དང་མཉམ་འདྲེས་ཐུབ་པ་དང་གོ་ལ་ཧྲིལ་པོའི་ཚན་རྩལ་བཅོས་སྐྱོང་བྱ་བར་ཤུགས་ཆེན་ཞུགས་ཏེ་འགོ་འཛུགས་ས་དེ་བས་མཐོ་བའི་སྒོ་ནས་རང་བརྟེན་གསར་གཏོད་ལ་སྐུལ་འདེད་གཏོང་དགོས་པའི་ཐད་ངོས་འཛིན་ཐུབ་པ་བྱེད་དགོས།
རང་སྟོབས་རང་སྤེལ་ནི་འཛམ་གླིང་མི་རིགས་ཀྱི་ཁྲོད་དུ་ཀྲུང་ཧྭ་མི་རིགས་འགྱིང་ངེར་གནས་པའི་འབད་འཐབ་ཀྱི་གཞི་རྩ་ཡིན་པ་དང་། རང་བརྟེན་གསར་གཏོད་ནི་ང་ཚོས་འཛམ་གླིང་ཚན་རྩལ་གྱི་ཡང་རྩེར་འཛེག་པའི་ལྡོག་མེད་ཀྱི་ལམ་བུ་ཞིག་ཡིན། རྒྱ་ཆེའི་ཚན་རྩལ་ལས་དོན་པས་གསར་གཏོད་ཀྱི་ཡིད་ཆེས་དང་ཆོད་སེམས་ཆེན་པོ་བཅངས་ཏེ་བློ་ཁོག་ཆེན་པོས་འགག་སྒྲོལ་དཀའ་སེལ་དང་། ཕུལ་བྱུང་དོན་གཉེར། རྒྱལ་ཁ་ཐོབ་ཐབས་བཅས་བྱས་ནས་ཚན་རྩལ་གྱི་འགྲན་རྩོད་དང་འབྱུང་འགྱུར་གོང་འཕེལ་གྱི་མཐོ་ས་བརྩོན་ལེན་ཧུར་ཐག་བྱས་ཚེ། ལོ་རྒྱུས་ཀྱིས་དཀྲིས་པའི་ལས་འགན་ལྕི་པོ་ཕྲག་ཏུ་ཁུར་ནས་འཛམ་གླིང་གི་ཚན་རྩལ་སྟོབས་ཆེན་རྒྱལ་ཁབ་བསྐྲུན་པའི་འོད་སྟོང་བྱས་རྗེས་གཏོད་ཐུབ་ངེས་རེད།
