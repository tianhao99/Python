བོད་རང་སྐད་ཀྱི་རྒྱུན་མཁོའི་ཕལ་ཚིག་དང་ཞེ་ཚིག
ཞེ་ཚིག་ཅེས་པ་ནི་གཞན་ལ་མོས་གུས་དང་ལྡན་པའི་སྒོ་ནས་བཀོལ་བའི་མིང་དང་ཚིག་དང་ཡིག་ཚོགས་ལ་གོ་དགོས་པ་ཡིན། 
          ཕ་ལ་ཡབ་དང་མ་ལ་ཡུམ། ། མིང་ལ་མཚན་དང་ལུས་ལ་སྐུ། ། གཟུགས་བརྙན་ལ་ནི་སྐུ་བརྙན་དང་། ། མགོ་ལ་དབུ་དང་ངོ་ལ་ཞལ། ། མིག་ལ་སྤྱན་དང་རྣ་ལ་སྙན། ། སྣ་ལ་ཤངས་དང་ཁ་ལ་ཞལ། ། སོ་ལ་ཚེམས་དང་ལྕེ་ལ་ལྗགས། །  སྐེ་ལ་མགུལ་དང་ལག་ལ་ཕྱག།རྐང་ལ་ཞབས་དང་ལྷམ་ཞབས་ཕྱགས། ། རྐང་པར་ཞབས་པད་ཅེས་ཀྱང་བརྗོད། ། མཛུ་གུ་ལ་ནི་ཕྱག་སོར་དང་། ། ལག་པའི་སེན་མོར་ཕྱག་སེན་བྱ། ། རྐང་མཛུབ་ལ་ནི་ཞབས་སོར་དང་། ། རྐང་པའི་སེན་སོར་ཞབས་སེན་བརྗོད། ། སེམས་ལ་ཐུགས་དང་སྐད་ལ་གསུང་། ། གླུ་ལ་མགུར་དང་ལེན་པར་བཞེངས། ། མདུན་ལ་དྲུང་དང་འཁྱག་ལ་བསིལ། ། བསམ་ལ་དགོངས་དང་ཤེས་ལ་མཁྱེན། ། ལྟ་ལ་གཟིགས་དང་ཉན་ལ་གསན། ། རྣ་བས་ཚོར་བར་སྙན་གྱིས་གསན། ། ཁྱོད་ལ་ཁྱེད་དང་ཁོ་ལ་ཁོང་། ། འགྲོ་ལ་བྱོན་དང་དང་འདུག་ལ་བཞུགས། ། ལས་ཀར་ཕྱག་ལས་སྙིང་ལ་ཐུགས། ། ལངས་ལ་བཞེངས་དང་ཉལ་ལ་གཟིམས། ། ཟ་ལ་གསོལ་དང་འཐུང་ལ་བཞེས། ། །ཇ་ལ་གསོལ་ཇ་ཆང་གསོལ་ཆང་། ། བཏུང་བར་སྐྱེམས་དང་གཉིད་ལ་མནལ། ། ཤ་ལ་ཀྲུམ་དང་ཁྲག་ལ་མཚལ། ། སྦྱིན་ལ་འབུལ་དང་བཟུང་ལ་སྣོམ། ། གྱོན་ལ་ནབ་དང་སླེབས་ལ་ཕེབས། ། དགའ་ལ་དགྱེས་དང་སྐྱོ་ལ་སུན། ། སྐྱེ་ལ་འཁྲུངས་དང་ཤི་ལ་འདས། ། ཤི་ལ་གྲོངས་གཤེགས་ཟེར་བའང་ཡོད། ། ངུ་ལ་བཤུམ་དང་བགད་ལ་བཞད། ། འདྲི་ལ་ཞུ་དང་བཤད་ལ་གསུངས། ། སྟེར་ལ་གནང་དང་བྱེད་ལ་མཛད། ། སྟེར་བར་བསྩལ་བ་ཞེས་ཀྱང་བྱ། ། ནད་ལ་སྙུན་དང་ན་ལ་སྙུང་། ། ནད་གཞི་ལ་ནི་སྙུང་གཞི་ཟེར། ། བསད་པ་ལ་ནི་བཀྲོངས་པ་དང་། ། ལས་དོན་ལ་ནི་མཛད་འཕྲིན་ཟེར། ། བུ་དང་བུ་མོར་སྲས་སྲས་མོ། ། མགོ་ཡི་ཞྭ་ལ་དབུ་ཞྭ་དང་། ། ལུས་ཀྱི་གོས་ལ་ན་བཟའ་ཞེས། ། སྐེ་རགས་མིང་དུ་སྐུ་ཆིངས་དང་། ། རྐུབ་བཀྱག་ལ་ནི་ཞབས་བཀྱག་ཟེར། ། རྐང་ཐང་ལ་ནི་ཞབས་ཐང་བརྗོད། ། མགོ་ཡི་སྐྲ་ལ་དབུ་སྐྲ་དང་། ། ཁ་སྤུ་ལ་ནི་ཞལ་ཚོམ་མོ། ། གྲོད་པ་ལ་ནི་གསོལ་གྲོད་དང་། ། སྡོད་ཁང་ལ་ནི་གཟིམ་ཆུང་ཟེར། ། ཉལ་སའི་ཁྲི་ལ་གཟིམས་ཁྲི་དང་། ། རྨི་ལམ་ལ་ནི་གཟིམས་ལམ་དང་། ། ཉལ་ཆས་དག་ལ་གཟིམས་ཆས་ཟེར། ། པར་རྒྱག་པ་ལ་སྐུ་པར་སྐྱོན། ། ལན་རྒྱག་པ་ལ་ལན་སྐྱོན་བྱ། ། ལྟ་ཚུལ་ལ་ནི་གཟིམས་ཚུལ་དང་། ། ལྟ་སྐོར་བྱེད་པར་གཟིགས་སྐོར་ཟེར། ། ཁྲོ་བའི་མིང་ལ་དགོངས་པ་འཚོམས། ། གཤེ་གཤེ་བཏང་བར་བཀའ་བཀྱོན་གནང་། ། བཟོད་སྒོམ་ཞུ་བར་དགོངས་དག་ཞུ། ། མི་ཚེ་ལ་ནི་སྐུ་ཚེ་དང་། ། ཤེས་རབ་མིང་དུ་མཁྱེན་རབ་ཟེར། ། ཤེལ་མིག་ལ་ནི་སྤྱན་ཤེལ་དང་། ། མིག་ཆུ་ལ་ནི་སྤྱན་ཆབ་ཅེས། ། ཆུ་ཡི་མིང་ལ་ཆབ་དང་ནི། ། ཆུ་ཡི་མཆོད་པར་ཡོན་ཆབ་ཟེར། ། གཅིན་པའི་མིང་དུ་གསང་ཆབ་དང་། ། ཕུང་པོ་རོ་ལ་སྐུ་གདུང་སྨྲ། ། རྒྱུ་བདག་མིང་ལ་ཕྱག་མཛོད་དང་། ། བཟའ་གྲོགས་མོ་ལ་ལྷ་ལྕམ་དང་། ། བཟའ་ཟླ་གཉིས་ལ་ཡབ་ཡུམ་བརྗོད། །
          སློབ་དཔོན་ཆེན་པོར་ཡོངས་འཛིན་ཞུ། ། བཞོན་པའི་རྟ་ལ་ཆིབས་རྟ་དང་། ། རྟ་ལྕག་ལ་ནི་ཆིབས་ལྕག་ཟེར། ། རྟ་བཞོན་པ་ལ་རྟ་ཆིབས་པ། ། རྟ་སྒ་ལ་ནི་ཆིབས་སྒ་དང་། ། འགྲོ་ལ་ཆིབས་བསྒྱུར་ཅེས་ཀྱང་བརྗོད། ། སྐྲ་བྲེག་པ་ལ་དབུ་སྐྲ་བསིལ། ། བྲིས་པའི་ཡི་གེར་ཕྱག་བྲིས་བརྗོད། ། ངོ་ཐུག་པ་ལ་མཇལ་འཕྲད་དང་། ། ཁྲུས་བྱེད་པ་ལ་ཁྲུས་གསོལ་ཟེར། ། འགོ་ཁྲིད་ལ་ནི་དབུ་ཁྲིད་དང་། ། འགོ་གཙོ་ལ་ནི་དབུ་གཙོ་བརྗོད། ། ལག་མགོར་བཞག་པར་ཕྱག་དབང་གནང་། ། འགོ་འཛུགས་པ་ལ་དབུ་འཛུགས་དང་། ། དབུ་བརྙེས་ཞེས་ཀྱང་ཟེར་སྲོལ་ཡོད། ། མགོ་ཡི་རྒྱན་དང་དབུ་རྒྱན་དང་། ། སྐེ་རྒྱན་མིང་དུ་མགུལ་རྒྱན་ཟེར། ། གཅོད་བྱེད་གྲི་ལ་ཕྱག་ཤན་བྱ། ། དཀའ་ཚེགས་ལ་ནི་འོ་བརྒྱལ་དང་། ། སྐུ་ཚེགས་སྐུ་མཉེལ་སོགས་ཀྱང་བརྗོད། ། སྐྲུན་པ་ལ་ནི་བཞེངས་པ་དང་། ། ཁས་བླངས་པ་ལ་ཞལ་བཞེས་དང་། ། ཡ་ཡ་ཟེར་བར་ལེགས་སོ་ཞུ། ། དོ་ཁུར་ཞེས་པར་ཐུགས་ཁུར་དང་། ། སེམས་ལ་ངེས་པར་ཐུགས་ལ་མངའ། ། བསམ་འཆར་ལ་ནི་དགོངས་འཆར་དང་། ། སྤུན་ཟླའི་མིང་ལ་སྐུ་མཆེད་འབོད། ། མགྲོན་འབོད་པ་ལ་གདན་འདྲེན་ཞུ། ། བྱ་དགའ་སྟེར་ལ་གསོལ་རས་གནང་། ། ཁ་ཆེམས་ལ་ནི་ཞལ་ཆེམས་དང་། ། ཐང་ག་ལ་ནི་ཞལ་ཐང་ཟེར། ། ལུས་ཁམས་མིང་ལ་སྐུ་ཁམས་དང་། ། མིང་བཏགས་པ་ལ་མཚན་གསོལ་བརྗོད། ། འགྲོ་རོགས་ལ་ནི་ཕེབས་གྲོགས་དང་། ། རྒད་པོ་རྒན་མོར་བགྲེས་སོང་ཟེར། ། ཁ་དྲོར་ཞལ་དྲོ་སྐུ་མཁྱེན་་དང་། ། ཁ་བཏགས་ལ་ནི་མཇལ་དར་ཞུ། ། མི་རབས་ལ་ནི་སྐུ་ཕྲེང་དང་། ། ག་ལེར་སོང་ལ་ག་ལེར་ཕེབས། ། ཚུར་ཤོག་ཅེས་པར་ཚུར་སྤྱོན་དང་། ། ལུས་ཤ་རྒྱགས་པར་སྐུ་ཤ་འབྱོར། ། སྡོད་སའི་གྲལ་ལ་བཞུགས་གྲལ་འབོད། ། ཁ་ཏ་ལ་ནི་ཞལ་ཏ་དང་། ། ཁ་འདོན་ལ་ནི་ཞལ་འདོན་གསུང་། ། སྡོད་རོགས་ཞེས་པར་བཞུགས་གདན་འཇགས། ། སྐྱེས་སྐར་ལ་ནི་འཁྲུངས་སྐར་དང་། ། སྐྱེས་པ་ལ་ནི་སྐུ་བལྟམས་པ། ། ཚིག་པ་ཟ་བར་དགོངས་འགལ་དང་། ། ཁོང་ཁྲོ་མ་ལང་ཞེས་པ་ལ། ། དགོངས་པ་མ་ཚོམས་ཞེས་ཀྱང་བརྗོད། ། ཁོག་རྒྱ་ལ་ནི་ཐུགས་རྒྱ་དང་། ། སེམས་ཁྲལ་ལ་ནི་ཐུགས་ཁྲལ་བྱ། ། སྲུང་མདུད་ལ་ནི་ཕྱག་མདུད་དང་།བསླབ་བྱ་ལ་ནི་བཀའ་སློབ་དང་། ། སེམས་དགའ་དགའ་ལ་ཐུགས་དགྱེས་དགྱེས། ། སྙིང་རྗེ་ལ་ནི་ཐུགས་རྗེ་དང་།གཏམ་བཤད་ལ་ནི་གསུང་བཤད་ཟེར། །
          དཀར་ཡོལ་ལ་ནི་ཞལ་དཀར་དང་། ། སྡེར་མ་ལ་ནི་གསོལ་སྡེར་ཞེས། ། སྡོད་གུར་མིང་ལ་བཞུགས་གུར་དང་། ། ཉལ་ཁང་ལ་ནི་གཟིམས་ཤག་ཟེར། ། ཕོ་བྲང་དང་ནི་ཁྱིམ་སོགས་ལ། ། ཁབ་ཅེས་འབོད་སྲོལ་ཡོད་པ་དཡིན། ། སྡོད་པའི་དུས་ལ་བཞུགས་རིང་དང་། ། དཔུང་རོགས་ལ་ནི་ཕྱག་རོགས་བྱ། ། གླེང་མོ་བྱེད་པར་བཀའ་མོལ་དང་། ། དོགས་ཟོན་ལ་ནི་ཐུགས་ཟོན་ཞེས། ། བྱས་རྗེས་ལ་ནི་མཛད་རྗེས་དང་། ། བྱ་གཞལ་ལ་ནི་མཛད་འཕྲིན་བྱ། ། བློ་འདྲི་བ་ལ་བཀའ་འདྲི་དང་། ། བློ་ལ་བབ་པར་ཐུགས་ལ་བབ། ། མིང་གྲགས་ལ་ནི་མཚན་སྙན་དང་། ། མིང་རྟགས་བཀོད་པར་མཚན་རྟགས་བཀོད། ། མིང་འཛིན་ལ་ནི་མཚན་འཛིན་དང་། ། མིང་ཤོག་ལ་ནི་མཚན་ཤོག་བྱ། ། རྩོམ་ཡིག་མིང་ལ་གསུང་རྩོམ་དང་། ། ཚ་ཞང་ལ་ནི་དབོན་ཞང་ཞེས། ། ཟས་རིན་ལ་ནི་གསོལ་ཡོན་དང་། ། ཟས་ལྷག་མིང་ལ་གསོལ་ལྷག་ཟེར། ། ལོ་གྲངས་ལ་ནི་དགོངས་ལོ་དང་། ། གོས་རྒྱན་སྤྲས་པར་སྒྲོན་ཞེས་བརྗོད། ། ཡོང་བའི་མིང་ལ་ཞབས་བཅགས་དང་། ། ཡོངས་ཤེས་ལ་ནི་ཀུན་མཁྱེན་ནམ། ། ཐམས་ཅད་མཁྱེན་པ་ཞེས་ཀྱང་བྱ། ། གཤིས་རྒྱུད་ལ་ནི་ཐུགས་རྒྱུད་དང་། ། བཤད་ཁྲིད་བྱེད་པར་ལྗགས་ཁྲིད་ཅེས། ། བསམ་གཞིགས་ལ་ནི་དགོངས་བཞེས་དང་། ། བསམ་ཚུལ་ལ་ནི་དགོངས་ཚུལ་བརྗོད། ། ལག་པའི་རྗེས་ལ་ཕྱག་རྗེས་དང་། ། ལག་བསྟར་མ་ལ་ཕྱག་བསྟར་མ། ། ལག་དངུལ་ལ་ནི་ཕྱག་དངུལ་དང་། ། ལོ་གསར་གནམ་ལོ་གསར་ཚེས་ཟེར། ། ཤེས་རྒྱ་ལ་ནི་མཁྱེན་རྒྱ་དང་། ། ཤེས་གསལ་ལ་ནི་མཁྱེན་གསལ་བྱ། ། མཉམ་འཛོམས་ལ་ནི་ལྷན་འཛོམས་དང་། ། སེམས་ལྷོད་ལྷོད་ལ་ཐུགས་ལྷོད་ལྷོད། ། ཨ་ཁུར་ཨ་ཁུ་ལགས་ཞེས་དང་། ། ཨ་ནེར་ཨ་ནེ་ལགས་ཞེས་བོས། ། ཁ་པར་ལ་ནི་ཞལ་པར་དང་། ། ལག་འཕྲིན་ལ་ནི་ཕྱག་འཕྲིན་བརྗོད། ། དེ་ལྟར་བྲིས་པའི་དགོས་པ་ཡང་། ། གཞན་ལ་གུས་པ་མཛོན་བྱའི་ཕྱིར། ། གུས་པའི་ཞེ་སའི་ཕྲེང་བ་འདི། ། ལྷན་རྒྱས་ཡིད་ལ་ངེས་པར་སྨོན། །
