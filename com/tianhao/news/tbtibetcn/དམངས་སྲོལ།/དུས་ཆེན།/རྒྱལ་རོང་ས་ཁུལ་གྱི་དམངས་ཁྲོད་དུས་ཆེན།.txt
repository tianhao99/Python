རྒྱལ་རོང་ས་ཁུལ་གྱི་དམངས་ཁྲོད་དུས་ཆེན།
སི་ཁྲོན་ཞིང་ཆེན་འབར་ཁམས་ཕྱོགས་ཀྱི་རྒྱལ་རོང་ས་ཁུལ་དུ་དམངས་ཁྲོད་ཀྱི་དུས་ཆེན་ཞིག་དར་སྲོལ་ཆེ་ལ་བོད་སྐད་དུ་རི་འགོ་འཛོམས་མྱོས་དང་ཡང་ན་སྤྱི་བསང་དུས་ཆེན་ཟེར། དེང་སང་རྒྱ་སྐད་དུ་མེ་ཏོག་ལ་ལྟ་བའི་དུས་ཆེན་ཟེར། ལོ་གཅིག་གི་རིང་དུ་ཐེངས་གཉིས་ལ་དུས་ཆེན་དེ་རོལ་སྲོལ་མཆིས་ཏེ།གཅིག་ནི་ལུགས་རྙིང་ལོའི་ཟླ་བ་ལྔ་པའི་ཚེས་བཞི་ཉིན་ཡིན་ལ་བོད་སྐད་དུ་རི་འགོ་འཛོམས་མྱོས་ཞེས་བརྗོད་དེ་ཕྱི་རོལ་ཡུལ་གྱི་རི་འགོ་དང་སྤང་ཐོག་ཕྱིན་ཏེ་མེ་ཏོག་ལ་སོགས་པ་མཛེས་ལྗོངས་ལ་རོལ་བའི་དོན་ཡིན། གཅིག་ནི་སྤྱི་ལོའི་ཟླ་བ་བཅུ་གཅིག་པའི་ནང་ཡིན་ལ་བོད་སྐད་དུ་དང་རབ་འཛོམས་མྱོས་ཞེས་ཟེར་ཏེ་སྟོན་ཀའི་དཔལ་ལ་རོལ་བའི་སྲོལ་རྒྱུན་གྱི་དུས་ཆེན་ཞེས་པའི་དོན་ཡིན། ཐེངས་གཉིས་ལས་དང་པོ་རི་འགོ་འཛོམ་མྱོས་ཞེས་པ་དེ་ཆེས་དགའ་བསུ་ཐོབ་ཅིང་གཙོ་བོར་འཛིན་པ་ཞིག་ཡིན། ལོ་རྒྱུས་ཀྱི་དཔྱད་གཞིའི་ཡིག་ཆ་ནང་བཀོད་པ་ལྟར་ན་ཟླ་བ་ལྔ་པའི་ཚེས་བཞི་ཉིན་ལ་མེས་པོ་རྣམས་ཀྱིས་སྨན་ལ་ཞེས་པའི་རི་བོར་རྡོ་ཡི་སྟེགས་བུ་ཞིག་བརྩིགས་ཏེ་དེ་ཡི་ཐོག་ལ་ཐམ་ག་ཞིག་བཞག  རྟ་རྒྱུགས་འགྲན་བསྡུར་བྱས་ཏེ་སུ་ཡིས་དང་པོར་ཐམ་ག་དེ་ལག་ཏུ་ལོན་ན་དེ་རྒྱལ་པོར་བསྐོ་རྒྱུའི་གྲོས་ཐག་བཅད། མེ་མདའ་འབེན་གཟེད་སོགས་ལྟད་མོ་སྣ་ཚོགས་ཀྱིས་དྲན་རྟེན་བྱེད་སྲོལ་ཆགས། རིམ་བཞིན་ད་ལྟའི་མེ་ཏོག་ལ་ལྟ་བའི་དུས་ཆེན་ཞེས་པར་གྱུར་པ་རེད་ཅེས་བཤད། གཞན་ཞིག་ནི་ཟླ་བ་ལྔ་པའི་ནང་རྒྱལ་རོང་ས་ཁུལ་དུ་མེ་ཏོག་རྣམ་པར་བཞད་དེ་བདེ་ལྡན་ལྷ་ཡི་ཞིང་ཁམས་ལྟ་བུའི་རྣམ་པར་ཤར་བས་བོད་མི་རྣམས་ཀྱིས་དེ་ནི་ལྷ་ཡི་དངོས་གྲུབ་ཏུ་བསྒོམས་ནས་མེས་པོ་རྣམས་ཀྱི་སྲོལ་རྒྱུན་ལྟར་ལྷ་འབོད་ཀླུ་མཆོད་ཀྱི་བསང་བཏང་བ་མ་ཚད་རི་མོ་ལྟ་བུའི་སྤང་ཐང་དང་ནགས་ཚལ་དུ་སྤྲིན་དཀར་པོའི་རྗེས་སུ་འབྲངས་པའི་གུར་སྣ་ཚོགས་ཕུབ་སྟེ་སྙན་འཇེབས་ཀྱི་གླུ་དབྱངས་ལེན་བཞིན་ཡིད་འཕྲོག་པའི་བྲོ་འཁྲབ་ནས་རང་བྱུང་ཁམས་ཀྱི་པང་བར་ཅི་འདོད་གང་དགར་གྱིས་ཡིད་སེམས་མྱོས་པར་བྱེད་དོ།།
