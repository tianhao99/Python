ལྷོ་ཁ་འཕྱོངས་རྒྱས་གར་བྲོ་སྤུན་བདུན་པའི་ལྷ་མོའི་ཟློས་གར།
ལྷོ་ཁ་འཕྱོངས་རྒྱས་གར་བྲོ་སྤུན་བདུན་ལྷ་མོའི་ཟློས་གར་ནི་འབག་དཀར་པོའི་ལུགས་ཀྱི་ལྷ་མོའི་ཟློས་གར་ཞིག་ཡིན་ལ། དེར་ལོ་ངོ་༦༠༠ལྷག་ཙམ་གྱི་ལོ་རྒྱུས་ལྡན་ཡོད།
        དེ་ཡང་ཤོད་རྒྱུན་དུ“སྤུན་བདུན་པ”ཞེས་པ་ནི་བོད་ཀྱི་ལྷ་མོའི་ཟློས་གར་གྱི་ཚོགས་པ་སྔ་ཤོས་ཤིག་ཡིན་པར་གྲགས་ཏེ། སྐབས་དེར་ལྷ་མོའི་སྲོལ་གཏོད་པ་པོ་གྲུབ་ཐོབ་ཐང་སྟོང་རྒྱལ་པོས་བོད་ཀྱི་ཆུ་ཀླུང་དང་གཙང་པོ་ཆེ་ཁག་ལ་ལྕགས་ཐག་གི་ཟམ་པ་གསར་བསྐྲུན་གནང་ཆེད། སྐུ་ངལ་འབུམ་ཕྲག་ཁྱད་དུ་བསད་ཅིང་། ཐ་ན་སྐུ་སྲོག་གི་ཉེན་ཁ་ལའང་མ་འཛེམས་པར་བོད་ཡུལ་སྟོད་སྨད་གང་སར་ལྕགས་རིགས་འཚོལ་བསྡུ་དང་སློང་མོ་གནང་བར་ཕེབས། དུས་དེར་སུམ་རྩེན་གནས་ནས་ཕེབས་པའི་ལྷ་མོ་ཞིག་གིས་ཁོང་གི་མཛད་སྤྱོད་རྣམས་གཟིགས་པས་ཐུགས་འགུལ་ཚད་མེད་ཐེབས་ཏེ ། གྲུབ་ཐོབ་ཁོང་ལ་མནལ་ལམ་དུ་ཡུལ་ལྷོ་ཁ་འཕྱོངས་རྒྱས་ཕྱོགས་སུ“གཅེན་གཅུང་སྤུན་བདུན”（བུ་མོ་སྤུན་བདུན་ཡང་ཟེར）འཚོལ་དགོས་པའི་ལུང་བསྟན་གནང་བས། ཁོང་གིས་ཀྱང་ས་དེར་ཕེབས་པས་དངོས་འབྲེལ་ཁ་འབྲས་མཚར་ལ་རྣམ་རིག་བཀྲ་བའི་གཞོན་ནུ་ཕོ་མོ་སྤུན་བདུན་ཡོད་པར་མཁྱེན། དེ་ནས་བཟུང་ཁོང་གིས་གཞོན་ནུ་སྤུན་བདུན་སྒྲིག་འཛུགས་གནང་སྟེ་ལྷ་མོའི་འཁྲབ་ཚན་གྱི་ཟློས་གར་འཁྲབ་སྟོན་བྱེད་པའི་སྲོལ་དར་བར་གྲགས།
        གྲུབ་ཆེན་ཁོང་གིས་སྤུན་བདུན་གྱི་ནང་ནས་ལོ་ན་རྒན་ཤོས་དེས“རྒྱ་ལུ”འཁྲབ་ཏུ་བཅུག་གནང་། “རྒྱ་ལུ”ཞེས་པ་ནི་དེ་སྔའི་ཚོ་ཁག་གི་འགོ་དཔོན་ཞིག་གི་མིང་ཡིན་ལ། དེའི་ཆ་ཆས་སུ་མགོ་ལ་མདོག་དམར་པོའི་ཞྭ་མོ་འབོག་ཏོ་ཞེས་པ་གྱོན་པ་དང་། ལག་ཏུ་འཁར་རྒྱུག་བཟུང་བའི་རྣམ་པ་ཅན་ཞིག་ཡིན་ལ། དབང་ཆ་དང་གོ་གནས་ངེས་ཅན་ཡོད་པའི་མི་ཞིག་གི་གཟུགས་བརྙན་མཚོན་པ་དང་། རྒྱ་ལུ་ནི་འགོ་སྣེ་ཁྲིད་མཁན་དུ་འཇོག་པ་དང་། དེ་ནས་བུ་མོ་སྤུན་གསུམ“ལྷ་མོ”འཁྲབ་ཏུ་བཅུག  བུ་སྤུན་གསུམ“རྔོན་པ”འཁྲབ་ཏུ་བཅུག་པ་མ་ཟད། གྲུབ་ཆེན་ཐང་སྟོང་རྒྱལ་པོ་ཁོང་ཉིད་ཀྱིས“དགེ་རྒན”གནང་སྟེ། ལྷ་མོའི་ཟློས་གར་གྱི་འཁྲབ་ཚན་མང་པོ་གསར་བཟོ་མཛད་ཅིང་། བོད་ཀྱི་ས་ཕྱོགས་གང་སར་ཟློས་གར་འཁྲབ་སྟོན་གནང་བས། ཁོང་ལ་ལྕགས་རིགས་གཙོས་པའི་ཟམ་བཟོའི་རྒྱུ་ཆ་མང་པོ་ཕྱག་སོན་བྱུང་མཐར། ལྕགས་ཐག་གི་ཟམ་པ་མང་པོ་བསྐྲུན་གནང་བ་རེད། 
        “སྤུན་བདུན་པ”ཞེས་པ་ནི་དོན་དུ“མིང་སྲིང་བདུན”ནམ“བུ་མོ་སྤུན་བདུན”ལ་ཟེར་ཞིང་། གར་བྲོ་སྤུན་བདུན་ལྷ་མོའི་ཟློས་གར་གྱིས་ཆེས་གནའ་བོའི་འཁྲབ་སྟོན་གྱི་སྒྱུ་རྩལ་ཉམས་འགྱུར་རྣམས་མ་ཉམས་རྒྱུན་འཛིན་བྱས་ཡོད། དེའི་འགོ་ཚུགས་འཁྲབ་ཚན་ལ“ཨ་ར་བ”ཞེས་ཟེར་ཞིང་། དེའི་དོན་ནི་ཨ་རའམ་ཁ་སྤུ་དཀར་པོའི་འབག་གྱོན་པའི་ཟློས་གར་བ་ཞེས་པའོ། རྒྱ་ལུས་གྱོན་པའི་ཞྭ་མོ་ནི་གནའ་བོའི་བོད་རིགས་ཀྱི“རྔ་བྲོའི”གར་འཁྲབ་སྐབས་སུ་རྔ་བརྡུང་མཁན་གྱིས་གྱོན་པའི“གར་རྒྱ”ཞེས་པའི་ཞྭ་མོ་སྒོར་སྒོར་དེ་ཡིན། ཁོ་ཚོའི་རྒྱན་གོས་སྤྲས་སྟངས་དང་ཞབས་བྲོའི་ཉམས་འགྱུར་རམ་སྟངས་སྟབས་སོགས་ནི་བསམ་ཡས་དགོན་པའི་ལྡེབས་རིས་ཐོག་གི་འབག་དཀར་པོའི་ཟློས་གར་གྱི་འཁྲབ་སྟངས་ཇི་མ་ཇི་བཞིན་རྒྱུན་འཛིན་བྱས་ཡོད། “རྔོན་པ”（ཉ་པའམ་རི་དྭགས་རྔོན་མཁན་གྱི་དོན）རྣམས་ཀྱིས་གྱོན་པའི་འབག་ནི་ར་དཀར་པོའི་པགས་པ་ལས་བཟོས་པ་ཞིག་ཡིན་ལ། ཁ་དོག་ཡང་དཀར་པོ་དེ་ག་རང་ཡིན། འབག་ངོའི་མིག་ཟུར་གྱེན་དུ་ཕྱོགས་པ་དང་མིག་ལྤགས་འོག་མས་གོང་མ་བཀབ་པ་སྟེ། ཤོད་རྒྱུན་དུ་རྒྱལ་དབང་ལྔ་པ་ཆེན་པོའི་སྤྱན་གཉིས་དེ་འདྲ་ཡིན་ལ། ཁོང་ཉིད་གར་བྲོ་སྤུན་བདུན་ལྷ་མོ་བར་ཧ་ཅང་མཉེས་པོ་གནང་རྐྱེན་ཕྱིས་སུ་ཚོགས་པ་དེས་རྒྱལ་དབང་གི་སྤྱན་ལ་དཔེ་བལྟས་ཏེ་དེ་ལྟར་བཟོས་པར་གྲགས། 
        གར་བྲོ་སྤུན་བདུན་ལྷ་མོའི་ཟློས་གར་ནི་དེ་སྔའི་བོད་ས་གནས་སྲིད་གཞུང་གིས་ལྷ་སའི་ཞོ་སྟོན་སྐབས་སུ《ཆོས་རྒྱལ་ནོར་བཟང་གི་རྣམ་ཐར》གྱི་དུམ་བུ་གཅིག་འཁྲབ་སྟོན་བྱས་ཆོག་པའི་གཏན་འབེབས་བྱས་རྐྱེན་ཞོ་སྟོན་སྐབས་དེའི་གཞུང་ཆ་ཚང་འཁྲབ་སྟོན་བྱེད་པ་ཅུང་དཀོན་པ་དང་། དེའི་འཁྲབ་སྟོན་གྱི་རྣམ་པ་དང་། ཉམས་འགྱུར་སོགས་ནི་དབུས་ཁུལ་གྱི་ས་ཕྱོགས་གང་སར་ཁྱབ་བརྡལ་བྱུང་ཡོད་པ་རེད།
        གར་བྲོ་སྤུན་བདུན་པས་གཙོ་འཁྲབ་བྱེད་པའི་ལྷ་མོའི་འཁྲབ་གཞུང་ནི《ཆོས་རྒྱལ་ནོར་བཟང་གི་རྣམ་ཐར》ཡིན་ཏེ། ཐོག་མར་དེའི་དུམ་བུ་ཁ་ཤས་འཁྲབ་པ་དང་། དུས་ཕྱིས་འབག་སྔོན་པའི་ལུགས་ཀྱི་ལྷ་མོའི་ཟློས་གར་ལ་དཔེ་བལྟས་པས། དེའི་གཞུང་ཆ་ཚང་འཁྲབ་སྲོལ་ཡོད་པ་རེད། ཟློས་གར་དེའི་གཞས་གདངས་ནི་ལྷོ་ཁའི་དམངས་ཁྲོད་གླུ་གཞས་དང་ཤིན་ཏུ་འདྲ་བ་མ་ཟད། འཁྲབ་སྟོན་བྱེད་པའི་རྒྱུད་རིམ་ནང་དམངས་ཁྲོད་ཀྱི་སྒོར་གཞས་སོགས་མང་པོ་བར་བཅུག་བྱེད་ཀྱི་ཡོད་པ་རེད། 
        ཡང་ཤོད་རྒྱུན་གཞན་ཞིག་ལ་སྤྱི་ལོའི་དུས་རབས་བཅུ་བཞི་པའི་དུས་འགོར། བོད་བརྒྱུད་ནང་བསྟན་ལས་བཀའ་བརྒྱུད་པའི་སྐྱེས་ཆེན་དམ་པ་གྲུབ་ཐོབ་ཐང་སྟོང་རྒྱལ་པོ་ཡར་ལུང་གཙང་པོའི་བརྒྱུད་ཀྱི་ལྕགས་ཟམ་ཆུ་བོ་རིར（བོད་ཀྱི་རི་ཆེན་བཞིའི་ཡ་གྱལ）སྒྲུབ་པ་ཉམས་ལེན་མཛད་བཞིན་བཞུགས་པའི་སྐབས། ཉིན་ཞིག་གཙང་པོའི་གཡས་གཡོན་གྱི་མི་རྣམས་ཕན་ཚུན་འགྲུལ་བཞུད་བྱེད་པར་གཙང་པོས་བར་གྱིས་ཆོད་པས་འགྲིམ་འགྲུལ་ཐད་དཀའ་ངལ་ཆེན་པོ་འཕྲད་བཞིན་ཡོད་པ་ཁོང་གིས་མངོན་སུམ་དུ་གཟིགས་པས། ཁོང་གིས་མང་ཚོགས་ཀྱི་དཀའ་ངལ་སེལ་བ་དང་། ཐོན་སྐྱེད་འཚོ་བའི་ཐད་སྟབས་བདེ་བསྐྲུན་ཆེད། བོད་ཀྱི་ཆུ་ཀླུང་དང་གཙང་པོ་ཁག་ལ་ལྕགས་ཐག་གི་ཟམ་པ་དང་གྲུ་ཤན་སོགས་བསྐྲུན་རྒྱུའི་དགོངས་པ་ཐུགས་ལ་འཁྲུངས། ཐོག་མར་ཁོང་གིས་མེས་རྒྱལ་ནང་ས་དང་། བལ་ཡུལ་སོགས་ས་ཕྱོགས་གང་སར་ཟམ་བསྐྲུན་གྱི་རྒྱུ་ཆ་བཙལ་འཚོལ་གནང་ཀྱང་ཐུགས་འདུན་ཇི་བཞིན་མ་འགྲུབ། ཉིན་ཞིག་ཁོང་ཕུ་འབྲོག（དེང་འཕན་པོ་ཟེར）ཞེས་པའི་སར་རང་ལུས་འཁར་རྒྱུག་ལ་བརྟེན་ཅིང་། སྐུ་ན་ཧ་ཅང་བགྲེས་པའི་རྒན་མོ་ཞིག་དང་མཇལ།（ཤོད་རྒྱུན་ལ་རྒན་མོ་དེ་སྤྱན་རས་གཟིགས་ཀྱི་རྣམ་སྤྲུལ་ཡིན་པར་གྲགས）རྒན་མོས་གྲུབ་ཐོབ་ཁོང་ལ་ལྕགས་སྐུད་རིལ་མ་བདུན་ཞལ་འདེབས་སུ་ཕུལ་བས། དེ་ནི་རྟགས་དང་རྟེན་འབྲེལ་བཟང་པོར་བརྩིས་གནང་ཏེ། མུ་མཐུད་སྐུ་ངལ་ལ་མ་འཛེམས་པར་ཕྱོགས་བཞི་མཚམས་བརྒྱད་ལ་ཕེབས་ཤིང་། མང་ཚོགས་སྐུལ་སློང་གནང་སྟེ་ལྕགས་ཟམ་དང་གྲུ་ཤན་སོགས་བསྐྲུན་རྒྱུའི་མི་ཤུགས་དང་དངོས་ཤུགས་ཀྱི་མཐུན་རྐྱེན་མང་པོ་སྒྲུབ་མཐར་གངས་ལྗོངས་ཡུལ་གྲུ་གང་སར་ལྕགས་ཟམ་དྲུག་ཅུ་ལ་ཉེ་བ་བསྐྲུན་གནང་བ་རེད། 
        གྲུབ་ཐོབ་ཐང་སྟོང་རྒྱལ་པོས་ཟམ་བཟོའི་མང་ཚོགས་ཁྲོད་ནས་ལྷོ་ཁ་འཕྱོངས་རྒྱས་ཁུལ་གྱི་ཁ་སྒོ་སྐྱེ་འབྲས་ཤིན་ཏུ་མཚར་བའི་བུ་མོ་སྤུན་བདུན་སྒྲིག་འཛུགས་གནང་སྟེ། ལྷ་མོའི་ཟློས་གར་གྱི་ཚོགས་པ་ཞིག་བཙུགས་གནང་བ་དང་། ཁོ་ཚོས་རང་འཁྲབ་རང་བཟོའི་ཐོག་ནས་བརྗོད་བྱ་སངས་རྒྱས་ཆོས་ལུགས་ཀྱི་གཏམ་རྒྱུད་ཁག་ལ་སྦྱར་ཏེ། གོ་བདེ་ཞིང་ནང་དོན་ཟབ་པའི་ལྷ་མོའི་འཁྲབ་ཚན་མང་པོ་གསར་བཅོས་གནང་བ་མ་ཟད། བོད་ཀྱི་ས་ཕྱོགས་གང་སར་ཟབ་བསྐྱོད་ཀྱིས་བསྐོར་བསྐྱོད་འཁྲབ་སྟོན་གནང་བས། རྒྱ་ཆེའི་མང་ཚོགས་ཀྱིས་དགའ་བསུ་ཚད་མེད་ཐོབ་ཅིང་ལྕགས་རིགས་སོགས་ཟམ་བསྐྲུན་རྒྱུ་ཆ་འབོར་ཆེན་ཕྱག་སོན་ཐུབ་ཡོད་ལ། ཁོང་གིས་ཟློས་གར་འཁྲབ་སྟོན་བྱེད་པའི་བརྒྱུད་རིམ་ནང་། མུ་མཐུད་ནང་དོན་དང་འཁྲབ་སྟངས་སོགས་ལ་ཁ་གསབ་དང་ལེགས་བཅོས་གང་ལེགས་གནང་རྐྱེན་ལྷ་མོའི་འཁྲབ་ཚན་ཡང་འཐུས་སྒོ་ཇེ་ཚང་དུ་སོང་ཡོད་པ་རེད། དུས་དེ་ནས་བཟུང་བོད་ཀྱི་ལྷ་མོའི་འཁྲབ་ཚན་ཐོག་མ་སྟེ“འཕྱོངས་རྒྱས་གར་བྲོ་བཀྲ་ཤིས་སྤུན་བདུན་པའི་ལྷ་མོའི་ཟློས་གར”ཞེས་པ་དེ་དངོས་སུ་་བྱུང་ཞིང་། ད་བར་ལོ་ངོ༦༠༠ལྷག་ཙམ་གྱི་ལོ་རྒྱུས་ལྡན་ཡོད། 
        དེ་ལྟར་འབག་དཀར་པོའི་ལྷ་མོའི་ཟློས་གར“འཕྱོངས་རྒྱས་གར་བྲོ་བཀྲ་ཤིས་སྤུན་བདུན”ཞེས་པའི་འཁྲབ་ཚན་དེ་དུམ་བུ་ཆེ་བ་གསུམ་ལ་དབྱེ་ཆོག་པ་སྟེ། འགོ་ཚུགས་པ་དང་། གཞུང་དོན་དངོས། མཇུག་བསྡུ་བ་བཅས་གསུམ་མོ། དུམ་བུ་དང་པོ་འགོ་ཚུགས་པར་ནང་གསེས་གསུམ་དུ་དབྱེ་བ་སྟེ། རྔོན་པའི་འདོན་དང་། རྒྱ་ལུའི་བྱིན་འབེབས། ལྷ་མོའི་གླུ་གར་བཅས་ཡིན། “རྔོན་པའི་འདོན”ནི་རྔོན་པས་ས་འདུལ་བྱེད་པའི་ཆོ་ག་ཞིག་སྟེ། འབག་དཀར་པོ་གྱོན་ཞིང་། ལུས་ལ་རྔོན་པའི་ཆས་སྤྲས་པ་དང་། ལག་ཏུ་ཚོན་ལྔའི་མདའ་དར་འཁྱེར་ཏེ་ས་གཞི་ངན་པ་འདུལ་བ་དང་། སྙན་པའི་དབྱངས་ལ་བརྟེན་ནས་བཀྲ་ཤིས་སྨོན་ཚིག་འདེབས་པར་བྱེད། དེ་རྗེས“རྒྱ་ལུའི་བྱིན་འབེབས”ཞེས་པ་རྒྱ་ལུ་ཕེབས་ནས་བྱིན་རླབས་གནས་དེར་འབེབས་པའི་དོན་ཏེ། རྒྱ་ལུའི་ཆས་གོས་སྤྲས་པའི་མི་ཞིག་གར་སྟེགས་ཐོག་དོན་ནས་གཟིགས་མོ་བར་བཀྲ་ཤིས་པ་དང་། བདེ་སྐྱིད་ཡོང་བའི་བྱིན་རླབས་སྐྱེལ་བ། མཇུག་ཏུ“ལྷ་མོའི་གླུ་གར”ཞེས་ནམ་མཁའི་ལྷ་མོས་གླུ་ལེན་གར་འཁྲབ་བྱེད་པ་སྟེ་ལྷ་མོའི་ཆས་སྤྲས་ཏེ་འཁྲབ་སྟོན་བྱས་པས། དེས་དྲི་ཟའི་ལྷ་མོ་རྣམས་ཀྱང་གནས་དེར་ཕེབས་ཏེ། ཕལ་པའི་འགྲོ་བ་མི་དང་ལྷན་དུ་གླུ་གར་ལ་རོལ་བ་མཚོན། དུམ་བུ་གཉིས་པ་ནི་གཞུང་དངོས་ཏེ། འཁྲབ་ཚན་གྱི་ནང་དོན་དངོས་གཞིར་འཇུག་པ་དང་། སྒྲུང་གི་བྱུང་རིམ་བརྗོད་པའོ། དུམ་བུ་གསུམ་པ་ནི“བཀྲ་ཤིས”ཏེ་མཇུག་བསྡུ་བའི་དོན་ཡིན། “འཕྱོངས་རྒྱས་གར་བྲོ་བཀྲ་ཤིས་སྤུན་བདུན”གྱིས་འཁྲབ་སྟོན་བྱེད་པའི་གཞུང་ནི་བོད་ཀྱི་ལྷ་མོའི་འཁྲབ་གཞུང་ཆེན་པོ་བརྒྱད་ལས་གཙོ་བོ《ཆོས་རྒྱལ་ནོར་བཟང་གི་རྣམ་ཐར》ཡིན།
        འཁྲབ་ཚན་དེའི་ནང་དོན་མདོར་བསྡུས་ནི། གནའ་སྔ་མོ་ལྷོ་ཕྱོགས་རིགས་ལྡན་པ་དང་། བྱང་ཕྱོགས་མངའ་ལྡན་པ་ཞེས་པའི་རྒྱལ་ཁབ་གཉིས་ཡོད། བྱང་ཕྱོགས་མངའ་ལྡན་རྒྱལ་ཁབ་ཀྱི་རྒྱལ་པོ་ཡབ་སྲས་གཉིས་བྱམས་སྙིང་རྗེས་ཕྱུག་པས། རྒྱལ་སྲིད་ཆོས་བཞིན་སྐྱོང་ལ། འབངས་མི་སེར་རྣམས་བདེ་སྐྱིད་དཔལ་ལ་འཁོད་ཡོད་པ་དང་། ལྷོ་ཕྱོགས་རིགས་ལྡན་རྒྱལ་པོ་ནི་ལས་འབྲས་རྩིས་མེད་དུ་གཏོང་ཞིང་། གདུག་རྩུབ་ཆེ་བས། རང་ཉིད་ལ་རྒྱུ་ནོར་ལོངས་སྤྱོད་ཐོབ་ཆེད། སྔགས་པ་དུ་མ་མངགས་ནས་བྱང་ཕྱོགས་རྒྱལ་པའི་བླ་མཚོའི་ནང་གི་ཀླུ་དབང་འཛིན་པར་བརྫངས་ཀྱང་། ཀླུ་དབང་དེ་རྔོན་པ་ཞིག་གི་རོགས་རམ་འོག་སྲོག་འགོ་ཐོན་ཐུབ་པ་བྱུང་བས། ཀླུ་དབང་གིས་རང་ཉིད་ཀྱི་རིན་ཆེན་ཚེ་སྲོག་སྐྱོབ་པའི་བཀའ་དྲིན་གསོལ་ཆེད། རྔོན་པ་དེ་ལ་ནོར་བུ་དགོས་འདོད་ཀུན་འབྱུང་ཕུལ་བ་རེད། འོན་ཏེ་རྔོན་པར་རྒྱུ་ནོར་མི་དགོས་པར་ནམ་མཁའི་ལྷ་མོ་ཞིག་བཙུན་མོར་བྱུང་ན་བསམ་སྟེ་ཀླུ་དབང་ལ་ལྷ་མོ་བཟུང་བྱེད་ཀྱི་ནོར་བུ་དོན་ཡོད་ཞགས་པ་ཞིག་གཡར་བས། དེས་ལྷ་ཡུལ་གྱི་རྫིང་བུའི་ནང་སྐུ་ཁྲུས་གསོལ་བའི་ཡིད་འཕྲོག་ལྷ་མོ་བཟུང་སྟེ། རང་ཉིད་ཀྱི་མནའ་མར་བསུ་རྩིས་བྱས། སྐབས་དེར་དབེན་གནས་དུ་སྒྲུབ་པ་ཉམས་ལེན་གནང་བཞིན་པའི་དྲང་སྲོང་ཞིག་ཕེབས་པས་ཁོང་གིས་མི་ཕལ་པས་ལྷ་མོ་མནའ་མར་ལེན་ན་བཀྲ་མི་ཤིས་ལ། ཆག་སྒོ་མང་བས། ལྷ་མོ་དེ་ཆོས་རྒྱལ་ནོར་བཟང་ལ་ཕུལ་ནས་རྒྱལ་པོའི་བཙུན་མོ་བྱ་རུ་བཅུག་ན་ལེགས་ཞེས་རྔོན་པར་ཁ་ཏ་མང་པོ་གནང་། རྔོན་པས་ཀྱང་དེ་ལྟར་ཁས་བླངས་ཏེ་ཡིད་འཕྲོག་ལྷ་མོ་ནོར་བཟང་ལ་ཕུལ་བས། རྒྱལ་སྲས་དང་ལྷ་མོ་ཞལ་འཛོམས་བྱུང་ཞིང་བདེ་སྐྱིད་རྫོགས་ལྡན་གྱི་འཚོ་བར་རོལ། སྐབས་དེར་རྒྱལ་པོའི་ཕོ་བྲང་ནང་གི་བཙུན་མོ་ལྔ་བརྒྱས་ཕྲག་དོག་དབང་གིས། ཐབས་བརྒྱ་ཇུས་སྟོང་ངང་ཡིད་འཕྲོག་ལྷ་མོ་རྩ་མེད་བཟོ་རྒྱུའི་ཇུས་ངན་བཏིང་ཡོད་དེ། ཐོག་མར་ནོར་བཟང་བྱང་ཕྱོགས་སུ་བདུད་འདུལ་བར་མངགས། དེ་རྗེས་རྒྱལ་པོའི་བསྙུང་གཞི་འདུལ་ཆེད་ཡིད་འཕྲོག་གི་སྙིང་དང་མཆིན་པས་གསོལ་སྨན་བཟོ་དགོས་པའི་ངན་ཇུས་བཤམས་ཏེ་ཡིད་འཕྲོག་མེད་པ་བཟོ་དགོས་སྙམ་ཡང་ཡིད་འཕྲོག་ལྷ་མོ་ནི་ཆོས་རྒྱལ་ནོར་བཟང་གི་ཡུམ་ཆེན་གྱི་ཕྱག་རོགས་འོག་སྟེང་ཕྱོགས་ལྷ་ཡུལ་དུ་ཕེབས་པ་རེད། དེ་ནས་ནོར་བཟང་དམག་ལ་རྒྱལ་ཁ་ཡོངས་སུ་ཐོབ་སྟེ་བྱང་ནས་ལོག་ཕེབས་སྐབས་ཡིད་འཕྲོག་བཞུགས་མེད་པ་མཁྱེན་ནས་ཐུགས་སེམས་ལྷག་པར་སྐྱོ་ཞིང་། ས་ཕྱོགས་གང་སར་ཡིད་འཕྲོག་འཚོལ་བར་ཕེབས། མཐར་དྲང་སྲོང་ཞིག་གི་ལུང་བསྟན་དང་རོགས་རམ་འོག་ལྷ་ཡུལ་དུ་ཕེབས་ཐུབ་པས་ཡིད་འཕྲོག་དང་མཇལ་འཛོམས་བྱུང་། འོན་ཏེ་ལྷ་མོའི་ཡབ་ཀྱིས་སྲས་མོ་མི་ཡུལ་དུ་གཏོང་འདོད་མ་གནང་བར་ནོར་བཟང་ལ་སྒྲུབ་དཀའ་བའི་བྱ་བ་དུ་མ་སྒྲུབ་ཏུ་བཅུག་ཀྱང་ནོར་བཟང་གིས་དེ་དག་རེ་རེ་ལས་རྣམ་པར་རྒྱལ་བས། མཐར་ནོར་བཟང་དང་ཡིད་འཕྲོག་གཉིས་ཀྱི་ཐུགས་བསམ་གྲུབ་སྟེ་མི་ཡུལ་དུ་ལོག་ཕེབས་པ་རེད། དེ་རྗེསཡབ་རྒྱལ་པོས་ཀྱང་རྒྱལ་སྲིད་ཀྱི་གསེར་ཁྲི་ནོར་བཟང་ལ་གཏད་ཅིང་། ཡིད་འཕྲོག་ལྷ་མོ་ཡང་བཙུན་མོར་བསུས། དུས་དེ་ནས་བཟུང་རྒྱལ་ཁབ་སྔར་ལས་མངའ་ཐང་གྲ་རྒྱས་ཤིང་། འབངས་རྣམས་སྔར་བཞིན་བདེ་སྐྱིད་དཔལ་ལ་རོལ་རྒྱུ་བྱུང་བ་རེད། 
        ལྷ་མོ་འཁྲབ་སྐབས་ཀྱི་ཡོ་བྱད་དང་དེའི་ཁྱད་ཆོས་མདོར་ཙམ་གླེང་ན། བོད་ཀྱི་ལྷ་མོའི་ཟློས་གར་འཁྲབ་སྟོན་མཐའ་དག་རྔ་སྦུབ་གཉིས་ལ་ལྟོས་དགོས་པ་ནི་སྨོས་མི་དགོས་པས། འཕྱོངས་རྒྱས་གར་བྲོ་བཀྲ་ཤིས་སྤུན་བདུན་གྱི་ཟློས་གར་ལ་ཡང་རྔ་དང་། སྦུག་ཆ་གཉིས་ངེས་པར་དུ་འཛོམས་དགོས་སོ། གཞན་དུ་དུང་བཟོ་ལྟ་མི་འདྲ་བའི་འཁྲབ་ཆས་ཀྱི་རིགས་སྣ་ཚོགས་དགོས།“འཕྱོངས་རྒྱས་གར་བྲོ་བཀྲ་ཤིས་སྤུན་བདུན”གྱིས་ལྷ་མོ་འཁྲབ་སྐབས་སུ་རྣམ་ཐར་གྱི་སྐད་གདངས་ནི་རྔམ་བརྗིད་ལྡན་པ་དང་། མི་སྣ་སོ་སོས་རང་རང་གི་གཞས་གདངས་རེ་གཏོང་ལེན་བྱེད་ལ། དེ་དག་གི་གདངས་ཀྱང་གཅིག་མཚུངས་མ་ཡིན། ལྷག་པར་དུ་ཟློས་གར་འཁྲབ་ཚན་དེའི་ནང་དུ“བཀྲ་ཤིས་དགུ་འཛོམས”ཞེས་པའི་གླུ་གར་དུམ་བུ་གཅིག་ཡོད་པས། དེ་ནི་ལྷ་མོའི་ཟློས་གར་གཞན་དུ་མཐོང་དཀོན་པ་ཞིག་ཡིན། ལོ་ངོ་བཅུ་ཕྲག་ཁ་ཤས་ནང་འཕེལ་རྒྱས་བཏང་བའི་འོག  དུས་ཕྱིས་ཟློས་གར་པ་ཚང་མ་སྐྱེས་པ་བུ་ཤ་སྟག་ཡིན་པ་གང་ཞིག འཁྲབ་ཚན་གྱི་དགོས་མཁོར་གཞིགས་ཏེ་བུས་བུ་མོའི་རྣམ་པ་འཁྲབ་པས། མི་གྲངས་ཐད་རྔོན་པ་འཁྲབ་མཁན༥རྒྱ་ལུ་འཁྲབ་མཁན༢། ལྷ་མོ་འཁྲབ་མཁན༢། རྔ་སྦུབ་མཉམ་དཀྲོལ་བྱེད་མཁན༡སྟེ་ཁྱོན་མི་བཅུ་དགོས། འཁྲབ་སྟོན་བྱེད་སྐབས་སྔོན་ལ་མི་གཅིག་གིས་གཞུང་སྲོང་བའི་ཚུལ་གྱིས་གཟིགས་མོ་བར་འཁྲབ་ཚན་གྱི་ནང་དོན་མདོར་བསྡུས་ངོ་སྤྲོད་བྱེད་པ་དང་། འཁྲབ་ཡུན་ཡང་རིན་ཐུང་སྣ་ཚོགས་ཡོད་དེ། སྐབས་རེ་ཆུ་ཚོད་ཁ་ཤས་འཁྲབ་པ་དང་། སྐབས་རེ་ཉིན་རེ་གཉིས་ཙམ་འཁྲབ་པ། ཡང་མཚམས་རེ་ཉིན་དྲུག་བདུན་ཙམ་འགོར་བའང་ཡོད། 
        འཕྱོངས་རྒྱས་གར་བྲོ་བཀྲ་ཤིས་སྤུན་བདུན་ནི་བོད་མི་རིགས་ཀྱི་དམིགས་བསལ་ཁྱད་ཆོས་ལྡན་པའི་དམངས་ཁྲོད་ཟློས་གར་ཞིག་ཡིན་ལ། དེ་ནི་བོད་མི་རིགས་ཀྱི་ཕུལ་བྱུང་མི་རིགས་ཤུལ་བཞག་རིག་གནས་ཤིག་ཡིན། དེས་ན་ཟློས་གར་དེ་མྱུར་སྐྱོབ་དང་། སླར་གསོ། སྲུང་སྐྱོབ་བཅས་ཀྱི་ལས་དོན་སྤེལ་ཚེ། དེས་བོད་མི་རིགས་ཀྱི་ཕུལ་བྱུང་རིག་གནས་དར་སྤེལ་དང་། མ་ཉམས་གོང་འཕེལ་གཏོང་རྒྱུར་སྐུལ་འདེད་ཀྱི་ནུས་པ་ཤུགས་ཆེན་ཐོན་ཐུབ་ལ། ས་ཁུལ་དེ་ཙམ་མ་ཟད། རྒྱལ་ཡོངས་ཀྱི་བསམ་པའི་དཔལ་ཡོན་འཛུགས་སྐྲུན་དང་། མི་དམངས་མང་ཚོགས་ཀྱི་རིག་གནས་འཚོ་བ་ཕུན་སུམ་ཇེ་ཚོགས་སུ་གཏོང་རྒྱུ། སྤྱི་ཚོགས་ཧྲིལ་པོའི་བྱང་ཚད་གོང་མཐོར་གཏོང་རྒྱུ། དེ་བཞིན་སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་ཞི་མཐུན་སྤྱི་ཚོགས་འཛུགས་སྐྲུན་ལ་སྐུལ་འདེད་ཀྱི་ནུས་པ་ཤུགས་ཆེན་ཐོན་ཐུབ། ལྷག་པར་དུ་འཕྱོངས་རྒྱས་གར་བྲོ་བཀྲ་ཤིས་སྤུན་བདུན་གྱི་ཟློས་གར་དེ་མྱུར་སྐྱོབ་ཐུབ་ཚེ། དེས་བོད་ཀྱི་ལྷ་མོའི་ཟློས་གར་གྱི་གྲུབ་ཚུལ་དང་། འཕེལ་རིམ་གྱི་ལོ་རྒྱུས་སོགས་ལ་ཞིབ་འཇུག་བྱེད་པར་དཔྱད་གཞིའི་རིན་ཐང་ཧ་ཅང་ཆེན་པོ་ལྡན་ཡོད་པ་རེད། 
དེས་ན། འཕྱོངས་རྒྱས་གར་བྲོ་བཀྲ་ཤིས་སྤུན་བདུན་གྱི་ཟློས་གར་རྒྱུན་འཛིན་དང་འཕེལ་རྒྱས་གཏོང་རྒྱུ་དང་། གོམ་གང་མདུན་སྤོས་ཀྱིས་བོད་མི་རིགས་ཀྱི་ཕུལ་བྱུང་སྲོལ་རྒྱུན་རིག་གནས་དར་སྤེལ་དང་། ཞིབ་འཇུག  སྔོག་འདོན་བཅས་ཧུར་ཐག་བྱས་ཏེ། མི་རིགས་སྲོལ་རྒྱུན་རིག་གནས་ཀྱི་གཟུགས་བརྙན་གསར་པ་བསྐྲུན་རྒྱུར་འབད་དགོས་སོ།།
