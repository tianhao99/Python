བོད་ཀྱི་སྲོལ་རྒྱུན་ལོ་གསར་དར་ཚུལ་སྐོར་རགས་ཙམ་གླེང་བ།
ལོ་གསར་ནི། ལོ་སྔ་མ་རྫོགས་ནས་ལོ་རྗེས་མའི་འགོ་ཚུགས་པ་ལ་ཟེར་ཞིང༌། བོད་མིའི་ངག་རྒྱུན་དུ་ཡོད་པའི་བླ་མ་ལོ་གསར་དང༌། རྒྱལ་པོ་ལོ་གསར། བསྟན་སྲུང་ལོ་གསར། བེའུ་ལོ་གསར། ཡུལ་པ་ལོ་གསར། སོ་ནམ་ལོ་གསར་རམ་ཞིང་པ་ལོ་གསར་སོགས་ལོ་འགོ་འཛིན་ཚུལ་མི་འདྲ་བའི་དབང་གིས་ལོ་གསར་ཡང་ཡུལ་སྲོལ་མི་འདྲ་བ་སྣ་ཚོགས་བྱུང་ཡོད་ཀྱང༌། བོད་ཀྱི་ཁྱད་ཆོས་ཚང་བའི་ལོ་གསར་གཏོང་བའི་ཐོག་མའི་ལམ་སྲོལ་དེ་དུས་ག་ཚོད་ཙམ་ཞིག་ནས་དར་སྲོལ་བྱུང་བའི་སྐོར། ཁོ་བོས་རང་རེའི་ལོ་རྒྱུས་དེབ་ཐེར་ཁག་ཅིག་དང༌། སྐར་ནག་རྩིས་རིག་ཐོག་ཏུ་ཡོད་པའི་བྱུང་རབས་ཐོར་བུ། རྒྱ་ཆེའི་ཞིང་འབྲོག་ཁུལ་གྱི་རྒན་པོའི་ངག་རྒྱུན་དུ་གྲགས་པའི་སྒྲུང་གཏམ་བཅས་ལ་གཞི་བཅོལ་གྱིས་དེའི་སྐོར་རགས་པ་ཙམ་ཞིག་གླེང་བར་འདོད།
          དེ་ཡང༌། གངས་རི་ར་བས་སྐོར་བའི་བོད་གངས་ཅན། འཕགས་པ་སྤྱན་རས་གཟིགས་ཀྱི་གདུལ་ཞིང་གནས་མཆོག་འདིར། བདག་ཅག་སྟོན་པ་བཀའ་དྲིན་ཅན་འདས་ལོའི་སྔོན་གྱི་ལོ་བརྒྱའི་ནང་བོད་རྗེ་གཉའ་ཁྲི་བཙན་པོའི་རྒྱལ་བརྒྱུད་པུ་དེ་གུང་རྒྱལ་གྱི་སྐབས་སུ་སྤང་ཐང་ཞིང་དུ་རྨོས། ཆུ་ལག་ཡུ་བར་དྲངས། ས་བོན་ཕུར་འདེབས་བྱེད། སོལ་བ་ལ་བརྟེན་ནས་དངུལ་དང་ཟངས། ལྕགས་བཅས་བཞུ་བ་སོགས་ཞིང་ལས་དང་ལག་ཤེས་བཟོ་ལས་བཅས་ཡར་རྒྱས་ཆེན་པོ་བྱུང་བའི་ སྐབས་དེར། ལྷོ་ཁ་ཡར་ལུང་ཁུལ་དུ་ཐགས་མཁན་རྒན་མོ་བལ་མ་ཞིག་གིས་ཟླ་བའི་དཀར་ཆ་འཕེལ་འགྲིབ་ཀྱི་ཆ་ཤས་ལ་བལྟས་ནས་ལོ་ཟླ་ཞག་དུས་རགས་ཙམ་རྩིས་སྲོལ་ཡོད་པ་དེར་རྒན་མོ་བལ་ མོའི་ཟླ་རྩིས་ཞེས་གྲགས། ཟླ་རྩིས་དེ་ལ་བརྟེན་ནས་ཟླ་བ་དང་ལོ་འཁོར་ཡང་རྩིས་སྲོལ་བྱུང་འདུག་ཀྱང༌། འབྱུང་བའི་འགྱུར་འགྲོས་ལ་བརྟེན་ནས་དྲོད་རླུང་ལྡང་ཞིང་ཤིང་རྩི་ཁ་འབུས། ལོ་མ་རྒྱས་ཏེ་ལོ་ལྟར་ཁམ་བུའི་མེ་ཏོག་རབ་ཏུ་ཤར་སྐབས། ནམ་དུས་དཔྱིད་ཀ་འཆར་བའི་དགའ་སྤྲོ་བྱེད་པ་ལས་དམིགས་བསལ་ལོ་གསར་ཞེས་དགའ་སྟོན་ བྱེད་སྲོལ་མེད་པར་སྣང༌།
གདོད་མའི་ལོ་གསར།
          དེ་ནས་རིམ་བཞིན་བོད་དུ་བོན་གྱི་བསྟན་པ་དར་ཚུལ་སྐོར། ད་ཡོད་ཀྱི་དཔྱད་གཞིའི་ཡིག་ཆ་གཞིར་བཟུང་ན་སྟོན་པ་འདས་ལོའི་སྔོན་གྱི་ལོ་བརྒྱ་ཕྲག་གཅིག་ཡས་མས་སུ་བོད་ཀྱི་རྒྱལ་པོ་ཐོག་མ་གཉའ་ཁྲི་བཙན་པོས་སྲིད་དབང་བཟུང་བའི་སྐབས་རྒྱུ་བོན་ཤེས་པ་ཅན་བཅུ་གཉིས་དར་ཞེས་གསལ་བ་དང༌། དེའི་ནང་ནས་སྐོར་ཤེས་རྩིས་མཁན་ནི་སྐར་རྩིས་ཤེས་མཁན་ཞིག་ཡིན་པ་བོན་གྱི་ལོ་རྒྱུས་ནང་འཁོད་ཡོད་ལ། རྒྱལ་པོ་པུ་དེ་གུང་རྒྱལ་གྱི་དུས་སུ་རྒྱུ་བོན་རིགས་བཞི་ཞེས་པ་བྱུང་བའི་ནང་ནས་དུར་གཤེན་མཚོན་ཆ་གསོས་གཤིས་གཉིས་ཀྱི་རྩིས་བྱེད་མཁན་ཞིག་ཡིན་པར་《རྒྱལ་རབས་གསལ་བའི་མེ་ལོང་》དང༌། 《ཆོས་འབྱུང་མཁས་པའི་དགའ་སྟོན་》སོགས་ལོ་རྒྱུས་དེབ་ཐེར་མང་པོའི་ནང་འཁོད་པ་ ལྟར། བོན་གྱི་བླ་ཆེན་དྲན་པའི་ཞལ་ནས་ཀྱང༌། “སྤྱིར་ན་སྟོང་གསུམ་འདུལ་བྱེད་སྟོན་པ་ཆེའི། །གང་འདུལ་མཛད་པའི་ཐབས་སུ་བཀའ་གསོལ་བཞི། །བོན་ཆོས་རྩིས་དཔྱད་བཞི་རུ་བསྟན་པ་ཡང༌། །དོན་དུ་ཐ་དད་མེད་དུ་ངོ་བོ་གཅིག །”ཅེས་གསུངས་ཡོད། དེས་ན། བོན་ཆོས་དང་མཐུན་པའི་ཀོང་རྩེ་འཕྲུལ་རྒྱལ་༡ཞེས་པའི་རྩིས་དང་ཤེས་རིག་མཁས་པ་ ཁག་གཅིག་གིས་ཙི་ནའི་ཡུལ་༢ནས་སྟོད་ཞང་ཞུང་བརྒྱུད་དེ་བོད་ལ་ཕེབས། བོད་དུ་ཁོང་རྣམ་པའི་ལུགས་དང་མཐུན་པའི་《འབྱུང་ལྡན་》༣རྩིས་ལུགས་སམ་ལོ་འཁོར་རྩིས་ཚུལ་དང་དགུན་ཉི་ལྡོག་མཚམས་ལོ་འཁོར་བའི་གདོད་མའི་ ལོ་གསར་གྱི་དགའ་སྟོན་གཏོང་བའི་ལམ་སྲོལ་ནི་བོན་གྱི་ལུགས་སྲོལ་ཡིན་པ་དཔེར་ན་ལུག་མགོ་ལ་གཡུང་དྲུང་དང་ཉི་ཟླའི་གཟུགས་དབྱིབས་བརྒྱན་ཏེ་ཕ་མེས་ཀྱི་དུར་ལྷ་ གསོལ་ནས། ཕ་མེས་ལ་དྲན་གསོ་ཞུ་བ། ས་དམར་དང་ས་དཀར་གྱི་གཡུང་དྲུང་འབྲི་བ། དེ་བཞིན་མི་ནོར་གཡང་ཆགས་པའི་ཆེད་དུ་མདའ་དར་དང་གྲོ་སོ་ཕྱེ་མར། ཆང་ཕུད། ལོ་ཕུད། གཡང་ཤ་ལག་པ་གཡས་པ། ཁ་ཟས་ཀྱི་སྡེར་ཁ་བཅས་ཡུལ་གྱི་ཐོན་ཁུངས་གང་ལེགས་བསྒྲིགས་ཏེ་ཚེ་རིང་ནད་མེད་བདེ་སྐྱིད་ལྡན་པའི་བསྔོ་སྨོན་བྱེད་པ་དང་སྦྲགས་བཀྲ་ཤིས་གཡང་ཆགས་ཀྱི་བྱེད་སྒོ་སྤེལ་བ། རིག་རྩལ་ཐོག་ཕོ་གཞོན་ཚོས་གཡག་ལུག་བསད་པའི་རྭ་ཅོའི་ཐོ་བཙུགས་པར་ཐག་རིང་ཙམ་ནས་ལག་རྡོ་འཕངས་ཏེ་སློག་ཐུབ་ན་དཔའ་སྐད་བསྒྲགས་པ། དེ་བཞིན་མདའ་གྲི་མདུང་དང་འུར་རྡོ་འགྲན་པ། མོ་གཞོན་ཚོས་གཞས་སྒོར་གྱིས་གླུ་ལེན་པ་དང༌། ཞབས་བྲོ་རྒྱག་པ། ཐེ་ཕེག་རྩེ་བ། གབ་རེས་འུ་རེས་རྩེ་བ་སོགས་བྱེད་པ། དེ་བཞིན་བོན་ཆོས་དང་མཐུན་པའི་རི་རྩེར་རང་བསྟེན་སྲུང་མའི་གཟུངས་དང་སྲོག་ཡིག་གི་གཤམ་དུ་ལས་བཅོལ་འཕྲིན་བསྐུལ་བཀོད་པའི་དར་ཚོན་སྣ་ལྔའམ་རང་མཐུན་ཚོན་མདོག་གི་དར་ལྕོག་འཛུགས་པ། རླུང་རྟ་ཡིན་ན། ཚོན་སྣ་ལྔའམ་སོ་སོའི་འབྱུང་ཁམས་མདོག་མཐུན་གྱི་དར་རས་ལ་རླུང་རྟའི་གཟུངས་སྔགས་དཔར་ཏེ་དར་ལྡིང་གཏོང་བ། ལྷ་གསོལ་བྱེད་པ་སོགས་ཀྱི་བྱེད་སྒོ་དུས་དེ་ཙམ་ནས་དར་སྲོལ་བྱུང་ཡོད་སྐད།
བེའུ་ལོ་གསར།
          ཕྱི་ལོ་༥༥༨ གནམ་རི་སྲོང་བཙན་གྱི་དུས་ཙི་ན་ལ་བོད་མི་རིག་པ་རྣོ་བའི་སློབ་བུ་བཞི་སྐར་རྩིས་སློབ་གཉེར་བྱེད་དུ་བརྫངས་ཤིང༌། ཁོང་ཚོ་ཕྱིར་ལོག་ནས་ཤེས་བྱ་རྣམས་ངག་རྒྱུན་ནས་བརྒྱུད་སྤེལ་བྱས་ཏེ་བོད་ཀྱི་གནམ་རིག་སྐར་རྩིས་རིག་པའི་འཕེལ་རྒྱས་ལ་ལག་རྗེས་གང་ལེགས་བཞག་ཡོད། ཕྱི་ལོ་༦༣༠ 《དཔའ་བོ་གཙུག་ལག་ཕྲེང་བ་》ལྟར་ན་བོད་རྗེ་གནམ་རི་སྲོང་བཙན་གྱི་སྲས་སྲོང་བཙན་སྒམ་པོར་རྒྱལ་སྲིད་ཕུལ་བ་དང༌། ཕྱི་ལོ་༦༣༢ བལ་བཟའ་ཁྲི་བཙུན་བོད་དུ་བསུས་ཏེ་ཇོ་བོ་མི་བསྐྱོད་རྡོ་རྗེ་བོད་དུ་ཕེབས། ཕྱི་ལོ་༦༣༨ ལོར་”བོད་རྗེ་སྲོང་བཙན་སྒམ་པོས་༤རྒྱ་(རྒྱ་ནག་གི་རྒྱལ་པོ་ཐང་ཐེ་ཙུང་གི་བུ་མོ )རྗེའི་སྲས་མོ་སློང་བའི་ཕོ་ཉ་མངགས་ཀྱང་རྒྱ་རྗེས་བུ་མོ་མ་བྱིན་པར་བོད་དུ་ ལོག་ཏེ་བཙན་པོར་རྫུན་ཞུས། རྒྱ་རྗེ་དེ་ངེད་རང་ལ་ཤིན་ཏུ་དགའ་ནས་བུ་མོ་སྟེར་བ་ཡོད་པ་ལ་ཧོར་སེར་ཐུ་ལུ་ ཧུན་གྱིས་རྒྱ་རྗེ་ལ་ཕྲ་མ་ཞུས་པས་ལན་ཚུལ་ཞུས་པས། བོད་ཀྱི་བཙན་པོ་ཁྲོས་ནས་བོད་དམག་ཁྲི་ཉི་ཤུ་ཁྲིད་དེ་ཟུང་ཅིའུ་(སུང་ཀྲའུ་)དེ་ ནས་བློན་པོ་གཡའ་ཐུང་བྱ་བ་ལ་དམག་སྐུལ་ནས། ཐུ་ལུ་ཧུན་གྱི་ཡུལ་འཇོམས་སུ་བཏང་བ་ལ་ཐུ་ལུ་ཧུན་རྣམས་གཙང་ཁའི་མཚོ་སྔོན་ཕན་ལ་བྲོས། ཤུལ་དུ་ལུས་པའི་ནོར་ཐམས་ཅད་བོད་ཀྱིས་འཁྱེར། དེ་ནས་བོད་ཀྱི་བཙན་པོས་བློན་པོ་སེ་ལེ་སྟོང་བཙན་ལ་རིན་པོ་ཆེ་སྣ་ཚོགས་བསྐུར་ནས། རྒྱ་རྗེའི་སྲས་མོ་སློང་དུ་བཏང་བར་འུན་ཤིང་ཀོང་ཇོར་རྟེན་སྐལ་ཇོ་བོ་ཤཱཀྱ་མུ་ནེ་སོགས་རྫོངས་གྱ་ནོམ་པ་དང་བཅས་བོད་དུ་སླེབས། ”རྒྱ་བཟའ་ཀོང་ཇོ་རྩིས་ལ་མཁས་པས་རྒྱའི་འབྱུང་རྩིས་སྤོར་ཐང་བརྒྱད་ཅུའི་སྐོར་མགོ་དང་རྒྱ་བཟའི་གུང་ཁུག་སོགས་བོད་ལ་མཚམས་སྦྱོར་གནང་བ་ནས་བཟུང༌། བོད་དུ་《སྔ་འགྱུར་རྒྱུ་རྩིས་རྙིང་པའི་ལུགས་》དང༌། ཡང་འགྱུར་《གསལ་སྒྲོན་སྦས་བཅད་མེད་པའི་ལུགས་》ཞེས་པའི་ནང༌། “དཔྱིད་ར་སྟག་ལ་ལོ་འགོ་འཛིན་ཞེས་པ་བོད་ཟླ་བཅུ་གཅིག་པའི་ཚེས་གཅིག་ལ་ལོ་གསར་གཏོང་སྲོལ་ཆགས་པ་”ཞེས་པའི་བརྗོད་ཚུལ་བྱུང་འདུག དེ་དང་གོང་དུ་དགུན་ཉི་ལྡོག་མཚམས་དང་མཐུན་ཤས་ཆེ་བ་ཡོད་པ་དེ་ནི། སྐབས་དེར་བོད་ས་མང་ཆེ་བ་འབྲོག་ལས་དང་ཡུལ་པ་ཕལ་ཆེ་བ་ས་མ་འབྲོག་ཡིན་པས་ནམ་དུས་གྲང་ངར་ཆེ་བས་འབྲོག་པ་རྣམས་ས་དྲོ་སར་རུ་བ་(རྩ་ར་)སྤོས་ཏེ། སྤུ་ནག་འབྲི་དང་གཡང་དཀར་ལུག་གི་ལུག་ཕྲུག་དང་བེའུ་བཙའ་རྒྱུའི་གྲབས་གཤོམ་བྱེད་པའི་དུས་ཚོད་ཡིན་ཞིང༌། དེའི་རིང་འབྲོག་པ་ཁག་ཅིག་ཡུལ་པའི་ནང་ཞིང་འབྲོག་ཐོན་རྫས་(ཤ་མར་ཚྭ་བུལ་ཏོག་ དང་འབྲུ་རིགས་བརྗེ་བ)བརྗེ་ལེན་གྱི་ཚུར་འཁོར་ཏེ་ནང་མི་ཚང་འཛོམས་ཀྱིས་བེའུ་ལོ་གསར་ཞེས་ལོ་གསར་གཏོང་སྲོལ་ཡོད་པ་དུས་ད་ལྟ་ཡང་བོད་ཀྱི་འབྲོག་ཁུལ་མང་ཆེ་བའི་ནང་འགྱུར་མེད་གནས་ཡོད་པ་གོ་ཐོས་འདུག
སོ་ནམ་ལོ་གསར།
          ཕྱི་ལོ་༧༥༥ དུས་རབས་བརྒྱད་པའི་ནང་ཆོས་རྒྱལ་ཁྲི་སྲོང་ལྡེ་བཙན་སྐབས། རྒྱའི་གྱིམ་ཤིང་ཀོང་ཇོ་བོད་དུ་བག་མར་ཕེབས་སྐབས་སྐར་རྩིས་དང་དཔེ་དེབ་མང་པོ་ཞིག་བསྣམས་ཕེབས་པ་དེ་དག་བོད་ཡིག་ཏུ་བསྒྱུར། བོད་ལ་བར་བསྒྱུར་《གསལ་སྒྲོན་སྦས་བཅད་》ཅན་གྱི་ལུགས་ཟེར་བ་རྒྱ་ནག་གི་ཞིང་རྩིས་དང་མཐུན་པའི་ལོ་གསར་གཏོང་རྒྱུར་བོད་རྗེ་ཁྲི་སྲོང་ལྡེ་བཙན་གྱིས་གྱིམ་ཤིང་ཀོང་ཇོར་ལོ་གསར་ག་དུས་གཏོང་རྒྱུའི་སྐོར་བཀའ་འདྲི་གནང་སྐབས། རྒྱ་རྗེ་ཀོང་ཇོས་ང་ལ་བཀའ་འདྲི་མ་གནང་སྐར་མ་རྒྱལ་ལ་བཀའ་གནང་རོགས་ཞུས་ནས་སྐར་མ་རྒྱལ་གྱི་ནམ་ཚོད་ཟིན་པའི་དུས་ནི། བོད་ཟླ་བཅུ་གཉིས་པའི་(རྒྱ་རྩིས་ལོ་གསར་གྱི་ཉིན་གཅིག་གི་རྗེས)ནང་ཡིན་པར་བརྟེན། བོད་ཟླ་བཅུ་གཉིས་པའི་ཚེས་གཅིག་ལ་ལོ་གསར་གཏོང་རྒྱུའི་ལམ་སྲོལ་ཆགས་པ་ཞེས་པའི་བརྗོད་ཚུལ་ཡང་བྱུང་འདུག
          དེའི་སྐོར་དམངས་ཁྲོད་ནང་བཤད་སྲོལ་དེ་འདྲ་ཞིག་ཡོད་འདུག་པ་སྟེ། ཡུལ་པ་ལོ་གསར། སོ་ནམ་ལོ་གསར། ཞིང་པ་ལོ་གསར། ཤར་ཕྱོགས་ལོ་གསར་ཞེས་བོད་ཀྱི་མདོ་དབུས་ཁམས་གསུམ་གྱི་རྒྱ་ཆེ་བའི་ཡུལ་ལུང་འདྲ་མིན་ནང་ཡུལ་སྐད་ཤས་ཆེ་བའི་དབང་གིས་འབོད་ཚུལ་ཅུང་མི་འདྲ་བ་ཙམ་ལས། དོན་དུ་རང་རེའི་སོ་ནམ་ཞིང་ཁུལ་གྱི་དུས་བརྟག་ལ་མྱོང་བ་ཆེ་བའི་རྒན་པོ་ཚོས་ ཉིན་མཚན་གྱི་དུས་བཞིར་ཕྱེ་བ་རྒྱུ་སྐར་ཉེར་བརྒྱད་ལ་བརྟག་པའི་ཐོག་ནས་གནམ་ལ་སྐར་མ་གང་འཆར་བའི་དུས་སུ་ས་ཐོག་ལ་འགྱུར་བ་གང་ཡོང་དང༌། དེའི་དུས་སུ་ཞིང་འབྲོག་ཐོན་སྐྱེད་བཟང་ངན་གཏན་འཁེལ་བའི་སྐར་རྩིས་ཀྱི་རྩིས་འབྲས་ལྟར་སོ་ནམ་ཞིང་ལས་ཀྱི་ལས་འགོ་བཙུགས་པའི་བྱེད་སྒོ་ཞིག་རིམ་པས་ལོ་གསར་དགའ་སྟོན་ལ་གྱུར་བར་སེམས། རྒྱུ་མཚན་ནི། སྐབས་དེར་ཡུལ་དགོན་སོ་སོའི་བླ་གྲྭ་གང་ཅོག་གདན་ཞུས་ཀྱིས་ཆོ་ག་གཞུང་ལྟར་ས་ ཆོག་རྣམས་ལེགས་པར་གྲུབ་མཚམས། ཡུལ་མི་རྣམས་ནས་དཀར་སྤྲོ་དང་གྲོ་སོ་ཕྱེ་མར། ཆང་ཕུད་འོ་ཕུད་སོགས་མཆོག་གསུམ་དང་ཡུལ་ལྷ་སོ་སོར་མཆོད་དེ། ཡུལ་གྱི་གཙོ་རྒན་གྱིས་ལག་ཏུ་རིན་ཆེན་སྣ་ཚོགས་བསྒྲོན་པའི་དར་སྣ་ཐོགས་ཅིང་ངག་ ནས། “ས་བདག་དབང་ཆེན་ཁྱབ་པ་དང༌། ས་ཡི་ལྷ་མོ་བརྟན་མའི་གཙོས། ལྷ་ཀླུ་གཞི་བདག་རྣམས་ཀྱིས་བདག་ཅག་འཁོར་དང་བཅས་པ་ལ། ནད་ཡམས་མུ་གེ་མཚོན་འཁྲུག་དང༌། ཙད་སད་སེར་གསུམ་ཐོག་མདའ་དང༌། གོད་ཁ་ཆོ་འཕྲུལ་ལྟས་ངན་དང༌། མི་ནད་ཕྱུགས་ནད་མ་གཏོང་ཞིག ཆར་ཆུ་དུས་སུ་འབབ་པ་དང༌། ལོ་ཕྱུགས་རྟག་ཏུ་ལེགས་པར་ཤོག བདག་སོགས་འདོད་པའི་བསམ་པ་འགྲུབ་པར་ཤོག ཀི། ཀི། བསྭོ། བསྭོ། ལྷ་རྒྱལ་ལོ་”ཞེས་བསང་གསོལ་ཀི་བསྭོ་ནམ་མཁར་བསྒྲགས་པ་བཞིན་བྱས་རྗེས། དབྱར་གྱི་སྔོ་ཁྲིམས་ཐོག་མ་བསྡམས་ཤིང༌། བོད་ཟླ་བཅུ་གཉིས་པའི་ཚེས་གཅིག་ནས་ཚེས་བཅུ་བར་ཡུལ་པ་ལོ་གསར་རམ་སོ་ནམ་ལོ་གསར་ ཞེས་རང་སོ་སོའི་སྡོད་ཁང་གི་ཕྱི་ནང་གང་སར་གད་ཕྱིས་གཙང་བཟོས་ཀྱིས་སྒོ་ར་དང་ཐབ་ཚང་ནང་ས་དཀར་གྱིས་གཡུང་དྲུང་དང་བཀྲ་ཤིས་རྟགས་བརྒྱད་སོགས་བྲིས་པ་དང༌། ཁང་པའི་ནང་མི་ཚང་སོ་སོའི་དོད་ཡོན་ཆེ་ཆུང་གི་དཀར་སྤྲོ་དང་ཤ་མར་ཐུད་སོགས་ཟས་རིགས་སྒྲིག་གཤོམ་བྱེད། རང་ལུས་གཟབ་མཆོར་དང་རྒྱན་ཆ་སྤྲས་ཏེ། ཚེས་གཅིག་ལ་ཞིང་སར་ཡུལ་མི་འདུ་འཛོམས་ཀྱིས་མཆོག་གསུམ་གྱི་མཚོན་བསང་མཆོད་དང་ཚེས་གསུམ་ལ་ཡུལ་ལྷ་གཞི་བདག་སོགས་ལྷ་སྲུང་གསོལ་ཆེད་རི་འགོར་བསང་གསོལ་དར་ལྡིང་བཙུགས་ཏེ་དགའ་སྟོན་གླུ་གར་འཁྲབ་པ་དང་ཟས་ཆང་ལ་རོལ། ཚེས་བཞི་ནས་བཅུ་བར་གྲོང་པ་ཁྱིམ་མཚེས་དང་དགའ་ཉེ་བློ་དཀར་ཕན་ཚུན་མགྲོན་འབོད་ ཀྱིས་ཟ་དགའ་འཐུང་སྐྱིད་དང་དགའ་སྟོན་གླུ་གར་འཁྲབ། ཚེས་བཅུ་ནས་ཉེ་འཁོར་ཡུལ་དགོན་སོ་སོའི་ནང་སྨོན་ལམ་ཚུགས་པ་སོགས་བྱེད་སྲོལ་དུས་ད་ལྟ་ཡང་དབུས་གཙང་ཁུལ་(ལྷ་སའི་ཉེ་འཁོར་རྫོང་གཞིས་ཁག་བོད་ཀྱི་ལོ་གསར་གཏོང་བ་ཕུད་)ཙམ་མ་ཟད། མདོ་སྟོད་སྨད་གཉིས་ལའང་སྔར་སྲོལ་མི་ཉམས་རྒྱུན་འཛིན་ཡོད་པ་མ་ཟད། བོད་ནས་ཉེ་ཆར་ཕེབས་པ་ཚོར་དེའི་སྐོར་བཅར་འདྲི་ཞུ་སྐབས་ཆོས་ལུགས་བྱེད་སྒོར་རྒྱ་གཞུང་ནས་འགོག་སྡོམ་བྱེད་པ་ལས་དེ་མིན་སྔར་སོར་གནས་སྐོར་གླེང་གི་འདུག
          མཚམས་འདིར་གསལ་བཤད་དགོས་པ་ཞིག་ལ། “ཡོངས་སུ་གྲགས་པའི་ལོ་རྒྱུས་མང་པོ་ཞིག་གི་ནང༌། ཆོས་རྒྱལ་སྲོང་བཙན་སྒམ་པོའི་ཡབ་གནམ་རི་སྲོང་བཙན་གྱི་དུས་ཐང་རྒྱལ་རབས་དང་བོད་གཉིས་འབྲེལ་བ་ཐོག་མར་བྱུང་ཞིང༌། རྒྱའི་སྐར་རྩིས་དང་སྨན་ཐོག་མར་དར་ཞེས་བཤད་སྲོལ་ཡོད་ཅིང༌། དེ་ནི་རྒྱ་ནག་ནས་བོད་དུ་དར་བའི་འབྱུང་རྩིས་དང་གསོ་རིག་ཐོག་མར་དར་བའི་དུས་སྐབས་དེ་ནས་འགོ་ཚུགས་པ་ཡིན་ཞེས་བཤད་པ་ཙམ་ལས་དེའི་གོང་དུ་བོད་ལ་སྐར་རྩིས་དང་གསོ་རིག་མེད་པའི་དོན་མེད་ཅིང༌། ཆོས་རྒྱལ་སྲོང་བཙན་སྒམ་པོ་དང༌། ཁྲི་སྲོང་ལྡེ་བཙན་གྱིས་དུས་སུ་རྒྱ་བཟའ་ཀོང་ཇོ་དང་གྱིམ་ཤིང་ཀོང་ཇོ་གཉིས་ ཀྱིས་རྒྱའི་རིག་གཞུང་གི་དཔེ་དེབ་མང་པོ་རྟེན་སྐལ་ཚུལ་དུ་བོད་ལ་འཁྱེར་ཡོད་པ་དང༌། དེའི་ནང་ནས་ཁག་གཅིག་བོད་སྐད་དུ་བསྒྱུར་བ། རྒྱ་ཡི་སློབ་དཔོན་ཧར་ནག་པོའི་སློབ་མ་ཁྱུང་ཤག་དར་སོགས་ཀྱིས་རྩིས་གཞུང་མང་པོ་བརྩམས་ཡོད་ཀྱང་འབྱུང་རྩིས་དང་ནག་རྩིས་(རྨེ་བ་དགུ་དང༌། སྤར་ཁ་བརྒྱད། ལོ་སྐོར་བཅུ་གཉིས། ཀེག་རྩིས། བག་རྩིས། གཤིན་རྩིས་སོགས་)ཀྱི་ཕྱོགས་ཡིན་པ་ལས་སྐར་རྩིས་ཀྱི་ཕྱོགས་འདྲ་བ་ཉུང་ཤས་ཤིག་ལས་མེད། དེར་བརྟེན། ཆོས་རྒྱལ་སྲོང་བཙན་སྒམ་པོའི་སྐབས་ད་ལྟ་ང་ཚོས་བེད་སྤྱོད་བྱེད་བཞིན་པའི་བོད་ཡིག་འདི་གསར་གཏོད་བྱས་པ་ནས། རབ་བྱུང་དང་པོའི་མེ་ཡོས་མ་ཤར་བར་གྱི་ལོ་གྲངས་༣༩༠ ལྷག་ཙམ་གྱི་བར་དུ་ནི་བོད་རང་ཉིད་ཀྱི་གནའ་རབས་སྐར་རྩིས་འགའ་ཞིག་དང་། ཐང་རྒྱལ་རབས་ཀྱི་སྐབས་བོད་དུ་དར་བའི་འབྱུང་རྩིས་ནག་རྩིས་ཉུང་ཤས་ཤིག་ལས་གལ་ཆེའི་གནམ་རིག་སྐར་རྩིས་ནི་དར་སྤེལ་བྱུང་མེད། རབ་བྱུང་དང་པོའི་མེ་ཡོས་(༡༠༢༧) ལོར་ཁ་ཆེ་ཟླ་བ་མགོན་པོས་དུས་འཁོར་གྱི་གཞུང་མང་པོ་བོད་དུ་གདན་དྲངས་ནས་བོད་ དུ་བསྒྱུར་འགོ་ཚུགས་པ་དང༌། སྐབས་དེར་དུས་འཁོར་རྩ་རྒྱུད་ཀྱི་ལེའུ་ལྔའི་ནང་ནས་དབང་མདོར་བསྟན་ཙམ་ལས་མ་བསྒྱུར་ཞིང༌། བསྡུས་རྒྱུད་ཡོངས་རྫོགས་དང་འགྲེལ་ཆེན་དྲ་མེད་འོད་བཅས་བོད་དུ་བསྒྱུར་ཀྱང་ དུས་འཁོར་རྩ་རྒྱུད་ཀྱི་གཞུང་འདི་སངས་རྒྱས་ཆོས་ལུགས་ཀྱི་གཞུང་ཡང་དག་པ་ཡིན་མིན་ཐད་རྩོད་པ་ལངས་ཏེ་ལོ་སུམ་བརྒྱ་ཙམ་ཞིག་བར་དར་སྤེལ་མ་ཐུབ། དེ་ནས་རབ་བྱུང་ལྔ་པའི་ས་རྟ་(༡༣༡༨) ལོར་ཀརྨ་རང་བྱུང་རྡོ་རྗེས་《རྩིས་ཀྱི་བསྟན་བཅོས་ཀུན་བཏུས་》བརྩམས་ཤིང་དེ་ནས་ལོ་གསུམ་སོང་བའི་ལྕགས་བྱ་(༡༣༢༡)ལོར་བུ་སྟོན་རིན་པོ་ཆེས་《རྩིས་ཀྱི་བསྟན་བཅོས་མཁས་པའི་དགའ་བྱེད་》བརྩམས་གནང་མཛད་པ་དེ་ནས་བཟུང་དུས་འཁོར་སངས་རྒྱས་ཀྱི་བཀའ་ཡང་དག་པ་ཡིན་མིན་གྱི་རྩོད་པ་དེ་འཇགས་ནས་རྒྱུད་དེ་དར་སྤེལ་བྱུང་བའི་འགོ་ཚུགས། རབ་བྱུང་བརྒྱད་པའི་མེ་ཡོས་༡༤༤༧ ལོར་ཕུག་པ་ལྷུན་གྲུབ་རྒྱ་མཚོ་དང༌། མཁས་གྲུབ་ནོར་བཟང་རྒྱ་མཚོ་གཉིས་ཀྱིས་《རྩིས་ཀྱི་བསྟན་བཅོས་པད་དཀར་ཞལ་ལུང་》 དང་《དྲི་མེད་འོད་རྒྱན་》སོགས་བརྩམས་ནས་ཕུག་ལུགས་ཕྱི་མའི་བཞེད་སྲོལ་དར་རྒྱས་ ཆེ་བ་བྱུང་བ་དང༌། རབ་བྱུང་བཅུ་པའི་མཇུག་དང་བཅུ་གཅིག་པའི་དཀྱིལ་བར་བོད་ཀྱི་སྐར་རྩིས་མཁས་པ་ སྨོན་འགྲོ་པཎ་ཆེན་ཡབ་སྲས་ལྷུན་པོ་དོན་གྲུབ་དབང་རྒྱལ། འདར་ལོ་ཕུན་ཚོགས་ལྷུན་གྲུབ། ལུ་འགོ་བླ་མཁྱེན། རྣམ་གླིང་དཀོན་མཆོག་ཆོས་གྲགས། ཡབ་སྐུ་སྐུ་སྐྱེ་དབང་གྲགས་རྡོ་རྗེ། ཕུ་ལུང་དགེ་སློང་བློ་བཟང་ཡོན་ཏན། བོད་མཁས་པ་མི་ཕམ་དགེ་ལེགས་རྣམ་རྒྱལ་སོགས་བྱུང་ཞིང༌། ལྷག་པར་དུ་སྡེ་སྲིད་སངས་རྒྱས་རྒྱ་མཚོ་ནས་སྔོན་བྱོན་རྒྱ་བོད་ཀྱི་རྩིས་རིག་མཁས་པ་རྣམས་ཀྱི་གཞུང་མང་པོ་བརྟག་དཔྱད་ཞིབ་འཇུག་གནང་སྟེ། 《ཕུག་ལུང་རྩིས་ཀྱི་ལེགས་བཤད་བཻཌཱུརྱ་དཀར་པོ་》མ་བུ་དང༌། དེའི་དོགས་སེལ་《གཡའ་སེལ་》གཉིས་བརྩམས་པ་ནས་གཞུང་དེ་བོད་ལྗོངས་ས་ཁུལ་མང་ཆེ་བར་དར་ཁྱབ་ཆེ་བ་བྱུང༌། དེ་རྗེས་རབ་བྱུང་བཅུ་གཉིས་པའི་ཤིང་རྟ་(༡༧༡༤) ལོར་སྨིན་གྲོལ་གླིང་ལོ་ཆེན་དྷརྨ་ཤྲིས་《རྩིས་ཀྱི་བསྟན་བཅོས་ཉིན་བྱེད་སྣང་བ་》 མ་ཡིག་དུང་ཡིག་དང༌། དེའི་འགྲེལ་པ་གསེར་གྱི་ཤིང་རྟ་བཅས་བརྩམས་གནང་མཛད་དེ་སྨིན་གྲོལ་གླིང་དུ་སྐར་ རྩིས་དང་སྒྲ་རིག་པ་སློབ་སྦྱོང་བྱེད་པའི་ཆེད་སྦྱོང་སློབ་གྲྭ་བཙུགས་པ་དེ་ནས་བཟུང༌། མདོ་སྟོད་སྨད་སོགས་སུ་མཁས་པ་མང་པོ་བྱུང་སྟེ་རྩིས་ཀྱི་བསྟན་བཅོས་བརྩམས་ཏེ་ དར་ཁྱབ་ཆེན་པོ་བྱུང༌། ྋགོང་ས་སྐུ་ཕྲེང་བཅུ་གསུམ་པའི་སྐབས་རབ་བྱུང་བཅོ་ལྔ་པའི་མེ་འབྲུག་(༡༩༡༦) ལོར་སྨན་རྩིས་ཁང་བཙུགས་ཤིང༌། མཁས་དབང་མཁྱེན་རབ་ནོར་བུ་མཆོག་གཙོ་བོའི་སློབ་དཔོན་དུ་བསྐོས་ཏེ་བོད་ཀྱི་གསོ་ རིག་དང་སྐར་རྩིས་ཀྱི་ཆེད་ལས་ཤེས་ཡོན་སློབ་འཁྲིད་གནང་ནས་ད་ལྟ་འཚོ་བཞུགས་ཀྱི་ སྨན་རྩིས་སློབ་དཔོན་མཁས་པ་རྣམས་བྱུང་སྟེ། ཁོང་རྣམས་ཀྱིས་གཞན་ལ་བརྟེན་མི་དགོས་པར་རང་རེའི་སྨན་བཅོས་འཕྲོད་བསྟེན་དང༌། ཞིང་འབྲོག་ཐོན་སྐྱེད་ཀྱི་ཆེད་ལོ་ལྟར་དུས་ཚིགས་ཉེར་བཞི་ལ་སོགས་པའི་སྐར་རྩིས་ ལོ་ཐོ་རྒྱས་འབྲིང་བསྡུས་གསུམ་བཏོན་གནང་ཡོད་པ་ལགས།
ཧོར་ཟླ་ཞེས་ལོ་ཟླ་བརྩི་སྲོལ་བྱུང་སྟངས།
          ཆུ་ཁྱི་སྟེ། ཕྱི་ལོ་༨༤༢ བོད་ཀྱི་བཙན་པོ་ཁྲི་རབས་ཞེ་གསུམ་པ་སྟེ། གླང་དར་མ་འུར་དུམ་བཙན་གྱི་རྗེས། བོད་ཆེན་པོའི་གོང་བུ་གཅིག་གྱུར་གྱི་བཙན་པོའི་སྟོབས་ཤུགས་ཉམས་ཏེ། བོད་ཀྱི་རྒྱལ་ཕྲན་ཁག་རྣམས་རིམ་པས་སྟོད་སྨད་དང་ལྷོ་བྱང་སོགས་ཁ་འཐོར་ཡ་གྱར་དུ་གནས་པའི་ལོ་བརྒྱ་ཕྲག་བཞི་ཙམ་རིང་རྒྱལ་ཕྲན་སོ་སོ་དང་དཔོན་ཚང་སོ་སོའི་འདོད་མོས་ལྟར་ལོ་གསར་གཏོང་སྲོལ་ཡང་མི་འདྲ་བ་བྱུང༌།
          ཕྱི་ལོ་༡༢༡༦ ཧོར་ཇིང་གེར་རྒྱལ་པོ་རང་ལོ་ང་ལྔ་པ་རབ་བྱུང་བཞི་པའི་མེ་བྱི་ལོར་(《ཧོར་ཆོས་ འབྱུང་》ལས་རེ་དྲུག་པ་མེ་ཕག་ལོར་ཞེས་གསལ་)སྟག་གི་ཟླ་བའི་ཡར་ངོའི་ཚེས་བཅོ་ བརྒྱད་ལ་བོད་མི་ཉག་ཡུལ་དུ་དམག་དྲངས་ནས་རབ་བྱུང་བཞི་པའི་མེ་ཕག་(༡༢༢༧) ལོར་མི་ཉག་རྒྱལ་ཐོག་དགུ་པ་རྡོ་རྗེ་དཔལ་ལམ་སོག་སྐད་དུ་ཐོ་ཅེལ་ཞི་དུར་གུ་ཏ་ གན་ཟེར་བ་བསད་ནས་ཡུལ་སྡེ་ཐམས་ཅད་མངའ་འོག་ཏུ་བསྡུས། དེའི་དགའ་སྟོན་རྟེན་འབྲེལ་བྱས་པའི་ཟླ་བ་དེ་ལ་ཧོར་ཟླ་དང་པོ་ཟེར་བའི་སྲོལ་དེ་ ནས་བྱུང༌།
          ཕྱི་ལོ་༡༢༥༣ ཧོར་ཇིང་གེར་རྒྱལ་པོའི་སྲས་”ཧུ་པི་ལི་སེ་ཆེན་ཧན་གྱིས་ས་སྐྱ་འགྲོ་མགོན་ཆོས་རྒྱལ་འཕགས་པ་མཆོད་གནས་སུ་གདན་དྲངས་པ་ལྟར་ཕེབས་ཏེ། ཧོར་རྒྱལ་པོས་ཞུས་པ་ལྟར་རྒྱལ་པོ་རང་ཉིད་ཀྱི་གཙོ་བྱས་པའི་མི་ཉི་ཤུ་རྩ་ལྔ་ལ་གསང་སྔགས་ཀྱི་དབང་ཐེངས་དང་པོ་གནང༌། དེའི་དབང་ཡོན་དུ་བོད་ཁྲི་སྐོར་བཅུ་གསུམ་གནང་བ་དང་ཏི་ཤྲིའི་གོ་གནས་དང༌། གཡང་ཌི་ལས་བྱས་པའི་ཐམ་ཀ མུ་ཏིག་གིས་བརྒྱན་པའི་སྣམ་སྦྱར། སྐུ་ཆས་སོགས་དངོས་འབུལ་མང་པོ་ཕུལ། སླར་ཡང་ཧོར་གོང་མས་གསང་སྔགས་ཀྱི་དབང་ཐེངས་གཉིས་པ་ཞུས་པའི་དབང་ཡོན་དུ་བོད་ཆོལ་ཁ་གསུམ་གྱི་སྲིད་དབང་ཡོངས་རྫོགས་ཀྱང་ཕུལ་ནས་བོད་ཀྱི་ཆོས་སྲིད་གཉིས་ཀའི་ དབང་གི་བདག་པོར་གྱུར། དབང་ཐེངས་གསུམ་པ་ཞུས་པའི་དབང་ཡོན་དུ༑ ཆོས་དུང་དཀར་པོ་རྒྱང་གྲགས་དང་རྒྱ་ཡི་མི་ཡུར་ཆེན་མོ་སོགས་ཕུལ།”ཞེས་གསལ་བ་བཞིན་བོད་དུ་ས་སྐྱས་དབང་བསྒྱུར་བྱེད་པ་ནས་ཧོར་ལུགས་ཀྱི་སྲོལ་ལྟར་ཟླ་བ་དང་པོ་ནས་བཅུ་གཉིས་བར་བོད་ཟླ་ཞེས་མི་བརྗོད་པར་ཧོར་ཟླ་ཞེས་ལོ་ཟླ་བརྩི་དགོས་པའི་མིང་གྱུར། ཧོར་ཟླ་དང་པོའི་ཚེས་གཅིག་ལ་ལོ་འཁོར་བ་དང་བོད་ལ་དུས་ཚོད་དེ་དང་མཚུངས་པར། རྒྱ་རྩིས་ཀྱི་ཟླ་བ་གསུམ་པའི་ནང་བོད་ཀྱི་རྒྱལ་པོ་ལོ་གསར་ཞེས་པ་གཏོང་བའི་ལམ་སྲོལ་བྱུང༌ཡོད་འདུགབོད་ཀྱི་སྲོལ་རྒྱུན་ལོ་གསར་དར་ཚུལ་སྐོར་རགས་ཙམ་གླེང་བ།
ལོ་གསར་ནི། ལོ་སྔ་མ་རྫོགས་ནས་ལོ་རྗེས་མའི་འགོ་ཚུགས་པ་ལ་ཟེར་ཞིང༌། བོད་མིའི་ངག་རྒྱུན་དུ་ཡོད་པའི་བླ་མ་ལོ་གསར་དང༌། རྒྱལ་པོ་ལོ་གསར། བསྟན་སྲུང་ལོ་གསར། བེའུ་ལོ་གསར། ཡུལ་པ་ལོ་གསར། སོ་ནམ་ལོ་གསར་རམ་ཞིང་པ་ལོ་གསར་སོགས་ལོ་འགོ་འཛིན་ཚུལ་མི་འདྲ་བའི་དབང་གིས་ལོ་གསར་ཡང་ཡུལ་སྲོལ་མི་འདྲ་བ་སྣ་ཚོགས་བྱུང་ཡོད་ཀྱང༌། བོད་ཀྱི་ཁྱད་ཆོས་ཚང་བའི་ལོ་གསར་གཏོང་བའི་ཐོག་མའི་ལམ་སྲོལ་དེ་དུས་ག་ཚོད་ཙམ་ཞིག་ནས་དར་སྲོལ་བྱུང་བའི་སྐོར། ཁོ་བོས་རང་རེའི་ལོ་རྒྱུས་དེབ་ཐེར་ཁག་ཅིག་དང༌། སྐར་ནག་རྩིས་རིག་ཐོག་ཏུ་ཡོད་པའི་བྱུང་རབས་ཐོར་བུ། རྒྱ་ཆེའི་ཞིང་འབྲོག་ཁུལ་གྱི་རྒན་པོའི་ངག་རྒྱུན་དུ་གྲགས་པའི་སྒྲུང་གཏམ་བཅས་ལ་གཞི་བཅོལ་གྱིས་དེའི་སྐོར་རགས་པ་ཙམ་ཞིག་གླེང་བར་འདོད།
      དེ་ཡང༌། གངས་རི་ར་བས་སྐོར་བའི་བོད་གངས་ཅན། འཕགས་པ་སྤྱན་རས་གཟིགས་ཀྱི་གདུལ་ཞིང་གནས་མཆོག་འདིར། བདག་ཅག་སྟོན་པ་བཀའ་དྲིན་ཅན་འདས་ལོའི་སྔོན་གྱི་ལོ་བརྒྱའི་ནང་བོད་རྗེ་གཉའ་ཁྲི་བཙན་པོའི་རྒྱལ་བརྒྱུད་པུ་དེ་གུང་རྒྱལ་གྱི་སྐབས་སུ་སྤང་ཐང་ཞིང་དུ་རྨོས། ཆུ་ལག་ཡུ་བར་དྲངས། ས་བོན་ཕུར་འདེབས་བྱེད། སོལ་བ་ལ་བརྟེན་ནས་དངུལ་དང་ཟངས། ལྕགས་བཅས་བཞུ་བ་སོགས་ཞིང་ལས་དང་ལག་ཤེས་བཟོ་ལས་བཅས་ཡར་རྒྱས་ཆེན་པོ་བྱུང་བའི་ སྐབས་དེར། ལྷོ་ཁ་ཡར་ལུང་ཁུལ་དུ་ཐགས་མཁན་རྒན་མོ་བལ་མ་ཞིག་གིས་ཟླ་བའི་དཀར་ཆ་འཕེལ་འགྲིབ་ཀྱི་ཆ་ཤས་ལ་བལྟས་ནས་ལོ་ཟླ་ཞག་དུས་རགས་ཙམ་རྩིས་སྲོལ་ཡོད་པ་དེར་རྒན་མོ་བལ་ མོའི་ཟླ་རྩིས་ཞེས་གྲགས། ཟླ་རྩིས་དེ་ལ་བརྟེན་ནས་ཟླ་བ་དང་ལོ་འཁོར་ཡང་རྩིས་སྲོལ་བྱུང་འདུག་ཀྱང༌། འབྱུང་བའི་འགྱུར་འགྲོས་ལ་བརྟེན་ནས་དྲོད་རླུང་ལྡང་ཞིང་ཤིང་རྩི་ཁ་འབུས། ལོ་མ་རྒྱས་ཏེ་ལོ་ལྟར་ཁམ་བུའི་མེ་ཏོག་རབ་ཏུ་ཤར་སྐབས། ནམ་དུས་དཔྱིད་ཀ་འཆར་བའི་དགའ་སྤྲོ་བྱེད་པ་ལས་དམིགས་བསལ་ལོ་གསར་ཞེས་དགའ་སྟོན་ བྱེད་སྲོལ་མེད་པར་སྣང༌།
      གདོད་མའི་ལོ་གསར།
      དེ་ནས་རིམ་བཞིན་བོད་དུ་བོན་གྱི་བསྟན་པ་དར་ཚུལ་སྐོར། ད་ཡོད་ཀྱི་དཔྱད་གཞིའི་ཡིག་ཆ་གཞིར་བཟུང་ན་སྟོན་པ་འདས་ལོའི་སྔོན་གྱི་ལོ་བརྒྱ་ཕྲག་གཅིག་ཡས་མས་སུ་བོད་ཀྱི་རྒྱལ་པོ་ཐོག་མ་གཉའ་ཁྲི་བཙན་པོས་སྲིད་དབང་བཟུང་བའི་སྐབས་རྒྱུ་བོན་ཤེས་པ་ཅན་བཅུ་གཉིས་དར་ཞེས་གསལ་བ་དང༌། དེའི་ནང་ནས་སྐོར་ཤེས་རྩིས་མཁན་ནི་སྐར་རྩིས་ཤེས་མཁན་ཞིག་ཡིན་པ་བོན་གྱི་ལོ་རྒྱུས་ནང་འཁོད་ཡོད་ལ། རྒྱལ་པོ་པུ་དེ་གུང་རྒྱལ་གྱི་དུས་སུ་རྒྱུ་བོན་རིགས་བཞི་ཞེས་པ་བྱུང་བའི་ནང་ནས་དུར་གཤེན་མཚོན་ཆ་གསོས་གཤིས་གཉིས་ཀྱི་རྩིས་བྱེད་མཁན་ཞིག་ཡིན་པར་《རྒྱལ་རབས་གསལ་བའི་མེ་ལོང་》དང༌། 《ཆོས་འབྱུང་མཁས་པའི་དགའ་སྟོན་》སོགས་ལོ་རྒྱུས་དེབ་ཐེར་མང་པོའི་ནང་འཁོད་པ་ ལྟར། བོན་གྱི་བླ་ཆེན་དྲན་པའི་ཞལ་ནས་ཀྱང༌། “སྤྱིར་ན་སྟོང་གསུམ་འདུལ་བྱེད་སྟོན་པ་ཆེའི། །གང་འདུལ་མཛད་པའི་ཐབས་སུ་བཀའ་གསོལ་བཞི། །བོན་ཆོས་རྩིས་དཔྱད་བཞི་རུ་བསྟན་པ་ཡང༌། །དོན་དུ་ཐ་དད་མེད་དུ་ངོ་བོ་གཅིག །”ཅེས་གསུངས་ཡོད། དེས་ན། བོན་ཆོས་དང་མཐུན་པའི་ཀོང་རྩེ་འཕྲུལ་རྒྱལ་༡ཞེས་པའི་རྩིས་དང་ཤེས་རིག་མཁས་པ་ ཁག་གཅིག་གིས་ཙི་ནའི་ཡུལ་༢ནས་སྟོད་ཞང་ཞུང་བརྒྱུད་དེ་བོད་ལ་ཕེབས། བོད་དུ་ཁོང་རྣམ་པའི་ལུགས་དང་མཐུན་པའི་《འབྱུང་ལྡན་》༣རྩིས་ལུགས་སམ་ལོ་འཁོར་རྩིས་ཚུལ་དང་དགུན་ཉི་ལྡོག་མཚམས་ལོ་འཁོར་བའི་གདོད་མའི་ ལོ་གསར་གྱི་དགའ་སྟོན་གཏོང་བའི་ལམ་སྲོལ་ནི་བོན་གྱི་ལུགས་སྲོལ་ཡིན་པ་དཔེར་ན་ལུག་མགོ་ལ་གཡུང་དྲུང་དང་ཉི་ཟླའི་གཟུགས་དབྱིབས་བརྒྱན་ཏེ་ཕ་མེས་ཀྱི་དུར་ལྷ་ གསོལ་ནས། ཕ་མེས་ལ་དྲན་གསོ་ཞུ་བ། ས་དམར་དང་ས་དཀར་གྱི་གཡུང་དྲུང་འབྲི་བ། དེ་བཞིན་མི་ནོར་གཡང་ཆགས་པའི་ཆེད་དུ་མདའ་དར་དང་གྲོ་སོ་ཕྱེ་མར། ཆང་ཕུད། ལོ་ཕུད། གཡང་ཤ་ལག་པ་གཡས་པ། ཁ་ཟས་ཀྱི་སྡེར་ཁ་བཅས་ཡུལ་གྱི་ཐོན་ཁུངས་གང་ལེགས་བསྒྲིགས་ཏེ་ཚེ་རིང་ནད་མེད་བདེ་སྐྱིད་ལྡན་པའི་བསྔོ་སྨོན་བྱེད་པ་དང་སྦྲགས་བཀྲ་ཤིས་གཡང་ཆགས་ཀྱི་བྱེད་སྒོ་སྤེལ་བ། རིག་རྩལ་ཐོག་ཕོ་གཞོན་ཚོས་གཡག་ལུག་བསད་པའི་རྭ་ཅོའི་ཐོ་བཙུགས་པར་ཐག་རིང་ཙམ་ནས་ལག་རྡོ་འཕངས་ཏེ་སློག་ཐུབ་ན་དཔའ་སྐད་བསྒྲགས་པ། དེ་བཞིན་མདའ་གྲི་མདུང་དང་འུར་རྡོ་འགྲན་པ། མོ་གཞོན་ཚོས་གཞས་སྒོར་གྱིས་གླུ་ལེན་པ་དང༌། ཞབས་བྲོ་རྒྱག་པ། ཐེ་ཕེག་རྩེ་བ། གབ་རེས་འུ་རེས་རྩེ་བ་སོགས་བྱེད་པ། དེ་བཞིན་བོན་ཆོས་དང་མཐུན་པའི་རི་རྩེར་རང་བསྟེན་སྲུང་མའི་གཟུངས་དང་སྲོག་ཡིག་གི་གཤམ་དུ་ལས་བཅོལ་འཕྲིན་བསྐུལ་བཀོད་པའི་དར་ཚོན་སྣ་ལྔའམ་རང་མཐུན་ཚོན་མདོག་གི་དར་ལྕོག་འཛུགས་པ། རླུང་རྟ་ཡིན་ན། ཚོན་སྣ་ལྔའམ་སོ་སོའི་འབྱུང་ཁམས་མདོག་མཐུན་གྱི་དར་རས་ལ་རླུང་རྟའི་གཟུངས་སྔགས་དཔར་ཏེ་དར་ལྡིང་གཏོང་བ། ལྷ་གསོལ་བྱེད་པ་སོགས་ཀྱི་བྱེད་སྒོ་དུས་དེ་ཙམ་ནས་དར་སྲོལ་བྱུང་ཡོད་སྐད།
      བེའུ་ལོ་གསར།
      ཕྱི་ལོ་༥༥༨ གནམ་རི་སྲོང་བཙན་གྱི་དུས་ཙི་ན་ལ་བོད་མི་རིག་པ་རྣོ་བའི་སློབ་བུ་བཞི་སྐར་རྩིས་སློབ་གཉེར་བྱེད་དུ་བརྫངས་ཤིང༌། ཁོང་ཚོ་ཕྱིར་ལོག་ནས་ཤེས་བྱ་རྣམས་ངག་རྒྱུན་ནས་བརྒྱུད་སྤེལ་བྱས་ཏེ་བོད་ཀྱི་གནམ་རིག་སྐར་རྩིས་རིག་པའི་འཕེལ་རྒྱས་ལ་ལག་རྗེས་གང་ལེགས་བཞག་ཡོད། ཕྱི་ལོ་༦༣༠ 《དཔའ་བོ་གཙུག་ལག་ཕྲེང་བ་》ལྟར་ན་བོད་རྗེ་གནམ་རི་སྲོང་བཙན་གྱི་སྲས་སྲོང་བཙན་སྒམ་པོར་རྒྱལ་སྲིད་ཕུལ་བ་དང༌། ཕྱི་ལོ་༦༣༢ བལ་བཟའ་ཁྲི་བཙུན་བོད་དུ་བསུས་ཏེ་ཇོ་བོ་མི་བསྐྱོད་རྡོ་རྗེ་བོད་དུ་ཕེབས། ཕྱི་ལོ་༦༣༨ ལོར་”བོད་རྗེ་སྲོང་བཙན་སྒམ་པོས་༤རྒྱ་(རྒྱ་ནག་གི་རྒྱལ་པོ་ཐང་ཐེ་ཙུང་གི་བུ་མོ )རྗེའི་སྲས་མོ་སློང་བའི་ཕོ་ཉ་མངགས་ཀྱང་རྒྱ་རྗེས་བུ་མོ་མ་བྱིན་པར་བོད་དུ་ ལོག་ཏེ་བཙན་པོར་རྫུན་ཞུས། རྒྱ་རྗེ་དེ་ངེད་རང་ལ་ཤིན་ཏུ་དགའ་ནས་བུ་མོ་སྟེར་བ་ཡོད་པ་ལ་ཧོར་སེར་ཐུ་ལུ་ ཧུན་གྱིས་རྒྱ་རྗེ་ལ་ཕྲ་མ་ཞུས་པས་ལན་ཚུལ་ཞུས་པས། བོད་ཀྱི་བཙན་པོ་ཁྲོས་ནས་བོད་དམག་ཁྲི་ཉི་ཤུ་ཁྲིད་དེ་ཟུང་ཅིའུ་(སུང་ཀྲའུ་)དེ་ ནས་བློན་པོ་གཡའ་ཐུང་བྱ་བ་ལ་དམག་སྐུལ་ནས། ཐུ་ལུ་ཧུན་གྱི་ཡུལ་འཇོམས་སུ་བཏང་བ་ལ་ཐུ་ལུ་ཧུན་རྣམས་གཙང་ཁའི་མཚོ་སྔོན་ཕན་ལ་བྲོས། ཤུལ་དུ་ལུས་པའི་ནོར་ཐམས་ཅད་བོད་ཀྱིས་འཁྱེར། དེ་ནས་བོད་ཀྱི་བཙན་པོས་བློན་པོ་སེ་ལེ་སྟོང་བཙན་ལ་རིན་པོ་ཆེ་སྣ་ཚོགས་བསྐུར་ནས། རྒྱ་རྗེའི་སྲས་མོ་སློང་དུ་བཏང་བར་འུན་ཤིང་ཀོང་ཇོར་རྟེན་སྐལ་ཇོ་བོ་ཤཱཀྱ་མུ་ནེ་སོགས་རྫོངས་གྱ་ནོམ་པ་དང་བཅས་བོད་དུ་སླེབས། ”རྒྱ་བཟའ་ཀོང་ཇོ་རྩིས་ལ་མཁས་པས་རྒྱའི་འབྱུང་རྩིས་སྤོར་ཐང་བརྒྱད་ཅུའི་སྐོར་མགོ་དང་རྒྱ་བཟའི་གུང་ཁུག་སོགས་བོད་ལ་མཚམས་སྦྱོར་གནང་བ་ནས་བཟུང༌། བོད་དུ་《སྔ་འགྱུར་རྒྱུ་རྩིས་རྙིང་པའི་ལུགས་》དང༌། ཡང་འགྱུར་《གསལ་སྒྲོན་སྦས་བཅད་མེད་པའི་ལུགས་》ཞེས་པའི་ནང༌། “དཔྱིད་ར་སྟག་ལ་ལོ་འགོ་འཛིན་ཞེས་པ་བོད་ཟླ་བཅུ་གཅིག་པའི་ཚེས་གཅིག་ལ་ལོ་གསར་གཏོང་སྲོལ་ཆགས་པ་”ཞེས་པའི་བརྗོད་ཚུལ་བྱུང་འདུག དེ་དང་གོང་དུ་དགུན་ཉི་ལྡོག་མཚམས་དང་མཐུན་ཤས་ཆེ་བ་ཡོད་པ་དེ་ནི། སྐབས་དེར་བོད་ས་མང་ཆེ་བ་འབྲོག་ལས་དང་ཡུལ་པ་ཕལ་ཆེ་བ་ས་མ་འབྲོག་ཡིན་པས་ནམ་དུས་གྲང་ངར་ཆེ་བས་འབྲོག་པ་རྣམས་ས་དྲོ་སར་རུ་བ་(རྩ་ར་)སྤོས་ཏེ། སྤུ་ནག་འབྲི་དང་གཡང་དཀར་ལུག་གི་ལུག་ཕྲུག་དང་བེའུ་བཙའ་རྒྱུའི་གྲབས་གཤོམ་བྱེད་པའི་དུས་ཚོད་ཡིན་ཞིང༌། དེའི་རིང་འབྲོག་པ་ཁག་ཅིག་ཡུལ་པའི་ནང་ཞིང་འབྲོག་ཐོན་རྫས་(ཤ་མར་ཚྭ་བུལ་ཏོག་ དང་འབྲུ་རིགས་བརྗེ་བ)བརྗེ་ལེན་གྱི་ཚུར་འཁོར་ཏེ་ནང་མི་ཚང་འཛོམས་ཀྱིས་བེའུ་ལོ་གསར་ཞེས་ལོ་གསར་གཏོང་སྲོལ་ཡོད་པ་དུས་ད་ལྟ་ཡང་བོད་ཀྱི་འབྲོག་ཁུལ་མང་ཆེ་བའི་ནང་འགྱུར་མེད་གནས་ཡོད་པ་གོ་ཐོས་འདུག
