རྒྱུན་ལྡན་དུ་མིས་བཟོད་བསྲན་བྱེད་ཐུབ་པའི་ཆེས་མཐོ་བའི་དྲོད་ཚད་ནི་ཏུའུ་གྲངས་ག་ཚོད་ཡིན་ནམ།
ཉེ་ལམ། རྒྱ་གར་གྱི་ས་ཁུལ་རེ་འགར་ཚད་པ་ཆེན་པོ་བྱུང་ནས་མི་འབོར ༡༡༠༠ ལྷག་རྐྱེན་ལམ་དུ་ཤོར་བའི་གནས་ཚུལ་ཐོས་རྒྱུ་འདུག དོན་དངོས་སུ་ཚ་བས་མིའི་ཚེ་སྲོག་ལ་ཉེན་ཁ་བཟོ་སྲིད་མིན་དང་། མིའི་བཟོད་བསྲན་བྱེད་ཐུབ་པའི་ཚ་བའི་དྲོད་ཚད་དེ་ཇི་འདྲ་ཡིན་མིན་ངཅས་གྱི་ཐད་ལ། དུས་རབས ༡༨ པའི་ནང་། དབྱིན་ཇིའི་ཚོད་ལྟའི་སྐྱེ་ཁམས་རིག་པའི་ཚན་རིག་པ་པོ་ལེ་ཏོན་བྱ་བས་རང་སྟེང་དུ་ཚོད་ལྟ་བྱས་པ་རེད། ཚད་ལྟའི་འབྲས་བུ་ལས་ཤེས་གསལ་ལྟར་ན། གལ་ཏེ་ཁོར་ཡུག་གི་མཁའ་དབུགས་སྐམ་ཤས་ཆེ་དུས། མིས་དྲོད་ཚད་ཏུའུ་གྲངས ༡༢༠℃ ཡིན་པའི་ཁོར་ཡུག་ཁྲོད་ནས་སྐར་མ ༡༥ ལ་རྒྱུན་ལྡན་ངང་སྡོད་ཐུབ་པ་དང་། གནས་ཚུལ་མི་ལེགས་པ་ཅི་ཡང་མེད་པར་ལུས་དྲོད་ཀྱང་སྔར་བཞིན་བརྟན་པོར་རྒྱུན་འཁྱོངས་བྱེད་ཐུབ་པ་རེད། རྒྱུན་ལྡན་གནས་ཚུལ་འོག་ཏུ་དྲོད་ཚད་མཐོན་པོ་འདིའི་འོག་ནས་སྐར་མ ༡༣ ཙམ་གྱི་ནང་སྐམས་ཤའང་བཙོ་ཐུབ་པར་བཤད། ཡིན་ཡང་བརླན་བཤེར་ཆེ་བའི་ཁོར་ཡུག་གི་གནས་ཚུལ་འོག་ནས། ཁོར་ཡུག་གི་དྲོད་ཚད་ཏུའུ ༤༨℃ ནས་ཏུའུ༥༠℃ཡི་འོག མིས་དུས་ཡུན་ཐུང་ངུ་ཙམ་ཞིག་ལས་བཟོད་བསྲན་བྱེད་མི་ཐུབ་པ་རེད། འདིའི་རྒྱ་མཚན་ནི་མིའི་རྔུལ་ཆུ་དུས་ལྟར་རླངས་འགྱུར་བྱེད་མ་ཐུབ་པའི་རྐྱེན་རེད།
          ཚན་རིག་པས་ད་དུང་མི་ལུས་ཀྱིས་སྐམ་ཤས་ཆེ་བའི་ཁོར་ཡུག་འོག་བཟོད་བསྲན་བྱེད་ཐུབ་པའི་དྲོད་ཚད་མཐོ་ཤོས་ལ་དངོས་བཤེར་བྱས་པ་རེད། དེ་ལས་ལུས་གཅེར་བུ་ཡིན་པའི་གནས་ཚུལ་འོག་ཏུ། དྲོད་ཚད་མགྱོགས་མྱུར་ངང་འཕར་དུས་མིས་བཟོད་བསྲན་བྱེད་ཐུབ་པའི་མཐར་སོན་པའི་དྲོད་ཚད་ནི་ཏུའུ་གྲངས ༢༡༠ ℃ ཡིན་པ་དང་། ལུས་སུ་ལྭ་བ་མཐུག་པོ་གྱོན་དུས། དྲོད་ཚད་གློ་བུར་དུ་འཕར་བར་མིས་བཟོད་བསྲན་བྱེད་ཐུབ་པའི་མཐར་སོན་པའི་དྲོད་ཚད་ནི་ཏུའུ་གྲངས ༢༧༠ ℃ ལ་སླེབས་ཐུབ་པ་རེད། ཡིན་ཡང་། མིའི་ལུས་པོའི་དྲོད་ཚད་ཡར་འཕར་དུས་མིས་བཟོད་བསྲན་གང་ཡང་བྱེད་ཐུབ་ཀྱིན་མི་འདུག ལུས་པོའི་དྲོད་ཚད་ཏུའུ ༤༢ ℃ ལས་བརྒལ་དུས། དབང་རྩའི་ལྟེ་བའི་མ་ལག་གི་བྱེད་ནུས་ལ་གནོད་སྐྱོན་ཐེབས་ནས། ལུས་ཕུང་ནང་གི་སྤྲི་དཀར་རྒྱུའི་ངོ་བོ་འགྱུར་བ་དང་དཀག་པར་འགྱུར་བ་བཅས་ཀྱི་རྐྱེན་གྱིས་མིའི་ཚེ་སྲོག་ལའང་ཉེན་ཁ་འབྱུང་སྲིད་པ་རེད། དེ་བས་ལུས་དྲོད་འཇལ་ཆས་ངོས་གྱི་ཆེས་མཐོ་བའི་དྲོད་ཚད་ནི་ཏུའུ་གྲངས ༤༢ ℃ ཡིན་པ་རེད། མིའི་ལུས་ཕུང་ལ་མཚོན་ན། ནང་རོལ་གྱི་དྲོད་ཚད་ཏུའུ ༤༠ ℃ ནི་ཉེན་ཁའི་བརྡ་རྟགས་ཡིན་པ་དང་། དྲོད་ཚད་ཏུའུ ༤༢ ℃ ལ་སླེབས་དུས། དབང་བོའི་མ་ལག་ལ་ཕྱིར་ལྡོག་མི་སྲིད་པའི་གནོད་སྐྱོན་ཚབས་ཆེན་ཐེབས་སྲིད་ལ། ཐ་ན་ཚེ་སྲོག་གི་ཉེན་ཁའང་འབྱུང་སྲིད་པ་རེད།
          ཨ་རིའི་ཚན་རིག་པ་ཁེ་ལོ་ཏི་ཕེན་ཐ་རྡི་ཞིས་གསལ་བཤད་བྱས་པ་ལྟར་ན། ཚ་བས་མིའི་ཚེ་སྲོག་ལ་གནོད་སྐྱོན་ཐེབས་པའི་སྣང་ཚུལ་ལ་གཤམ་གསལ་རིགས་འགའ་ཡོད་དེ།
          དང་པོ། རྒྱུན་ལྡན་གྱི་གནས་ཚུལ་འོག་ཏུ། མིའི་ལུས་པོས་ཚ་བ་བརྒྱུད་འདྲེན་བྱས་པར་བརྟེན་ནས་ཕུང་བོའི་ནང་གི་ཚ་བ་དེ་སྐྱི་ལྤགས་ཀྱི་ཕྱི་རུ་འཕུད་སྲིད་པ་དང་། དེ་ནས་ཕན་ཚུན་བལྟོས་རྒྱུག་གིས་ཚ་དྲོད་དེ་མཁའ་དབུགས་སུ་ཕུད་ནས་ལུས་དྲོད་དམའ་རུ་གཏོང་སྲིད། ཡིན་ཡང་དྲོད་ཚད་ཇི་ལྟར་མཐོ་ན་ཚ་དྲོད་ཕྱིར་འཕུད་པ་དེ་བས་ཁག་པོ་འདུག ཕུང་བོའི་དྲོད་ཚད་ཏུའུ ༣༨ ℃ ལ་ཉེ་བ་དང་བསྟུན་ནས། མ་ལག་ཡོད་ཚད་ལྡོག་ཕྱོགས་སུ་འཁོར་སྲིད་དེ། ཁོར་ཡུག་གི་ཚ་དྲོད་དེ་མིའི་ལུས་ཕུང་ནང་ལ་བརྒྱུད་འདྲེན་བྱེད་ཐུབ་པ་དང་། མིའི་ཕུང་བོའི་དྲོད་ཚད་དེ་ཏུའུ ༤༢ ℃ ལས་བརྒལ་དུས། ཚེ་སྲོག་ལའང་ཉེན་ཁ་ཡོད་པ་རེད།
          གཉིས་པ། ལུས་སྟེང་གི་རྔུལ་ཆུ་ནི་མིའི་ལུས་ཕུང་གྲང་མོར་བཟོ་བའི་ནང་རྐྱེན་གཉིས་པ་ཡིན་ཏེ། གཤེར་གཟུགས་ཀྱི་རྔུལ་ཆུ་དེར་དྲོད་བཏང་བ་བརྒྱུད་ནས་ཆ་རྡུལ་གྱི་འགུལ་སྐྱོད་ཤུགས་དྲག་ཏུ་གྱུར་ཏེ་ཆུའི་རླངས་པར་འགྱུར་སྲིད་པ་དང་། ཆུའི་རླངས་པས་ལུས་ཕུང་ནང་གི་དྲོད་ཚད་དེ་ཕྱིར་ཁྱེར་ཏེ་་དྲོད་ཚད་ཇེ་དམའ་རུ་གཏོང་ནུས་ལ། རྩ་བའི་དྲོད་ཚད་དེའང་དམའ་རུ་གཏོང་ནུས་པ་རེད། ཡིན་ཡང་ཅུང་མཐོ་བའི་གཤེར་ཚད་ཀྱིས་མ་ལག་འདིའི་འཁོར་སྐྱོད་ལ་ཤུགས་རྐྱེན་བཟོ་སྲིད་པ་དང་། དེ་ནི་མཁའ་དབུགས་ཁྲོད་ཀྱི་གཤེར་ཚད་ཚད་ལོངས་ཡོད་དུས། རྔུལ་ཆུ་ཕྱིར་རླངས་པར་འགྱུར་མ་ཐུབ་པས་མིའི་ལུས་དྲོད་ཀྱང་དེར་བསྟུན་ནས་ཡར་འཕར་སྲིད་པའི་རྐྱེན་རེད།
          གསུམ་པ། གལ་ཏེ་མིའི་འཐུང་བའི་སྐོམ་ཆུའི་ཚད་མ་འདང་ཚེ་ལུས་ཆུ་ཟད་སྲིད་པ་དང་། སྐྱི་ལྤགས་ཀྱི་ཁྲག་རྒྱུན་འཁོར་སྐྱོད་ཀྱང་ཉུང་དུ་འགྲོ་ལ་རྔུལ་ཆུ་ཕྱིར་འཕུད་པའི་ནུས་པའང་ཉམས་འགྲོ་བས། ལུས་དྲོད་དེའང་རང་བཞིན་གྱིས་ཡར་འཕར་སྲིད་པ་རེད།
          མཇུག་མཐར། ལུས་ཕུང་ནང་གི་ཆུ་རླངས་དེ་དུས་ལྟར་ཕྱིར་འཕུད་མ་ཐུབ་པར་ཕུང་བོའི་ནང་ལུས་དུས། མིར་ཚ་གདུག་ཕོག་ནས་ཚེ་སྲོག་གི་ཉེན་ཁའང་འབྱུང་སྲིད་པ་རེད། ཁྱད་པར་དུ་སྔར་ལུས་ཕུང་དུ་སྙིང་ནད་དང་ཁྲག་སྦུག་གི་ནད་སོགས་ནད་རིགས་ག་གེ་མོ་འདུས་པའི་མི་ལ་མཚོན་ན། ཚ་གདུག་ཕོག་རྗེས་ནད་བསླང་ནས་རྐྱེན་ལམ་དུ་ཤོར་བའི་ཉེན་ཁ་ཧ་ཅང་ཆེའོ།།
     ཁ་བྱང་ཕྱི་མོ། རྒྱུན་ལྡན་དུ་མིས་བཟོད་བསྲན་བྱེད་ཐུབ་པའི་ཆེས་མཐོ་བའི་དྲོད་ཚད་ནི་ཏུའུ་གྲངས་ག་ཚོད་ཡིན་ནམ།
