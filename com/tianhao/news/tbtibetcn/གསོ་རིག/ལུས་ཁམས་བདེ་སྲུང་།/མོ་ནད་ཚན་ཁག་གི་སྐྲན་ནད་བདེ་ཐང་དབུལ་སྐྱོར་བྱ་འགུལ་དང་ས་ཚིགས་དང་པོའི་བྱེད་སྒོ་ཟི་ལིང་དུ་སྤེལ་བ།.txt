མོ་ནད་ཚན་ཁག་གི་སྐྲན་ནད་བདེ་ཐང་དབུལ་སྐྱོར་བྱ་འགུལ་དང་ས་ཚིགས་དང་པོའི་བྱེད་སྒོ་ཟི་ལིང་དུ་སྤེལ་བ།
ཟླ10པའི་ཚེས18ཉིན། ཀྲུང་གོ་འབྲས་སྐྲན་ཐེབས་རྩ་ཚོགས་པས་གཙོ་གཉེར་བྱས་ཤིང་། ཞིང་ཆེན་མི་དམངས་སྨན་ཁང་དང་ཞིང་ཆེན་མི་དམངས་སྨན་ཁང་ལྔ་པ། མཚོ་སྔོན་རྒྱ་གྲམ་དམར་པོ་སྨན་ཁང་། མཚོ་སྔོན་སློབ་ཆེན་གྱི་ཁོང་གཏོགས་སྨན་ཁང་བཅས་ཀྱིས་མཉམ་འབྲེལ་གྱིས་རོགས་སྒྲུབ་གནང་བའི་“བརྩེ་བའི་ཆེད་དུ་ནུབ་ཕྱོགས་སུ་བསྐྱོད་པ་དང་། ནུས་ཤུགས་ཡོད་རྒུས་རོགས་རམ་བྱེད་པ”ཞེས་པའི་མོ་ནད་ཚན་ཁག་གི་སྐྲན་ནད་བདེ་ཐང་དབུལ་སྐྱོར་བྱ་འགུལ་དང་ས་ཚིགས་དང་པོའི་བྱེད་སྒོ་ཟི་ལིང་དུ་སྤེལ་བ་རེད། ཞིང་ཆེན་ཕྱི་ནང་གི་སྐྲན་ནད་སྨན་ཁང་མང་པོ་ཞིག་གི་མོ་ནད་ཚན་ཁག་གི་སྐྲན་ནད་ཆེད་མཁས་པ100ལྷག་ཙམ་ཚོགས་འདུར་ཞུགས་ཡོད་པ་རེད།
          རྒྱུས་ལོན་བྱུང་བ་ལྟར་ན། “བརྩེ་བའི་ཆེད་དུ་ནུབ་ཕྱོགས་སུ་བསྐྱོད་པ་དང་། ནུས་ཤུགས་ཡོད་རྒུས་རོགས་སྐྱོར་བྱེད་པ་”ཞེས་པའི་མོ་ནད་ཚན་ཁག་གི་སྐྲན་ནད་བདེ་ཐང་དབུལ་སྐྱོར་བྱ་འགུལ་སྤེལ་བའི་དམིགས་ཡུལ་ནི་རྒྱལ་ཡོངས་ཀྱི་ཆུ་ཚད་མཐོ་བའི་སྨན་ཁང་མང་པོའི་མོ་ནད་ཚན་ཁག་གི་སྐྲན་ནད་ཆེད་མཁས་པ་དང་། ནུབ་རྒྱུད་ས་ཁུལ་གྱི་སྨན་ཁང་གཙོ་བོ་ཁག་ཏུ་ཕུགས་ཕན་གནད་སྨིན་རོགས་སྐྱོར་ལམ་སྲོལ་བཙུགས་ནས་རྒྱང་ཁྲིད་སྨན་བཅོས་དང་རྒྱུན་མཐུད་སློབ་ཁྲིད། ནད་ཐོག་ཞིབ་འཇུག་སོགས་ཀྱི་ལས་སྒོའི་མཛུབ་ཁྲིད་ལས་དོན་ཧུར་ཐག་གིས་སྤེལ་ཆེད་ཡིན་པས། ནུབ་རྒྱུད་ས་ཁུལ་གྱི་ནད་པ་རྣམས་ཀྱིས་རང་གི་སྒོ་འགྲམ་དུ་རྒྱལ་ནང་གི་ཆེད་མཁས་པ་རྩེ་གྲའི་ནད་བརྟག་སྨན་བཅོས་ཀྱི་ཞབས་ཞུ་ལོངས་སུ་སྤྱོད་ཐུབ་པ། དུས་མཚུངས་སུ་ནུབ་རྒྱུད་ས་ཁུལ་གྱི་སྨན་དོན་མི་སྣས་ཕུགས་ཕན་གནད་འཁེལ་རོགས་སྐྱོར་ལམ་སྲོལ་བརྒྱུད་དེ། རྒྱལ་ནང་གི་གསོ་རིག་ཆེད་མཁས་པ་ཚོགས་པའི་རྗེས་འབྲངས་ནས་ལས་སྒོའི་ཆུ་ཚད་མགྱོགས་མྱུར་གྱིས་མཐོ་རུ་བཏང་གཏོང་བ་དང་། ནུབ་རྒྱུད་ས་ཁུལ་གྱི་སྨན་ཁང་དང་རྒྱལ་ནང་གི་སྨན་ཁང་རྩེ་གྲ་དབར་གྱི་བར་ཐག་ཇེ་ཐུང་དུ་བཏང་ནས། ནུབ་རྒྱུད་ཀྱི་བདེ་ཐང་དབུལ་སྐྱོར་ལས་དོན་ཧུར་ཐག་གིས་སྤེལ་བར་རོགས་རམ་བྱེད་རྒྱུ་ཡིན་པ་རེད། །
