རུས་པ་ཆག་པ་མྱུར་སྐྱོབ་བྱ་ཐབས།
          རུས་པ་ཆག་ཚེ་མྱུར་སྐྱོབ་ཇི་ལྟར་བྱ་དགོས་སམ་ཞེ་ན། འདིར་བརྗོད་པ་ནི་གཙོ་བོ་ཡན་ལག་བཞིའི་རུས་པ་ཆག་པ་མྱུར་སྐྱོབ་བྱ་ཐབས་ཡིན་ཞིང་། དེ་ཡང་བརྡབས་ཆག་བྱུང་བ་དང་ཅ་ལག་བར་ལ་བཙིར་བ་སོགས་ཀྱིས་རྐྱེན་པས་ཡན་ལག་བཞིའམ་ཡང་ན་ལུས་པོའི་རུས་པ་གང་རུང་ཞིག་ཆག་སྐྱོན་བྱུང་བའི་ནད་པར་རྨ་བཟོས་ཡོད་ན་སྐབས་འཕྲལ་ཁྲག་བཀག་དགོས་པ་དང་། ན་ཟུག་ཆེ་ན་ན་ཟུག་གཅོག་པའི་སྨན་ཁྱུང་ལྔ་རིལ་བུ་5ནས་7བར་ལུས་སྟོབས་ལ་གཞིགས་ཏེ་སྟེར་དགོས་པ(སྨན་འདི་བཟི་ཆེ་བས་མང་དྲགས་རྒྱུ་མེད)དང་། ཡང་ན་ཕྱི་ལུགས་སྨན་ཆུས་ཐུང་ཕེན(去痛片)རིལ་བུ་1གམ་ཧྥན་པི་ཏེ(分必得)རིལ་བུ་1གང་རུང་ཞིག་སྟེར་ནས་རྨས་སྐྱོན་བྱུང་སར་རྨ་རས་དཀྲིས་དགོས་པ་དང་ཆག་ཆིངས་བྱ་དགོས་པ་ལས་གང་བྱུང་འགུལ་སྐྱོད་བྱ་རྒྱུ་མེད།
          ཆག་ཆིངས་ཇི་ལྟར་བྱ་དགོས་སམ་ཞེ་ན།
          དེ་ཡང་རྨ་ཡོད་པའི་རུས་ཆག་ནད་པར་སྐབས་འཕྲལ་རྨ་ངོས་སུ་སང་རས་གཙང་མ་ཞིག་གིས་རྨ་ཁེབས་བརྒྱབ་རྗེས་སྲིང་བལ་ལམ་ཡང་ན་རས་ཆ་གང་རུང་ཞིག་རྨ་ཁའི་ཐོག་དཀྲིས་ཏེ། རུས་པ་ཆག་པའི་ཡན་ལག་དེ་ཉིད་ཅུང་ཟད་འཐེན་པ་དང་འབྲེལ་པང་ལེབ་སྲབ་མོ་ཞིག་གིས་རུས་པ་ཆག་སའི་ཕྱོགས་བཞི་ནས་ཆག་ཆིངས་བྱེད་དགོས། རུས་པ་ཆག་ས་དེ་མི་འགྱུར་བའི་ཆེད་དུ་གོང་འོག་གི་ཚིགས་གཞི་བརྒལ་ཏེ་ཆག་ཆིངས་བྱ་དགོས། དེ་ཡང་ཐོག་མར་རུས་པ་ཆག་སའི་གོང་འོག་བར་གསུམ་བཅིངས་རས་ཀྱི་དམ་ལྷོད་སྙོམས་པོ་བཅིངས་དགོས་པ་མ་ཟད། ལྷག་པར་དུ་རུས་ཆག་བྱུང་སའི་བཅིངས་རས་དམ་ལྷོད་སྙོམས་པོ་ཡོང་བ་བྱ་དགོས། རྨ་མེད་པའི་རུས་ཆག་ནད་པར་རུས་ཆག་བྱུང་བའི་ཡན་ལག་དེ་ཅུང་ཟད་འཐེན་པ་དང་འབྲེལ་ཡན་ལག་དེ་ཉིད་རས་དང་སྲིང་བལ་སོགས་གང་རུང་གིས་དཀྲིས་ནས་ཆག་ཆིངས་བྱ་དགོས། དེ་ལྟར་བྱས་རྗེས་བརྡབས་འཁྲུག་མི་ཐེབས་པ་བྱས་ནས་སྨན་ཁང་དུ་བསྐྱལ་ཏེ་བརྟག་དཔྱད་དང་སྨན་བཅོས་ཡག་པོ་བྱ་དགོས།
          དོ་སྣང་བྱ་དགོས་པ་གཤམ་གསལ།
          ཐོད་པ་དང་། མཇིང་ཚིགས། རྐེད་ཚིགས། སྒལ་ཚིགས། མཚང་རུས་བཅས་རུས་ཆག་བྱུང་ཚེ་ཐད་ཀར་སྲོག་ལ་ཉེན་ཁ་ཡོད་པས། ངེས་པར་དུ་ནད་པ་བརྡབས་འཁྲུག་མ་ཐེབས་པའི་ཐོག སྒོ་དང་པང་ལེབ་སོགས་ཀྱི་སྟེང་བསྙལ་ཏེ་འཁྲུག་མ་ཐེབས་པ་བྱས་ནས། གང་མགྱོགས་ཉེ་འཁོར་སྨན་ཁང་དུ་བསྐྱལ་ཏེ་བརྟག་དཔྱད་བྱ་དགོས།
