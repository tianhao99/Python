བོད་ལུགས་ཀྱི་སྨན་སྒྲོ་དང་སྨན་བཅོས་ཀྱི་སྤྱོད་ཆས།
༡ བོད་ལུགས་སྨན་སྒྲོ།
           བོད་སྔ་རབས་སུ་གསོ་རིག་གཤག་བཅོས་ཀྱི་ལག་རྩལ་དར་ཞིང་། དེ་མཚུངས་ཀྱི་གཤག་བཅོས་ལག་ཆས་ཡོད་ཀྱང་། དེང་དུས་མེ་བུམ་དང་། མེ་བཙའ་འཇོག་བྱེད། གཏར་གྲི་ཙམ་ལས་བེད་སྤྱོད་མཁན་ཉུང་མོད། བོད་ལུགས་གསོ་རིག་གི་སྨན་འཇོག་སྣོད་སྨན་སྒྲོ་ནི་རྒྱུ་དང་བཟོ་ཐབས་ཁྱད་པར་ཅན་ལ་སྨན་རྫས་ཡུན་དུ་ལྡུག་ཀྱང་དྲི་མ་མི་འཆོར། ནུས་པ་མི་ཉམས་པར་མ་ཟད་ཕྱིན་དྲུག་དང་། སྨན་གྱི་རྒྱུ་བཞི་ཆ་ལག་དང་བཅས་པ་མཚོན་པས་བོད་ཀྱི་རིག་གནས་ལག་ཤེས་དང་བཅས་པའི་ཤེས་བྱ་དུ་མ་འདུས་པའི་བཟོའི་དངོས་པོ་ཞིག་ཡིན་ཕྱིར། སྨན་ཆེན་བསྟན་འཛིན་ཕུན་ཚོགས་ཀྱིས་མཛད་པའི་སྨན་སྒྲོའི་བཟོ་ཐབས་ཙམ་བཀོད་པ་ནི། རྒྱུད་དང་མན་ངག་གསང་ཚིག་ལས། །སྨན་སྒྲོའི་རྒྱུ་ཡི་དབྱེ་བ་ནི། ། སྟག་གཟིག་དོམ་དྲེད་གུང་སོགས་རབ། ། གྲུམ་དང་རྒྱ་གོར་ར་རྒོད་འབྲིང་། ། ཏིང་ལོ་འཕྱི་བ་ཐ་མར་བཤད། ། དེ་དག་རྒྱུ་བཅའ་མ་གྲུབ་ཚེ། །སྨན་གྱི་ཕྱི་སྣོད་མི་བཏུབ་པས། ། ཆུ་དར་སྣལ་མ་དགུ་རིམ་དུ། །བྱས་ཀྱང་སྨན་ཡལ་བཅུད་མི་ཐུབ། ། དེ་བས་གཅན་གཟན་གྲྭ་སྤུ་གཅེས། །ཕྱི་གྲྭ་མ་ཉམས་སྨན་མི་བརྫི། །སྦུབས་གྲྭ་མ་ཉམས་སྨན་མི་རྒོག །ཀླད་སྲང་མ་ཉམས་དྲི་མི་འཆོར། ། དེ་ལྟར་རྒྱུ་བཅའ་མཉེ་ལེགས་པར། །རང་རང་ཤེས་རབ་རིགས་བརྒྱུད་དང་། །དཔག་སྟེ་བཟོ་ལུགས་འདི་ལྟར་བྱ། ། སྨན་སྒྲོ་ཁྲུ་གང་བྱས་པ་དེ། །རིགས་བརྒྱུད་སྨན་པ་ཡིན་པའི་རྟགས། །ཁ་ཧེལ་ཁྲུ་གང་བྱས་པ་དེ། །ཡོན་མཆོད་རྟེན་འབྲེལ་འགྲིག་པའི་རྟགས། །མགོ་ལྤགས་ཞབས་རྒྱ་བྱས་པ་དེ། །ཀུན་གྱི་སྤྱི་བོར་ཁུར་བའི་བརྡ། །འདི་ལ་ལུང་བཞི་བཏགས་པ་དེ། །རྒྱུད་བཞི་ཁོང་དུ་ཆུད་པའི་བརྡ། ། ཕྲན་ཚར་བཅོ་བརྒྱད་བཏགས་པ་དེ། །ཆ་ལག་བཅོ་བརྒྱད་ཤེས་པའི་བརྡ། །གྱེན་ལ་སྦུ་གུ་འཐེན་པ་འདི། །འབྲེལ་ཚད་མཐོ་རིས་འདྲེན་པའི་རྟགས།།སག་རི་སྔོ་དམར་སྐེད་དཀྲིས་འདི། །པུ་ཏི་སྐེ་རགས་བཅིངས་འདྲའི་བརྡ། ། དར་ཚོན་སྣ་ལྔ་བཏགས་པ་དེ། །ཉོན་མོངས་དུག་ལྔ་འཇོམས་པའི་རྟགས། །ཕུར་མ་སྣ་ཚོགས་བཀྲ་བ་དེ། །ཀུན་གྱིས་འགྲན་ཟླ་བྲལ་བའི་བརྡ། །དྲིལ་བུ་གཡེར་ཁ་བཏགས་པ་དེ། །སྙན་པ་ཕྱོགས་བཅུར་སྒྲོག་པའི་རྟགས། །ལུང་བཞི་ཧེལ་གེ་བྱས་པ་དེ། །སྨན་པ་གཤིས་རྒྱུད་འཇམ་པའི་རྟགས། ། ལུང་བཞི་སོར་གསུམ་བྱས་པ་དེ། །དུག་གསུམ་ཉེས་གསུམ་འཇོམས་པའི་བརྡ། །སྤྲེལ་ལུང་ཟུང་ཆབ་བརྡབ་པ་དེ། །ཐབས་ཤེས་ཟུང་དུ་འཇུག་པའི་རྟགས། །ཆབ་ལུང་སོར་དྲུག་བྱས་པ་དེ། །སྨན་པའི་རྒྱུ་དྲུག་ཚང་བའི་རྟགས། །ཆབ་མ་ཁྲུ་གང་བྱས་པ་དེ། །ཀུན་དང་མཐུན་པར་འཇུག་པའི་རྟགས། །ཁ་གཉིས་གཅིག་སྒྲིལ་བསྡམས་པ་དེ། །ཆོས་ཀུན་གཅིག་འདུས་དོན་ཤེས་རྟགས། ། ཁ་ཐིག་དར་སྐུད་གྲུ་གསུམ་བྱས། །ཆོས་སྐུ་སྐྱེ་མེད་མཚོན་པ་ཡིན། །དེ་འདྲ་ཟུང་ལྡན་ཁྲུ་ཚད་དྲུག ། ཕ་རོལ་ཕྱིན་དྲུག་སྤྱོད་པའི་རྟགས། །ཁ་ཐིག་གཡས་སུ་བསྒྲིགས་པ་དེ། །ལོངས་སྐུ་འགགས་མེད་མཚོན་པ་ཡིན། །འཕྲུལ་ཚར་སོ་སོར་བཀྲམ་པ་དེ། །སྤྲུལ་སྐུས་འགྲོ་དོན་མཛད་པའི་རྟགས། །སྐུ་གསུམ་ལྡན་པའི་ཁ་ཐིག་དེར། །ཚོན་སྐུད་སྣ་ལྔར་ཡོད་པ་འདི། །ཡེ་ཤེས་ལྔ་ཡང་ལྡན་པའི་རྟགས། །ཁ་དེ་བྱ་རྒོད་མགྲིན་ལྟར་བཅིངས། །ཕ་མེས་སྨན་བརྒྱུད་བཟང་བའི་རྟགས། །ཁ་ལ་མདུད་རྒྱས་གདབ་པ་དེ། །མན་ངག་གབ་ཏུ་སྦས་པའི་རྟགས། ། ལྕགས་གདུབ་རག་ཐག་སྒྲོག་ཆིངས་དེ། ། གཞན་གྱིས་མ་ཁྱེར་ཟེར་བའི་བརྡ། ། དེ་ལ་གཟེ་ལྕགས་བརྒྱབ་པ་དེ། ། འགྲོ་བའི་མགོན་སྐྱབས་ཡིན་པའི་བརྡ། ། གཟན་ཕྱར་གྲུ་བཞིས་བཀབ་པ་དེ། ། སྣོད་མིན་རྣམས་ལ་སྦས་པ་ཡིན། །སྨན་ཕྱར་ཞིང་བཞི་བྱས་པ་དེ། །ཚད་མེད་བཞི་དང་ལྡན་པའི་བརྡ། ། ཞིང་རེར་སོར་བཅུ་ཡོད་པ་དེ། །འབྱོར་བ་བཅུ་ཡིས་ཕྱུག་པའི་རྟགས། །སྲིད་དུ་གྲུ་བཞི་ཡོད་པ་དེ། །མི་འཇིགས་བཞི་ཡིས་དཔའ་བའི་རྟགས། །སྣ་གཉིས་ཕྲན་ཚར་གྲངས་མེད་དེ། །ཕན་ཐོག་བསམ་མི་ཁྱབ་པའི་རྟགས། །བཟོ་དབྱིབས་ཟླུམ་པོར་འདྲིལ་བ་དེ། །སྲིད་གསུམ་གཙུག་ན་ཁུར་བའི་བརྡ། །སེང་གེའི་ཐོད་པ་ལྟར་བཅོས་པ། །སྨྲ་བ་ཀུན་ལ་མཁས་པའི་རྟགས། །སྟེང་འོག་གླེགས་ཤིང་ལྟར་སྦྱར་བ། ། ཤེས་བྱ་ཀུན་ལ་མཁས་པའི་རྟགས། །བར་དུ་ཞབས་རྒྱ་ཡར་བརྒྱུད་པ། །མཐོ་དམན་དྲག་ཞན་ཀུན་འདུས་རྟགས། །སྟེང་ཁེབ་གཅན་ལྤགས་གྲྭ་བཟང་བགྱིས། །གོང་དུ་སྡོད་མཁན་གཞན་མེད་བརྡ། །སེང་ཐོད་ཞབས་རྒྱའི་རྩེ་གཉིས་མ། ། འཇའ་འགྲིག་ཅོད་པན་གཉིས་འབྱུང་དེ། །ལུས་ཅན་ཀུན་ལ་སྲོག་འབྱིན་པས། །གཙུག་གི་ཅོད་པན་ཡིན་པའི་རྟགས། །དེ་འོག་འགྲམ་གཉིས་འཇའ་འགྲིག་གཉིས། །སྨན་པ་ཚོགས་གཉིས་བསགས་པའི་རྟགས། །དེ་ཡི་ཞབས་ཀྱི་གདོང་དཀྱིལ་ལ། །འཇའ་ཡི་ཀ་བ་གཉིས་བརྩེགས་འབྱུང་། །སྨན་པ་སྒྲིབ་གཉིས་སྤངས་པའི་རྟགས། །སྒྲོག་གདན་འཇའ་དབུས་དགུ་བོ་ལ། །གསེར་གྱི་འཕྲུལ་ལོང་དགུ་མཛེས་པ། །ཐེག་པ་རིམ་དགུ་རྫོགས་པའི་བརྡ། །བཤད་དང་ལྡན་པ་སྨན་སྒྲོའི་མཆོག །ཡོན་ཏན་ཁྱད་འཕགས་དེའི་ནང་ན། །སྨན་སྣོད་བཞི་བརྒྱ་རྩ་བཞི་འདི། །འགྲོ་བས་མི་བཟོད་ནད་ཀྱི་རིགས། །བཞི་བརྒྱ་རྩ་བཞི་འཇོམས་པའི་རྟགས། །སྨན་རིགས་སྣ་ཚོགས་བསགས་པ་དེ། །སྣ་ཚོགས་ནད་ལ་བསྔགས་པའི་རྟགས། །སྨན་སྣོད་བཟོ་རིགས་སྣ་ཚོགས་བྱས། །མན་ངག་སྣ་ཚོགས་ཤེས་པའི་རྟགས། །ཁ་ལ་ཤིང་བྱང་བཏགས་པ་དེ། །འགྱུར་མེད་དོན་དང་ལྡན་པའི་རྟགས། །འབྱེད་བསྡམ་འཇུག་པ་བདེ་བ་འདི། །ཟླུམ་སྐོར་པད་མ་ལྟར་བསྡུས་པ། ། ཕྱོགས་མེད་རྒྱལ་ཁམས་བསྐོར་བའི་བརྡ། །སྣོད་དུ་སྨན་རྣམས་གཞུག་ཚུལ་ནི། ཛ་ཏི་ལི་ཤི་ཅུ་གང་སོགས། །རྩིས་ལས་བྱུང་བའི་སྨན་རྣམས་ནི། །སྟག་གཟིག་གཅན་ལྤགས་ཁུག་མར་སྦ། །ཨ་རུ་བ་རུ་སྐྱུ་རུ་སོགས། །ཤིང་ལས་བྱུང་བའི་འབྲས་དང་ཁྲོབ། །ཤས་བཟང་ཟླུམ་སྐོར་རལ་གར་སྦ། །ལོ་མ་མེ་ཏོག་ས་སྨན་ཕལ། །ཤྭ་གནའ་གླ་ཧེལ་སྣོད་དུ་སྦ། །རྒྱ་ཚྭ་རྒྱམ་ཚྭ་ཚྭ་ལ་སོགས། །ཆུ་ལས་བྱུང་བའི་སྨན་རྣམས་ནི། །རས་ཁུག་རྐང་མཁྲེགས་དཀར་པོར་སྦ། །བྲག་ཞུན་སེ་འབྲུ་སྟར་བུ་སོགས། །རླན་སྣུམ་འབྱར་བག་ལྡན་པ་རྣམས། །ཐགས་བཟང་ཕ་ཚེ་གཙང་མར་སྦ། །ཙན་དན་དཀར་དམར་ཨ་གར་སོགས། །ཤིང་སྡོང་ལས་བྱུང་ཚལ་པ་རྣམས། །སག་རིའི་དཔའ་ཆབ་རིང་པོར་སྦ། །དངུལ་ཆུ་དང་ནི་སྦྱར་གྲུབ་པའི། །རྣོ་དང་དྲི་ལྡན་བྱིན་རླབས་ཅན། །ལྕགས་པར་སྨྱུག་སྦུབས་དམ་པར་སྦ། །མཁྲིས་པ་ཁཎྜ་མར་ཚིལ་སོགས། །རྭའམ་ཤིང་གི་པར་བུ་སྦ། །ཐར་ནུ་དུར་བྱིད་དན་ད་སོགས། །རྣོ་སྨན་རྩི་སྣུམ་འབྱར་བག་ཅན། །དུག་བསད་རྩི་ལྡན་སྣོད་དུ་སྦ། །སྦལ་སྦྲུལ་ལ་སོགས་ཤ་ཁྲག་རིགས། །རས་བཟང་སྐམ་པོར་དྲིལ་ནན་བྱ། །གང་ཡང་ནུས་མཐུན་སོ་སོར་སྦྲེལ། །མི་མཐུན་བསིལ་དྲོད་མེ་དང་མར། །ཇི་བཞིན་ལྷན་ཅིག་མི་བཞག་གསུངས། །སྨན་ཡང་རྩི་དུག་སྦྲགས་མི་རུང་། །བཞག་ན་ངན་པ་བློ་རྣོ་བས། །བཟང་པོའི་བློ་འཕྲོག་བྱེད་མཚུངས་པས། །སྤང་བར་བྱ་ཞེས་རྒྱུད་ལས་གསུངས། །
བོད་ལུགས་ཀྱི་སྨན་སྒྲོ་དང་སྨན་བཅོས་ཀྱི་སྤྱོད་ཆས།
           དེ་ལྟར་ཤེས་པའི་སྨན་མཛོད་དེར། །མན་ངག་གླེགས་བམ་བཅུག་པ་དེ། །འཕགས་པ་འཇམ་དཔལ་མཚོན་པ་ཡིན། །ལྡེམ་ཚེ་དཀར་པོ་པད་མ་ཅན། །འཇིག་རྟེན་དབང་ཕྱུག་མཚོན་པ་ཡིན། །དྲག་རྩུབ་ཕག་ནག་ཟེ་བ་དེ། །མཐུ་སྟོབས་ཕྱག་རྡོར་མཚོན་པ་ཡིན། །དེ་གསུམ་ཕྱོགས་གཅིག་བཅུག་པ་དེ། །རིགས་གསུམ་མགོན་པོ་མི་འབྲལ་འཛོམས། །སྨན་པས་ལམ་བགྲོད་དམ་ཚིག་ནི། །ཉམ་ཐག་འབྱོར་མེད་ནད་པ་ལ། །ཕྱོགས་རིས་མེད་པར་སྨན་སྦྱིན་པ། །སྦྱིན་པའི་ཕ་རོལ་ཕྱིན་པ་ཡིན། །བུད་མེད་ཁྱི་སོགས་མ་དག་པས། །འགོམ་འབོག་བཙོག་གྲིབ་མ་གོས་པ། །ཚུལ་ཁྲིམས་ཕ་རོལ་ཕྱིན་པ་ཡིན། །ཡོན་ཟས་ལ་འཁོར་སྨན་རིན་རྩོད། །མཁས་འདྲ་མཐོ་གནོན་བྱེད་སྤང་བ། །བཟོད་པའི་ཕ་རོལ་ཕྱིན་པའོ། །སྨན་དེ་སྦྱར་བར་བརྩམ་པ་ན། །སྨན་ཀློག་ཁ་ཡན་མི་འཇོག་པ། །གློ་བུར་དགྲ་ཇག་བྱུང་བ་སོགས། །གལ་སྲིད་ཡན་པར་བཞག་དགོས་ཚེ། །ཟླུམ་ལ་སྨན་སྒྲོ་ཆ་སྦྱར་བཀབ། །དེ་སྟེང་རིགས་གསུམ་མཚོན་རྒྱས་བཏབ། །དེ་བཞག་དེ་ནི་རུང་ཙམ་ཡིན། །སྨན་སྒྲོ་གཞི་བྱེས་གང་དུ་ཡང་། །ཙོག་པུར་བཞུགས་པ་མ་གཏོགས་པ། །བསྙལ་དང་ས་རྗེན་ཐོག་མི་འཇོག །དེས་ནི་བསམ་གཏན་ཕར་ཕྱིན་མཚོན། །མགོ་ཆ་གཏར་ཆ་བསྲེགས་ཆ་སོགས། །རྣོ་མཚོན་གླེགས་བམ་རྒྱུན་མི་གཅོད། །དེ་ཡིས་ཤེས་རབ་ཕར་ཕྱིན་མཚོན། །སྦྱིན་བདག་ཚོགས་བསགས་དམ་ཚིག་ནི། །འདི་ལྟར་ཡོན་ཏན་དུ་མ་དང་། །ལྡན་པ་སྨན་སྒྲོའི་ནོར་བུའི་མཛོད། །རིགས་བརྒྱུད་ལྡན་པའི་བླ་མ་ཡིས། །བྱིན་རླབས་རབ་ཏུ་གནས་བྱས་ཤིང་། །འདི་ལ་མཆོད་པ་སུས་འབུལ་བ། །རླུང་ནད་དྲག་པོས་མི་གཟིར་རོ། །འདི་ལ་གསོལ་བ་སུས་བཏབ་པ། །མཁྲིས་ནད་གནོད་པས་མི་ཚུགས་སོ། །འདི་ལ་བཀུར་སྟི་སུས་བྱས་པ། །ལྡན་འདུས་ནད་ཀྱིས་མི་འཆིའོ། །འདི་ལ་ཁྲི་སྟན་སུས་བཤམས་པ།
