སྨད་འཚོང་བྱེད་སྤྱོད་སྤེལ་བས་ན། ཁྲིམས་ཀྱི་ཆད་པ་མི་ཕོག་ཅི།
ཉེ་ཆར། ལྷ་ས་གྲོང་ཁྱེར་ཁྲེང་ཀོན་ཆུས་ཞིབ་དཔྱོད་ཁང་གིས་ཁྲིམས་ལྟར་གཞུང་གཏུག་བརྒྱབ་པའི་ལྷ་སའི“སྤྲིན་ཚོམ་སྤྲོ་ཁང”གིས་སྨད་འཚོང་བྱེད་སྤྱོད་སྤེལ་བར་འདྲི་གཅོད་དང་པོ་བྱས་ཏེ། ཞུ་སྦྱོར་ཕྱི་མ་5ལ་དུས་བཀག་བཙོན་འཇུག་གི་ཆད་པ་བཅད་པ་རེད། ཤེས་རྟོགས་བྱུང་བར་གཞིགས་ན། གྱོད་གཞི་དེ་ནི་རྒྱལ་ཡོངས་ནག་སེལ་གཤེད་བཅོམ་གཞུང་དོན་ཁང（ཀྲུང་དབྱང་ལྟ་སྐུལ་མཛུབ་ཁྲིད་ཚོགས་ཆུང）གིས་གཙོ་གནད་ལྟ་སྐུལ་སྒྲུབ་གཉེར་བྱེད་པའི་གྱོད་གཞིའི་གྲས་ཤིག་རེད།
          ལྷ་ས་གྲོང་ཁྱེར་ཁྲེང་ཀོན་ཆུས་ཞིབ་དཔྱོད་ཁང་གིས་ཞིབ་དཔྱད་བྱས་པ་བརྒྱུད། ཞུ་སྦྱོར་ཕྱི་མ་ལིའུ་ཆེ་གེ་མོས་ཁེ་བཟང་རག་ཆེད། རང་ཉིད་ཀྱིས་ལྷ་ས་གྲོང་ཁྱེར་ཁྲེང་ཀོན་ཆུས་ཆུ་མིག་ལམ་དུ་གཉེར་བའི“སྤྲིན་ཚོམ་སྤྲོ་ཁང”ཞེས་པའི་གཟུགས་པོ་འཁྲུ་སར། 2017ལོའི་ཟླ་བ་དང་པོ་ནས་2018ལོའི་ཟླ་བ་དྲུག་པ་བར་ལྐོག་ཏུ་བུད་མེད་སྒྲིག་འཛུགས་བྱས་ནས་སྨད་འཚོང་བྱེད་དུ་བཅུག དེའི་རིང་ཞུ་སྦྱོར་ཕྱི་མ་ཝང་ཆེ་གེ་མོ་དང་། བློ་ཆེ་གེ་མོ། ཐང་ཆེ་གེ་མོ། ལིའུ་ཆེ་གེ་མོ་བཅས་ཀྱིས་རོགས་པ་བྱས་ཏེ་སྨད་འཚོང་སྒྲིག་འཛུགས་བྱས་པས། སྨད་འཚོང་སྒྲིག་འཛུགས་དང་གཞོགས་འདེགས་བྱས་པའི་ཉེས་པ་བསགས།
          འདྲི་གཅོད་བྱེད་སྐབས། ཉེས་པ་བསགས་ལུགས་དང་། བྱེད་སྤྱོད་ཀྱི་ཁྱད་ཆོས། གནོད་འཚེའི་མཇུག་འབྲས་སོགས་ལ་གཞུང་གཏུག་རྒྱག་མཁན་ཕྱོགས་ཀྱིས་སྐྱོན་འདོན་ཞུ་གཏུག་བརྒྱབ། ཁྲེང་ཀོན་ཆུས་ཁྲིམས་ཁང་གིས་ཞིབ་དཔྱད་བྱས་རྗེས། ཁྲེང་ཀོན་ཆུས་ཞིབ་དཔྱོད་ཁང་གི་སྐྱོན་འདོན་ཞུ་གཏུག་གི་བསམ་འཆར་དང་ལེན་བྱས་ཏེ། སྨད་འཚོང་སྒྲིག་འཛུགས་ཀྱི་ཉེས་པ་བསགས་པར་བརྟེན། ལིའུ་ཆེ་གེ་མོ་དུས་བཀག་བཙོན་འཇུག་ལོ་ལྔ་དང་དངུལ་ཆད་སྒོར་6000བཏང་བ་དང་། སྨད་འཚོང་སྒྲིག་འཛུགས་ལ་གཞོགས་འདེགས་བྱེད་མཁན་ཞུ་སྦྱོར་ཕྱི་མ་ཝང་ཆེ་གེ་མོ་དུས་བཀག་བཙོན་འཇུག་ལོ་གཅིག་དང་ཟླ་བ་གཉིས་དང་དངུལ་ཆད་སྒོར་2000བཏང་བ། བློ་ཆེ་གེ་མོ་དུས་བཀག་བཙོན་འཇུག་ལོ་གཅིག་དང་དངུལ་ཆད་སྒོར་2000བཏང་བ། ཐང་ཆེ་གེ་མོ་དུས་བཀག་བཙོན་འཇུག་ལོ་གཅིག་དང་དངུལ་ཆད་སྒོར་2000བཏང་བ། ལིའུ་ཆེ་གེ་མོ་དུས་བཀག་བཙོན་འཇུག་ལོ་གཅིག་དང་ཟླ་བ་བཞི་དང་དངུལ་ཆད་སྒོར་3000བཏང་བ་རེད།
          གྱོད་གཞི་དེ་སྒྲུབ་རིང་། ལྷ་ས་གྲོང་ཁྱེར་ཁྲེང་ཀོན་ཆུས་ཞིབ་དཔྱོད་ཁང་གིས་ཞིབ་དཔྱོད་ལས་ཁུངས་ཀྱི་ཉེས་དོན་གཏུག་བཤེར་གཙོ་ཁྲིད་ནུས་པ་འདོན་སྤེལ་གང་ལེགས་དང་། ཁྲིམས་ལྟར་སྔ་སྣུར་དུ་གྱོད་གཞིར་མྱུལ་ཞིབ། སྤྱི་བདེ་ལས་ཁུངས་ཀྱི་མྱུལ་ཞིབ་དཔང་ལེན་ལ་ཁྲིད་སྟོན་བཅས་བྱས་པར་བརྟེན། གྱོད་གཞིའི་འབྲེལ་ཡོད་མི་སྣ་མགྱོགས་མྱུར་བཟུང་བ་དང་། མགྱོགས་མྱུར་ཞུ་གཏུག མགྱོགས་མྱུར་འདྲི་གཅོད་ཐུབ་ཡོད་ལ། ཁྲིམས་ལྟར་དང་། ནུས་ལྡན། མགྱོགས་མྱུར་བཅས་ཀྱི་སྒོ་ནས་ཉེས་གསོག་ལ་རྡུང་རྡེག་གཏོང་ཐུབ་ཡོད།
          རང་སྐྱོང་ལྗོངས་ཞིབ་དཔྱོད་ཁང་ནག་སེལ་གཤེད་བཅོམ་གཞུང་དོན་ཁང་གི་འགན་འཁུར་བས། གོམ་སྟབས་རྗེས་མར། ལྗོངས་ཡོངས་ཀྱི་ཞིབ་དཔྱོད་ལས་ཁུངས་ཀྱིས་རྒྱལ་ཡོངས་ཀྱི་ནག་སེལ་གཤེད་བཅོམ་གཞུང་ལས་ཁང་གི“གཙང་སེལ་དྲུག”གི་བྱ་སྤྱོད་བླང་བྱ་གཞིར་བཟུང“ནག་སེལ་གཤེད་བཅོམ་དང་འབྲེལ་བའི་གྱོད་གཞི་སྒྲུབ་པའི་ཉིན་བརྒྱའི་འགག་སྒྲོལ་དམག་འཐབ”ལས་དོན་སྤེལ་ནས། དུས་བཀག་ལྟར“གྱོད་གཞི་གཙང་སེལ”གྱི་ལས་འགན་ལེགས་སྒྲུབ་བྱས་ཏེ། རང་ལྗོངས་ཀྱི་ནག་སེལ་གཤེད་བཅོམ་དང་ལོག་སེལ་ཟིང་བཅོས་ཀྱི་ཆེད་དོན་འཐབ་རྩོད་ལ་ཕྱོགས་ཡོངས་ནས་རྒྱལ་ཁ་ལེན་ཆེད་ཞིབ་དཔྱོད་ཀྱི་སྟོབས་ཤུགས་ལེགས་སྐྱེས་འབུལ་རྒྱུ་རེད་ཅེས་བརྗོད།
