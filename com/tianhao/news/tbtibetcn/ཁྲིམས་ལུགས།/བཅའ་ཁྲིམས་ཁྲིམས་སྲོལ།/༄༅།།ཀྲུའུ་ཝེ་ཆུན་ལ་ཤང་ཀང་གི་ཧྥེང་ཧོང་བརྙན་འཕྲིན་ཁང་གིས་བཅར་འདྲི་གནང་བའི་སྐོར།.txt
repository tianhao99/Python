༄༅།།ཀྲུའུ་ཝེ་ཆུན་ལ་ཤང་ཀང་གི་ཧྥེང་ཧོང་བརྙན་འཕྲིན་ཁང་གིས་བཅར་འདྲི་གནང་བའི་སྐོར།
（སྨད་ཆ།）
　                 “ཁྱོད་ཀྱི་འཛུམ་མདངས་ལས་མིའི་ཡིད་དབང་འཕྲོག་སྲིད་ལ་ཁྱོད་ཀྱི་བྱ་སྤྱོད་ལས་མི་ལ་གནོད་པ་བྱེད་ངེས།”
     རྒྱལ་སྤྱིའི་གླེང་བརྗོད་ཁྲོད་དུ་བཤད་ཚུལ་འདི་འདྲ་ཞིག་ཡོད་པ་དང་།ཀྲུང་གོ་སྲིད་གཞུང་ལ་ཧ་ཅང་ཚོར་སྣང་སྐྱེན་པའི་གནད་དོན་གཉིས་ཡོད་པ་རེད།དེ་གཉིས་ནི་དབྱིན་ཡིག་གི་ཡི་གེTའགོ་ན་ཡོད་པའི་གནད་དོན་དང་འབྲེལ་བ་ཡོད་དེ།T གཅིག ་ནི་Taiwanགནད་དོན་དང་འབྲེལ་བ་ཡོད། T གཞན་ཞིག ནིTibetཡིན་ལ། བོད་ལྗོངས་ཀྱི་གནད་དོན་ཞིག་ཀྱང་རེད།
     ཀྲུའུ་ཝེ་ཆུན་གྱིས་འདིའི་ཁྲོད་ཀྱི་ཚོར་དབང་གཅིག་ལ་གཏིང་ཟབ་པའི་བྱུང་ཁུངས་ཡོད་པ་དང་།  ༡༩༩༡ལོར་ཀྲུའུ་ཝེ་ཆུན་ལགས《མི་དམངས་ཉིན་རེའི་ཚགས་པར》གྱི་གསར་འགོད་པ་ཞིག་ཡིན་པའི་ཆ་ནས།   བོད་ལྗོངས་ཞི་བས་བཅིངས་འགྲོལ་བཏང་བའི་ཚུན་གྱི་ལོ༤༠རིང་བོད་ལྗོངས་སུ་བསྐྱོད་ནས་ཟླ་གཅིག་ལྷག་ལ་བཅར་འདྲི་གནང་བའི་རིང་དུ་བོད་ལྗོངས་དང་བརྩེ་འདང་ཆགས་པ་རེད།   དེ་ནས་ཀྲུང་གུང་ཀྲུང་དབྱང་གཞུང་ལས་ཐིང་ནས་ལས་དོན་གཉེར་བའི་རིང་ལ།   ཁོང་གིས་སྐུ་དངོས་སུ་པཎ་ཆེན་སྐུ་ཕྲེང་བཅུ་བའི་ཡང་སྲིད་འཚོལ་བའི་ལས་དོན་གཉེར་བ་དང་།   ༡༩༩༨ལོར་ཀྲུའུ་ཝེ་ཆུན་གྱིས་ཀྲུང་དབྱང་འཐབ་ཕྱོགས་གཅིག་གྱུར་པུའུ་ཡི་པུའུ་ཀྲང་གཞོན་པའི་ལས་གནས་གཉེར་རྗེས་བོད་འབྲེལ་ལས་དོན་དེ་ཉིད་སྔར་ལས་ལྷག་པའི་སྒོ་ནས་ཁོང་གི་ཉིན་རེར་སྒྲུབ་བཞིན་པའི་བྱ་བའི་གྲས་ཤིག་ཏུ་གྱུར་ཡོད་པ་རེད།
    ཀྲུའུ་ཝེ་ཆུན་གྱིས་འདི་སྐད་དུ།Tརིགས་གཉིས་ཀྱི་བཤད་ཚུལ་འདི་ནི་ངས་མཐའ་གཅིག་ཏུ་ཁས་ལེན་མི་སྲིད་པ་ཡིན།   རྒྱུ་མཚན་ཅིའི་ཕྱིར་ཞེ་ན། བསྡུར་ཐབས་མེད་པའི་ཁྱད་པར་ཆེན་པོ་ཡོད་པས་སོ།   ཐའེ་ཝན་དུ་དེང་སྐབས་ད་དུང་མེས་རྒྱལ་གྱི་སྐམ་ས་ཆེན་མོ་དང་འདྲ་བར་ཁྱོན་ཡོངས་ནས་གཅིག་གྱུར་བྱེད་པ་མངོན་འགྱུར་བྱུང་མེད་ལ།   ལོ་རྒྱུས་སྟེང་དུ་བོད་ལྗོངས་ནི་གནའ་སྔ་མོ་ནས་བཟུང་ཀྲུང་གོ་རྒྱལ་ཁབ་ཀྱི་མངའ་ཁོངས་དང་ཆ་ཤས་ཤིག་ཡིན་པ་མ་ཟད།    ད་དུང་ཀྲུང་དབྱང་དང་སྐབས་རིམ་བྱུང་གི་ཀྲུང་དབྱང་སྲིད་གཞུང་གིས་ཐོག་མཐའ་བར་གསུམ་དུ་ས་དེར་བདག་དབང་གཙོ་བོར་བྱེད་ཀྱི་ཡོད།   ངས་བལྟས་ན་“བོད་ལྗོངས་ཀྱི་གནད་དོན་”ཞེས་བྱ་བ་ཞིག་མེད་པ་ནི་ཅང་སུའུ་ཞེས་་བྱ་བའི་གནད་དོན་དང་།   ཧུའུ་པེ་ཞེས་བྱ་བའི་གནད་དོན།   སི་ཁྲོན་ཞེས་བྱ་བའི་གནད་དོན་མེད་པ་དང་གཅིག་མཚུངས་རེད།    དེས་ན་གནད་དོན་ཅི་ཞིག་གནས་ཡོད་དམ་ཞེ་ན།   ཏཱ་ལའི་བླ་མའི་གནད་དོན་གཅིག་པུ་ལུས་ཡོད་ཅེས་གསུངས་པ་རེད།
    བར་བཅུག་འགྲེལ་བརྗོད།   ཟླ༡༢པའི་ཚེས༦ཉིན་ཕོ་ལན་དུ་ཧྥ་གོའི་ཙུང་ཐུང་ས་ཁོ་ཆི་ཡིས་ཏཱ་ལའི་དང་མཇལ་འཕྲད་གནང་ཞིང་།   ཉིན་དེའི་དགོང་མོར་ཀྲུང་གོའི་ཕྱི་འབྲེལ་པུའུ་ཡི་པུའུ་ཀྲང་གཞོན་པ་ཧོ་ཡ་ཧྥེ་ཡིས་ཀྲུང་གོར་སྡོད་པའི་ ཧྥ་གོ་རྒྱལ་ཁབ ་ཀྱི་དབུ་བཞུགས་བློན་ཆེན་དང་མཇལ་འཕྲད་གནང་ནས་སྐད་ཆ་ནན་བཤད་གནང་བ་རེད།   ཧོ་ཡ་ཧྥེ་ཡིས་ས་ཁོ་ཆི་ཡིས་ཡོ་རོབ་མནའ་འབྲེལ་གྱི་རེ་མོས་སུ་སྡོད་པའི་ཀྲུའུ་ཞིའི་ཐོབ་ཐང་ལ་བརྟེན་ནས་ཏཱ་ལར་མཇལ་འཕྲད་གནང་ནས་ངན་པ་ཐ་ཤལ་གྱི་ལུགས་གསར་ཞིག་བཏོད་ཡོད།    མཉམ་མཇོག་བྱེད་དགོས་རྒྱུ་ཞིག་ནི་ཏཱ་ལའི་ཡིས་ཐེངས་འདིར་འཚམས་འདྲི་གནང་བའི་ཐོག་མའི་ས་ཚིགས་ཀྱི་དམིགས་ཡུལ་ནི་ཅེ་ཁེ་ཡིན་པ་དང་།དེས་སང་ལོའི་ལོ་སྟོད་དུ་ ཡོ་རོབ་མནའ་འབྲེལ་གྱི་རེ་མོས་སུ་སྡོད་པའི ་ཀྲུའུ་ཞིའི་རྒྱལ་ཁབ་ཀྱི་འགན་ཁུར་རྒྱུ་རེད།
ཀྲུའུ་ཝེ་ཆུན་གྱིས་འདི་སྐད་དུ།གནས་ལུགས་ཧ་ཅང་རྟོགས་སླ་བ་ཡིན།   རྒྱུ་མཚན་ནི་ཏཱ་ལའི་བླ་མས་འཐུས་ཚབ་བྱེད་པ་ནི་ཁ་བྲལ་རིང་ལུགས་ཀྱི་ཆབ་སྲིད་ཚོགས་པ་ཞིག་ཡིན་ཞིང་།   དེང་སྐབས་ཕྱི་རྒྱལ་གྱི་གླེང་བརྗོད་ཁ་ཤས་ཤིག་དང་སྲིད་དོན་གཙོ་བོ་ཁ་ཤས་ཤིག་གིས་ཏཱ་ལའི་བླ་མར་སྐད་ཆ་ཡག་པོ་མང་པོ་ཞིག་ཤོད་ཀྱི་ཡོད་པ་དང་།   ཐ་ན་ངོ་འཕྲད་བྱུང་སྐབས་སུའང་དེ་འདྲའི་སྐད་ཆ་ཡག་པོ་ལས་མ་འདས་པ་ཡིན།   མི་མང་པོ་ཞིག་གནས་ཚུལ་དངོས་ལ་ཆ་རྒྱུས་མེད་པའི་ཡིད་ཆེས་ང་རང་ལ་ཡོད་མོད།   ཡིན་ནའང་ངས་བསམ་ན་མི་ཁ་ཤས་ཤིག་ལ་དྲན་ཚུན་གཞན་ཞིག་ཡོད་པ་རེད།   ང་ཚོ་ཀྲུང་གོ་བས་མི་འདི་རིགས་མི་གྲས་སུ་མི་འཇོག་པ་དང་།   མི་དེའི་རིགས་ལ་གཞན་ལ་གནོད་པ་བྱས་ནས་རང་ལ་ཕན་འདོགས་པ་ཞེས་པ་དེ་ཡིན་ཞེས་གསུངས་པ་རེད།

    བར་བཅུག་འགྲེལ་བརྗོད།ད་ལོའི་ཟླ༤པའི་ནང་དུ་ལྷ་སར་དོན་རྐྱེན་བྱུང་རྗེས་ཏཱ་ལའི་བླ་མས་ཐོག་མར་ཨ་མེ་རི་ཁར་འཚམས་འདྲི་གནང་བར་ཕྱིན་པ་དང་།   ཞི་ཡ་ཐུའུ་རུ་མི་མང་པོ་འདུ་འཛོམས་བྱེད་སྐབས་ཕྱིར་སྡོད་ཀྲུང་གོའི་མི་རྣམས་ཀྱིས་རྔམ་སྟོན་ཁྲོམ་སྐོར་བྱས་ནས་ངོ་རྒོལ་གནང་བ་དང་།   ཁྲོམ་སྐོར་བས་གླ་བཀོལ་གནམ་གྲུར་བསྡད་ནས་གནམ་གྲུའི་སྦྱར་ཤོག་སྟེང་དུ「ཏཱ་ལའི་བླ་མ།   ཁྱོད་ཀྱི་འཛུམ་མདངས་ལས་མིའི་ཡིད་དབང་འཕྲོག་སྲིད་མོད།   འོན་ཀྱང་ཁྱོད་ཀྱི་བྱ་སྤྱོད་མཐའ་དག་ལས་མི་རྣམས་ལ་གནོད་པ་བྱེད་ངེས」ཞེས་བྲིས་ཡོད་པ་མི་རྣམས་ལ་བསྟན་ནས་མཁའ་དབྱིངས་སུ་ཡང་ནས་ཡང་དུ་སྐོར་བ་རྒྱབ་པ་རེད།

    ཀྲུའུ་ཝེ་ཆུན་གྱིས་འདི་སྐད་དུ།ཕུལ་དུ་བྱུང་བ་རེད་ཅེས་བརྗོད་རྒྱུ་ཤིན་ཏུ་ཁག་པོ་འདུག་པ་དང་།    ངས་བསམ་ན་རྒྱལ་ཁབ་ཁ་ཤས་ཤིག་གིས་ཁོ་རང་བེད་སྤྱོད་བྱེད་པ་དང་།    ཁོ་རང་ཉིད་ཀྱིས་ཀྱང་རྒྱལ་ཁབ་ཁ་ཤས་ཤིག་དང་ཟིལ་ཤུགས་ཁ་ཤས་ཤིག་བེད་སྤྱོད་གཏོང་བར་སྤྲོ་སྣང་ཡོད་པ་རེད་ཅེས་གསུངས་པ་རེད།
འགོ་འཛིན་པས་འདི་སྐད་དུ།ཏཱ་ལའི་བླ་མས་བརྗོད་རྒྱུར་ཁོ་རང་ལ་ལག་པ་གཉིས་ཡོད་པ་དང་།   ལག་པ་ཡ་གཅིག་ཀྲུང་གོ་ཀྲུང་དབྱང་སྲིད་གཞུང་ལ་བསྲིངས་ཡོད་པ་དང་།   ལག་པ་ཡ་གཅིག་རྒྱལ་སྤྱིའི་སྤྱི་ཚོགས་སུ་བསྲིངས་ཡོད་མོད།   འོན་ཀྱང་ཁོས་མཐོང་བ་ནི་གལ་ཏེ་ཀྲུང་གོ་ཀྲུང་དབྱང་སྲིད་གཞུང་གིས་ཁོ་ལ་སྦྱིན་པ་ནི་གནས་ཚུལ་མི་ལེགས་པ་ཞིག་ཡིན་ན།ཁོ་རང་རྒྱལ་སྤྱིའི་སྤྱི་ཚོགས་སུ་ཕྱོགས་ངེས་པ་ཡིན།   ཁྱེད་ཀྱིས་ཁོའི་བཤད་ཚུལ་འདིར་ལྟ་ཚུལ་ཅི་ཞིག་འཛིན་ནམ། 
ཀྲུའུ་ཝེ་ཆུན་གྱིས་འདི་སྐད་དུ།    ཏཱ་ལའི་བླ་མས་ཐོག་མཐའ་བར་གསུམ་དུ་ལག་པ་ཟུང་བཀོལ་བཞིན་ཡོད་པ་འདི་ནི་ཁོའི་བཤད་ཚུལ་ཡང་དག་ཅིག་ཡིན་པ་དང་།   ང་ཚོའི་ཕྱོགས་སུ་བསྲིངས་པའི་ལག་པ་འདི་ནི་ང་ཚོའི་བོད་ལྗོངས་ལ་གཏོར་བརླགས་གཏོང་བའི་ལག་པ་ཞིག་ཡིན་ལ།   ད་དུང་ཁོས་བཤད་པ་ནང་བཞིན་གྱི་ལག་པ་གཅིག་ནི་རྒྱལ་སྤྱིའི་སྟེང་གི་རྒྱལ་ཁབ་ཁ་ཤས་ཤིག་ཏུ་བསྲིངས་ཡོད་ཅེས་གསུངས་པ་རེད།       ཁོས་ནམ་ཡིན་ཡང་ནུབ་ཕྱོགས་རྒྱལ་ཁབ་དང་ཡང་ན་མི་ཁ་ཤས་ཤིག་གི་རྒྱབ་སྐྱོར་ལ་བརྟེན་ནས་རང་ཉིད་ལ་སྟོབས་ཤུགས་འཕེལ་འདོད་ཡོད་པ་དང་།    དོན་དངོས་ཐོག་ང་དང་རྒྱ་རི་བློ་གྲོས་གཉིས་ཐེངས་དང་པོར་ངོ་འཕྲད་གནང་ནས་གྲོས་མོལ་གནང་སྐབས་གསལ་བོར་བརྗོད་མྱོང་བ་ཡིན།    ཀྲུང་གོ་ཀྲུང་ཁྲན་ཏང་གནའ་ནས་ད་བར་དུ་ཕྱི་ཕྱོགས་གནོན་ཤུགས་ལ་སྐྲག་མ་མྱོང་པ་ཁྱོད་ཀྱིས་གསལ་བོར་རྟོགས་དགོས་ཤིང་།   ཕྱི་རྒྱལ་གྱི་ཟིལ་ཤུགས་ཁ་ཤས་ཀྱིས་ཏཱ་ལར་སྙིང་རྗེ་བ་དང་རོགས་སྐྱོར་ཐོབ་པ་ཡིས་ཏཱ་ལའི་ཁོ་རང་ལ་སྟོབས་ཤུགས་ཆེན་པོ་ཡོད་སྙམ་པའི་ཡི་ནོར་འཁྲུལ་ཆེན་པོ་འདུག་པ་དང་།“བོད་རང་བཙན་”དང་རྣམ་པ་འགྱུར་བའི་རང་བཙན།    རང་བཙན་ཕྱེད་ཀ་ཞེས་པ་མངོན་འགྱུར་ཡོང་སྙམ་པའི་ནོར་འཁྲུལ་གྱི་བསམ་བློ་ཡོད་པས།  ནོར་འཁྲུལ་གྱི་ལམ་བུ་ཇེ་ཟབ་ནས་ཇེ་ཟབ་དུ་འགྲོ་བཞིན་ཡོད་ལ།  མཇུག་འབྲས་ནི་ཕ་ཡུལ་གྱི་ལམ་ཐག་དང་ཇེ་རིང་ནས་ཇེ་རིང་དུ་འགྲོ་བཞིན་ཡོད་པས། མདོར་ན་ཕྱི་རྒྱལ་པ་འདི་དག་གིས་དོན་དངོས་ཐོག་ཏུ་ཏཱ་ལར་གནོད་པ་ཆེན་པོ་སྐྱེལ་བཞིན་ཡོད་ཅེས་གསུངས་པ་རེད།                                                                  མུ་འགོད།
