གཉེན་སྒྲིག་ཡི་གེ་མ་བླངས་ཚེ་ཁྲིམས་མཐུན་བཟའ་ཟླ་ཡིན་ནམ།
གྱོད་དཔེ།
          ན་ནིང་གི་ཟླ་གསུམ་པའི་ནང་། ངེད་ཚང་གི་ཕུ་བོ་དང་སྡེ་བ་གཅིག་པའི་བུ་མོ་གཡུ་སྒྲོན་གཉིས་ཕན་ཚུན་གཉེན་རྟགས་བརྒྱབ་ནས་ང་ཚོ་བོད་ཀྱི་དཔེ་ལྟར་གཉེན་སྒྲིག་གི་ཆོ་གའང་རྒྱ་ཆེན་པོ་ཞིག་སྤེལ། ཡང་ནའང་སྡེ་བའི་མི་ཕལ་མོ་ཆེ་ཞིག་གཉེན་སྒྲིག་བྱས་ཀྱང་གཉེན་སྒྲིག་ཡི་གེ་མེད་པས་ངེད་ཚང་གིས་ཀྱང་དེར་མཐོང་ཆེན་མ་བྱས་སྟབས་གཉེན་སྒྲིག་ཡི་གེ་ད་ལྟའི་བར་མ་བླངས། ད་ལོའི་ཟླ་བཅུ་བར། མནའ་མ་གཡུ་སྒྲོན་གྱིས་གློ་བུར་དུ་ཁོ་གཉིས་ཀྱི་གཉེན་སྒྲིག་ཁས་ལེན་མི་བྱེད་པར་མ་ཟད་ཁོ་མོ་རང་ཡུལ་དུ་ལོག་ནས་ཕྱིར་མ་ཡོང་། ངེད་ཚང་གིས་ཐེངས་འགར་བར་བ་མངགས་ནས་འབོད་དུ་ཕྱིན་ཡང་མོས་ཁོ་གཉིས་ཀྱི་གཉེན་སྒྲིག་ཁས་ལེན་མ་བྱས་པས་ཕ་མ་གཉིས་ཀྱང་ཅི་བྱ་གཏོལ་མེད་དུ་གྱུར།
ཁྲིམས་བཤད།
          ཁྱོད་ཀྱིས་བཤད་པའི་དོན་དངོས་འདི་ལྟ་བུ་ང་ཚོའི་བོད་ཁུལ་ཡོངས་སུ་རྒྱུན་འབྱམས་སུ་སོང་འདུག ང་ཚོ་བོད་རིགས་རྣམས་ལ་མཚོན་ན། གཙོ་བོ་གཉེན་རྟགས་དང་གཉེན་སྒྲིག་གི་ཆོ་ག་གཉིས་གཙིགས་སུ་འཛིན་པ་ལས་གཉེན་སྒྲིག་ཡི་གེ་ལེན་པར་དེ་འདྲའི་མཐོང་ཆེན་བྱེད་ཀྱིན་མེད་སྟབས་འདི་ལྟ་བུའི་སྣང་ཚུལ་བྱུང་བ་རེད། ཁྱོད་ཀྱི་ཕོ་བོ་དང་བུ་མོ་གཡུ་སྒྲོན་གཉིས་ཀྱི་གཉེན་སྒྲིག་གི་འབྲེལ་བ་དེ་བཅའ་ཁྲིམས་སྟེང་ཁས་ལེན་འཐོབ་མི་སྲིད། རྒྱུ་མཚན་ནི《གཉེན་སྒྲིག་བཅའ་ཁྲིམས》ཀྱི་དོན་ཚན་བརྒྱད་པའི་ནང་། “གཉེན་སྒྲིག་བྱས་པའི་ཕོ་མོ་ཟུང་ངེས་པར་དུ་རང་ཉིད་དངོས་སུ་གཉེན་སྒྲིག་ཐོ་འགོད་སྡེ་ཁག་ཏུ་སོང་ནས་གཉེན་སྒྲིག་ཐོ་འགོད་བྱེད་དགོས། བཅའ་ཁྲིམས་འདིའི་གཏན་ཁེལ་དང་མཐུན་ཚེ་ཐོ་འགོད་བྱས་ཆོག་ལ་གཉེན་སྒྲིག་ཡི་གེ་ཡང་སྟེར་ངེས། གཉེན་སྒྲིག་ཡི་གེ་བླངས་རྗེས་དོན་དངོས་ཀྱི་གཉེན་སྒྲིག་འབྲེལ་བ་བྱུང་བ་རེད།”ཅེས་གསལ་བོར་གཏན་ཁེལ་བྱས་འདུག ཁྱོད་ཀྱི་ཕུ་བོ་དང་གཡུ་སྒྲོན་གཉིས་ལ་གཉེན་སྒྲིག་གི་ཆོ་ག་སྤེལ་ཟིན་ནའང་ཁོ་གཉིས་ཀྱི་གཉེན་སྒྲིག་ཐོ་འགོད་མ་བྱས་པར་མ་ཟད་གཉེན་སྒྲིག་ཡི་གེ་ཡང་མ་བླངས་པས། དེ་ནི་ཁྲིམས་ལུགས་སྟེང་དུ་གཉེན་སྒྲིག་འབྲེལ་བ་ཆགས་མེད་པར་ངོས་འཛིན་གྱིན་ཡོད་ལ། ཁྲིམས་ལུགས་ཀྱིས་རྒྱབ་སྐྱོར་མི་བྱེད། ཁོ་གཉིས་ཀྱི་གཉེན་སྒྲིག་ནི་དོན་དངོས་གཉེན་སྒྲིག་གི་ཁོངས་སུ་མི་གཏོགས། དེ་བས་ཁྲིམས་འགལ་མཉམ་སྡོད་ཀྱི་འབྲེལ་བ་དེ་གཏུག་བཤེར་བརྒྱུད་ནས་མེད་པར་བཟོ་མི་ཐུབ། ཆེས་མཐོ་བའི་མི་དམངས་ཁྲིངས་ཁང་གིས《<གཉེན་སྒྲིག་བཅའ་ཁྲིམས>ཀྱི་དོན་ཚན་འགའི་གསལ་བཤད་གཉིས་པའི》དོན་ཚན་དང་པོའི་ནང་དུ། “དོ་བདག་པའི་གཏུག་བཤེར་གྱི་རེ་བ་བཏོན་དེ་ཁྱོ་ཤུག་མཉམ་སྡོད་ཀྱི་འབྲེལ་བ་མེད་པར་བཟོ་རྒྱུ་ཡིན་ན། དེ་ལ་མི་དམངས་ཁྲིམས་ཁང་གིས་བདག་སྤྲོད་མི་བྱེད། དོ་བདག་པས་མཉམ་སྡོད་བྱས་པའི་རིང་གི་རྒྱུ་ནོར་བགོ་བཤའ་དང་ཡང་ན་བྱིས་པ་སྙོར་སྐྱོང་སོགས་རྩོད་གཞིར་གྱུར་པའི་ཞུ་གཏུག་ཡིན་ན། མི་དམངས་ཁྲིམས་ཁང་གིས་བདག་སྤྲོད་བྱེད་དགོས”ཞེས་གསལ་བོར་གཏན་ཁེལ་བྱས་ཡོད།
