སློབ་གསོ་པུས་སློབ་གྲྭ་འབྲིང་ཆུང་གི་སློབ་མའི་སློབ་ཕྱིའི་ཀློག་དེབ་སློབ་གྲྭར་ཁྱབ་རྒྱུའི་དོ་དམ་བྱ་ཐབས་བཏོན་པ།
སློབ་ཕྱིའི་ཀློག་དེབ་ནི་མི་གསོ་སྐྱོང་བྱེད་པའི་གཞི་རྟེན་གལ་ཆེན་གྲས་ཤིག་ཡིན་པ་དང་། དེས་སློབ་མའི་ཚན་རིག་ཤེས་བྱ་འཕེལ་བ་དང་། ལྟ་རྒྱ་ཐོས་རྒྱ་ཆེ་རུ་འགྲོ་བ། ཕྱོགས་བསྡུས་ཀྱི་བྱང་ཚད་མཐོ་རུ་འགྲོ་བ་སོགས་ཐད་ནུས་པ་གལ་ཆེན་ཐོན་གྱི་ཡོད། ཡིན་ནའང་མིག་སྔར་ཁྲོམ་རའི་ཐོག་སློབ་ཕྱིའི་ཀློག་དེབ་ཀྱི་ནང་དོན་ལ་ཡག་ཉེས་སྣ་ཚོགས་གནས་པའི་སྣང་ཚུལ་ཡོད་པས། ཚེས་2ཉིན་སློབ་གསོ་པུས་གསར་འགྱུར་བརྡ་གཏོང་ཚོགས་འདུ་འཚོགས་ནས། སློབ་གྲྭ་འབྲིང་ཆུང་གི་སློབ་མའི་སློབ་ཕྱིའི་ཀློག་དེབ་སློབ་གྲྭར་ཁྱབ་རྒྱུའི་ཐད“འགོག་ཟོན”བྱ་ཆེད། སློབ་གསོ་པུས《སློབ་གྲྭ་འབྲིང་ཆུང་གི་སློབ་མའི་སློབ་ཕྱིའི་ཀློག་དེབ་སློབ་གྲྭར་ཁྱབ་རྒྱུའི་དོ་དམ་བྱ་ཐབས》བཀྲམ་པའི་ནང་། ཚད་གཞིའི་བླང་བྱ་ཁ་གསལ་བཏོན་ནས་སློབ་ཕྱིའི་ཀློག་དེབ་སློབ་གྲྭར་ཁྱབ་པའི་བརྒྱུད་རིམ་ཚད་ལྡན་ཡོང་བ་བྱས་ཡོད།
          སློབ་གསོ་པུའི་བསླབ་གཞི་ཅུས་ཀྱི་ཅུས་ཀྲང་ཐན་ཧུད་ཧྲེང་གིས་ངོ་སྤྲོད་བྱས་པར་གཞིགས་ན། དོ་དམ་བྱ་ཐབས་ནང་། རྒྱལ་ཁབ་དང་། ས་གནས། སློབ་གྲྭ་བཅས་ཀྱིས་སློབ་ཕྱིའི་ཀློག་དེབ་སློབ་གྲྭར་ཁྱབ་པའི་བརྒྱུད་རིམ་ཁྲོད་ཀྱི་དོ་དམ་གྱི་འགན་འཁྲི་དང་དེའི་དོ་དམ་གྱི་དབྱེ་མཚམས་ཁ་གསལ་བཟོས་པ་དེས། སློབ་གྲྭར་ཁྱབ་རྒྱུའི་སློབ་ཕྱིའི་ཀློག་དེབ་འོས་སྦྱོར་བྱེད་དབང་སློབ་གྲྭར་སྤྲད་ནས་བརྒྱུད་ལམ་གཞན་ནས་སློབ་ཕྱིའི་ཀློག་དེབ་སློབ་གྲྭར་ཁྱབ་པར་དོ་དམ་བྱེད་པའི་སློབ་གྲྭའི་འགན་འཁྲི་ནན་དུ་གཏོང་རྒྱུ་རེད། དོ་དམ་བྱ་ཐབས་ནང་། ཁ་ཕྱོགས་རང་བཞིན་དང་། ཁྱོན་ཡོངས་རང་བཞིན། འཚམ་མཐུན་རང་བཞིན། སྣ་མང་རང་བཞིན། ཚད་ལ་རན་པའི་རང་བཞིན་བཅས་ཀྱི་འོས་སྦྱོར་རྩ་དོན་མཐའ་འཁྱོངས་བྱས་ཏེ། བརྗོད་དོན་ཁ་གསལ་བ་དང་། ནང་དོན་ཡར་སྐུལ་ལྡན་པ། ཀློག་རིན་ཆེ་བ། ཤེས་རབ་འཕེལ་བ་བཅས་ཀྱི་འོས་སྦྱོར་ཚད་གཞི་དང་མཐུན་པ་བྱེད་དགོས་པའི་བླང་བྱ་བཏོན་ནས་སློབ་གྲྭ་འབྲིང་ཆུང་གི་སློབ་མའི་སློབ་ཕྱིའི་ཀློག་དེབ་ལ་འོས་སྦྱོར་རམ་འདེམས་བྱེད་མི་ཆོག་པའི་ལྡོག་ཕྱོགས་ཀྱི་གསལ་ཐོ་12བཏོན་ཡོད་པ་རེད།
          ཐན་ཧུད་ཧྲེང་གིས། དོ་དམ་བྱ་ཐབས་ནང་། ད་དུང་སློབ་གྲྭར་ཁྱབ་པའི་སློབ་ཕྱིའི་སློབ་དེབ་འོས་སྦྱོར་གྱི་ནང་དོན་གཙོ་བོ་དང་གདམ་ག་ཆོག་པའི་འོས་སྦྱོར་གྱི་དཀར་ཆག་གི་ཁྱབ་ཁོངས་ཁ་གསལ་བཟོས་ཏེ། “ཐོག་མར་འདེམས་པ་དང་། གྲོས་ཞིབ་བྱེད་པ། གཏན་འཁེལ་བྱེད་པ། ཐོ་འགོད་བྱེད་པ་བཅས”ཀྱི་བརྒྱུད་རིམ་བཟོས་པ་མ་ཟད། སློབ་ཕྱིའི་ཀློག་དེབ་ཐད་རང་མོས་ཉོ་སྒྲུབ་ཀྱི་རྩ་དོན་མཐའ་འཁྱོངས་བྱེད་དགོས་པ་ཁ་གསལ་བཏོན་ནས་སློབ་གྲྭའི་ནང་སློབ་ཕྱིའི་ཀློག་དེབ་འཚོང་བའི་བྱེད་སྤྱོད་ཚད་ལྡན་དུ་བཏང་སྟེ། ཞལ་འདེབས་བརྒྱབ་པའི་ཀློག་དེབ་ལ་ཞིབ་དཔྱད་ཀྱི་འགག་སྒོ་ནན་པོ་སྲུང་དགོས་པ་མ་ཟད། འགག་དོགས་ཟས་སྤང་གི་ཕྱོགས་ལྷུང་སྔོན་འགོག་བྱ་ཆེད་དོ་དམ་གྱི་བྱ་ཐབས་ནང་ལྕག་སྐུལ་ལམ་སྲོལ་བཙུགས་ནས་སློབ་ཕྱིའི་ཀློག་དེབ་སློབ་གྲྭར་ཁྱབ་པའི་ཡར་སྐུལ་གྱི་ཁྲིད་སྟོན་འབོད་སྐུལ་ཤུགས་ཆེན་བྱེད་དགོས་པའི་བླང་བྱའང་བཏོན་ཡོད་ཅེས་ངོ་སྤྲད་བྱས་པ་རེད། 
རྩོམ་སྒྲིག་པ། བདེ་སྒྲོལ།
