ལྷོ་ཁའི་ལས་རིགས་ལག་རྩལ་སློབ་གྲྭ།
སློབ་མ་ཚོས་ས་གདན་འཐག་བཞིན་པ། 
རླངས་འཁོར་བསྐོར་ཚུལ་སྦྱོང་བྱེད་ཀྱི་ལད་ཟློས་འཕྲུལ་ཆས། 
གཞོན་ཤ་དར་བའི་ན་ཆུང་རྣམས་ཀྱིས་དྲར་འཕེན་སྤོ་ལོ་རྩེད་ཀྱིན་ཡོད་པ། 
ཚེས་༡༥ཉིན།  རྒྱལ་ཡོངས་ཀྱི་དྲ་བའི་སྨྱན་སྦྱོར་གཙོ་བོ་ཁག་བོད་ལྗོངས་སུ་བསྐྱོད་པའི་བྱེད་སྒོ་སྐབས་གསུམ་པའི་གསར་འགོད་ཚོགས་པས་ལྷོ་ཁ་ས་ཁུལ་གྱི་ལས་རིགས་ལག་རྩལ་སློབ་གྲྭར་བཅར་འདྲི་བྱས་ཤིང། ཤེས་རྟོགས་བྱུང་བ་ལྟར་ན། སློབ་གྲྭ་དེར་ད་ཆ་སློབ་མ་༤༢༡༣ཡོད་ཅིང་། ཡུན་རིང་སྦྱོང་ཚན་རིགས་༢༠ཙམ་དང་། ཡུན་ཐུང་སྦྱོང་ཚན་རིགས༡༠ལྷག་ཡོད་པ་དང་། ད་སྔའི་བར་ལག་རྩལ་ཤེས་ལྡན་པ་༨༠༠༠ལྷག་གསོ་སྐྱོང་གནང་བ་མ་ཟད། ད་དུང་ཁོང་ཚོར་ལས་ཀའང་བཙལ་རོགས་བྱས་པ་རེད། 
　　སློབ་མས་ཁོང་དུ་ཆུད་པའི་ཤེས་བྱ་རྣམས་བེད་སྦྱོད་ལེགས་པོ་གཏོང་ཆེད། སློབ་གྲྭས་ཐབས་ཤེས་མི་ཉུང་བར་བརྟེན་ནས་བོད་ལྗོངས་སློབ་གྲྭ་ཆེན་མོ་དང་། བོད་སྨན་སློབ་གླིང་། གསོལ་མགྲོན་ཁང་ཆེ་ཁག་བཅས་དང་མཉམ་འབྲེལ་བྱས་ཏེ་ཐོན་ཁུངས་མཉམ་སྤྱོད་བྱེད་བཞིན་ཡོད་པ་དང་། དེ་མ་ཟད། རླངས་འཁོར་གསོ་བཅོས་བྱེད་ཚུལ་སྦྱོང་བཞིན་པའི་སློབ་མ་རྣམས་ལོ་རེ་བཞིན་དགེ་རྒན་གྱིས་ཁྲིད་དེ་གཞི་རིམ་དུ་ཕྱིན་ནས་ཞིང་འབྲོག་མི་སེར་རྣམས་ལ་འདྲུད་འཐེན་འཁོར་ལོ་དང་། འཕྲུལ་ཆས་གཞན་པ་སོགས་གསོ་བཅོས་བྱེད་བཞིན་ཡོད་ལ། ལག་ལེན་དངོས་ལ་བརྟེན་ནས་རང་ཉིད་སྦྱོང་བརྡར་བྱེད་བཞིན་ཡོད་པ་རེད། 
སློབ་མར་སློབ་སྦྱོང་དང་ལག་ལེན་གཉིས་ལྡན་དགོས་པ་ཕུད། ད་དུང་སློབ་སྦྱོང་བྱེད་པའི་བརྩོན་སེམས་ཤིག་ཅིས་ཀྱང་ཡོད་དགོས་པའི་གནས་ཚུལ་ལ་གཞིགས་ནས། སློབ་གྲྭས་སློབ་མ་རྣམས་ལ་ཆེད་དུ་སློབ་མའི་བརྩམས་ཆོས་བཤམས་སྟོན་ཁང་གཉིས་བཙུགས་ཏེ། སློབ་མའི་ཕུལ་བྱུང་བརྩམས་ཆོས་རྣམས་གཞན་ལ་སྟོན་བཞིན་ཡོད་པ་མ་ཟད། ད་དུང་བརྩམས་ཆོས་རྣམས་འབྲེལ་ཡོད་འགྲན་བསྡུར་དུ་ཞུགས་པ་དང་། ཐ་ན་བརྩམས་ཆོས་རེ་འགའ་གཅེས་ཉར་བྱེད་པ་སོགས་བརྒྱུད་སློབ་མའི་སློབ་སྦྱོང་གི་བརྩོན་སེམས་དང་གསར་གཏོད་བྱེད་པར་སྐུལ་མ་གནང་བ་དང་། ཐམས་ཅད་སློབ་མའི་ཆེད་དུ་ཞེས་པའི་སློབ་གྲྭའི་དམིགས་ཡུལ་རིམ་བཞིན་སྒྲུབ་ཀྱིན་ཡོད་པ་རེད།
