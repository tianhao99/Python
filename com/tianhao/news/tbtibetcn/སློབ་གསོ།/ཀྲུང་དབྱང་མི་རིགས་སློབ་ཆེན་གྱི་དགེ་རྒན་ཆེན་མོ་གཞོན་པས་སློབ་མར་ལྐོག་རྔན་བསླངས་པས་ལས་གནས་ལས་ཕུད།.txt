ཟླ ༦ ཚེས ༢༥ ཉིན།  ཀྲུང་དབྱང་མི་རིགས་སློབ་ཆེན་གྱིས་སཱ་རུའུ་ལཱ་ཡིས་ལྐོག་ངན་བསླངས་པར་ཐེར་འདོན་གོང་ཞུ་ཐེབས་པ་དེར།  སྐྱོན་ཐོ་འགོད་པ་དང་ལས་གནས་ལས་ཕུད་པའི་ཆད་པ་བཅད་པ་ཁྱབ་བསྒྲགས་བྱས་ལ།  ད་ལོའི་ལོ་མགོར།  སློབ་གྲྭ་དེའི་སློབ་མས་སཱ་རུའུ་ལཱར་རྩད་གཅོད་ཅེས་པ་སྐད་འཕྲིན་དུ་རྒྱ་ཆེར་སྤེལ་ཞིང་།  དཔལ་འབྱོར་སློབ་གླིང་གི་དགེ་རྒན་ཆེན་མོ་གཞོན་པ་སཱ་རུའུ་ལཱ་ཡིས་བཙན་གྱིས་སློབ་མར་དཔེ་ཆ་ཉོ་རུ་འཇུག་པ་དང་།  དུ་བ་སློང་བ།  གཞན་ད་དུང་འབྲེལ་ཡོད་ཀྱི་པར་སོགས་བཀོད་དེ་ཁོས་སྤྱོད་སྟངས་ལ་ཐེར་འདོན་ནན་མོ་བྱས།
           སློབ་མས་ཐེར་འདོན་གོང་ཞུ་བྱས་རྗེས།  ཟླ ༡ ཚེས ༩ ཉིན།  སློབ་གྲྭའི་དྲིལ་བསྒྲགས་པུའི་འབྲེལ་ཡོད་འགན་ལེན་མི་སྣས་པེ་ཅིང་ན་གཞོན་ཚགས་པར་གྱི་གསར་འགོད་པར་སཱ་རུའུ་ལཱ་ཡི་ལས་མཚམས་གནས་སྐབས་ཀྱིས་བཞག་ཡོད་པ་བསྒྲགས་ལ།  སློབ་གླིང་གི་བརྟག་དཔྱད་ལ་བསྟུན་དགོས་པར་སྤེལ།  ཁ་སང་།  སློབ་གྲྭས《བློ་མཐུན་སཱ་རུའུ་ལཱ་ཡིས་ལྐོག་རྔན་བླངས་པ་དང་།  དཔེ་ཆ་བཙོང་བ་བཅས་ཀྱི་གནད་དོན་རྩད་ཞིབ་བྱས་པའི་བརྡ་ཁྱབ》སྤེལ་ལ། བརྡ་ཁྱབ་ཏུ།  རྩད་ཞིབ་བྱས་པར་གཞིགས་ན།  བློ་མཐུན་སཱ་རུའུ་ལཱ་ཡིས ༢༠༡༣ ལོའི་ཟླ ༩ བ་ནས ༢༠༡༥ ལོའི་ཟླ ༡ པའི་བར།  སློབ་ཁྲིད་ཀྱི་དཔེ་དེབ་དང་དཔྱད་གཞི་ཡིག་ཆ་མ་ཡིན་པར་སློབ་མར་རང་ཉིད་ཀྱི་དཔེ་ཆ་བཙོང་བ་དང་།  སློབ་མའི་ལྐོག་རྔན་བླངས་པས།  སློབ་གྲྭའི་སློབ་གཙོའི་གཞུང་བསྒྲུབ་ཚོགས་འདུས ༢༠༡༥ ལོའི་ཟླ ༥ བའི་ཚེས ༡༤ ཉིན་གྱི་ཐག་གཅོད་ལྟར་ན།  སཱ་རུའུ་ལཱ་སྐྱོན་ཐོར་འགོད་པ་དང་།  ལས་གནས་ལས་ཕུད་པའི་ཆད་པ་གཅོད་དགོས་པ་བསྒྲགས།
          སཱ་རུའུ་ལཱ་ནི་ཀྲུང་དབྱང་མི་རིགས་སློབ་ཆེན་དཔལ་འབྱོར་སློབ་གླིང་གི་དགེ་རྒན་ཆེན་མོ་གཞོན་པ་དང་།  ལོ་གང་མང་ལ་མི་ཚེའི་ལས་གནས་འཆར་འགོད་དང་།  མི་ཤུགས་ཀྱི་འབྱུང་ཁུངས་གསར་འབྱེད་དང་དོ་དམ།  མི་གྲངས་དཔལ་འབྱོར་རིག་པ།  བོད་ལྗོངས་ཀྱི་མི་གྲངས་དང་དཔལ་འབྱོར་གནད་དོན་ཐད་ཀྱི་ཞིབ་འཇུག་དང་ལག་ལེན་གྱི་ལས་བསྒྲུབ་མཁན་ཡིན་ལ།   ཀྲུང་གོའི་མི་ཚེའི་ལས་གནས་འཆར་འགོད་ཀྱི་མི་དང་པོ་དང་།  བསམ་པའི་གཙོ་འཛིན་དུ་གྲགས།
          ད་ལོའི་ཟླ ༡ པའི་ཚེས ༥ ཉིན།  སློབ་གྲྭའི་སློབ་མ་མང་པོས་དྲ་འཕྲིན་དང་།  སྐད་འཕྲིན་སོགས་བརྒྱུད་ཆ་འཕྲིན་དང་།  འདྲ་པར་གང་མང་སྤེལ་ཏེ།  སཱ་རུའུ་ལཱ་ཡིས་ཐེངས་མང་དུ་སློབ་མར་རང་གི་དཔེ་ཆ་བཙོང་བ་དང་།  ལྐག་རྔན་བསླངས་ཤིང་།  ཁོའི་ཚོགས་པར་ཞུགས་པར་ལོ་རེར་སྒོར ༢༠༠ དང ༤༠༠ སྤྲད་ན་ད་གཟོད VIP རུ་འགྱུར་ཐུབ།
           ཟླ ༡ པའི་ཚེས ༧ ཉིན་ནས་བཟུང་།  སློབ་མ་མང་པོས་སྔ་གཞུག་ཏུ་སློབ་གྲྭའི་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཨུ་ཡོན་ལྷན་ཁང་དུ་སཱ་རུའུ་ལཱ་གཏུག་བཤེར་བྱས་ལ།  ཁོ་རང་གིས་བརྩམས་པའི 《འཆར་གཞི་མེད་པའི་མི་མཐར་ཕམ་ངེས་ཡིན།  སཱ་རུའུ་ལཱ་ཡིས་ན་གཞོན་ཚོར་སྐུལ་བའི་བསླབ་བྱ》ནི་རྒྱུགས་གཞིའི་ནང་དོན་ཡིན་པར་བསྒྲགས་ནས་སློབ་མ་ཚོར་བཙན་གྱིས་ཉོ་རུ་བཅུག་པ་དང་།  སྒོར ༢༨ ཡིན་འདུག  འོན་ཀྱང་སློབ་ཁྲིད་དང་རྒྱུགས་གཞིའི་ཁྲོད།  འབྲེལ་ཡོད་ཀྱི་ནང་དོན་གཅིག་ཀྱང་མེད་པ་རེད།  ཡང་སློབ་མ་ལ་ལས་སཱ་རུའུ་ལཱར་སྒོར ༡༠༠༠ གི་ལེགས་སྐྱེས་ཉོས་པའི་བདེན་དཔང་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཨུ་ཡོན་ལྷན་ཁང་དུ་སྐྱེལ་ལ།  ཁོས་དཔྱད་རྩོམ་དག་བཅོས་བྱེད་པར་ངེས་པར་དུ་སྔོན་ལ་ལེགས་སྐྱེས་སྟེར་དགོས་ཟེར།
          སློར་མས་ཐེར་འདོན་གོང་ཞུ་བྱས་པའི་དོན་རྐྱེན་ཐད།  ཁོས་དྲ་འཕྲིན་ལས་བཤད་རྒྱུར།  ཀྲུང་གོར་མི་ཆོས་རིང་ལུགས་རབ་ཏུ་དར་བས།  ལེགས་སྐྱེས་སྟེར་བ་ནི་དྲིན་དྲན་པ་དང་རང་འཐད་ཡིན་ཟེར།  འོན་ཀྱང ༢༠༡༣ ལོའི་དགེ་རྒན་དུས་ཆེན་ཐོན་ལ་ཉེ་དུས།  ཁྱིམ་བདག་གིས་དགེ་རྒན་ལ་ལེགས་སྐྱེས་སྟེར་བར་དམིགས་ནས།  ཁོས་ཀྲུང་གོ་དྲ་བའི་བཅར་འདྲི་བྱེད་སྐབས་བཤད་རྒྱུར།  དེ་ནི་དངོས་ཡོད་ཀྱི་སྤྱི་ཚོགས་སྣང་ཚུལ་དང་།  དུས་ཡུན་རིང་བོ་བརྒྱུད་གྲུབ་པའི་སྲོལ་ངན་ཞིག་རེད།  དགེ་རྒན་ཞིག་ལ་མཚོན་ན་སློབ་མའི་རྒྱུགས་འབྲས་ལེགས་པོ་ནི་ལེགས་སྐྱེས་ཆེས་མཆོག་པ་རེད།  དགེ་རྒན་ཡིན་ཕྱིན།  ངེས་པར་དུ་ཀུན་སྤྱོད་ལེགས་པོ་དང་ལྡན་དགོས་ཟེར།
      ཁ་བྱང་ཕྱི་མོ། མི་རིགས་སློབ་ཆེན་གྱི་དགེ་རྒན་ཆེན་མོ་གཞོན་པས་སློབ་མར་ལྐོག་རྔན་བསླངས་པས་ལས་གནས་ལས་ཕུད།
