          བྱེ་བྲག་ཏུ། སློབ་མ་རྣམས་དངོས་ལུགས་ཀྱི་དངོས་བཤེར་ལ་སྤྲོ་བ་ཡོད་མེད་ལ་དཔྱད་དུས། (གཤམ་བཀོད་རེའུ་མིག་ལྟར) སློབ་མ90%དངོས་བཤེར་ལ་དགའ་ཞིང་།“དངོས་བཤེར་གྱི་ནང་དོན་མང་པོ་ཞིག་འཚོ་བ་དངོས་ལ་འབྲེལ་ཡོད་ཅིང་། གནད་དོན་དངོས་ཐག་གཅོད་པའི་གྲོགས་སུ་འགྲོ”ཞེས་པ་དགའ་བའི་རྒྱུ་མཚན་དུ་འཛིན་པའི་སློབ་མ38དང་དེས54%ཟིན་ཞིང་།“རང་ལག་འགུལ་བར་དགའ”ཞེས་པ་དགའ་བའི་རྒྱུ་མཚན་དུ་འཛིན་པའི་སློབ་མ28དང་དེས40%ཟིན། དེ་ལས་སྤྱིར་བཏང་དངོས་ལུགས་རིག་པ་སྦྱོང་དཀའ་བ་ཞིག་ཡིན་ནའང་། སློབ་མ་རྣམས་དེ་ལ་སྤྲོ་བ་གཏན་ནས་མེད་པ་མ་ཡིན་པར་དེ་ལས་ལྡོག་སྟེ་སྤྲོ་བ་ཡོད་པའི་སློབ་མ་མང་ཤོས་རེད།ངས་ལོ་མང་པོ་ཞིག་གི་རིང་ལ་སློབ་ཁྲིད་བྱས་པའི་མཉམ་མྱོང་ལྟར་ན། སློབ་མ་རྣམས་དངོས་ལུགས་ལ་སྤྲོ་བ་མི་དམན་པ་ཡོད་དེ། ཁོ་ཚོའི་ཚོར་བ་ཞིག་ལ་དངོས་ལུགས་ཀྱི་ཤེས་བྱ་ནི་དུས་རྒྱུན་གྱི་འཚོ་བ་དང་འབྲེལ་ཐུབ་པ་ལས་གཞན་ངོ་མཚར་བའི་ཤེས་བྱ་སྐྱིད་པོ་ཡོད་པ་དང་། དངོས་བཤེར་སྒྲུབ་པའི་སྐབས་སུ་རང་ལག་འགུལ་བར་དགའ་ཞིང་། ལྷག་པར་དུ་རིག་པ་གསལ་བའི་སློབ་མ་མང་ཆེ་ཤོས་དངོས་ལུགས་སྦྱོང་བར་སྤྲོ་བ་ཤིན་ཏུ་ཡོད།

          སློབ་ཁྲིད་ཅིག་ལ་མཚོན་ན་དགེ་རྒན་དང་སློབ་མ། སློབ་དེབ་གསུམ་ནི་མེད་མི་རུང་བའི་རྒྱུ་གསུམ་ཡིན། དེས་ན་སློབ་མ་ཚོའི་གྲངས་རིག་གི་རྨང་གཞི་བཟང་མིན་དེ་དགེ་རྒན་གྱི་གྲངས་རིག་ཆུ་ཚད་ལ་འབྲེལ་བ་དམ་པོ་ཡོད། གཤམ་དུ་བཀོད་པའི་རེའུ་མིག་ནི་རྔ་བ་ཁུལ་གྱི་སློབ་གསོ་ཅུའུ་ཡིས་བསྒྲུབས་པའི་གསོ་སྦྱོང་འཛིན་གྲྭའི་དགེ་རྒན48ལ་སློབ་ཆུང་གི་གྲངས་རིག་རྒྱུགས་བླངས་པའི་རྒྱུགས་འབྲས་ཡིན། རྒྱུགས་གཞི་དེ་ནི་སྡོམ་སྐར150ཅན་གྱི་རྒྱུགས་གཞི་ཞིག་ཡིན་ལ་སྐར་མ90ལ་ལོངས་ན་རྒྱུགས་འཕྲོད་ཡོད། རྒྱུགས་འབྲས་དེ་ལས་ཐོབ་སྐར་ཚད་ཆེས་དམའ་བ་ནི་ཁ་སྐོང་དྲི་གཞི་སྟེ39%དང་། དེའི་རྗེས་ནས་འགྲོལ་ལན་འདེབས་པའི་དྲི་གཞི་སྟེ53%ཡིན་པ་ཤེས་ཐུབ། དཔེ་རིས་འབྲི་བ་དང་གདམ་ང་གཉིས་ཀྱི་སྐར་མའི་ཐོབ་ཚད་ཅུང་ཙམ་མཐོ། གནས་ཚུལ་དེ་ལྟ་བུ་བྱུང་བའི་རྒྱུ་མཚན་ནི་ཁ་སྐོང་ནི་གཅིག་ཡིན་ན་གཅིག་དང་གཉིས་ཡིན་ན་གཉིས་ཡིན་དགོས་ཤིང་། འགྲོལ་ལན་དྲི་གཞི་ལ་རྩིས་རྒྱག་རྒྱུ་མང་པས་རྨང་གཞི་མ་བཟང་ན་སྐར་མ་འཐོབ་དཀའ། ཐེངས་འདིའི་རྒྱུགས་ལ་ཞུགས་པའི་མི48ནང་དུ་མི13མ་གཏོགས་ཚད་གཞིར་མ་ལོངས་ཤིང་། དེས་རྒྱུགས་ཞུགས་མི་གྲངས་ཀྱི27%མ་གཏོགས་ཟིན་མེད་པས། དཔྱོད་རྒྱུགས་དེ་ལས་དགེ་རྒན་རྣམས་ཀྱི་གྲངས་རིག་གི་ཆུ་ཚད་དམའ་བ་མཚོན་ཡོད། (གཤམ་བཀོད་རེའུ་མིག་ལྟར)

          “དངོས་ལུགས”ཞེས་པ་འདི་ནི་ཐོག་མར་ཀེ་རི་སེའི་ཡི་གེ“φυσικ”ལས་བྱུང་ཞིང་། དེ་རང་བྱུང་ལ་བསྟན་པ་ཡིན། རྗེས་སུ་ཡོ་རོབ་གླིང་གིས་དངོས་ལུགས་རིག་པར“རང་བྱུང་མཚན་ཉིད་རིག་པ”ཞེས་ཟེར། དངོས་ལུགས་རིག་པ་ཟེར་བའི་མིང་ནི་ལྗར་ཡིག་དང་རྒྱ་ཡིག་ལས་ཚུར་བསྒྱུར་བ་ཡིན། ད་ལྟ། དངོས་ལུགས་རིག་པ་ནི་རང་བྱུང་ཚན་རིག་གི་རྨང་གཞིའི་རིག་ཚན་གཙོ་བོ་ཞིག་ཡིན་ཞིང་། དངོས་ལུགས་རིག་པའི་ཆོས་ཉིད་ཀྱི་བདེན་པའི་རང་བཞིན་ནི་གཟབ་ནན་གྱི་དངོས་བཤེར་འབོར་ཆེན་དང་གྲངས་རིག་པའི་ཐབས་ཤེས་ལ་བརྟེན་ནས་ཡང་དང་བསྐྱར་དུ་ར་སྤྲོད་པ་དང་མཚོན་བཞིན་ཡོད། གཤམ་དུ་སློབ་འབྲིང་གི་དངོས་ལུགས་སྦྱོང་བའི་ཐད་ལ་གྲོས་གཞི་འགའ་བཏོན་ན།
          ༡. སློབ་ཆུང་སྐབས་ནས་བཟུང“དངོས་ལུགས”ཀྱི་རྨང་གཞི་འཇོག་ཚུལ་སྣ་ལྔ།
          གྲངས་རིག་ནི་རྩིས་ཚན་གྱི་རིག་ཚན་གཞན་པ་སྦྱོང་བའི་རྨང་གཞི་ཡིན་པར་མ་ཟད། དངོས་ལུགས་སྦྱོང་བའི་རྟེན་གཞི་གཙོ་བོ་ཡིན་ཞིང་དངོས་ལུགས་སྦྱོང་ན་གྲངས་རིག་པ་དང་ཁ་འབྲལ་ཐབས་མེད་དེ། གྲངས་རིག་པ་དང་དངོས་ལུགས་རིག་པ་གཉིས་ནི་ཁྱིམ་ཚང་གཅིག་གི་བཟའ་མི་ཡིན་ཞེས་བཤད་སྲོལ་ཡོད། གལ་སྲིད་སློབ་མར་དངོས་ལུགས་ཀྱི་རྨང་གཞི་བཟང་བོ་ཞིག་འཇོག་དགོས་ན་ཐོག་མར་གྲངས་རིག་གི་རྨང་གཞི་བཟང་བོ་ཞིག་ཡོད་དགོས་ཤིང་། སློབ་མ་ལ་གྲངས་རིག་གི་རྨང་གཞི་བརྟན་པོ་ཞིག་འཇོག་ན་སློབ་ཆུང་གི་སྐབས་ནས་བཟུང་མགོ་འཛིན་དགོས་པ་ཡིན་ལ། དེ་ད་དུང་དགེ་རྒན་དང་འབྲེལ་བ་དམ་པོ་ཡོད་པས། ཐོག་མར་དགེ་རྒན་ལ་གྲངས་རིག་གི་རྨང་གཞི་བཟང་བོ་ཞིག་ཡོད་རྒྱུ་ནི་ཁ་ཚ་དགོས་གཏུག་གི་དོན་ཞིག་ཡིན་ཞིང་། དགེ་རྒན་ལ་གྲངས་རིག་གི་རྨང་གཞི་བཟང་བོ་ཞིག་ཡོད་མེད་དེ་མི་རང་ཉིད་ལས་གཞན་ད་དུང་ས་གནས་ཀྱི་སློབ་གསོའི་ལས་ཁུངས་ལ་འབྲེལ་བ་དམ་པོ་ཡོད། གཤམ་ནས་དེའི་སྐོར་རགས་ཙམ་གླེང་བར་བྱ།
          གཅིག་ནི་ས་གནས་སོ་སོའི་རྫོང་རིམ་པ་དང་།ཁུལ་རིམ་པའི་སློབ་གསོ་ཅུའུ་དང་སློབ་དཔྱོད་ཁང་།དགེ་བསྐྱེད་ཁང་སོགས་སློབ་གསོའི་ལས་ཁུངས་ཀྱིས་སློབ་ཆུང་གི་གྲངས་རིག་དགེ་རྒན་རྣམས་ལ་དུས་ཡུན་རིང་ཐུང་གཉིས་ཀྱི་གསོ་སྦྱོང་གི་འཛིན་གྲྭ་འཛུགས་པ་དང་། སློབ་སྦྱོང་གི་མཇུག་རྫོགས་དུས་ནན་ཏན་སྒོས་རྒྱུགས་བླངས་ཏེ་རྒྱུགས་འཕྲོད་མིན་ལྟར་མཐར་ཕྱིན་འཛིན་ཡིག་བྱིན་ན་དགེ་རྒན་ཚོས་རང་ཤུགས་སྒོས་སློབ་སྦྱོང་ལ་ནན་ཏན་བྱེད་ངེས་པས་གསོ་སྦྱོང་གི་ལས་ཀ་དེ་ཆུད་ཟོས་སུ་མི་འགྲོ། གཉིས་ནི་དགེ་རྒན་ཚོས་སློབ་དེབ་ཀྱི་ཤེས་བྱ་ཁོང་དུ་ཆུད་པའི་ཚད་ལ་དཔྱད་པའི་ཆེད་དུ་ལོ་རེ་རེར་སློབ་དེབ་ཀྱི་ཆུ་ཚད་དཔྱད་རྒྱུགས་ལེན་དགོས་ཤིང་། རྒྱུགས་ལེན་པའི་འཆར་གཞི་འགོད་དུས་མ་མཐའ་ཡང་ལོ་ཕྱེད་ཀྱི་སྔོན་དུ་རྒྱུགས་ལེན་པའི་བརྡ་ཐོ་གཏོང་དགོས་ཤིང་། དེ་ལྟར་བྱེད་དོན་ནི་དགེ་རྒན་རྣམས་ལ་རང་སྦྱོང་བྱེད་པའི་དུས་ཚོད་བྱིན་པ་ཡིན། རྒྱུགས་ལེན་པའི་དམིགས་ཡུལ་ཡང་རྒྱུགས་རང་ཉིད་མ་ཡིན་པར་དགེ་རྒན་རྣམས་ཀྱིས་རང་འགུལ་གྱིས་དཔེ་ཆར་བལྟ་བར་སྐུལ་མ་གཏོང་བའི་ཆེད་དུ་ཡིན་ཞིང་། དེ་ལྟར་བསྒྲུབས་ན་དགེ་རྒན་ཚོའི་གྲངས་རིག་གི་ཆུ་ཚད་ཇེ་བཟང་དུ་འགྲོ་བར་ཕན་ཐོགས་ཆེན་པོ་ཡོད། གསུམ་ནི་གྲངས་རིག་པའི་བགྲོ་གླེང་ཚོགས་འདུ་དང་ཟབ་སྦྱོང་འཛིན་གྲྭར་ཞུགས་ཏེ་སློབ་ཁྲིད་མཉམ་མྱོང་བརྗེ་རེས་དང་འཁྲིད་ཐབས་གསར་བར་སློབ་སྦྱོང་བྱེད་པ་དང་། གྲངས་རིག་པའི་གལ་ཆེན་རང་བཞིན་སོགས་བགྲོ་གླེང་བྱས་ནས་གྲངས་རིག་པའི་ངོས་འཛིན་ཇེ་ཟབ་ཏུ་གཏོང་དགོས། བཞི་ནི་ལོ་རེའི་མཇུག་རྒྱུགས་སྐབས་སུ་གྲངས་རིག་རྒྱུགས་འབྲས་ལ་དབྱེ་ཞིབ་བྱས་ཏེ། བཟང་ན་བཟང་བའི་རྒྱུ་མཚན་དང་ངན་ན་ངན་པའི་རྒྱུ་རྐྱེན་ལ་དཔྱད་པ་བཏང་ནས་གྲོས་གཞི་འདོན་དགོས། ལྔ་ནི་བྱིས་པ་རྣམས་སློབ་ཆུང་ལོ་རིམ་དང་པོ་ནས་གྲངས་རིག་གི་རྨང་གཞིའི་ཤེས་བྱ་ལ་འཇུག་པའི་སྐབས་སུ་རང་གི་ཕ་སྐད་དེ་བོད་ཡིག་བོད་སྐད་ཀྱིས་ཁྲིད་ན་བཟང་སྟེ། ཕ་སྐད་ནི་སློབ་མ་རང་གི་ཕ་མས་བཤད་པའི་སྐད་ཆ་དང་། རང་ཉིད་ཆེས་རྒྱུས་ཆེ་ཞིང་ཆེས་གོ་སླ་བའི་སྐད་ཆ་ཡིན་པས། བོད་ཡིག་བོད་སྐད་ཀྱིས་ཁྲིད་ན་སློབ་མས་གྲངས་རིག་གི་ཤེས་བྱ་ཐད་ཀར་རྒྱུད་ལ་ལེན་ཐུབ།
          ༢. སློབ་འབྲིང་གི་དངོས་ལུགས་ནི་ཆེད་ལས་དགེ་རྒན་གྱིས་འཁྲིད་དགོས་པའི་གལ་ཆེན་རང་བཞིན།
          དམའ་འབྲིང་གི་ལོ་རིམ་གཉིས་པ་ནས་བཟུང་དངོས་ལུགས་རིག་པའི་སློབ་ཁྲིད་ཀྱི་མགོ་ཚུགས་ཤིང་།དམའ་འབྲིང་དང་མཐོ་འབྲིང་གཉིས་ཀྱི་དུས་ཡུན་ལོ་ལྔའི་རིང་དངོས་ལུགས་རིག་པའི་འཁྲིད་ཚན་ཡོད། དེའི་ནང་ལ་དངོས་ལུགས་རིག་པའི་ཁོངས་སུ་གཏོགས་པའི་སྒྲ་རིག་པ། འོད་རིག་པ། ཚ་བ་རིག་པ། ཤུགས་རིག་པ། གློག་སྡུད་རིག་པ། མ་རྡུལ་གཞུང་ལུགས་སོགས་ནང་དོན་འདུས་ཡོད་པར་མ་ཟད། ནང་དོན་སོ་སོར་འབྲེལ་ཡོད་ཀྱི་དམར་ཁྲིད་དང་དངོས་བཤེར་ཡོད། ལྷག་པར་དུ་མཐོ་འབྲིང་གི་དངོས་ལུགས་ལ་ནང་དོན་ཤིན་ཏུ་མང་ཞིང་ཅུང་གཏིང་ཟབ་མོ་ཡིན་པ་དང་། འབྲེལ་ཡོད་ཀྱི་དངོས་བཤེར་ཅུང་མང་ཞིང་རྙོག་དྲ་ཆེ།ལྷག་པར་དུ་མཐོ་འབྲིང་གི་ཤུགས་རིག་པ་ལ་གྲངས་རིག་པའི་རང་བཞིན་ཤུགས་ཆེར་ལྡན་ཏེ། རྩ་བའི་ཤུགས་གསུམ་དང་། ཉིའུ་ཏུན་གཏན་སྲོལ། འཁྱོག་ཐིག་འགུལ་སྐྱོད། ཀུན་ལྡན་འགུག་ཤུགས་གཏན་སྲོལ། ཁོའུ་ལོམ་གཏན་སྲོལ་སོགས་ནང་དོན་ལ་གོ་དོན་ཚུལ་དང་ངེས་གཏན་ཚུལ་མང་པོ་ཡོད། གལ་སྲིད་དགེ་རྒན་ལ་གྲངས་རིག་གི་རྨང་གཞི་བཟང་བོ་ཞིག་མེད་ན་ནང་དོན་དེ་རིགས་འཁྲིད་རྒྱུ་ཤིན་ཏུ་ཁག་པོ་ཡིན། དངོས་ལུགས་ཆེད་ལས་ཡིན་པའི་དགེ་རྒན་ཞིག་ཡིན་ན་སློབ་འབྲིང་སྐབས་སུ་ནང་དོན་དེ་རིགས་ལ་སྦྱོང་བ་བྱས་ཡོད་པ་གཞིར་བྱས་ཏེ། སློབ་ཆེན་གྱི་སྐབས་སུ་བྱེ་བྲག་ཏུ་ཤུགས་རིག་པ་དང་། ཚ་བའི་རིག་པ། འོད་རིག་པ། གློག་སྡུད་རིག་པ། དངོས་ལུགས་དཀྱུས་མའི་དངོས་བཤེར་སོགས་སློབ་དེབ་ཆེད་ཁྲིད་བྱས་ཡོད་པར་མ་ཟད། མཐོ་རིམ་གྲངས་རིག་ཀྱང་ངེས་པར་དུ་སྦྱང་དགོས་པའི་ཆེད་དེབ་ཡིན། གང་ལྟར་ཆེད་ལས་ཤིག་ཡིན་ན་ལོ་གང་མང་ལ་ནང་དོན་དེ་རིགས་ལ་སྦྱོང་བ་བྱས་ཡོད་པས་བག་ཆགས་ཟབ་མོ་ཞིག་བཞག་ཡོད་ངེས་ཡིན་ལ། སློབ་འབྲིང་གི་དངོས་ལུགས་ཁྲིད་ན་བཟང་ཆ་མང་པོ་ལྡན། གལ་སྲིད་ཆེད་ལས་མིན་པའི་དགེ་རྒན་ཞིག་ཡིན་ན་དམའ་འབྲིང་ལས་མཐོ་འབྲིང་གི་དངོས་ལུགས་བསླབ་མ་མྱོང་ཞིང་། (སློབ་གྲྭ་མང་ཤོས་ཀྱིས་མཐོ་འབྲིང་ལོ་རིམ་དང་པོ་ནས་བཟུང་རྩིས་ཚན་དང་རིག་ཚན་ཕྱེས་ཡོད)ལྷག་པར་དུ་སློབ་ཆེན་སྐབས་ཀྱི་དངོས་ལུགས་ཀྱི་ཤེས་བྱ་དང་དེ་མིན་མཐོ་རིམ་གྲངས་རིག་གི་རྨང་གཞི་དེ་འདྲ་མང་པོ་ཞིག་སྟོང་ཆར་ལུས་ཡོད་པས། སློབ་འབྲིང་གི་དངོས་ལུགས་ཁྲིད་ན་ཐོག་མར་དགེ་རྒན་རང་ཉིད་ཀྱིས་ནང་དོན་དེ་རིགས་སློབ་དགོས་པས། དེའི་རིང་ལ་སེམས་ཤུགས་གཙོ་བོ་དེའི་ཕྱོགས་ལ་བཏོན་ཚེ་སློབ་མ་ཆུད་ཟོས་སུ་གཏོང་བའི་ཉེན་ཁ་ཤིན་ཏུ་ཆེ།
          གཞན། ཤུགས་རིག་པ་དང་གློག་སྡུད་རིག་པ། འོད་རིག་པ་སོགས་ལ་དམར་ཁྲིད་དང་། རྩད་ཞིབ། དངོས་བཤེར་སོགས་མང་པོ་ཡོད་དེ། འགྲོས་ཚད་སྙོམས་འགྱུར་དྲང་ཐིག་འགུལ་སྐྱོད་ལ་ཞིབ་འཇུག་བྱེད་པའི་དངོས་བཤེར་དང་། ཉིའུ་ཏུན་འགུལ་སྐྱོད་གཏན་སྲོལ་ར་སྤྲོད་བྱེད་པའི་དངོས་བཤེར། གློག་གེགས་ཕྱོད་འཇལ་གཅོད་བྱེད་པའི་དངོས་བཤེར། གློག་ཁུངས་ཀྱི་གློག་སྒུལ་སྟོབས་དང་ནང་གེགས་འཇལ་གཅོད་བྱེད་པའི་དངོས་བཤེར། གློག་ཏོག་ཆུང་ངུའི་ཝོལ་ཨམ་ཁྱད་གཤིས་འཁྱོག་ཐིག་འབྲི་བའི་དངོས་བཤེར་སོགས་ནི་ཆེད་ལས་ཀྱི་རང་བཞིན་ཤིན་ཏུ་དྲག་པའི་དངོས་བཤེར་ཡིན་པས། དངོས་བཤེར་གྱི་ལག་ལེན་མཉམ་མྱོང་ཅུང་ཕུན་སུམ་ཚོགས་པོ་ཡོད་པའི་ཆེད་ལས་དགེ་རྒན་གྱིས་ཁྲིད་ན་ད་གཟོད་ཁྲིད་འབྲས་ལེགས་ངེས་ཡིན། དེ་མིན་པར་ཆེད་ལས་མ་ཡིན་པའི་དགེ་རྒན་ཆེ་གེ་མོས་ཁྲིད་ན་དངོས་བཤེར་ཚོགས་ཐུན་སྐབས་སུ། དངོས་བཤེར་སྒྲུབ་པའི་ཐབས་ཤེས་དང་།ཡོ་བྱད་བཀོལ་སྤྱོད་བྱེད་སྟངས། ལྟ་ཞིབ་བྱེད་སྟངས། དངོས་བཤེར་སྣང་ཚུལ་ལ་དབྱེ་ཞིབ་བྱེད་ཚུལ། ཟིན་བྲིས་འགོད་ཚུལ། དངོས་བཤེར་གཞི་འཛིན་གྲངས་ཐག་གཅོད་ཚུལ། དཔྱད་འཇོག་བྱེད་པ་སོགས་གང་ཐད་ལ་ནོར་འཁྲུལ་གང་མང་ཞིག་འབྱུང་སྲིད་པར་མ་ཟད། ནོར་འཁྲུལ་དེ་ནོར་འཁྲུལ་ཡིན་པ་ངོས་མ་ཟིན་པར་ནོར་བ་རྒྱུན་འབྱམས་སུ་སྤྱོད་པའི་ཉེན་ཁ་མཆིས་སོ།།
          ༣. 《རྒྱུགས་ལེན་རྩ་འཛིན》དང་མཐོ་རྒྱུགས་བར་གྱི་འབྲེལ་བ་དམ་པོ།
          ལོ་རེ་རེར་རྒྱལ་ཁབ་རིམ་པ་དང་ཞིང་ཆེན་རིམ་པ། ཁུལ་རིམ་པའི་སློབ་གསོ་ཞིབ་འཇུག་སྡེ་ཁག་གིས་མཐོ་རྒྱུགས་ལ་དབྱེ་ཞིབ་བྱེད་པའི་བགྲོ་གླེང་ཚོགས་འདུ་བསྡུ་བཞིན་ཡོད་ཅིང་། ཚོགས་འདུ་རུ་རྒྱུགས་གཞི་དང་རྒྱུགས་འབྲས་ལ་ཞིབ་འཇུག་བྱེད་པ་ལས་གཞན་ད་དུང་རྒྱལ་ཁབ་སློབ་གསོ་པུའུ་ཡིས་རྩོམ་སྒྲིག་བྱས་པའི《མཐོ་འབྲིང་དཀྱུས་མའི་བསླབ་ཚན་ཚད་གཞི》ལ་ཟུར་ལྟ་བྱས་ཏེ། རྒྱལ་ཡོངས་གཅིག་གྱུར་གྱི《རྒྱུགས་ལེན་རྩ་འཛིན》དང་དེ་གཞིར་བྱས་ཏེ་ཞིང་ཆེན་སོ་སོའི་མཐོ་རྒྱུགས་ཀྱི་འཆར་འགོད་དང་སློབ་ཁྲིད་ཀྱི་གནས་ཚུལ་དངོས་དང་བསྟུན་ནས་གཏན་ལ་ཕབ་པའི《རྒྱུགས་ལེན་གསལ་བཤད》གཉིས་ལ་དབྱེ་ཞིབ་བྱེད་བཞིན་ཡོད། 《རྒྱུགས་ལེན་རྩ་འཛིན》དང《རྒྱུགས་ལེན་གསལ་བཤད》ནང་དུ་རྒྱུགས་ཀྱི་ངོ་བོ་དང་ནང་དོན། རྣམ་པ་གཏན་ཕབ་ཡོད་ལ། བྱེ་བྲག་ཏུ་རྒྱུགས་བཤེར་དམིགས་ཚད་དང་། རྒྱུགས་ལེན་ཁྱབ་ཁོངས། ཤེས་བྱའི་བླང་བྱ་སོགས་རེའུ་མིག་གི་ལམ་ནས་གསལ་བོར་བྲིས་ཡོད་པ་དང་རིགས་འགྲེའི་དཔེར་བརྗོད་བཞག་སྟེ་གསལ་བཤད་བྱས་ཡོད་པས། ཞིབ་འཇུག་བྱེད་སྐབས་ཤིན་ཏུ་སྟབས་བདེ་བོ་ཡིན། 《རྒྱུགས་ལེན་རྩ་འཛིན》དང《རྒྱུགས་ལེན་གསལ་བཤད》གཉིས་ནི་རྒྱུགས་གཞི་སྒྲིག་པའི་གཞི་འཛིན་ས་ཡིན་པར་མ་ཟད་སློབ་མས་བསྐྱར་སྦྱོང་བྱེད་པའི་གཞི་འཛིན་སའང་ཡིན།
          《རྒྱུགས་ལེན་རྩ་འཛིན》དང《རྒྱུགས་ལེན་གསལ་བཤད》ནང་དུ་ཞིང་ཆེན་གྱིས་གཏན་ལ་ཕབ་པའི་སྦྱང་བྱའི་སློབ་གཞི་སྟེ་ངེས་སྦྱོང་དང་འདེམ་སྦྱོང་གསལ་བོར་བཀར་ཡོད་པ་ལས་གཞན་ད་དུང་བྱེ་བྲག་ཏུ་ནང་དོན་ལ་གོ་བ་ལེན་པའི་ཚད་ལ་བླང་བྱ་རིམ་པⅠདངⅡགཉིས་སུ་བཀར་ནས་ཤེས་བྱ་ལ་གཙོ་ཕལ་གྱི་དབྱེ་བ་ཕྱེས་ཡོད་པ་དང་།Ⅰཡི་ཁོངས་སུ་བསྡུས་པའི་ཤེས་བྱ་ནི་འབྲེལ་ཡོད་དྲི་གཞིའི་ནང་དུ་ངོས་ཟིན་པ་དང་ཐད་ཀར་བེད་སྤྱོད་བྱས་པས་ཆོག་སྟེ། དེ་ནི་བསླབ་ཚན་ཚད་གཞིའི་ནང་གི“རྒྱུས་ལེན་པ”དང“ངོས་འཛིན”དང་འདྲ།Ⅱཡི་ཁོངས་སུ་བསྡུས་པའི་ཤེས་བྱ་ནི་ནང་དོན་གསལ་བོར་ཤེས་དགོས་པར་མ་ཟད་ཤེས་བྱ་གཞན་དང་ཟུང་དུ་སྦྲེལ་ཤེས་དགོས་ལ། གནད་དོན་དངོས་སུ་དབྱེ་ཞིབ་དང་ཕྱོགས་བསྡུས། རིགས་འདེད་བྱེད་པ་སྟེ། བསླབ་ཚན་ཚད་གཞིའི་ནང་གི“གོ་བ་ལེན་པ”དང“བེད་སྤྱོད་པ”དང་འདྲ། མཐོ་འབྲིང་གི་དངོས་ལུགས་དགེ་རྒན་ཞིག་ཡིན་ཙ་ན་དེར་དབྱེ་ཞིབ་ཞིབ་མོ་བྱས་ན་ད་གཟོད་སློབ་གཞིའི་ནང་དོན་འབོར་ཆེན་ལས་བླང་དོར་གང་ལེགས་བྱས་ཏེ་དཀའ་གཙོ་གནད་དང་། གཙོ་ཕལ། བླང་བྱའི་རིམ་པ་སོགས་ཡང་དག་པར་འབྱེད་ནུས་པ་དང་། མུན་ནག་མདའ་འཕེན་ལྟར་མ་ཡིན་པར་དབྱེ་ཞིབ་དང་ཞིབ་འཇུག་བྱས་པས་ལོ་རེ་རེའི་མཐོ་རྒྱུགས་ཀྱི་ཁ་ཕྱོགས་ངེས་ཅན་ཞིག་ཀྱང་རྙེད་སྲིད་པས། སློབ་མའི་རྒྱུགས་འབྲས་ཡར་སྐྱེད་ཡོང་བ་དང་སློབ་མས་དངོས་ལུགས་སྦྱོང་བའི་སྤྲོ་བ་བསྐྱེད་པའི་ཕྱོགས་ལ་ཡང་ཕན་ཐོགས་ཆེན་པོ་ཞིག་ཡོད་ཅིང་། ལོ་གསུམ་རིང་གི་ཤེས་བྱ་འབོར་ཆེན་ལས་བླང་དོར་གང་ལེགས་བྱས་ཏེ་དགེ་སློབ་ཟུང་གི་སློབ་སྦྱོང་གི་འགན་ཁུར་སྐྱོང་བའི་ཐད་ལ་ཡང་ཕན་ནུས་གལ་ཆེན་ཡོད་པར་མ་ཟད། རྒྱུགས་འབྲས་ཇེ་བཟང་དུ་འགྲོ་བར་ཕན་པ་ངེས་ཅན་ཞིག་ཡོད་དོ།།

[3] 2015ལོར་རྔ་བ་ཁུལ་སློབ་གསོ་ཅུའུ་ཡིས་དགེ་རྒན་གསོ་སྦྱོང་འཛིན་གྲྭར་བླངས་པའི་སློབ་ཆུང་གྲངས་རིག་གི་རྒྱུགས་འབྲས།
[4] 2014ལོར་རྔ་བ་ཁུལ་གྱི་མི་རིགས་མཐོ་རིམ་སློབ་འབྲིང་གི་མཐོ་འབྲིང་ལོ་རིམ་གཉིས་པ་དང་ལོ་རིམ་དང་པོར་ཡིག་ཐོག་རྟོག་དཔྱོད་བྱས་པའི་དཔྱད་འབྲས།
[5] 2014ལོའི་མཐོ་རིམ་སློབ་གྲྭར་སློབ་མ་བསྡུ་བའི་རྒྱལ་ཡོངས་གཅིག་གྱུར་གྱི་རྒྱུགས་ལེན་རྩ་འཛིན།(རྩིས་ཚན•བསླབ་ཚན་གསར་བཅོས་ཚོད་ལྟའི་དེབ)མཐོ་རིམ་སློབ་གསོ་དཔེ་སྐྲུན་ཁང་།2014ལོར།
དཔལ་འབུམ་རྒྱལ། སི་ཁྲོན་ཞིང་ཆེན་རྔ་བ་ཁུལ་སློབ་གསོ་ཚན་རིག་ཞིབ་འཇུག་ཁང་།
